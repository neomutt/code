50enum CommandResult parse_score(const struct Command *cmd, struct Buffer *line,
51 const struct ParseContext *pc, struct ParseError *pe)
52{
53 struct Buffer *err = pe->message;
54
55 if (!MoreArgs(line))
56 {
57 buf_printf(err, _("%s: too few arguments"), cmd->name);
58 return MUTT_CMD_WARNING;
59 }
60
61 struct Score *ptr = NULL, *last = NULL;
62 char *pattern = NULL, *patchar = NULL;
63 struct Buffer *token = buf_pool_get();
64 enum CommandResult rc = MUTT_CMD_ERROR;
65
66 parse_extract_token(token, line, TOKEN_NO_FLAGS);
67 if (!MoreArgs(line))
68 {
69 buf_printf(err, _("%s: too few arguments"), cmd->name);
70 goto done;
71 }
72 pattern = buf_strdup(token);
73 parse_extract_token(token, line, TOKEN_NO_FLAGS);
74 if (MoreArgs(line))
75 {
76 buf_printf(err, _("%s: too many arguments"), cmd->name);
77 goto done;
78 }
79
80 /* look for an existing entry and update the value, else add it to the end
81 * of the list */
82 for (ptr = ScoreList, last = NULL; ptr; last = ptr, ptr = ptr->next)
83 if (mutt_str_equal(pattern, ptr->str))
84 break;
85
86 if (ptr)
87 {
88 /* 'token' arg was cleared and 'pattern' holds the only reference;
89 * as here 'ptr' != NULL -> update the value only in which case
90 * ptr->str already has the string, so pattern should be freed. */
91 FREE(&pattern);
92 }
93 else
94 {
95 struct MailboxView *mv_cur = get_current_mailbox_view();
96 struct PatternList *pat = mutt_pattern_comp(mv_cur, pattern, MUTT_PC_NO_FLAGS, err);
97 if (!pat)
98 {
99 goto done;
100 }
101 ptr = MUTT_MEM_CALLOC(1, struct Score);
102 if (last)
103 last->next = ptr;
104 else
105 ScoreList = ptr;
106 ptr->pat = pat;
107 ptr->str = pattern;
108 pattern = NULL;
109 }
110
111 patchar = token->data;
112 if (*patchar == '=')
113 {
114 ptr->exact = true;
115 patchar++;
116 }
117
118 if (!mutt_str_atoi_full(patchar, &ptr->val))
119 {
120 buf_strcpy(err, _("Error: score: invalid number"));
121 goto done;
122 }
123 OptNeedRescore = true;
124
125 rc = MUTT_CMD_SUCCESS;
126
127done:
128 buf_pool_release(&token);
129 FREE(&pattern);
130 return rc;
131}

139enum CommandResult parse_unscore(const struct Command *cmd, struct Buffer *line,
140 const struct ParseContext *pc, struct ParseError *pe)
141{
142 struct Buffer *err = pe->message;
143
144 if (!MoreArgs(line))
145 {
146 buf_printf(err, _("%s: too few arguments"), cmd->name);
147 return MUTT_CMD_WARNING;
148 }
149
150 struct Score *tmp = NULL, *last = NULL;
151 struct Buffer *token = buf_pool_get();
152
153 while (MoreArgs(line))
154 {
155 parse_extract_token(token, line, TOKEN_NO_FLAGS);
156 if (mutt_str_equal("*", buf_string(token)))
157 {
158 for (tmp = ScoreList; tmp;)
159 {
160 last = tmp;
161 tmp = tmp->next;
162 mutt_pattern_free(&last->pat);
163 FREE(&last);
164 }
165 ScoreList = NULL;
166 }
167 else
168 {
169 for (tmp = ScoreList; tmp; last = tmp, tmp = tmp->next)
170 {
171 if (mutt_str_equal(buf_string(token), tmp->str))
172 {
173 if (last)
174 last->next = tmp->next;
175 else
176 ScoreList = tmp->next;
177 mutt_pattern_free(&tmp->pat);
178 FREE(&tmp);
179 /* there should only be one score per pattern, so we can stop here */
180 break;
181 }
182 }
183 }
184 }
185
186 OptNeedRescore = true;
187 buf_pool_release(&token);
188 return MUTT_CMD_SUCCESS;
189}
