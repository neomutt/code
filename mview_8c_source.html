49void mview_free(struct MailboxView **ptr)
50{
51 if (!ptr || !*ptr)
52 return;
53
54 struct MailboxView *mv = *ptr;
55
56 struct EventMview ev_m = { mv };
57 mutt_debug(LL_NOTIFY, "NT_MVIEW_DELETE: %p\n", (void *) mv);
58 notify_send(mv->notify, NT_MVIEW, NT_MVIEW_DELETE, &ev_m);
59
60 if (mv->mailbox)
61 {
62 notify_observer_remove(mv->mailbox->notify, mview_mailbox_observer, mv);
63
64 // Disconnect the Emails before freeing the Threads
65 for (int i = 0; i < mv->mailbox->msg_count; i++)
66 {
67 struct Email *e = mv->mailbox->emails[i];
68 if (!e)
69 continue;
70 e->thread = NULL;
71 e->threaded = false;
72 }
73 }
74
75 mutt_thread_ctx_free(&mv->threads);
76 notify_free(&mv->notify);
77 FREE(&mv->pattern);
78 mutt_pattern_free(&mv->limit_pattern);
79
80 *ptr = NULL;
81 FREE(&mv);
82}

138void mview_update(struct MailboxView *mv)
139{
140 if (!mv || !mv->mailbox)
141 return;
142
143 struct Mailbox *m = mv->mailbox;
144
145 mutt_hash_free(&m->subj_hash);
146 mutt_hash_free(&m->id_hash);
147
148 /* reset counters */
149 m->msg_unread = 0;
150 m->msg_flagged = 0;
151 m->msg_new = 0;
152 m->msg_deleted = 0;
153 m->msg_tagged = 0;
154 m->vcount = 0;
155 m->changed = false;
156
157 mutt_clear_threads(mv->threads);
158
159 const bool c_score = cs_subset_bool(NeoMutt->sub, "score");
160 struct Email *e = NULL;
161 for (int msgno = 0; msgno < m->msg_count; msgno++)
162 {
163 e = m->emails[msgno];
164 if (!e)
165 continue;
166
167 if (WithCrypto)
168 {
169 /* NOTE: this _must_ be done before the check for mailcap! */
170 e->security = crypt_query(e->body);
171 }
172
173 if (mview_has_limit(mv))
174 {
175 e->vnum = -1;
176 }
177 else
178 {
179 m->v2r[m->vcount] = msgno;
180 e->vnum = m->vcount++;
181 }
182 e->msgno = msgno;
183
184 if (e->env->supersedes)
185 {
186 struct Email *e2 = NULL;
187
188 if (!m->id_hash)
189 m->id_hash = mutt_make_id_hash(m);
190
191 e2 = mutt_hash_find(m->id_hash, e->env->supersedes);
192 if (e2)
193 {
194 e2->superseded = true;
195 if (c_score)
196 mutt_score_message(mv->mailbox, e2, true);
197 }
198 }
199
200 /* add this message to the hash tables */
201 if (m->id_hash && e->env->message_id)
202 mutt_hash_insert(m->id_hash, e->env->message_id, e);
203 if (m->subj_hash && e->env->real_subj)
204 mutt_hash_insert(m->subj_hash, e->env->real_subj, e);
205 mutt_label_hash_add(m, e);
206
207 if (c_score)
208 mutt_score_message(mv->mailbox, e, false);
209
210 if (e->changed)
211 m->changed = true;
212 if (e->flagged)
213 m->msg_flagged++;
214 if (e->deleted)
215 m->msg_deleted++;
216 if (e->tagged)
217 m->msg_tagged++;
218 if (!e->read)
219 {
220 m->msg_unread++;
221 if (!e->old)
222 m->msg_new++;
223 }
224 }
225
226 /* rethread from scratch */
227 mutt_sort_headers(mv, true);
228}

234static void update_tables(struct MailboxView *mv)
235{
236 if (!mv || !mv->mailbox)
237 return;
238
239 struct Mailbox *m = mv->mailbox;
240
241 int i, j, padding;
242
243 /* update memory to reflect the new state of the mailbox */
244 m->vcount = 0;
245 mv->vsize = 0;
246 m->msg_tagged = 0;
247 m->msg_deleted = 0;
248 m->msg_new = 0;
249 m->msg_unread = 0;
250 m->changed = false;
251 m->msg_flagged = 0;
252 padding = mx_msg_padding_size(m);
253 const bool c_maildir_trash = cs_subset_bool(NeoMutt->sub, "maildir_trash");
254 for (i = 0, j = 0; i < m->msg_count; i++)
255 {
256 if (!m->emails[i])
257 break;
258 if (!m->emails[i]->quasi_deleted &&
259 (!m->emails[i]->deleted || ((m->type == MUTT_MAILDIR) && c_maildir_trash)))
260 {
261 if (i != j)
262 {
263 m->emails[j] = m->emails[i];
264 m->emails[i] = NULL;
265 }
266 m->emails[j]->msgno = j;
267 if (m->emails[j]->vnum != -1)
268 {
269 m->v2r[m->vcount] = j;
270 m->emails[j]->vnum = m->vcount++;
271 struct Body *b = m->emails[j]->body;
272 mv->vsize += b->length + b->offset - b->hdr_offset + padding;
273 }
274
275 m->emails[j]->changed = false;
276 m->emails[j]->env->changed = false;
277
278 if ((m->type == MUTT_MAILDIR) && c_maildir_trash)
279 {
280 if (m->emails[j]->deleted)
281 m->msg_deleted++;
282 }
283
284 if (m->emails[j]->tagged)
285 m->msg_tagged++;
286 if (m->emails[j]->flagged)
287 m->msg_flagged++;
288 if (!m->emails[j]->read)
289 {
290 m->msg_unread++;
291 if (!m->emails[j]->old)
292 m->msg_new++;
293 }
294
295 j++;
296 }
297 else
298 {
299 if ((m->type == MUTT_NOTMUCH) || (m->type == MUTT_MH) ||
300 (m->type == MUTT_MAILDIR) || (m->type == MUTT_IMAP))
301 {
302 mailbox_size_sub(m, m->emails[i]);
303 }
304 /* remove message from the hash tables */
305 if (m->subj_hash && m->emails[i]->env->real_subj)
306 mutt_hash_delete(m->subj_hash, m->emails[i]->env->real_subj, m->emails[i]);
307 if (m->id_hash && m->emails[i]->env->message_id)
308 mutt_hash_delete(m->id_hash, m->emails[i]->env->message_id, m->emails[i]);
309 mutt_label_hash_remove(m, m->emails[i]);
310
311 if (m->type == MUTT_IMAP)
312 imap_notify_delete_email(m, m->emails[i]);
313
314 mailbox_gc_add(m->emails[i]);
315 m->emails[i] = NULL;
316 }
317 }
318 m->msg_count = j;
319}

324int mview_mailbox_observer(struct NotifyCallback *nc)
325{
326 if (nc->event_type != NT_MAILBOX)
327 return 0;
328 if (!nc->global_data)
329 return -1;
330
331 struct MailboxView *mv = nc->global_data;
332
333 switch (nc->event_subtype)
334 {
335 case NT_MAILBOX_DELETE:
336 mutt_clear_threads(mv->threads);
337 mview_clean(mv);
338 break;
339 case NT_MAILBOX_INVALID:
340 mview_update(mv);
341 break;
342 case NT_MAILBOX_UPDATE:
343 update_tables(mv);
344 break;
345 case NT_MAILBOX_RESORT:
346 mutt_sort_headers(mv, true);
347 break;
348 default:
349 return 0;
350 }
351
352 mutt_debug(LL_DEBUG5, "mailbox done\n");
353 return 0;
354}

480bool mutt_limit_current_thread(struct MailboxView *mv, struct Email *e)
481{
482 if (!mv || !mv->mailbox || !e)
483 return false;
484
485 struct Mailbox *m = mv->mailbox;
486
487 struct MuttThread *me = top_of_thread(e);
488 if (!me)
489 return false;
490
491 m->vcount = 0;
492 mv->vsize = 0;
493 mv->collapsed = false;
494
495 for (int i = 0; i < m->msg_count; i++)
496 {
497 e = m->emails[i];
498 if (!e)
499 break;
500
501 e->vnum = -1;
502 e->visible = false;
503 e->collapsed = false;
504 e->num_hidden = 0;
505
506 if (top_of_thread(e) == me)
507 {
508 struct Body *body = e->body;
509
510 e->vnum = m->vcount;
511 e->visible = true;
512 m->v2r[m->vcount] = i;
513 m->vcount++;
514 mv->vsize += (body->length + body->offset - body->hdr_offset);
515 }
516 }
517 return true;
