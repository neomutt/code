357static int up_n_lines(int nlines, struct Line *info, int cur, bool hiding)
358{
359 while ((cur > 0) && (nlines > 0))
360 {
361 cur--;
362 if (!hiding || !COLOR_QUOTED(info[cur].cid))
363 nlines--;
364 }
365

376bool jump_to_bottom(struct PagerPrivateData *priv, struct PagerView *pview)
377{
378 if (!(priv->lines[priv->cur_line].offset < (priv->st.st_size - 1)))
379 {
380 return false;
381 }
382
383 int line_num = priv->cur_line;
384 /* make sure the types are defined to the end of file */
385 while (display_line(priv->fp, &priv->bytes_read, &priv->lines, line_num,
386 &priv->lines_used, &priv->lines_max,
387 priv->has_types | (pview->flags & MUTT_PAGER_NOWRAP),
388 &priv->quote_list, &priv->q_level, &priv->force_redraw,
389 &priv->search_re, priv->pview->win_pager, &priv->ansi_list) == 0)
390 {
391 line_num++;
392 }
393 priv->top_line = up_n_lines(priv->pview->win_pager->state.rows, priv->lines,
394 priv->lines_used, priv->hide_quoted);
395 notify_send(priv->notify, NT_PAGER, NT_PAGER_VIEW, priv);
396 return true;
397}

578static int op_pager_search(struct IndexSharedData *shared,
579 struct PagerPrivateData *priv, const struct KeyEvent *event)
580{
581 struct PagerView *pview = priv->pview;
582
583 int rc = FR_NO_ACTION;
584 struct Buffer *buf = buf_pool_get();
585
586 buf_strcpy(buf, priv->search_str);
587 const int op = event->op;
588 if (mw_get_field(((op == OP_SEARCH) || (op == OP_SEARCH_NEXT)) ? _("Search for: ") : _("Reverse search for: "),
589 buf, MUTT_COMP_CLEAR, HC_PATTERN, &CompletePatternOps, NULL) != 0)
590 {
591 goto done;
592 }
593
594 if (mutt_str_equal(buf_string(buf), priv->search_str))
595 {
596 if (priv->search_compiled)
597 {
598 struct KeyEvent event_s = { 0, OP_NULL };
599
600 /* do an implicit search-next */
601 if (op == OP_SEARCH)
602 event_s.op = OP_SEARCH_NEXT;
603 else
604 event_s.op = OP_SEARCH_OPPOSITE;
605
606 priv->wrapped = false;
607 op_pager_search_next(shared, priv, &event_s);
608 }
609 }
610
611 if (buf_is_empty(buf))
612 goto done;
613
614 mutt_str_copy(priv->search_str, buf_string(buf), sizeof(priv->search_str));
615
616 /* leave search_back alone if op == OP_SEARCH_NEXT */
617 if (op == OP_SEARCH)
618 priv->search_back = false;
619 else if (op == OP_SEARCH_REVERSE)
620 priv->search_back = true;
621
622 if (priv->search_compiled)
623 {
624 regfree(&priv->search_re);
625 for (size_t i = 0; i < priv->lines_used; i++)
626 {
627 FREE(&(priv->lines[i].search));
628 priv->lines[i].search_arr_size = -1;
629 }
630 }
631
632 uint16_t rflags = mutt_mb_is_lower(priv->search_str) ? REG_ICASE : 0;
633 int err = REG_COMP(&priv->search_re, priv->search_str, REG_NEWLINE | rflags);
634 if (err != 0)
635 {
636 regerror(err, &priv->search_re, buf->data, buf->dsize);
637 mutt_error("%s", buf_string(buf));
638 for (size_t i = 0; i < priv->lines_max; i++)
639 {
640 /* cleanup */
641 FREE(&(priv->lines[i].search));
642 priv->lines[i].search_arr_size = -1;
643 }
644 priv->search_flag = 0;
645 priv->search_compiled = false;
646 }
647 else
648 {
649 priv->search_compiled = true;
650 /* update the search pointers */
651 int line_num = 0;
652 while (display_line(priv->fp, &priv->bytes_read, &priv->lines, line_num,
653 &priv->lines_used, &priv->lines_max,
654 MUTT_SEARCH | (pview->flags & MUTT_PAGER_NOWRAP) | priv->has_types,
655 &priv->quote_list, &priv->q_level, &priv->force_redraw,
656 &priv->search_re, priv->pview->win_pager, &priv->ansi_list) == 0)
657 {
658 line_num++;
659 }
660
661 if (priv->search_back)
662 {

783 (priv->has_types && !COLOR_QUOTED(priv->lines[i].cid))) &&
784 !priv->lines[i].cont_line && (priv->lines[i].search_arr_size > 0))
785 {
786 break;
787 }
788 }
789
790 const bool c_wrap_search = cs_subset_bool(NeoMutt->sub, "wrap_search");
791 if (i >= 0)
792 {
793 priv->top_line = i;
794 }
795 else if (priv->wrapped || !c_wrap_search)

797 mutt_error(_("Not found"));
798 }
799 else
800 {
801 mutt_message(_("Search wrapped to bottom"));

804 }
805 }
806
807 if (priv->lines[priv->top_line].search_arr_size > 0)
808 {
809 priv->search_flag = MUTT_SEARCH;
810 /* give some context for search results */
811 if (priv->top_line - priv->searchctx > 0)
812 priv->top_line -= priv->searchctx;
813 }
814
815 notify_send(priv->notify, NT_PAGER, NT_PAGER_VIEW, priv);

817 }
