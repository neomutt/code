110bool mutt_body_cmp_strict(const struct Body *b1, const struct Body *b2)
111{

114

117 !mutt_str_equal(b1->description, b2->description) ||
118 !mutt_param_cmp_strict(&b1->parameter, &b2->parameter) || (b1->length != b2->length))
