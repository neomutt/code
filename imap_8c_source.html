91enum CommandResult parse_subscribe_to(const struct Command *cmd, struct Buffer *line,
92 const struct ParseContext *pc, struct ParseError *pe)
93{
94 struct Buffer *err = pe->message;
95
96 if (!MoreArgs(line))
97 {
98 buf_printf(err, _("%s: too few arguments"), cmd->name);
99 return MUTT_CMD_WARNING;
100 }
101
102 struct Buffer *token = buf_pool_get();
103 enum CommandResult rc = MUTT_CMD_WARNING;
104
105 buf_reset(err);
106
107 parse_extract_token(token, line, TOKEN_NO_FLAGS);
108
109 if (MoreArgs(line))
110 {
111 buf_printf(err, _("%s: too many arguments"), cmd->name);
112 goto done;
113 }
114
115 // Expand and subscribe
116 expand_path(token, false);
117 if (imap_subscribe(buf_string(token), true) != 0)
118 {
119 buf_printf(err, _("Could not subscribe to %s"), buf_string(token));
120 rc = MUTT_CMD_ERROR;
121 goto done;
122 }
123
124 mutt_message(_("Subscribed to %s"), buf_string(token));
125 rc = MUTT_CMD_SUCCESS;
126
127done:
128 buf_pool_release(&token);
129 return rc;
130}

231static char *get_flags(struct ListHead *hflags, char *s)
232{
233 /* sanity-check string */
234 const size_t plen = mutt_istr_startswith(s, "FLAGS");
235 if (plen == 0)
236 {
237 mutt_debug(LL_DEBUG1, "not a FLAGS response: %s\n", s);
238 return NULL;
239 }
240 s += plen;
241 SKIPWS(s);
242 if (*s != '(')
243 {
244 mutt_debug(LL_DEBUG1, "bogus FLAGS response: %s\n", s);
245 return NULL;
246 }
247
248 /* update caller's flags handle */
249 while (*s && (*s != ')'))
250 {
251 s++;
252 SKIPWS(s);
253 const char *flag_word = s;
254 while (*s && (*s != ')') && !mutt_isspace(*s))
255 s++;
256 const char ctmp = *s;
257 *s = '\0';
258 if (*flag_word)
259 mutt_list_insert_tail(hflags, mutt_str_dup(flag_word));
260 *s = ctmp;
261 }
262
263 /* note bad flags response */
264 if (*s != ')')
265 {
266 mutt_debug(LL_DEBUG1, "Unterminated FLAGS response: %s\n", s);
267 mutt_list_free(hflags);
268
269 return NULL;
270 }
271
272 s++;
273
274 return s;
275}

332static int select_email_uids(struct Email **emails, int num_emails, enum MessageType flag,
333 bool changed, bool invert, struct UidArray *uida)
334{
335 if (!emails || !uida)
336 return -1;
337
338 for (int i = 0; i < num_emails; i++)
339 {
340 struct Email *e = emails[i];
341 if (changed && !e->changed)
342 continue;
343
344 /* don't include pending expunged messages.
345 *
346 * TODO: can we unset active in cmd_parse_expunge() and
347 * cmd_parse_vanished() instead of checking for index != INT_MAX. */
348 if (!e || !e->active || (e->index == INT_MAX))
349 continue;
350
351 struct ImapEmailData *edata = imap_edata_get(e);
352
353 bool match = false;
354 switch (flag)
355 {
356 case MUTT_DELETED:
357 if (e->deleted != edata->deleted)
358 match = invert ^ e->deleted;
359 break;
360 case MUTT_FLAG:
361 if (e->flagged != edata->flagged)
362 match = invert ^ e->flagged;
363 break;
364 case MUTT_OLD:
365 if (e->old != edata->old)
366 match = invert ^ e->old;
367 break;
368 case MUTT_READ:
369 if (e->read != edata->read)
370 match = invert ^ e->read;
371 break;
372 case MUTT_REPLIED:
373 if (e->replied != edata->replied)
374 match = invert ^ e->replied;
375 break;
376 case MUTT_TRASH:
377 if (e->deleted && !e->purge)
378 match = true;
379 break;
380 default:
381 break;
382 }
383
384 if (match)
385 ARRAY_ADD(uida, edata->uid);
386 }
387
388 return ARRAY_SIZE(uida);
389}

402static int sync_helper(struct Mailbox *m, struct Email **emails, int num_emails,
403 AclFlags right, enum MessageType flag, const char *name)
404{
405 struct ImapAccountData *adata = imap_adata_get(m);
406 struct ImapMboxData *mdata = imap_mdata_get(m);
407 if (!adata || !mdata)
408 return -1;
409
410 if ((m->rights & right) == 0)
411 return 0;
412
413 if ((right == MUTT_ACL_WRITE) && !imap_has_flag(&mdata->flags, name))
414 return 0;
415
416 int count = 0;
417 char buf[1024] = { 0 };
418
419 struct UidArray uida = ARRAY_HEAD_INITIALIZER;
420
421 // Set the flag (+FLAGS) on matching emails
422 select_email_uids(emails, num_emails, flag, true, false, &uida);
423 snprintf(buf, sizeof(buf), "+FLAGS.SILENT (%s)", name);
424 int rc = imap_exec_msg_set(adata, "UID STORE", buf, &uida);
425 if (rc < 0)
426 return rc;
427 count += rc;
428 ARRAY_FREE(&uida);
429
430 // Clear the flag (-FLAGS) on non-matching emails
431 select_email_uids(emails, num_emails, flag, true, true, &uida);
432 buf[0] = '-';
433 rc = imap_exec_msg_set(adata, "UID STORE", buf, &uida);
434 if (rc < 0)
435 return rc;
436 count += rc;
437 ARRAY_FREE(&uida);
438
439 return count;
440}

635static void imap_logout(struct ImapAccountData *adata)
636{
637 if (adata->status != IMAP_FATAL)
638 {
639 /* we set status here to let imap_handle_untagged know we _expect_ to
640 * receive a bye response (so it doesn't freak out and close the conn) */

642 {

653 ; // do nothing
654 }
655 }
656 mutt_socket_close(adata->conn);
657 adata->state = IMAP_DISCONNECTED;

702int imap_read_literal(FILE *fp, struct ImapAccountData *adata,
703 unsigned long bytes, struct Progress *progress)
704{
705 char c;
706 bool r = false;
707 struct Buffer buf = { 0 }; // Do not allocate, maybe it won't be used
708
709 const short c_debug_level = cs_subset_number(NeoMutt->sub, "debug_level");
710 if (c_debug_level >= IMAP_LOG_LTRL)
711 buf_alloc(&buf, bytes + 1);
712
713 mutt_debug(LL_DEBUG2, "reading %lu bytes\n", bytes);
714
715 for (unsigned long pos = 0; pos < bytes; pos++)
716 {
717 if (mutt_socket_readchar(adata->conn, &c) != 1)
718 {
719 mutt_debug(LL_DEBUG1, "error during read, %lu bytes read\n", pos);
720 adata->status = IMAP_FATAL;
721
722 buf_dealloc(&buf);
723 return -1;
724 }
725
726 if (r && (c != '\n'))
727 fputc('\r', fp);
728
729 if (c == '\r')
730 {
731 r = true;
732 continue;
733 }
734 else
735 {
736 r = false;
737 }
738
739 fputc(c, fp);
740
741 if ((pos % 1024) == 0)
742 progress_update(progress, pos, -1);
743 if (c_debug_level >= IMAP_LOG_LTRL)
744 buf_addch(&buf, c);
745 }
746
747 if (c_debug_level >= IMAP_LOG_LTRL)
748 {
749 mutt_debug(IMAP_LOG_LTRL, "\n%s", buf.data);
750 buf_dealloc(&buf);
751 }
752 return 0;
753}

781void imap_expunge_mailbox(struct Mailbox *m, bool resort)
782{
783 struct ImapAccountData *adata = imap_adata_get(m);
784 struct ImapMboxData *mdata = imap_mdata_get(m);
785 if (!adata || !mdata)
786 return;
787
788 struct Email *e = NULL;
789
790#ifdef USE_HCACHE
791 imap_hcache_open(adata, mdata, false);
792#endif
793
794 for (int i = 0; i < m->msg_count; i++)
795 {
796 e = m->emails[i];
797 if (!e)
798 break;
799
800 if (e->index == INT_MAX)
801 {
802 mutt_debug(LL_DEBUG2, "Expunging message UID %u\n", imap_edata_get(e)->uid);
803
804 e->deleted = true;
805
806 imap_cache_del(m, e);
807#ifdef USE_HCACHE
808 imap_hcache_del(mdata, imap_edata_get(e)->uid);
809#endif
810
811 mutt_hash_int_delete(mdata->uid_hash, imap_edata_get(e)->uid, e);
812
813 imap_edata_free((void **) &e->edata);
814 }
815 else
816 {
817 /* NeoMutt has several places where it turns off e->active as a
818 * hack. For example to avoid FLAG updates, or to exclude from
819 * imap_exec_msg_set.
820 *
821 * Unfortunately, when a reopen is allowed and the IMAP_EXPUNGE_PENDING
822 * flag becomes set (e.g. a flag update to a modified header),
823 * this function will be called by imap_cmd_finish().
824 *
825 * The ctx_update_tables() will free and remove these "inactive" headers,
826 * despite that an EXPUNGE was not received for them.
827 * This would result in memory leaks and segfaults due to dangling
828 * pointers in the msn_index and uid_hash.
829 *
830 * So this is another hack to work around the hacks. We don't want to
831 * remove the messages, so make sure active is on. */
832 e->active = true;
833 }
834 }
835
836#ifdef USE_HCACHE
837 imap_hcache_close(mdata);
838#endif
839
840 mailbox_changed(m, NT_MAILBOX_UPDATE);
841 if (resort)
842 {
843 mailbox_changed(m, NT_MAILBOX_RESORT);
844 }
845}

853int imap_open_connection(struct ImapAccountData *adata)
854{
855 if (mutt_socket_open(adata->conn) < 0)
856 return -1;
857
858 adata->state = IMAP_CONNECTED;
859
860 if (imap_cmd_step(adata) != IMAP_RES_OK)
861 {
862 imap_close_connection(adata);
863 return -1;
864 }
865
866 if (mutt_istr_startswith(adata->buf, "* OK"))
867 {
868 if (!mutt_istr_startswith(adata->buf, "* OK [CAPABILITY") && check_capabilities(adata))
869 {
870 goto bail;
871 }
872#ifdef USE_SSL
873 /* Attempt STARTTLS if available and desired. */
874 const bool c_ssl_force_tls = cs_subset_bool(NeoMutt->sub, "ssl_force_tls");
875 if ((adata->conn->ssf == 0) &&
876 (c_ssl_force_tls || (adata->capabilities & IMAP_CAP_STARTTLS)))
877 {
878 enum QuadOption ans;
879
880 if (c_ssl_force_tls)
881 {
882 ans = MUTT_YES;
883 }
884 else if ((ans = query_quadoption(_("Secure connection with TLS?"),
885 NeoMutt->sub, "ssl_starttls")) == MUTT_ABORT)
886 {
887 goto bail;
888 }
889 if (ans == MUTT_YES)
890 {
891 enum ImapExecResult rc = imap_exec(adata, "STARTTLS", IMAP_CMD_SINGLE);
892 // Clear any data after the STARTTLS acknowledgement
893 mutt_socket_empty(adata->conn);
894
895 if (rc == IMAP_EXEC_FATAL)
896 goto bail;
897 if (rc != IMAP_EXEC_ERROR)
898 {
899 if (mutt_ssl_starttls(adata->conn))
900 {
901 mutt_error(_("Could not negotiate TLS connection"));
902 goto bail;
903 }
904 else
905 {
906 /* RFC2595 demands we recheck CAPABILITY after TLS completes. */
907 if (imap_exec(adata, "CAPABILITY", IMAP_CMD_NO_FLAGS) != IMAP_EXEC_SUCCESS)
908 goto bail;
909 }
910 }
911 }
912 }
913
914 if (c_ssl_force_tls && (adata->conn->ssf == 0))
915 {
916 mutt_error(_("Encrypted connection unavailable"));
917 goto bail;
918 }
919#endif
920 }
921 else if (mutt_istr_startswith(adata->buf, "* PREAUTH"))
922 {
923#ifdef USE_SSL
924 /* Unless using a secure $tunnel, an unencrypted PREAUTH response may be a
925 * MITM attack. The only way to stop "STARTTLS" MITM attacks is via
926 * $ssl_force_tls: an attacker can easily spoof "* OK" and strip the
927 * STARTTLS capability. So consult $ssl_force_tls, not $ssl_starttls, to
928 * decide whether to abort. Note that if using $tunnel and
929 * $tunnel_is_secure, adata->conn->ssf will be set to 1. */
930 const bool c_ssl_force_tls = cs_subset_bool(NeoMutt->sub, "ssl_force_tls");
931 if ((adata->conn->ssf == 0) && c_ssl_force_tls)
932 {
933 mutt_error(_("Encrypted connection unavailable"));
934 goto bail;
935 }
936#endif
937
938 adata->state = IMAP_AUTHENTICATED;
939 if (check_capabilities(adata) != 0)
940 goto bail;
941 FREE(&adata->capstr);
942 }
943 else
944 {
945 imap_error("imap_open_connection()", adata->buf);
946 goto bail;
947 }
948
949 return 0;
950
951bail:
952 imap_close_connection(adata);
953 FREE(&adata->capstr);
954 return -1;
955}

1038int imap_sync_message_for_copy(struct Mailbox *m, struct Email *e,
1039 struct Buffer *cmd, enum QuadOption *err_continue)
1040{
1041 struct ImapAccountData *adata = imap_adata_get(m);
1042 struct ImapEmailData *edata = imap_edata_get(e);
1043
1044 if (!adata || (adata->mailbox != m) || !e)
1045 return -1;
1046
1047 if (!compare_flags_for_copy(e))
1048 {
1049 if (e->deleted == edata->deleted)
1050 e->changed = false;
1051 return 0;
1052 }
1053
1054 buf_printf(cmd, "UID STORE %u", edata->uid);
1055
1056 struct Buffer *flags = buf_pool_get();
1057
1058 set_flag(m, MUTT_ACL_SEEN, e->read, "\\Seen ", flags);
1059 set_flag(m, MUTT_ACL_WRITE, e->old, "Old ", flags);
1060 set_flag(m, MUTT_ACL_WRITE, e->flagged, "\\Flagged ", flags);
1061 set_flag(m, MUTT_ACL_WRITE, e->replied, "\\Answered ", flags);
1062 set_flag(m, MUTT_ACL_DELETE, edata->deleted, "\\Deleted ", flags);
1063
1064 if (m->rights & MUTT_ACL_WRITE)
1065 {
1066 /* restore system flags */
1067 if (edata->flags_system)
1068 buf_addstr(flags, edata->flags_system);
1069
1070 /* set custom flags */
1071 struct Buffer *tags = buf_pool_get();
1072 driver_tags_get_with_hidden(&e->tags, tags);
1073 if (!buf_is_empty(tags))
1074 buf_addstr(flags, buf_string(tags));
1075 buf_pool_release(&tags);
1076 }
1077
1078 mutt_str_remove_trailing_ws(flags->data);
1079 buf_fix_dptr(flags);
1080
1081 /* UW-IMAP is OK with null flags, Cyrus isn't. The only solution is to
1082 * explicitly revoke all system flags (if we have permission) */
1083 if (buf_is_empty(flags))
1084 {
1085 set_flag(m, MUTT_ACL_SEEN, true, "\\Seen ", flags);
1086 set_flag(m, MUTT_ACL_WRITE, true, "Old ", flags);
1087 set_flag(m, MUTT_ACL_WRITE, true, "\\Flagged ", flags);
1088 set_flag(m, MUTT_ACL_WRITE, true, "\\Answered ", flags);
1089 set_flag(m, MUTT_ACL_DELETE, !edata->deleted, "\\Deleted ", flags);
1090
1091 /* erase custom flags */
1092 if ((m->rights & MUTT_ACL_WRITE) && edata->flags_remote)
1093 buf_addstr(flags, edata->flags_remote);
1094
1095 mutt_str_remove_trailing_ws(flags->data);
1096 buf_fix_dptr(flags);
1097
1098 buf_addstr(cmd, " -FLAGS.SILENT (");
1099 }
1100 else
1101 {
1102 buf_addstr(cmd, " FLAGS.SILENT (");
1103 }
1104
1105 buf_addstr(cmd, buf_string(flags));
1106 buf_addstr(cmd, ")");
1107
1108 int rc = -1;
1109
1110 /* after all this it's still possible to have no flags, if you
1111 * have no ACL rights */
1112 if (!buf_is_empty(flags) &&
1113 (imap_exec(adata, cmd->data, IMAP_CMD_NO_FLAGS) != IMAP_EXEC_SUCCESS) &&
1114 err_continue && (*err_continue != MUTT_YES))
1115 {
1116 *err_continue = imap_continue("imap_sync_message: STORE failed", adata->buf);
1117 if (*err_continue != MUTT_YES)
1118 goto done;
1119 }
1120
1121 /* server have now the updated flags */
1122 FREE(&edata->flags_remote);
1123 struct Buffer *flags_remote = buf_pool_get();
1124 driver_tags_get_with_hidden(&e->tags, flags_remote);
1125 edata->flags_remote = buf_strdup(flags_remote);
1126 buf_pool_release(&flags_remote);
1127
1128 if (e->deleted == edata->deleted)
1129 e->changed = false;
1130
1131 rc = 0;
1132
1133done:
1134 buf_pool_release(&flags);
1135 return rc;
1136}

1144enum MxStatus imap_check_mailbox(struct Mailbox *m, bool force)
1145{
1146 if (!m || !m->account)
1147 return MX_STATUS_ERROR;
1148
1149 struct ImapAccountData *adata = imap_adata_get(m);
1150 struct ImapMboxData *mdata = imap_mdata_get(m);
1151 if (!adata || !mdata)
1152 return MX_STATUS_ERROR;
1153
1154 /* overload keyboard timeout to avoid many mailbox checks in a row.
1155 * Most users don't like having to wait exactly when they press a key. */
1156 int rc = 0;
1157
1158 /* try IDLE first, unless force is set */
1159 const bool c_imap_idle = cs_subset_bool(NeoMutt->sub, "imap_idle");
1160 const short c_imap_keep_alive = cs_subset_number(NeoMutt->sub, "imap_keep_alive");
1161 if (!force && c_imap_idle && (adata->capabilities & IMAP_CAP_IDLE) &&
1162 ((adata->state != IMAP_IDLE) || (mutt_date_now() >= adata->lastread + c_imap_keep_alive)))
1163 {
1164 if (imap_cmd_idle(adata) < 0)
1165 return MX_STATUS_ERROR;
1166 }
1167 if (adata->state == IMAP_IDLE)
1168 {
1169 while ((rc = mutt_socket_poll(adata->conn, 0)) > 0)
1170 {
1171 if (imap_cmd_step(adata) != IMAP_RES_CONTINUE)
1172 {
1173 mutt_debug(LL_DEBUG1, "Error reading IDLE response\n");
1174 return MX_STATUS_ERROR;
1175 }
1176 }
1177 if (rc < 0)
1178 {
1179 mutt_debug(LL_DEBUG1, "Poll failed, disabling IDLE\n");
1180 adata->capabilities &= ~IMAP_CAP_IDLE; // Clear the flag
1181 }
1182 }
1183
1184 const short c_timeout = cs_subset_number(NeoMutt->sub, "timeout");
1185 if ((force || ((adata->state != IMAP_IDLE) && (mutt_date_now() >= adata->lastread + c_timeout))) &&
1186 (imap_exec(adata, "NOOP", IMAP_CMD_POLL) != IMAP_EXEC_SUCCESS))
1187 {
1188 return MX_STATUS_ERROR;
1189 }
1190
1191 /* We call this even when we haven't run NOOP in case we have pending
1192 * changes to process, since we can reopen here. */
1193 imap_cmd_finish(adata);
1194
1195 enum MxStatus check = MX_STATUS_OK;
1196 if (mdata->check_status & IMAP_EXPUNGE_PENDING)
1197 check = MX_STATUS_REOPENED;
1198 else if (mdata->check_status & IMAP_NEWMAIL_PENDING)
1199 check = MX_STATUS_NEW_MAIL;
1200 else if (mdata->check_status & IMAP_FLAGS_PENDING)
1201 check = MX_STATUS_FLAGS;
1202 else if (rc < 0)
1203 check = MX_STATUS_ERROR;
1204
1205 mdata->check_status = IMAP_OPEN_NO_FLAGS;
1206
1207 if (force)
1208 m->last_checked = 0; // force a check on the next mx_mbox_check() call
1209 return check;
1210}

1219static int imap_status(struct ImapAccountData *adata, struct ImapMboxData *mdata, bool queue)
1220{
1221 char *uidvalidity_flag = NULL;
1222 char cmd[2048] = { 0 };
1223
1224 if (!adata || !mdata)
1225 return -1;
1226
1227 /* Don't issue STATUS on the selected mailbox, it will be NOOPed or
1228 * IDLEd elsewhere.
1229 * adata->mailbox may be NULL for connections other than the current
1230 * mailbox's. */
1231 if (adata->mailbox && (adata->mailbox->mdata == mdata))
1232 {
1233 adata->mailbox->has_new = false;
1234 return mdata->messages;
1235 }
1236
1237 if (adata->mailbox && !adata->mailbox->poll_new_mail)
1238 return mdata->messages;
1239
1240 if (adata->capabilities & IMAP_CAP_IMAP4REV1)
1241 {
1242 uidvalidity_flag = "UIDVALIDITY";
1243 }
1244 else if (adata->capabilities & IMAP_CAP_STATUS)
1245 {
1246 uidvalidity_flag = "UID-VALIDITY";
1247 }
1248 else
1249 {
1250 mutt_debug(LL_DEBUG2, "Server doesn't support STATUS\n");
1251 return -1;
1252 }
1253
1254 snprintf(cmd, sizeof(cmd), "STATUS %s (UIDNEXT %s UNSEEN RECENT MESSAGES)",
1255 mdata->munge_name, uidvalidity_flag);
1256
1257 int rc = imap_exec(adata, cmd, queue ? IMAP_CMD_QUEUE : IMAP_CMD_POLL);
1258 if (rc != IMAP_EXEC_SUCCESS)
1259 {
1260 mutt_debug(LL_DEBUG1, "Error queueing command\n");
1261 return rc;
1262 }
1263 return mdata->messages;
1264}

1395{

1397 struct ImapMboxData *mdata = NULL;
1398 char tmp[2048] = { 0 };
1399 struct ImapList listresp = { 0 };
1400 struct Buffer *completion_buf = NULL;
1401 size_t clen;
1402 int completions = 0;
1403 int rc;
1404
1405 if (imap_adata_find(path, &adata, &mdata) < 0)
1406 {
1407 buf_strcpy(buf, path);
1408 return complete_hosts(buf);
1409 }
1410
1411 /* fire off command */
1412 const bool c_imap_list_subscribed = cs_subset_bool(NeoMutt->sub, "imap_list_subscribed");

1414 c_imap_list_subscribed ? "LSUB" : "LIST", mdata->real_name);
1415
1416 imap_cmd_start(adata, tmp);

1418 /* and see what the results are */
1419 completion_buf = buf_pool_get();
1420 buf_strcpy(completion_buf, mdata->name);
1421 imap_mdata_free((void *) &mdata);
1422
1423 adata->cmdresult = &listresp;
1424 do
1425 {
1426 listresp.name = NULL;
1427 rc = imap_cmd_step(adata);

1432 * to enter it on second tab. */

1434 {
1435 clen = strlen(listresp.name);
1436 listresp.name[clen++] = listresp.delim;
1437 listresp.name[clen] = '\0';
1438 }

1440 if (!completions)
1441 {
1442 buf_strcpy(completion_buf, listresp.name);
1443 completions++;
1444 continue;
1445 }
1446
1447 longest_common_prefix(completion_buf, listresp.name, 0);

1451 adata->cmdresult = NULL;
1452
1453 if (completions)
1454 {
1455 /* reformat output */
1456 imap_buf_qualify_path(buf, &adata->conn->account, completion_buf->data);
1457 pretty_mailbox(buf);
1458 buf_fix_dptr(buf);
1459 buf_pool_release(&completion_buf);
1460 return 0;
1461 }

1463 buf_pool_release(&completion_buf);

1475int imap_fast_trash(struct Mailbox *m, const char *dest)
1476{
1477 char prompt[1024] = { 0 };
1478 int rc = -1;
1479 bool triedcreate = false;
1480 enum QuadOption err_continue = MUTT_NO;
1481
1482 struct ImapAccountData *adata = imap_adata_get(m);
1483 if (!adata)
1484 return -1;
1485
1486 struct ImapAccountData *dest_adata = NULL;
1487 struct ImapMboxData *dest_mdata = NULL;
1488
1489 if (imap_adata_find(dest, &dest_adata, &dest_mdata) < 0)
1490 return -1;
1491
1492 struct Buffer *sync_cmd = buf_pool_get();
1493
1494 /* check that the save-to folder is in the same account */
1495 if (!imap_account_match(&(adata->conn->account), &(dest_adata->conn->account)))
1496 {
1497 mutt_debug(LL_DEBUG3, "%s not same server as %s\n", dest, mailbox_path(m));
1498 goto out;
1499 }
1500
1501 for (int i = 0; i < m->msg_count; i++)
1502 {
1503 struct Email *e = m->emails[i];
1504 if (!e)
1505 break;
1506 if (e->active && e->changed && e->deleted && !e->purge)
1507 {
1508 rc = imap_sync_message_for_copy(m, e, sync_cmd, &err_continue);
1509 if (rc < 0)
1510 {
1511 mutt_debug(LL_DEBUG1, "could not sync\n");
1512 goto out;
1513 }
1514 }
1515 }
1516
1517 /* loop in case of TRYCREATE */
1518 do
1519 {
1520 struct UidArray uida = ARRAY_HEAD_INITIALIZER;
1521 select_email_uids(m->emails, m->msg_count, MUTT_TRASH, false, false, &uida);
1522 ARRAY_SORT(&uida, imap_sort_uid, NULL);
1523 rc = imap_exec_msg_set(adata, "UID COPY", dest_mdata->munge_name, &uida);
1524 if (rc == 0)
1525 {
1526 mutt_debug(LL_DEBUG1, "No messages to trash\n");
1527 rc = -1;
1528 goto out;
1529 }
1530 else if (rc < 0)
1531 {
1532 mutt_debug(LL_DEBUG1, "could not queue copy\n");
1533 goto out;
1534 }
1535 else if (m->verbose)
1536 {
1537 mutt_message(ngettext("Copying %d message to %s...", "Copying %d messages to %s...", rc),
1538 rc, dest_mdata->name);
1539 }
1540 ARRAY_FREE(&uida);
1541
1542 /* let's get it on */
1543 rc = imap_exec(adata, NULL, IMAP_CMD_NO_FLAGS);
1544 if (rc == IMAP_EXEC_ERROR)
1545 {
1546 if (triedcreate)
1547 {
1548 mutt_debug(LL_DEBUG1, "Already tried to create mailbox %s\n", dest_mdata->name);
1549 break;
1550 }
1551 /* bail out if command failed for reasons other than nonexistent target */
1552 if (!mutt_istr_startswith(imap_get_qualifier(adata->buf), "[TRYCREATE]"))
1553 break;
1554 mutt_debug(LL_DEBUG3, "server suggests TRYCREATE\n");
1555 snprintf(prompt, sizeof(prompt), _("Create %s?"), dest_mdata->name);
1556 const bool c_confirm_create = cs_subset_bool(NeoMutt->sub, "confirm_create");
1557 if (c_confirm_create &&
1558 (query_yesorno_help(prompt, MUTT_YES, NeoMutt->sub, "confirm_create") != MUTT_YES))
1559 {
1560 mutt_clear_error();
1561 goto out;
1562 }
1563 if (imap_create_mailbox(adata, dest_mdata->name) < 0)
1564 break;
1565 triedcreate = true;
1566 }
1567 } while (rc == IMAP_EXEC_ERROR);
1568
1569 if (rc != IMAP_EXEC_SUCCESS)
1570 {
1571 imap_error("imap_fast_trash", adata->buf);
1572 goto out;
1573 }
1574
1575 rc = IMAP_EXEC_SUCCESS;
1576
1577out:
1578 buf_pool_release(&sync_cmd);
1579 imap_mdata_free((void *) &dest_mdata);
1580
1581 return ((rc == IMAP_EXEC_SUCCESS) ? 0 : -1);
1582}

1593enum MxStatus imap_sync_mailbox(struct Mailbox *m, bool expunge, bool close)
1594{
1595 if (!m)
1596 return -1;
1597
1598 struct Email **emails = NULL;
1599 int rc;
1600
1601 struct ImapAccountData *adata = imap_adata_get(m);
1602 struct ImapMboxData *mdata = imap_mdata_get(m);
1603 if (!adata || !mdata)
1604 return MX_STATUS_ERROR;
1605
1606 if (adata->state < IMAP_SELECTED)
1607 {
1608 mutt_debug(LL_DEBUG2, "no mailbox selected\n");
1609 return -1;
1610 }
1611
1612 /* This function is only called when the calling code expects the context
1613 * to be changed. */
1614 imap_allow_reopen(m);
1615
1616 enum MxStatus check = imap_check_mailbox(m, false);
1617 if (check == MX_STATUS_ERROR)
1618 return check;
1619
1620 /* if we are expunging anyway, we can do deleted messages very quickly... */
1621 if (expunge && (m->rights & MUTT_ACL_DELETE))
1622 {
1623 struct UidArray uida = ARRAY_HEAD_INITIALIZER;
1624 select_email_uids(m->emails, m->msg_count, MUTT_DELETED, true, false, &uida);
1625 ARRAY_SORT(&uida, imap_sort_uid, NULL);
1626 rc = imap_exec_msg_set(adata, "UID STORE", "+FLAGS.SILENT (\\Deleted)", &uida);
1627 ARRAY_FREE(&uida);
1628 if (rc < 0)
1629 {
1630 mutt_error(_("Expunge failed"));
1631 return rc;
1632 }
1633
1634 if (rc > 0)
1635 {
1636 /* mark these messages as unchanged so second pass ignores them. Done
1637 * here so BOGUS UW-IMAP 4.7 SILENT FLAGS updates are ignored. */
1638 for (int i = 0; i < m->msg_count; i++)
1639 {
1640 struct Email *e = m->emails[i];
1641 if (!e)
1642 break;
1643 if (e->deleted && e->changed)
1644 e->active = false;
1645 }
1646 if (m->verbose)
1647 {
1648 mutt_message(ngettext("Marking %d message deleted...",
1649 "Marking %d messages deleted...", rc),
1650 rc);
1651 }
1652 }
1653 }
1654
1655#ifdef USE_HCACHE
1656 imap_hcache_open(adata, mdata, true);
1657#endif
1658
1659 /* save messages with real (non-flag) changes */
1660 for (int i = 0; i < m->msg_count; i++)
1661 {
1662 struct Email *e = m->emails[i];
1663 if (!e)
1664 break;
1665
1666 if (e->deleted)
1667 {
1668 imap_cache_del(m, e);
1669#ifdef USE_HCACHE
1670 imap_hcache_del(mdata, imap_edata_get(e)->uid);
1671#endif
1672 }
1673
1674 if (e->active && e->changed)
1675 {
1676#ifdef USE_HCACHE
1677 imap_hcache_put(mdata, e);
1678#endif
1679 /* if the message has been rethreaded or attachments have been deleted
1680 * we delete the message and reupload it.
1681 * This works better if we're expunging, of course. */
1682 if (e->env->changed || e->attach_del)
1683 {
1684 /* L10N: The plural is chosen by the last %d, i.e. the total number */
1685 if (m->verbose)
1686 {
1687 mutt_message(ngettext("Saving changed message... [%d/%d]",
1688 "Saving changed messages... [%d/%d]", m->msg_count),
1689 i + 1, m->msg_count);
1690 }
1691 bool save_append = m->append;
1692 m->append = true;
1693 mutt_save_message_mbox(m, e, SAVE_MOVE, TRANSFORM_NONE, m);
1694 m->append = save_append;
1695 e->env->changed = false;
1696 }
1697 }
1698 }
1699
1700#ifdef USE_HCACHE
1701 imap_hcache_close(mdata);
1702#endif
1703
1704 /* presort here to avoid doing 10 resorts in imap_exec_msg_set */
1705 emails = MUTT_MEM_MALLOC(m->msg_count, struct Email *);
1706 memcpy(emails, m->emails, m->msg_count * sizeof(struct Email *));
1707 mutt_qsort_r(emails, m->msg_count, sizeof(struct Email *), imap_sort_email_uid, NULL);
1708
1709 rc = sync_helper(m, emails, m->msg_count, MUTT_ACL_DELETE, MUTT_DELETED, "\\Deleted");
1710 if (rc >= 0)
1711 rc |= sync_helper(m, emails, m->msg_count, MUTT_ACL_WRITE, MUTT_FLAG, "\\Flagged");
1712 if (rc >= 0)
1713 rc |= sync_helper(m, emails, m->msg_count, MUTT_ACL_WRITE, MUTT_OLD, "Old");
1714 if (rc >= 0)
1715 rc |= sync_helper(m, emails, m->msg_count, MUTT_ACL_SEEN, MUTT_READ, "\\Seen");
1716 if (rc >= 0)
1717 rc |= sync_helper(m, emails, m->msg_count, MUTT_ACL_WRITE, MUTT_REPLIED, "\\Answered");
1718
1719 FREE(&emails);
1720
1721 /* Flush the queued flags if any were changed in sync_helper. */
1722 if (rc > 0)
1723 if (imap_exec(adata, NULL, IMAP_CMD_NO_FLAGS) != IMAP_EXEC_SUCCESS)
1724 rc = -1;
1725
1726 if (rc < 0)
1727 {
1728 if (close)
1729 {
1730 if (query_yesorno(_("Error saving flags. Close anyway?"), MUTT_NO) == MUTT_YES)
1731 {
1732 adata->state = IMAP_AUTHENTICATED;
1733 return 0;
1734 }
1735 }
1736 else
1737 {
1738 mutt_error(_("Error saving flags"));
1739 }
1740 return -1;
1741 }
1742
1743 /* Update local record of server state to reflect the synchronization just
1744 * completed. imap_read_headers always overwrites hcache-origin flags, so
1745 * there is no need to mutate the hcache after flag-only changes. */
1746 for (int i = 0; i < m->msg_count; i++)
1747 {
1748 struct Email *e = m->emails[i];
1749 if (!e)
1750 break;
1751 struct ImapEmailData *edata = imap_edata_get(e);
1752 edata->deleted = e->deleted;
1753 edata->flagged = e->flagged;
1754 edata->old = e->old;
1755 edata->read = e->read;
1756 edata->replied = e->replied;
1757 e->changed = false;
1758 }
1759 m->changed = false;
1760
1761 /* We must send an EXPUNGE command if we're not closing. */
1762 if (expunge && !close && (m->rights & MUTT_ACL_DELETE))
1763 {
1764 if (m->verbose)
1765 mutt_message(_("Expunging messages from server..."));
1766 /* Set expunge bit so we don't get spurious reopened messages */
1767 mdata->reopen |= IMAP_EXPUNGE_EXPECTED;
1768 if (imap_exec(adata, "EXPUNGE", IMAP_CMD_NO_FLAGS) != IMAP_EXEC_SUCCESS)
1769 {
1770 mdata->reopen &= ~IMAP_EXPUNGE_EXPECTED;
1771 imap_error(_("imap_sync_mailbox: EXPUNGE failed"), adata->buf);
1772 return -1;
1773 }
1774 mdata->reopen &= ~IMAP_EXPUNGE_EXPECTED;
1775 }
1776
1777 if (expunge && close)
1778 {
1779 adata->closing = true;
1780 imap_exec(adata, "CLOSE", IMAP_CMD_NO_FLAGS);
1781 adata->state = IMAP_AUTHENTICATED;
1782 }
1783
1784 const bool c_message_cache_clean = cs_subset_bool(NeoMutt->sub, "message_cache_clean");
1785 if (c_message_cache_clean)
1786 imap_cache_clean(m);
1787
1788 return check;
1789}

1901int imap_login(struct ImapAccountData *adata)
1902{
1903 if (!adata)
1904 return -1;
1905
1906 if (adata->state == IMAP_DISCONNECTED)
1907 {
1908 buf_reset(&adata->cmdbuf); // purge outstanding queued commands
1909 imap_open_connection(adata);
1910 }
1911 if (adata->state == IMAP_CONNECTED)
1912 {
1913 if (imap_authenticate(adata) == IMAP_AUTH_SUCCESS)
1914 {
1915 adata->state = IMAP_AUTHENTICATED;
1916 FREE(&adata->capstr);
1917 if (adata->conn->ssf != 0)
1918 {
1919 mutt_debug(LL_DEBUG2, "Communication encrypted at %d bits\n",
1920 adata->conn->ssf);

1922 }
1923 else
1924 {
1925 mutt_account_unsetpass(&adata->conn->account);
1926 }
1927 }
1928 if (adata->state == IMAP_AUTHENTICATED)
1929 {

1933#ifdef USE_ZLIB
1934 /* RFC4978 */
1935 const bool c_imap_deflate = cs_subset_bool(NeoMutt->sub, "imap_deflate");

1940 adata->conn->account.host);
1941 mutt_zstrm_wrap_conn(adata->conn);
1942 }
1943#endif

1946 const bool c_imap_send_id = cs_subset_bool(NeoMutt->sub, "imap_send_id");
1947 if (c_imap_send_id && (adata->capabilities & IMAP_CAP_ID))
1948 {
1949 imap_exec(adata, "ID (\"name\" \"NeoMutt\" \"version\" \"" PACKAGE_VERSION "\")",
1950 IMAP_CMD_QUEUE);
1951 }

1956 imap_exec(adata, "ENABLE UTF8=ACCEPT", IMAP_CMD_QUEUE);
1957
1958 /* enable QRESYNC. Advertising QRESYNC also means CONDSTORE
1959 * is supported (even if not advertised), so flip that bit. */
1960 if (adata->capabilities & IMAP_CAP_QRESYNC)
1961 {
1962 adata->capabilities |= IMAP_CAP_CONDSTORE;
1963 const bool c_imap_qresync = cs_subset_bool(NeoMutt->sub, "imap_qresync");
1964 if (c_imap_rfc5161 && c_imap_qresync)
1965 imap_exec(adata, "ENABLE QRESYNC", IMAP_CMD_QUEUE);
1966 }
1967
1968 /* get root delimiter, '/' as default */
1969 adata->delim = '/';
1970 imap_exec(adata, "LIST \"\" \"\"", IMAP_CMD_QUEUE);
1971
1972 /* we may need the root delimiter before we open a mailbox */
1973 imap_exec(adata, NULL, IMAP_CMD_NO_FLAGS);
1974
1975 /* select the mailbox that used to be open before disconnect */
1976 if (adata->mailbox)
1977 {
1978 imap_mbox_select(adata->mailbox);
1979 }
1980 }
1981
1982 if (adata->state < IMAP_AUTHENTICATED)
1983 return -1;
1984
1985 return 0;

2318static int imap_tags_edit(struct Mailbox *m, const char *tags, struct Buffer *buf)
2319{
2320 struct ImapMboxData *mdata = imap_mdata_get(m);
2321 if (!mdata)
2322 return -1;
2323
2324 char *new_tag = NULL;
2325 char *checker = NULL;
2326
2327 /* Check for \* flags capability */
2328 if (!imap_has_flag(&mdata->flags, NULL))
2329 {
2330 mutt_error(_("IMAP server doesn't support custom flags"));
2331 return -1;
2332 }
2333
2334 buf_reset(buf);
2335 if (tags)
2336 buf_strcpy(buf, tags);
2337
2338 if (mw_get_field("Tags: ", buf, MUTT_COMP_NO_FLAGS, HC_OTHER, NULL, NULL) != 0)
2339 return -1;
2340
2341 /* each keyword must be atom defined by rfc822 as:
2342 *
2343 * atom = 1*<any CHAR except specials, SPACE and CTLs>
2344 * CHAR = ( 0.-127. )
2345 * specials = "(" / ")" / "<" / ">" / "@"
2346 * / "," / ";" / ":" / "\" / <">
2347 * / "." / "[" / "]"
2348 * SPACE = ( 32. )
2349 * CTLS = ( 0.-31., 127.)
2350 *
2351 * And must be separated by one space.
2352 */
2353
2354 new_tag = buf->data;
2355 checker = buf->data;
2356 SKIPWS(checker);
2357 while (*checker != '\0')
2358 {
2359 if ((*checker < 32) || (*checker >= 127) || // We allow space because it's the separator
2360 (*checker == 40) || // (
2361 (*checker == 41) || // )
2362 (*checker == 60) || // <
2363 (*checker == 62) || // >
2364 (*checker == 64) || // @
2365 (*checker == 44) || // ,
2366 (*checker == 59) || // ;
2367 (*checker == 58) || // :
2368 (*checker == 92) || // backslash
2369 (*checker == 34) || // "
2370 (*checker == 46) || // .
2371 (*checker == 91) || // [
2372 (*checker == 93)) // ]
2373 {
2374 mutt_error(_("Invalid IMAP flags"));
2375 return 0;
2376 }
2377
2378 /* Skip duplicate space */
2379 while ((checker[0] == ' ') && (checker[1] == ' '))
2380 checker++;
2381
2382 /* copy char to new_tag and go the next one */
2383 *new_tag++ = *checker++;
2384 }
2385 *new_tag = '\0';
2386 new_tag = buf->data; /* rewind */
2387 mutt_str_remove_trailing_ws(new_tag);
2388
2389 return !mutt_str_equal(tags, buf_string(buf));
2390}
