82void maildir_parse_flags(struct Email *e, const char *path)
83{
84 char *q = NULL;
85
86 e->flagged = false;
87 e->read = false;
88 e->replied = false;
89
90 struct MaildirEmailData *edata = maildir_edata_get(e);
91 if (!edata)
92 {
93 e->edata = maildir_edata_new();
94 edata = e->edata;
95 }
96
97 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
98 char *p = strrchr(path, c_maildir_field_delimiter);
99 if (p && mutt_str_startswith(p + 1, "2,"))
100 {
101 p += 3;
102
103 mutt_str_replace(&edata->custom_flags, p);
104 q = edata->custom_flags;
105
106 while (*p)
107 {
108 switch (*p)
109 {
110 case 'F': // Flagged
111 e->flagged = true;
112 break;
113
114 case 'R': // Replied
115 e->replied = true;
116 break;
117
118 case 'S': // Seen
119 e->read = true;
120 break;
121
122 case 'T': // Trashed
123 {
124 const bool c_flag_safe = cs_subset_bool(NeoMutt->sub, "flag_safe");
125 if (!e->flagged || !c_flag_safe)
126 {
127 e->trash = true;
128 e->deleted = true;
129 }
130 break;
131 }
132
133 default:
134 *q++ = *p;
135 break;
136 }
137 p++;
138 }
139 }
140
141 if (q == edata->custom_flags)
142 FREE(&edata->custom_flags);
143 else if (q)
144 *q = '\0';
145}

158bool maildir_parse_stream(FILE *fp, const char *fname, bool is_old, struct Email *e)
159{
160 if (!fp || !fname || !e)
161 return false;
162
163 const long size = mutt_file_get_size_fp(fp);
164 if (size == 0)
165 return false;
166
167 e->env = mutt_rfc822_read_header(fp, e, false, false);
168
169 if (e->received == 0)
170 e->received = e->date_sent;
171
172 /* always update the length since we have fresh information available. */
173 e->body->length = size - e->body->offset;
174
175 e->index = -1;
176
177 /* maildir stores its flags in the filename, so ignore the
178 * flags in the header of the message */
179 e->old = is_old;
180 maildir_parse_flags(e, fname);
181
182 return e;
183}

216static int maildir_move_to_mailbox(struct Mailbox *m, const struct MdEmailArray *mda)
217{
218 if (!m)
219 return 0;
220
221 int oldmsgcount = m->msg_count;
222
223 struct MdEmail *md = NULL;
224 struct MdEmail **mdp = NULL;
225 ARRAY_FOREACH(mdp, mda)
226 {
227 md = *mdp;
228 mutt_debug(LL_DEBUG2, "Considering %s\n", NONULL(md->canon_fname));
229 if (!md->email)
230 continue;
231
232 mutt_debug(LL_DEBUG2, "Adding header structure. Flags: %s%s%s%s%s\n",
233 md->email->flagged ? "f" : "", md->email->deleted ? "D" : "",
234 md->email->replied ? "r" : "", md->email->old ? "O" : "",
235 md->email->read ? "R" : "");
236 mx_alloc_memory(m, m->msg_count);
237
238 m->emails[m->msg_count] = md->email;
239 m->emails[m->msg_count]->index = m->msg_count;
240 mailbox_size_add(m, md->email);
241
242 md->email = NULL;
243 m->msg_count++;
244 }
245
246 int num = 0;
247 if (m->msg_count > oldmsgcount)
248 num = m->msg_count - oldmsgcount;
249
250 return num;
251}

274static int maildir_parse_dir(struct Mailbox *m, struct MdEmailArray *mda,
275 const char *subdir, struct Progress *progress)
276{
277 struct dirent *de = NULL;
278 int rc = 0;
279 bool is_old = false;
280 struct MdEmail *entry = NULL;
281 struct Email *e = NULL;
282
283 struct Buffer *buf = buf_pool_get();
284
285 buf_printf(buf, "%s/%s", mailbox_path(m), subdir);
286 is_old = mutt_str_equal("cur", subdir);
287
288 DIR *dir = mutt_file_opendir(buf_string(buf), MUTT_OPENDIR_CREATE);
289 if (!dir)
290 {
291 rc = -1;
292 goto cleanup;
293 }
294
295 while (((de = readdir(dir))) && !SigInt)
296 {
297 if (*de->d_name == '.')
298 continue;
299
300 mutt_debug(LL_DEBUG2, "queueing %s\n", de->d_name);
301
302 e = maildir_email_new();
303 e->old = is_old;
304 maildir_parse_flags(e, de->d_name);
305
306 progress_update(progress, ARRAY_SIZE(mda) + 1, -1);
307
308 buf_printf(buf, "%s/%s", subdir, de->d_name);
309 e->path = buf_strdup(buf);
310
311 entry = maildir_entry_new();
312 entry->email = e;
313 entry->inode = de->d_ino;
314 ARRAY_ADD(mda, entry);
315 }
316
317 closedir(dir);
318
319 if (SigInt)
320 {
321 SigInt = false;
322 return -2; /* action aborted */
323 }
324
325 ARRAY_SORT(mda, maildir_sort_inode, NULL);
326
327cleanup:
328 buf_pool_release(&buf);
329
330 return rc;
331}

339static void maildir_delayed_parsing(struct Mailbox *m, struct MdEmailArray *mda,
340 struct Progress *progress)
341{
342 char fn[PATH_MAX] = { 0 };
343
344 struct HeaderCache *hc = maildir_hcache_open(m);
345
346 struct MdEmail *md = NULL;
347 struct MdEmail **mdp = NULL;
348 ARRAY_FOREACH(mdp, mda)
349 {
350 md = *mdp;
351 if (!md || !md->email || md->header_parsed)
352 continue;
353
354 progress_update(progress, ARRAY_FOREACH_IDX_mdp, -1);
355
356 snprintf(fn, sizeof(fn), "%s/%s", mailbox_path(m), md->email->path);
357
358 struct Email *e = maildir_hcache_read(hc, md->email, fn);
359 if (e)
360 {
361 email_free(&md->email);
362 md->email = e;
363 }
364 else
365 {
366 if (maildir_parse_message(fn, md->email->old, md->email))
367 {
368 md->header_parsed = true;
369 maildir_hcache_store(hc, md->email);
370 }
371 else
372 {
373 email_free(&md->email);
374 }
375 }
376 }
377
378 maildir_hcache_close(&hc);

390static void maildir_check_dir(struct Mailbox *m, const char *dir_name,
391 bool check_new, bool check_stats)
392{
393 DIR *dir = NULL;
394 struct dirent *de = NULL;
395 char *p = NULL;
396 struct stat st = { 0 };
397
398 struct Buffer *path = buf_pool_get();
399 struct Buffer *msgpath = buf_pool_get();
400 buf_printf(path, "%s/%s", mailbox_path(m), dir_name);
401
402 /* when $mail_check_recent is set, if the new/ directory hasn't been modified since
403 * the user last exited the mailbox, then we know there is no recent mail. */
404 const bool c_mail_check_recent = cs_subset_bool(NeoMutt->sub, "mail_check_recent");
405 if (check_new && c_mail_check_recent)
406 {
407 if ((stat(buf_string(path), &st) == 0) &&
408 (mutt_file_stat_timespec_compare(&st, MUTT_STAT_MTIME, &m->last_visited) < 0))
409 {
410 check_new = false;
411 }
412 }
413
414 if (!(check_new || check_stats))
415 goto cleanup;
416
417 dir = mutt_file_opendir(buf_string(path), MUTT_OPENDIR_CREATE);
418 if (!dir)
419 {
420 m->type = MUTT_UNKNOWN;
421 goto cleanup;
422 }
423
424 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
425
426 char delimiter_version[8] = { 0 };
427 snprintf(delimiter_version, sizeof(delimiter_version), "%c2,", c_maildir_field_delimiter);
428 while ((de = readdir(dir)))
429 {
430 if (*de->d_name == '.')
431 continue;
432
433 p = strstr(de->d_name, delimiter_version);
434 if (p && strchr(p + 3, 'T'))
435 continue;
436
437 if (check_stats)
438 {
439 m->msg_count++;
440 if (p && strchr(p + 3, 'F'))
441 m->msg_flagged++;
442 }
443 if (!p || !strchr(p + 3, 'S'))
444 {
445 if (check_stats)
446 m->msg_unread++;
447 if (check_new)
448 {
449 if (c_mail_check_recent)
450 {
451 buf_printf(msgpath, "%s/%s", buf_string(path), de->d_name);
452 /* ensure this message was received since leaving this m */
453 if ((stat(buf_string(msgpath), &st) == 0) &&
454 (mutt_file_stat_timespec_compare(&st, MUTT_STAT_CTIME, &m->last_visited) <= 0))
455 {
456 continue;
457 }
458 }
459 m->has_new = true;
460 if (check_stats)
461 {
462 m->msg_new++;
463 }
464 else
465 {
466 break;
467 }
468 }
469 }
470 }
471
472 closedir(dir);
473
474cleanup:
475 buf_pool_release(&path);
476 buf_pool_release(&msgpath);
477}

486static int maildir_read_dir(struct Mailbox *m, const char *subdir)
487{
488 if (!m)
489 return -1;
490
491 mutt_path_tidy(&m->pathbuf, true);
492
493 struct Progress *progress = NULL;
494
495 if (m->verbose)
496 {
497 progress = progress_new(MUTT_PROGRESS_READ, 0);
498 progress_set_message(progress, _("Scanning %s..."), mailbox_path(m));
499 }
500
501 struct MaildirMboxData *mdata = maildir_mdata_get(m);
502 if (!mdata)
503 {
504 mdata = maildir_mdata_new();
505 m->mdata = mdata;
506 m->mdata_free = maildir_mdata_free;
507 }
508
509 struct MdEmailArray mda = ARRAY_HEAD_INITIALIZER;
510 int rc = maildir_parse_dir(m, &mda, subdir, progress);
511 progress_free(&progress);
512 if (rc < 0)
513 return -1;
514
515 if (m->verbose)
516 {
517 progress = progress_new(MUTT_PROGRESS_READ, ARRAY_SIZE(&mda));
518 progress_set_message(progress, _("Reading %s..."), mailbox_path(m));
519 }
520 maildir_delayed_parsing(m, &mda, progress);
521 progress_free(&progress);
522
523 maildir_move_to_mailbox(m, &mda);
524 maildirarray_clear(&mda);
525
526 if (!mdata->umask)
527 mdata->umask = maildir_umask(m);
528
529 return 0;
530}

544static enum MxStatus maildir_check(struct Mailbox *m)
545{
546 struct stat st_new = { 0 }; /* status of the "new" subdirectory */
547 struct stat st_cur = { 0 }; /* status of the "cur" subdirectory */
548 int changed = MMC_NO_DIRS; /* which subdirectories have changed */
549 bool occult = false; /* messages were removed from the mailbox */
550 int num_new = 0; /* number of new messages added to the mailbox */
551 bool flags_changed = false; /* message flags were changed in the mailbox */
552 struct HashTable *hash_names = NULL; // Hash Table: "base-filename" -> MdEmail
553 struct MaildirMboxData *mdata = maildir_mdata_get(m);
554
555 const bool c_check_new = cs_subset_bool(NeoMutt->sub, "check_new");
556 if (!c_check_new)
557 return MX_STATUS_OK;
558
559 struct Buffer *buf = buf_pool_get();
560 buf_printf(buf, "%s/new", mailbox_path(m));
561 if (stat(buf_string(buf), &st_new) == -1)
562 {
563 buf_pool_release(&buf);
564 return MX_STATUS_ERROR;
565 }
566
567 buf_printf(buf, "%s/cur", mailbox_path(m));
568 if (stat(buf_string(buf), &st_cur) == -1)
569 {
570 buf_pool_release(&buf);
571 return MX_STATUS_ERROR;
572 }
573
574 /* determine which subdirectories need to be scanned */
575 if (mutt_file_stat_timespec_compare(&st_new, MUTT_STAT_MTIME, &mdata->mtime) > 0)
576 changed = MMC_NEW_DIR;
577 if (mutt_file_stat_timespec_compare(&st_cur, MUTT_STAT_MTIME, &mdata->mtime_cur) > 0)
578 changed |= MMC_CUR_DIR;
579
580 if (changed == MMC_NO_DIRS)
581 {
582 buf_pool_release(&buf);
583 return MX_STATUS_OK; /* nothing to do */
584 }
585
586 /* Update the modification times on the mailbox.
587 *
588 * The monitor code notices changes in the open mailbox too quickly.
589 * In practice, this sometimes leads to all the new messages not being
590 * noticed during the SAME group of mtime stat updates. To work around
591 * the problem, don't update the stat times for a monitor caused check. */
592#ifdef USE_INOTIFY
593 if (MonitorCurMboxChanged)
594 {
595 MonitorCurMboxChanged = false;
596 }
597 else
598#endif
599 {
600 mutt_file_get_stat_timespec(&mdata->mtime_cur, &st_cur, MUTT_STAT_MTIME);
601 mutt_file_get_stat_timespec(&mdata->mtime, &st_new, MUTT_STAT_MTIME);
602 }
603
604 /* do a fast scan of just the filenames in
605 * the subdirectories that have changed. */
606 struct MdEmailArray mda = ARRAY_HEAD_INITIALIZER;
607 if (changed & MMC_NEW_DIR)
608 maildir_parse_dir(m, &mda, "new", NULL);
609 if (changed & MMC_CUR_DIR)
610 maildir_parse_dir(m, &mda, "cur", NULL);
611
612 /* we create a hash table keyed off the canonical (sans flags) filename
613 * of each message we scanned. This is used in the loop over the
614 * existing messages below to do some correlation. */
615 hash_names = mutt_hash_new(ARRAY_SIZE(&mda), MUTT_HASH_NO_FLAGS);
616
617 struct MdEmail *md = NULL;
618 struct MdEmail **mdp = NULL;
619 ARRAY_FOREACH(mdp, &mda)
620 {
621 md = *mdp;
622 maildir_canon_filename(buf, md->email->path);
623 md->canon_fname = buf_strdup(buf);
624 mutt_hash_insert(hash_names, md->canon_fname, md);
625 }
626
627 /* check for modifications and adjust flags */
628 for (int i = 0; i < m->msg_count; i++)
629 {
630 struct Email *e = m->emails[i];
631 if (!e)
632 break;
633
634 maildir_canon_filename(buf, e->path);
635 md = mutt_hash_find(hash_names, buf_string(buf));
636 if (md && md->email)
637 {
638 /* message already exists, merge flags */
639
640 /* check to see if the message has moved to a different
641 * subdirectory. If so, update the associated filename. */
642 if (!mutt_str_equal(e->path, md->email->path))
643 mutt_str_replace(&e->path, md->email->path);
644
645 /* if the user hasn't modified the flags on this message, update
646 * the flags we just detected. */
647 if (!e->changed)
648 if (maildir_update_flags(m, e, md->email))
649 flags_changed = true;
650
651 if (e->deleted == e->trash)
652 {
653 if (e->deleted != md->email->deleted)
654 {
655 e->deleted = md->email->deleted;
656 flags_changed = true;
657 }
658 }
659 e->trash = md->email->trash;
660
661 /* this is a duplicate of an existing email, so remove it */
662 email_free(&md->email);
663 }
664 /* This message was not in the list of messages we just scanned.
665 * Check to see if we have enough information to know if the
666 * message has disappeared out from underneath us. */
667 else if (((changed & MMC_NEW_DIR) && mutt_strn_equal(e->path, "new/", 4)) ||
668 ((changed & MMC_CUR_DIR) && mutt_strn_equal(e->path, "cur/", 4)))
669 {
670 /* This message disappeared, so we need to simulate a "reopen"
671 * event. We know it disappeared because we just scanned the
672 * subdirectory it used to reside in. */
673 occult = true;
674 e->deleted = true;
675 e->purge = true;
676 }
677 else
678 {
679 /* This message resides in a subdirectory which was not
680 * modified, so we assume that it is still present and
681 * unchanged. */
682 }
683 }
684
685 /* destroy the file name hash */
686 mutt_hash_free(&hash_names);
687
688 /* If we didn't just get new mail, update the tables. */
689 if (occult)
690 mailbox_changed(m, NT_MAILBOX_RESORT);
691
692 /* do any delayed parsing we need to do. */
693 maildir_delayed_parsing(m, &mda, NULL);
694
695 /* Incorporate new messages */
696 num_new = maildir_move_to_mailbox(m, &mda);
697 maildirarray_clear(&mda);
698
699 if (num_new > 0)
700 {
701 mailbox_changed(m, NT_MAILBOX_INVALID);
702 m->changed = true;
703 }
704
705 buf_pool_release(&buf);
706
707 ARRAY_FREE(&mda);
708 if (occult)
709 return MX_STATUS_REOPENED;
710 if (num_new > 0)
711 return MX_STATUS_NEW_MAIL;
712 if (flags_changed)
713 return MX_STATUS_FLAGS;
714 return MX_STATUS_OK;
715}

843enum MxStatus maildir_mbox_sync(struct Mailbox *m)
844{
845 enum MxStatus check = maildir_check(m);
846 if (check == MX_STATUS_ERROR)
847 return check;
848
849 struct HeaderCache *hc = maildir_hcache_open(m);
850
851 struct Progress *progress = NULL;
852 if (m->verbose)
853 {
854 progress = progress_new(MUTT_PROGRESS_WRITE, m->msg_count);
855 progress_set_message(progress, _("Writing %s..."), mailbox_path(m));
856 }
857
858 for (int i = 0; i < m->msg_count; i++)
859 {
860 progress_update(progress, i, -1);
861
862 struct Email *e = m->emails[i];
863 if (!maildir_sync_mailbox_message(m, e, hc))
864 {
865 progress_free(&progress);
866 goto err;
867 }
868 }
869 progress_free(&progress);
870 maildir_hcache_close(&hc);
871
872 /* XXX race condition? */
873
874 maildir_update_mtime(m);
875
876 /* adjust indices */
877
878 if (m->msg_deleted)
879 {
880 const bool c_maildir_trash = cs_subset_bool(NeoMutt->sub, "maildir_trash");
881 for (int i = 0, j = 0; i < m->msg_count; i++)
882 {
883 struct Email *e = m->emails[i];
884 if (!e)
