682 static struct SmimeKey *smime_parse_key(char *buf)
683 {
684  char *pend = NULL, *p = NULL;
685  int field = 0;
686 
687  struct SmimeKey *key = mutt_mem_calloc(1, sizeof(struct SmimeKey));
688 
689  for (p = buf; p; p = pend)
690  {
691  /* Some users manually maintain their .index file, and use a tab
692  * as a delimiter, which the old parsing code (using fscanf)
693  * happened to allow. smime_keys uses a space, so search for both. */
694  if ((pend = strchr(p, ' ')) || (pend = strchr(p, '\t')) || (pend = strchr(p, '\n')))
695  *pend++ = 0;
696 
697  /* For backward compatibility, don't count consecutive delimiters
698  * as an empty field. */
699  if (!*p)
700  continue;
701 
702  field++;
703 
704  switch (field)
705  {
706  case 1: /* mailbox */
707  key->email = mutt_str_strdup(p);
708  break;
709  case 2: /* hash */
710  key->hash = mutt_str_strdup(p);
711  break;
712  case 3: /* label */
713  key->label = mutt_str_strdup(p);
714  break;
715  case 4: /* issuer */
716  key->issuer = mutt_str_strdup(p);
717  break;
718  case 5: /* trust */
719  key->trust = *p;
720  break;
721  case 6: /* purpose */
722  while (*p)
723  {
724  switch (*p++)
725  {
726  case 'e':
727  key->flags |= KEYFLAG_CANENCRYPT;
728  break;
729 
730  case 's':
731  key->flags |= KEYFLAG_CANSIGN;
732  break;
733  }
734  }
735  break;
736  }
737  }
738 
739  /* Old index files could be missing issuer, trust, and purpose,
740  * but anything less than that is an error. */
741  if (field < 3)
742  {
743  smime_key_free(&key);
744  return NULL;
745  }
746 
747  if (field < 4)
748  key->issuer = mutt_str_strdup("?");
749 
750  if (field < 5)
751  key->trust = 't';
752 
753  if (field < 6)
754  key->flags = (KEYFLAG_CANENCRYPT | KEYFLAG_CANSIGN);
755 
756  return key;
757 }
