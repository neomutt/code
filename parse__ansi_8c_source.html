77int ansi_color_seq_length(const char *str)
78{
79 if (!str || !*str)
80 return 0;
81
82 if ((str[0] != '\033') || (str[1] != '[')) // Escape
83 return 0;
84
85 int i = 2;
86 while ((str[i] != '\0') && (mutt_isdigit(str[i]) || (str[i] == ';')))
87 {
88 i++;
89 }
90
91 if (str[i] == 'm')
92 return i + 1;
93
94 return 0;
95}
