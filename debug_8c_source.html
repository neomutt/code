143const char *color_debug_log_attrs(int attrs)
144{
145 static char text[64];
146 struct Mapping attr_names[] = {
147 { "\033[5mBLI\033[0m", A_BLINK },
148 { "\033[1mBLD\033[0m", A_BOLD },
149 { "\033[0mNOR\033[0m", A_NORMAL },
150 { "\033[7mREV\033[0m", A_REVERSE },
151 { "\033[1mSTD\033[0m", A_STANDOUT },
152 { "\033[4mUND\033[0m", A_UNDERLINE },
153 { NULL, 0 },
154 };
155
156 int offset = 0;
157 text[0] = '\0';
158 for (int i = 0; attr_names[i].name; i++)
159 {
160 if (attrs & attr_names[i].value)
161 {
162 offset += snprintf(text + offset, sizeof(text) - offset, "%s ",
163 attr_names[i].name);
164 }
165 }
166 return text;
167}

515
