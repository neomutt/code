135  const char *value, struct Buffer *err)
136 {
137  if (!cs || !cdef)
138  return CSR_ERR_CODE; /* LCOV_EXCL_LINE */
139 
140  intptr_t id = -1;
141  int flags = 0;
142 
143  if (!value || !value[0])
144  {
145  mutt_buffer_printf(err, _("Option %s may not be empty"), cdef->name);
146  return CSR_ERR_INVALID | CSR_INV_TYPE;
147  }
148 
149  size_t plen = 0;
150  plen = mutt_str_startswith(value, "reverse-", CASE_MATCH);
151  if (plen != 0)
152  {
153  flags |= SORT_REVERSE;
154  value += plen;
155  }
156 
157  plen = mutt_str_startswith(value, "last-", CASE_MATCH);
158  if (plen != 0)
159  {
160  flags |= SORT_LAST;
161  value += plen;
162  }
163 
164  switch (cdef->type & DT_SUBTYPE_MASK)
165  {
166  case DT_SORT_INDEX:
167  id = mutt_map_get_value(value, SortMethods);
168  break;
169  case DT_SORT_ALIAS:
170  id = mutt_map_get_value(value, SortAliasMethods);
171  break;
172  case DT_SORT_AUX:
173  id = mutt_map_get_value(value, SortAuxMethods);
174  break;
175  case DT_SORT_BROWSER:
176  id = mutt_map_get_value(value, SortBrowserMethods);
177  break;
178  case DT_SORT_KEYS:
179  id = mutt_map_get_value(value, SortKeyMethods);
180  break;
181  case DT_SORT_SIDEBAR:
182  id = mutt_map_get_value(value, SortSidebarMethods);
183  break;
184  default:
185  mutt_debug(LL_DEBUG1, "Invalid sort type: %u\n", cdef->type & DT_SUBTYPE_MASK);
186  return CSR_ERR_CODE;
187  break;
188  }
189 
190  if (id < 0)
191  {
192  mutt_buffer_printf(err, _("Invalid sort name: %s"), value);
193  return CSR_ERR_INVALID | CSR_INV_TYPE;
194  }
195 
196  id |= flags;
197

199  {

201  return CSR_SUCCESS | CSR_SUC_NO_CHANGE;
202 
203  if (cdef->validator)
204  {
205  int rc = cdef->validator(cs, cdef, (intptr_t) id, err);
206 
207  if (CSR_RESULT(rc) != CSR_SUCCESS)
208  return rc | CSR_INV_VALIDATOR;
209  }
210 
211  *(short *) var = id;
212  }

216  }
