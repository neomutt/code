187static int op_browser_subscribe(struct BrowserPrivateData *priv, int op)
188{
189 if (OptNews)
190 {
191 struct NntpAccountData *adata = CurrentNewsSrv;
192 int index = menu_get_index(priv->menu);
193
194 if (ARRAY_EMPTY(&priv->state.entry))
195 {
196 mutt_error(_("No newsgroups match the mask"));
197 return FR_ERROR;
198 }
199
200 int rc = nntp_newsrc_parse(adata);
201 if (rc < 0)
202 return FR_ERROR;
203
204 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
205 if (op == OP_BROWSER_SUBSCRIBE)
206 mutt_newsgroup_subscribe(adata, ff->name);
207 else
208 mutt_newsgroup_unsubscribe(adata, ff->name);
209
210 menu_set_index(priv->menu, index + 1);
211
212 if (rc > 0)
213 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
214 nntp_newsrc_update(adata);
215 nntp_clear_cache(adata);
216 nntp_newsrc_close(adata);
217 }
218 else
219 {
220 if (ARRAY_EMPTY(&priv->state.entry))
221 {
222 mutt_error(_("There are no mailboxes"));
223 return FR_ERROR;
224 }
225
226 struct Buffer *buf = buf_pool_get();
227 int index = menu_get_index(priv->menu);
228 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
229 buf_strcpy(buf, ff->name);
230 buf_expand_path(buf);
231 imap_subscribe(buf_string(buf), (op == OP_BROWSER_SUBSCRIBE));
232 buf_pool_release(&buf);
233 }
234 return FR_SUCCESS;
235}

264static int op_browser_view_file(struct BrowserPrivateData *priv, int op)
265{
266 if (ARRAY_EMPTY(&priv->state.entry))
267 {
268 mutt_error(_("No files match the file mask"));
269 return FR_ERROR;
270 }
271
272 int index = menu_get_index(priv->menu);
273 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
274 if (ff->selectable)
275 {
276 buf_strcpy(priv->file, ff->name);

278 return FR_DONE;
279 }
280 else if (S_ISDIR(ff->mode) ||
281 (S_ISLNK(ff->mode) && link_is_dir(buf_string(&LastDir), ff->name)))
282 {
283 mutt_error(_("Can't view a directory"));

285 }
286 else
287 {

289 buf_concat_path(path, buf_string(&LastDir), ff->name);

294 mutt_body_free(&b);
295 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
296 }
297 else
298 {
299 mutt_error(_("Error trying to view file"));
300 }
301 buf_pool_release(&path);
302 }

309static int op_catchup(struct BrowserPrivateData *priv, int op)
310{
311 if (!OptNews)
312 return FR_NOT_IMPL;
313
314 struct NntpMboxData *mdata = NULL;
315
316 int rc = nntp_newsrc_parse(CurrentNewsSrv);
317 if (rc < 0)
318 return FR_ERROR;
319
320 int index = menu_get_index(priv->menu);
321 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
322 if (op == OP_CATCHUP)
323 mdata = mutt_newsgroup_catchup(priv->mailbox, CurrentNewsSrv, ff->name);
324 else
325 mdata = mutt_newsgroup_uncatchup(priv->mailbox, CurrentNewsSrv, ff->name);
326
327 if (mdata)
328 {
329 nntp_newsrc_update(CurrentNewsSrv);
330 index = menu_get_index(priv->menu) + 1;
331 if (index < priv->menu->max)
332 menu_set_index(priv->menu, index);
333 }
334
335 if (rc != 0)
336 menu_queue_redraw(priv->menu, MENU_REDRAW_INDEX);
337
338 nntp_newsrc_close(CurrentNewsSrv);
339 return FR_ERROR;
340}

349static int op_change_directory(struct BrowserPrivateData *priv, int op)
350{
351 if (OptNews)
352 return FR_NOT_IMPL;
353
354 struct Buffer *buf = buf_pool_get();
355 buf_copy(buf, &LastDir);
356 if (!priv->state.imap_browse)
357 {
358 /* add '/' at the end of the directory name if not already there */
359 size_t len = buf_len(buf);
360 if ((len > 0) && (buf_string(&LastDir)[len - 1] != '/'))
361 buf_addch(buf, '/');
362 }
363
364 if (op == OP_CHANGE_DIRECTORY)
365 {
366 struct FileCompletionData cdata = { false, priv->mailbox, NULL, NULL, NULL };
367 int rc = mw_get_field(_("Chdir to: "), buf, MUTT_COMP_NO_FLAGS, HC_FILE,
368 &CompleteMailboxOps, &cdata);
369 if ((rc != 0) && buf_is_empty(buf))
370 {
371 buf_pool_release(&buf);
372 return FR_NO_ACTION;
373 }
374 }
375 else if (op == OP_GOTO_PARENT)
376 {
377 mutt_get_parent_path(buf_string(buf), buf->data, buf->dsize);
378 }
379
380 if (!buf_is_empty(buf))
381 {
382 priv->state.is_mailbox_list = false;
383 buf_expand_path(buf);
384 if (imap_path_probe(buf_string(buf), NULL) == MUTT_IMAP)
385 {
386 buf_copy(&LastDir, buf);
387 destroy_state(&priv->state);
388 init_state(&priv->state);
389 priv->state.imap_browse = true;
390 imap_browse(buf_string(&LastDir), &priv->state);
391 browser_sort(&priv->state);
392 browser_highlight_default(&priv->state, priv->menu);
393 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
394 }
395 else
396 {
397 if (buf_string(buf)[0] != '/')
398 {
399 /* in case dir is relative, make it relative to LastDir,
400 * not current working dir */
401 struct Buffer *tmp = buf_pool_get();
402 buf_concat_path(tmp, buf_string(&LastDir), buf_string(buf));
403 buf_copy(buf, tmp);
404 buf_pool_release(&tmp);
405 }
406 /* Resolve path from <chdir>
407 * Avoids buildup such as /a/b/../../c
408 * Symlinks are always unraveled to keep code simple */
409 if (mutt_path_realpath(buf) == 0)
410 {
411 buf_pool_release(&buf);
412 return FR_ERROR;
413 }
414
415 struct stat st = { 0 };
416 if (stat(buf_string(buf), &st) == 0)
417 {
418 if (S_ISDIR(st.st_mode))
419 {
420 destroy_state(&priv->state);
421 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
422 buf_string(buf), buf_string(priv->prefix)) == 0)
423 {
424 buf_copy(&LastDir, buf);
425 }
426 else
427 {
428 mutt_error(_("Error scanning directory"));
429 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
430 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
431 {
432 priv->done = true;
433 return FR_ERROR;
434 }
435 }
436 browser_highlight_default(&priv->state, priv->menu);
437 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
438 }
439 else
440 {
441 mutt_error(_("%s is not a directory"), buf_string(buf));
442 }
443 }
444 else
445 {
446 mutt_perror("%s", buf_string(buf));
447 }
448 }
449 }
450 buf_pool_release(&buf);
451 return FR_ERROR;
452}

484static int op_delete_mailbox(struct BrowserPrivateData *priv, int op)
485{
486 int index = menu_get_index(priv->menu);
487 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
488 if (!ff->imap)
489 {
490 mutt_error(_("Delete is only supported for IMAP mailboxes"));
491 return FR_ERROR;
492 }
493
494 char msg[128] = { 0 };
495
496 // TODO(sileht): It could be better to select INBOX instead. But I
497 // don't want to manipulate Mailboxes/mailbox->account here for now.
498 // Let's just protect neomutt against crash for now. #1417
499 if (mutt_str_equal(mailbox_path(priv->mailbox), ff->name))
500 {
501 mutt_error(_("Can't delete currently selected mailbox"));
502 return FR_ERROR;
503 }
504
505 snprintf(msg, sizeof(msg), _("Really delete mailbox \"%s\"?"), ff->name);
506 if (query_yesorno(msg, MUTT_NO) != MUTT_YES)
507 {
508 mutt_message(_("Mailbox not deleted"));
509 return FR_NO_ACTION;
510 }
511
512 if (imap_delete_mailbox(priv->mailbox, ff->name) != 0)
513 {
514 mutt_error(_("Mailbox deletion failed"));
515 return FR_ERROR;
516 }
517
518 /* free the mailbox from the browser */
519 FREE(&ff->name);
520 FREE(&ff->desc);
521 /* and move all other entries up */
522 ARRAY_REMOVE(&priv->state.entry, ff);
523 mutt_message(_("Mailbox deleted"));
524 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
525
526 return FR_SUCCESS;
527}

532static int op_enter_mask(struct BrowserPrivateData *priv, int op)
533{
534 const struct Regex *c_mask = cs_subset_regex(NeoMutt->sub, "mask");
535 struct Buffer *buf = buf_pool_get();
536 buf_strcpy(buf, c_mask ? c_mask->pattern : NULL);
537 if (mw_get_field(_("File Mask: "), buf, MUTT_COMP_NO_FLAGS, HC_OTHER, NULL, NULL) != 0)
538 {
539 buf_pool_release(&buf);
540 return FR_NO_ACTION;
541 }
542
543 buf_fix_dptr(buf);
544
545 priv->state.is_mailbox_list = false;
546 /* assume that the user wants to see everything */
547 if (buf_is_empty(buf))
548 buf_strcpy(buf, ".");
549
550 struct Buffer *errmsg = buf_pool_get();
551 int rc = cs_subset_str_string_set(NeoMutt->sub, "mask", buf_string(buf), errmsg);
552 buf_pool_release(&buf);
553 if (CSR_RESULT(rc) != CSR_SUCCESS)
554 {
555 if (!buf_is_empty(errmsg))
556 {
557 mutt_error("%s", buf_string(errmsg));
558 buf_pool_release(&errmsg);
559 }
560 return FR_ERROR;
561 }
562 buf_pool_release(&errmsg);
563
564 destroy_state(&priv->state);
565 if (priv->state.imap_browse)
566 {
567 init_state(&priv->state);
568 priv->state.imap_browse = true;
569 imap_browse(buf_string(&LastDir), &priv->state);
570 browser_sort(&priv->state);
571 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
572 }
573 else if (examine_directory(priv->mailbox, priv->menu, &priv->state,
574 buf_string(&LastDir), NULL) == 0)
575 {
576 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
577 }
578 else
579 {
580 mutt_error(_("Error scanning directory"));
581 priv->done = true;
582 return FR_ERROR;
583 }
584 priv->kill_prefix = false;
585 if (ARRAY_EMPTY(&priv->state.entry))
586 {
587 mutt_error(_("No files match the file mask"));
588 return FR_ERROR;
589 }
590 return FR_SUCCESS;
591}

642static int op_generic_select_entry(struct BrowserPrivateData *priv, int op)
643{
644 if (ARRAY_EMPTY(&priv->state.entry))
645 {
646 mutt_error(_("No files match the file mask"));
647 return FR_ERROR;
648 }
649
650 int index = menu_get_index(priv->menu);
651 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
652 if ((priv->menu->tag_prefix) && (op == OP_GENERIC_SELECT_ENTRY))
653 {
654 // Do nothing
655 }
656 else if (S_ISDIR(ff->mode) ||
657 (S_ISLNK(ff->mode) && link_is_dir(buf_string(&LastDir), ff->name)) || ff->inferiors)
658 {
659 /* make sure this isn't a MH or maildir mailbox */
660 struct Buffer *buf = buf_pool_get();
661 if (priv->state.is_mailbox_list)
662 {
663 buf_strcpy(buf, ff->name);
664 buf_expand_path(buf);
665 }
666 else if (priv->state.imap_browse)
667 {
668 buf_strcpy(buf, ff->name);
669 }
670 else
671 {
672 buf_concat_path(buf, buf_string(&LastDir), ff->name);
673 }
674
675 enum MailboxType type = mx_path_probe(buf_string(buf));
676 buf_pool_release(&buf);
677
678 const bool dot_dot = mutt_str_equal(ff->name, "..");
679
680 // Descend into a directory, if:
681 // 1) User selected '..'
682 // 2) User pressed <descend-directory>
683 // 3) User is looking for a Mailbox and this directory isn't one
684 if (dot_dot || (op == OP_DESCEND_DIRECTORY) || (priv->folder && (type <= MUTT_UNKNOWN)))
685 {
686 /* save the old directory */
687 buf_copy(priv->old_last_dir, &LastDir);
688
689 if (dot_dot)
690 {
691 size_t lastdirlen = buf_len(&LastDir);
692 if ((lastdirlen > 1) && mutt_str_equal("..", buf_string(&LastDir) + lastdirlen - 2))
693 {
694 buf_addstr(&LastDir, "/..");
695 }
696 else
697 {
698 char *p = NULL;
699 if (lastdirlen > 1)
700 p = strrchr(LastDir.data + 1, '/');
701
702 if (p)
703 {
704 *p = '\0';
705 buf_fix_dptr(&LastDir);
706 }
707 else
708 {
709 if (buf_string(&LastDir)[0] == '/')
710 buf_strcpy(&LastDir, "/");
711 else
712 buf_addstr(&LastDir, "/..");
713 }
714 }
715 }
716 else if (priv->state.is_mailbox_list)
717 {
718 buf_strcpy(&LastDir, ff->name);
719 buf_expand_path(&LastDir);
720 }
721 else if (priv->state.imap_browse)
722 {
723 buf_strcpy(&LastDir, ff->name);
724 /* tack on delimiter here */
725
726 /* special case "" needs no delimiter */
727 struct Url *url = url_parse(ff->name);
728 if (url && url->path && (ff->delim != '\0'))
729 {
730 buf_addch(&LastDir, ff->delim);
731 }
732 url_free(&url);
733 }
734 else
735 {
736 struct Buffer *tmp = buf_pool_get();
737 buf_concat_path(tmp, buf_string(&LastDir), ff->name);
738 buf_copy(&LastDir, tmp);
739 buf_pool_release(&tmp);
740 }
741
742 destroy_state(&priv->state);
743 if (priv->kill_prefix)
744 {
745 buf_reset(priv->prefix);
746 priv->kill_prefix = false;
747 }
748 priv->state.is_mailbox_list = false;
749 if (priv->state.imap_browse)
750 {
751 init_state(&priv->state);
752 priv->state.imap_browse = true;
753 imap_browse(buf_string(&LastDir), &priv->state);
754 browser_sort(&priv->state);
755 }
756 else
757 {
758 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
759 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
760 {
761 /* try to restore the old values */
762 buf_copy(&LastDir, priv->old_last_dir);
763 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
764 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
765 {
766 buf_strcpy(&LastDir, NeoMutt->home_dir);
767 priv->done = true;
768 return FR_DONE;
769 }
770 }
771 /* resolve paths navigated from GUI */
772 if (mutt_path_realpath(&LastDir) == 0)
773 return FR_ERROR;
774 }
775
776 browser_highlight_default(&priv->state, priv->menu);
777 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
778 priv->goto_swapper[0] = '\0';
779 return FR_SUCCESS;
780 }
781 }
782 else if (op == OP_DESCEND_DIRECTORY)
783 {
784 mutt_error(_("%s is not a directory"), ARRAY_GET(&priv->state.entry, index)->name);
785 return FR_ERROR;
786 }
787
788 if (priv->state.is_mailbox_list || OptNews)
789 {
790 buf_strcpy(priv->file, ff->name);
791 buf_expand_path(priv->file);
792 }
793 else if (priv->state.imap_browse)
794 {
795 buf_strcpy(priv->file, ff->name);
796 }
797 else
798 {
799 buf_concat_path(priv->file, buf_string(&LastDir), ff->name);
800 }
801
802 return op_exit(priv, op);
803}

885static int op_sort(struct BrowserPrivateData *priv, int op)
886{
887 bool resort = true;
888 int sort = -1;
889 int reverse = (op == OP_SORT_REVERSE);
890
891 switch (mw_multi_choice((reverse) ?
892 /* L10N: The highlighted letters must match the "Sort" options */
893 _("Reverse sort by (d)ate, (a)lpha, si(z)e, d(e)scription, (c)ount, ne(w) count, or do(n)'t sort?") :
894 /* L10N: The highlighted letters must match the "Reverse Sort" options */
895 _("Sort by (d)ate, (a)lpha, si(z)e, d(e)scription, (c)ount, ne(w) count, or do(n)'t sort?"),
896 /* L10N: These must match the highlighted letters from "Sort" and "Reverse Sort" */
897 _("dazecwn")))
898 {
899 case -1: /* abort */
900 resort = false;
901 break;
902
903 case 1: /* (d)ate */
904 sort = BROWSER_SORT_DATE;
905 break;
906
907 case 2: /* (a)lpha */
908 sort = BROWSER_SORT_ALPHA;
909 break;
910
911 case 3: /* si(z)e */
912 sort = BROWSER_SORT_SIZE;
913 break;
914
915 case 4: /* d(e)scription */
916 sort = BROWSER_SORT_DESC;
917 break;
918
919 case 5: /* (c)ount */
920 sort = BROWSER_SORT_COUNT;
921 break;
922
923 case 6: /* ne(w) count */
924 sort = BROWSER_SORT_NEW;
925 break;
926
927 case 7: /* do(n)'t sort */
928 sort = BROWSER_SORT_UNSORTED;
929 break;
930 }
931
932 if (!resort)
933 return FR_NO_ACTION;
934
935 sort |= reverse ? SORT_REVERSE : 0;
936 cs_subset_str_native_set(NeoMutt->sub, "browser_sort", sort, NULL);
937 browser_sort(&priv->state);
938 browser_highlight_default(&priv->state, priv->menu);

950static int op_subscribe_pattern(struct BrowserPrivateData *priv, int op)
951{
952 if (!OptNews)
953 return FR_NOT_IMPL;
954
955 struct NntpAccountData *adata = CurrentNewsSrv;
956 regex_t rx = { 0 };
957 int index = menu_get_index(priv->menu);
958
959 char tmp2[256] = { 0 };
960
961 struct Buffer *buf = buf_pool_get();
962 if (op == OP_SUBSCRIBE_PATTERN)
963 snprintf(tmp2, sizeof(tmp2), _("Subscribe pattern: "));
964 else
965 snprintf(tmp2, sizeof(tmp2), _("Unsubscribe pattern: "));
966 /* buf comes from the buffer pool, so defaults to size 1024 */
967 if ((mw_get_field(tmp2, buf, MUTT_COMP_NO_FLAGS, HC_PATTERN, &CompletePatternOps, NULL) != 0) ||
968 buf_is_empty(buf))
969 {
970 buf_pool_release(&buf);
971 return FR_NO_ACTION;
972 }
973
974 int err = REG_COMP(&rx, buf->data, REG_NOSUB);
975 if (err != 0)
976 {
977 regerror(err, &rx, buf->data, buf->dsize);
978 regfree(&rx);
979 mutt_error("%s", buf_string(buf));
980 buf_pool_release(&buf);
981 return FR_ERROR;
982 }
983 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
984 index = 0;
985 buf_pool_release(&buf);
986
987 int rc = nntp_newsrc_parse(adata);
988 if (rc < 0)
989 return FR_ERROR;
990
991 struct FolderFile *ff = NULL;
992 ARRAY_FOREACH_FROM(ff, &priv->state.entry, index)
993 {
994 if (regexec(&rx, ff->name, 0, NULL, 0) == 0)
995 {
996 if (op == OP_SUBSCRIBE_PATTERN)
997 mutt_newsgroup_subscribe(adata, ff->name);
998 else
999 mutt_newsgroup_unsubscribe(adata, ff->name);
1000 }
1001 }
1002
1003 if (op == OP_SUBSCRIBE_PATTERN)
1004 {
1005 for (size_t j = 0; j < adata->groups_num; j++)
1006 {
1007 struct NntpMboxData *mdata = adata->groups_list[j];
1008 if (mdata && mdata->group && !mdata->subscribed)
1009 {
1010 if (regexec(&rx, mdata->group, 0, NULL, 0) == 0)
1011 {
1012 mutt_newsgroup_subscribe(adata, mdata->group);
1013 browser_add_folder(priv->menu, &priv->state, mdata->group, NULL, NULL, NULL, mdata);
1014 }
1015 }
1016 }
1017 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
1018 }
1019 if (rc > 0)
1020 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
1021 nntp_newsrc_update(adata);
1022 nntp_clear_cache(adata);
1023 nntp_newsrc_close(adata);
1024 regfree(&rx);
1025 return FR_SUCCESS;
1026}

1035static int op_toggle_mailboxes(struct BrowserPrivateData *priv, int op)
1036{
1037 if (priv->state.is_mailbox_list)
1038 {
1039 priv->last_selected_mailbox = menu_get_index(priv->menu);
1040 }
1041
1042 if (op == OP_TOGGLE_MAILBOXES)
1043 {
1044 priv->state.is_mailbox_list = !priv->state.is_mailbox_list;
1045 }
1046
1047 if (op == OP_BROWSER_GOTO_FOLDER)
1048 {
1049 /* When in mailboxes mode, disables this feature */
1050 const char *const c_folder = cs_subset_string(NeoMutt->sub, "folder");
1051 if (c_folder)
1052 {
1053 mutt_debug(LL_DEBUG3, "= hit! Folder: %s, LastDir: %s\n", c_folder,
1054 buf_string(&LastDir));
1055 if (priv->goto_swapper[0] == '\0')
1056 {
1057 if (!mutt_str_equal(buf_string(&LastDir), c_folder))
1058 {
1059 /* Stores into goto_swapper LastDir, and swaps to `$folder` */
1060 mutt_str_copy(priv->goto_swapper, buf_string(&LastDir), sizeof(priv->goto_swapper));
1061 buf_copy(&LastDirBackup, &LastDir);
1062 buf_strcpy(&LastDir, c_folder);
1063 }
1064 }
1065 else
1066 {
1067 buf_copy(&LastDirBackup, &LastDir);
1068 buf_strcpy(&LastDir, priv->goto_swapper);
1069 priv->goto_swapper[0] = '\0';
1070 }
1071 }
1072 }
1073 destroy_state(&priv->state);
1074 buf_reset(priv->prefix);
1075 priv->kill_prefix = false;
1076
1077 if (priv->state.is_mailbox_list)
1078 {
1079 examine_mailboxes(priv->mailbox, priv->menu, &priv->state);
1080 }
1081 else if (imap_path_probe(buf_string(&LastDir), NULL) == MUTT_IMAP)
1082 {
1083 init_state(&priv->state);
1084 priv->state.imap_browse = true;
1085 imap_browse(buf_string(&LastDir), &priv->state);
1086 browser_sort(&priv->state);
1087 }
1088 else if (examine_directory(priv->mailbox, priv->menu, &priv->state,
1089 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
1090 {
1091 priv->done = true;
1092 return FR_ERROR;
1093 }
1094 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
1095 if (priv->state.is_mailbox_list)
1096 menu_set_index(priv->menu, priv->last_selected_mailbox);
1097 return FR_ERROR;
1098}

1143int browser_function_dispatcher(struct MuttWindow *win, int op)
1144{
1145 // The Dispatcher may be called on any Window in the Dialog

1148 return FR_ERROR;

1152 if (!priv)
1153 return FR_ERROR;
1154
1155 int rc = FR_UNKNOWN;
1156 for (size_t i = 0; BrowserFunctions[i].op != OP_NULL; i++)
1157 {
1158 const struct BrowserFunction *fn = &BrowserFunctions[i];
1159 if (fn->op == op)
1160 {
1161 rc = fn->function(priv, op);
1162 break;
1163 }
1164 }
1165
1166 return rc;
