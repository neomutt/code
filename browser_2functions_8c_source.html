326
327 if (rc != 0)
328 menu_queue_redraw(priv->menu, MENU_REDRAW_INDEX);
329
330 nntp_newsrc_close(CurrentNewsSrv);
331 return FR_ERROR;

341static int op_change_directory(struct BrowserPrivateData *priv, int op)
342{
343 if (OptNews)
344 return FR_NOT_IMPL;
345
346 struct Buffer *buf = buf_pool_get();
347 buf_copy(buf, &LastDir);
348 if (!priv->state.imap_browse)
349 {
350 /* add '/' at the end of the directory name if not already there */
351 size_t len = buf_len(buf);
352 if ((len > 0) && (buf_string(&LastDir)[len - 1] != '/'))
353 buf_addch(buf, '/');
354 }
355
356 if (op == OP_CHANGE_DIRECTORY)
357 {
358 struct FileCompletionData cdata = { false, priv->mailbox, NULL, NULL };
359 int rc = mw_get_field(_("Chdir to: "), buf, MUTT_COMP_NO_FLAGS, HC_FILE,
360 &CompleteMailboxOps, &cdata);
361 if ((rc != 0) && buf_is_empty(buf))
362 {
363 buf_pool_release(&buf);
364 return FR_NO_ACTION;
365 }
366 }
367 else if (op == OP_GOTO_PARENT)
368 {
369 mutt_get_parent_path(buf_string(buf), buf->data, buf->dsize);
370 }
371
372 if (!buf_is_empty(buf))
373 {
374 priv->state.is_mailbox_list = false;
375 buf_expand_path(buf);
376 if (imap_path_probe(buf_string(buf), NULL) == MUTT_IMAP)
377 {
378 buf_copy(&LastDir, buf);
379 destroy_state(&priv->state);
380 init_state(&priv->state);
381 priv->state.imap_browse = true;
382 imap_browse(buf_string(&LastDir), &priv->state);
383 browser_sort(&priv->state);
384 browser_highlight_default(&priv->state, priv->menu);
385 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
386 }
387 else
388 {
389 if (buf_string(buf)[0] != '/')
390 {
391 /* in case dir is relative, make it relative to LastDir,
392 * not current working dir */
393 struct Buffer *tmp = buf_pool_get();
394 buf_concat_path(tmp, buf_string(&LastDir), buf_string(buf));
395 buf_copy(buf, tmp);
396 buf_pool_release(&tmp);
397 }
398 /* Resolve path from <chdir>
399 * Avoids buildup such as /a/b/../../c
400 * Symlinks are always unraveled to keep code simple */
401 if (mutt_path_realpath(buf) == 0)
402 {
403 buf_pool_release(&buf);
404 return FR_ERROR;
405 }
406
407 struct stat st = { 0 };
408 if (stat(buf_string(buf), &st) == 0)
409 {
410 if (S_ISDIR(st.st_mode))
411 {
412 destroy_state(&priv->state);
413 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
414 buf_string(buf), buf_string(priv->prefix)) == 0)
415 {
416 buf_copy(&LastDir, buf);
417 }
418 else
419 {
420 mutt_error(_("Error scanning directory"));
421 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
422 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
423 {
424 priv->done = true;
425 return FR_ERROR;
426 }
427 }
428 browser_highlight_default(&priv->state, priv->menu);
429 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
430 }
431 else
432 {
433 mutt_error(_("%s is not a directory"), buf_string(buf));
434 }
435 }
436 else
437 {
438 mutt_perror("%s", buf_string(buf));
439 }
440 }
441 }
442 buf_pool_release(&buf);
443 return FR_ERROR;
444}

476static int op_delete_mailbox(struct BrowserPrivateData *priv, int op)
477{
478 int index = menu_get_index(priv->menu);
479 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
480 if (!ff->imap)
481 {
482 mutt_error(_("Delete is only supported for IMAP mailboxes"));
483 return FR_ERROR;
484 }
485
486 char msg[128] = { 0 };
487
488 // TODO(sileht): It could be better to select INBOX instead. But I
489 // don't want to manipulate Mailboxes/mailbox->account here for now.
490 // Let's just protect neomutt against crash for now. #1417

492 {
493 mutt_error(_("Can't delete currently selected mailbox"));
494 return FR_ERROR;
495 }
496

498 if (query_yesorno(msg, MUTT_NO) != MUTT_YES)

503
504 if (imap_delete_mailbox(priv->mailbox, ff->name) != 0)
505 {
506 mutt_error(_("Mailbox deletion failed"));
507 return FR_ERROR;
508 }
509
510 /* free the mailbox from the browser */
511 FREE(&ff->name);
512 FREE(&ff->desc);
513 /* and move all other entries up */
514 ARRAY_REMOVE(&priv->state.entry, ff);
515 mutt_message(_("Mailbox deleted"));
516 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
517
518 return FR_SUCCESS;
519}

524static int op_enter_mask(struct BrowserPrivateData *priv, int op)
525{
526 const struct Regex *c_mask = cs_subset_regex(NeoMutt->sub, "mask");
527 struct Buffer *buf = buf_pool_get();
528 buf_strcpy(buf, c_mask ? c_mask->pattern : NULL);
529 if (mw_get_field(_("File Mask: "), buf, MUTT_COMP_NO_FLAGS, HC_OTHER, NULL, NULL) != 0)
530 {
531 buf_pool_release(&buf);
532 return FR_NO_ACTION;
533 }
534
535 buf_fix_dptr(buf);
536
537 priv->state.is_mailbox_list = false;
538 /* assume that the user wants to see everything */
539 if (buf_is_empty(buf))
540 buf_strcpy(buf, ".");
541
542 struct Buffer *errmsg = buf_pool_get();
543 int rc = cs_subset_str_string_set(NeoMutt->sub, "mask", buf_string(buf), errmsg);
544 buf_pool_release(&buf);
545 if (CSR_RESULT(rc) != CSR_SUCCESS)
546 {
547 if (!buf_is_empty(errmsg))
548 {
549 mutt_error("%s", buf_string(errmsg));
550 buf_pool_release(&errmsg);
551 }
552 return FR_ERROR;
553 }
554 buf_pool_release(&errmsg);
555
556 destroy_state(&priv->state);
557 if (priv->state.imap_browse)
558 {
559 init_state(&priv->state);
560 priv->state.imap_browse = true;
561 imap_browse(buf_string(&LastDir), &priv->state);
562 browser_sort(&priv->state);
563 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
564 }
565 else if (examine_directory(priv->mailbox, priv->menu, &priv->state,
566 buf_string(&LastDir), NULL) == 0)
567 {
568 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
569 }
570 else
571 {
572 mutt_error(_("Error scanning directory"));
573 priv->done = true;
574 return FR_ERROR;
575 }
576 priv->kill_prefix = false;
577 if (ARRAY_EMPTY(&priv->state.entry))
578 {
579 mutt_error(_("No files match the file mask"));
580 return FR_ERROR;
581 }
582 return FR_SUCCESS;
583}

634static int op_generic_select_entry(struct BrowserPrivateData *priv, int op)
635{
636 if (ARRAY_EMPTY(&priv->state.entry))
637 {
638 mutt_error(_("No files match the file mask"));
639 return FR_ERROR;
640 }
641
642 int index = menu_get_index(priv->menu);
643 struct FolderFile *ff = ARRAY_GET(&priv->state.entry, index);
644 if ((priv->menu->tag_prefix) && (op == OP_GENERIC_SELECT_ENTRY))
645 {
646 // Do nothing
647 }
648 else if (S_ISDIR(ff->mode) ||
649 (S_ISLNK(ff->mode) && link_is_dir(buf_string(&LastDir), ff->name)) || ff->inferiors)
650 {
651 /* make sure this isn't a MH or maildir mailbox */
652 struct Buffer *buf = buf_pool_get();
653 if (priv->state.is_mailbox_list)
654 {
655 buf_strcpy(buf, ff->name);
656 buf_expand_path(buf);
657 }
658 else if (priv->state.imap_browse)
659 {
660 buf_strcpy(buf, ff->name);
661 }
662 else
663 {
664 buf_concat_path(buf, buf_string(&LastDir), ff->name);
665 }
666
667 enum MailboxType type = mx_path_probe(buf_string(buf));
668 buf_pool_release(&buf);
669
670 if ((op == OP_DESCEND_DIRECTORY) || (type == MUTT_MAILBOX_ERROR) ||
671 (type == MUTT_UNKNOWN) || ff->inferiors)
672 {
673 /* save the old directory */
674 buf_copy(priv->old_last_dir, &LastDir);
675
676 if (mutt_str_equal(ff->name, ".."))
677 {
678 size_t lastdirlen = buf_len(&LastDir);
679 if ((lastdirlen > 1) && mutt_str_equal("..", buf_string(&LastDir) + lastdirlen - 2))
680 {
681 buf_addstr(&LastDir, "/..");
682 }
683 else
684 {
685 char *p = NULL;
686 if (lastdirlen > 1)
687 p = strrchr(LastDir.data + 1, '/');
688
689 if (p)
690 {
691 *p = '\0';
692 buf_fix_dptr(&LastDir);
693 }
694 else
695 {
696 if (buf_string(&LastDir)[0] == '/')
697 buf_strcpy(&LastDir, "/");
698 else
699 buf_addstr(&LastDir, "/..");
700 }
701 }
702 }
703 else if (priv->state.is_mailbox_list)
704 {
705 buf_strcpy(&LastDir, ff->name);
706 buf_expand_path(&LastDir);
707 }
708 else if (priv->state.imap_browse)
709 {
710 buf_strcpy(&LastDir, ff->name);
711 /* tack on delimiter here */
712
713 /* special case "" needs no delimiter */
714 struct Url *url = url_parse(ff->name);
715 if (url && url->path && (ff->delim != '\0'))
716 {
717 buf_addch(&LastDir, ff->delim);
718 }
719 url_free(&url);
720 }
721 else
722 {
723 struct Buffer *tmp = buf_pool_get();
724 buf_concat_path(tmp, buf_string(&LastDir), ff->name);
725 buf_copy(&LastDir, tmp);
726 buf_pool_release(&tmp);
727 }
728
729 destroy_state(&priv->state);
730 if (priv->kill_prefix)
731 {
732 buf_reset(priv->prefix);
733 priv->kill_prefix = false;
734 }
735 priv->state.is_mailbox_list = false;
736 if (priv->state.imap_browse)
737 {
738 init_state(&priv->state);
739 priv->state.imap_browse = true;
740 imap_browse(buf_string(&LastDir), &priv->state);
741 browser_sort(&priv->state);
742 }
743 else
744 {
745 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
746 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
747 {
748 /* try to restore the old values */
749 buf_copy(&LastDir, priv->old_last_dir);
750 if (examine_directory(priv->mailbox, priv->menu, &priv->state,
751 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
752 {
753 buf_strcpy(&LastDir, NeoMutt->home_dir);
754 priv->done = true;
755 return FR_DONE;
756 }
757 }
758 /* resolve paths navigated from GUI */
759 if (mutt_path_realpath(&LastDir) == 0)
760 return FR_ERROR;
761 }
762
763 browser_highlight_default(&priv->state, priv->menu);
764 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
765 priv->goto_swapper[0] = '\0';
766 return FR_SUCCESS;
767 }
768 }
769 else if (op == OP_DESCEND_DIRECTORY)
770 {
771 mutt_error(_("%s is not a directory"), ARRAY_GET(&priv->state.entry, index)->name);
772 return FR_ERROR;
773 }
774
775 if (priv->state.is_mailbox_list || OptNews)
776 {
777 buf_strcpy(priv->file, ff->name);
778 buf_expand_path(priv->file);
779 }
780 else if (priv->state.imap_browse)
781 {
782 buf_strcpy(priv->file, ff->name);
783 }
784 else
785 {
786 buf_concat_path(priv->file, buf_string(&LastDir), ff->name);
787 }
788
789 return op_exit(priv, op);
790}

795static int op_load_active(struct BrowserPrivateData *priv, int op)
796{
797 if (!OptNews)
798 return FR_NOT_IMPL;
799
800 struct NntpAccountData *adata = CurrentNewsSrv;
801
802 if (nntp_newsrc_parse(adata) < 0)
803 return FR_ERROR;
804
805 for (size_t i = 0; i < adata->groups_num; i++)

872static int op_sort(struct BrowserPrivateData *priv, int op)
873{
874 bool resort = true;
875 int sort = -1;
876 int reverse = (op == OP_SORT_REVERSE);
877
878 switch (mw_multi_choice((reverse) ?
879 /* L10N: The highlighted letters must match the "Sort" options */
880 _("Reverse sort by (d)ate, (a)lpha, si(z)e, d(e)scription, (c)ount, ne(w) count, or do(n)'t sort?") :
881 /* L10N: The highlighted letters must match the "Reverse Sort" options */
882 _("Sort by (d)ate, (a)lpha, si(z)e, d(e)scription, (c)ount, ne(w) count, or do(n)'t sort?"),
883 /* L10N: These must match the highlighted letters from "Sort" and "Reverse Sort" */
884 _("dazecwn")))
885 {
886 case -1: /* abort */
887 resort = false;
888 break;
889
890 case 1: /* (d)ate */
891 sort = BROWSER_SORT_DATE;
892 break;
893
894 case 2: /* (a)lpha */
895 sort = BROWSER_SORT_ALPHA;
896 break;
897
898 case 3: /* si(z)e */
899 sort = BROWSER_SORT_SIZE;
900 break;
901
902 case 4: /* d(e)scription */
903 sort = BROWSER_SORT_DESC;
904 break;
905
906 case 5: /* (c)ount */
907 sort = BROWSER_SORT_COUNT;
908 break;
909
910 case 6: /* ne(w) count */
911 sort = BROWSER_SORT_NEW;
912 break;
913
914 case 7: /* do(n)'t sort */
915 sort = BROWSER_SORT_UNSORTED;
916 break;
917 }
918
919 if (!resort)
920 return FR_NO_ACTION;
921
922 sort |= reverse ? SORT_REVERSE : 0;
923 cs_subset_str_native_set(NeoMutt->sub, "browser_sort", sort, NULL);
924 browser_sort(&priv->state);
925 browser_highlight_default(&priv->state, priv->menu);
926 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
927 return FR_SUCCESS;

937static int op_subscribe_pattern(struct BrowserPrivateData *priv, int op)
938{
939 if (!OptNews)
940 return FR_NOT_IMPL;
941
942 struct NntpAccountData *adata = CurrentNewsSrv;
943 regex_t rx = { 0 };
944 int index = menu_get_index(priv->menu);
945
946 char tmp2[256] = { 0 };
947
948 struct Buffer *buf = buf_pool_get();
949 if (op == OP_SUBSCRIBE_PATTERN)
950 snprintf(tmp2, sizeof(tmp2), _("Subscribe pattern: "));
951 else
952 snprintf(tmp2, sizeof(tmp2), _("Unsubscribe pattern: "));
953 /* buf comes from the buffer pool, so defaults to size 1024 */
954 if ((mw_get_field(tmp2, buf, MUTT_COMP_NO_FLAGS, HC_PATTERN, &CompletePatternOps, NULL) != 0) ||
955 buf_is_empty(buf))
956 {
957 buf_pool_release(&buf);
958 return FR_NO_ACTION;
959 }
960
961 int err = REG_COMP(&rx, buf->data, REG_NOSUB);
962 if (err != 0)
963 {
964 regerror(err, &rx, buf->data, buf->dsize);
965 regfree(&rx);
966 mutt_error("%s", buf_string(buf));
967 buf_pool_release(&buf);
968 return FR_ERROR;
969 }
970 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
971 index = 0;
972 buf_pool_release(&buf);
973
974 int rc = nntp_newsrc_parse(adata);
975 if (rc < 0)
976 return FR_ERROR;
977
978 struct FolderFile *ff = NULL;
979 ARRAY_FOREACH_FROM(ff, &priv->state.entry, index)
980 {
981 if (regexec(&rx, ff->name, 0, NULL, 0) == 0)
982 {
983 if (op == OP_SUBSCRIBE_PATTERN)
984 mutt_newsgroup_subscribe(adata, ff->name);
985 else
986 mutt_newsgroup_unsubscribe(adata, ff->name);
987 }
988 }
989
990 if (op == OP_SUBSCRIBE_PATTERN)
991 {
992 for (size_t j = 0; j < adata->groups_num; j++)
993 {
994 struct NntpMboxData *mdata = adata->groups_list[j];
995 if (mdata && mdata->group && !mdata->subscribed)
996 {
997 if (regexec(&rx, mdata->group, 0, NULL, 0) == 0)
998 {
999 mutt_newsgroup_subscribe(adata, mdata->group);
1000 browser_add_folder(priv->menu, &priv->state, mdata->group, NULL, NULL, NULL, mdata);
1001 }
1002 }
1003 }
1004 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
1005 }
1006 if (rc > 0)
1007 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
1008 nntp_newsrc_update(adata);
1009 nntp_clear_cache(adata);
1010 nntp_newsrc_close(adata);
1011 regfree(&rx);
1012 return FR_SUCCESS;
1013}

1022static int op_toggle_mailboxes(struct BrowserPrivateData *priv, int op)
1023{
1024 if (priv->state.is_mailbox_list)
1025 {
1026 priv->last_selected_mailbox = menu_get_index(priv->menu);
1027 }
1028
1029 if (op == OP_TOGGLE_MAILBOXES)
1030 {
1031 priv->state.is_mailbox_list = !priv->state.is_mailbox_list;
1032 }
1033
1034 if (op == OP_BROWSER_GOTO_FOLDER)
1035 {
1036 /* When in mailboxes mode, disables this feature */
1037 const char *const c_folder = cs_subset_string(NeoMutt->sub, "folder");
1038 if (c_folder)
1039 {
1040 mutt_debug(LL_DEBUG3, "= hit! Folder: %s, LastDir: %s\n", c_folder,
1041 buf_string(&LastDir));
1042 if (priv->goto_swapper[0] == '\0')
1043 {
1044 if (!mutt_str_equal(buf_string(&LastDir), c_folder))
1045 {
1046 /* Stores into goto_swapper LastDir, and swaps to `$folder` */
1047 mutt_str_copy(priv->goto_swapper, buf_string(&LastDir), sizeof(priv->goto_swapper));
1048 buf_copy(&LastDirBackup, &LastDir);
1049 buf_strcpy(&LastDir, c_folder);
1050 }
1051 }
1052 else
1053 {
1054 buf_copy(&LastDirBackup, &LastDir);
1055 buf_strcpy(&LastDir, priv->goto_swapper);
1056 priv->goto_swapper[0] = '\0';
1057 }
1058 }
1059 }
1060 destroy_state(&priv->state);
1061 buf_reset(priv->prefix);
1062 priv->kill_prefix = false;
1063
1064 if (priv->state.is_mailbox_list)
1065 {
1066 examine_mailboxes(priv->mailbox, priv->menu, &priv->state);
1067 }
1068 else if (imap_path_probe(buf_string(&LastDir), NULL) == MUTT_IMAP)
1069 {
1070 init_state(&priv->state);
1071 priv->state.imap_browse = true;
1072 imap_browse(buf_string(&LastDir), &priv->state);
1073 browser_sort(&priv->state);
1074 }
1075 else if (examine_directory(priv->mailbox, priv->menu, &priv->state,
1076 buf_string(&LastDir), buf_string(priv->prefix)) == -1)
1077 {
1078 priv->done = true;
1079 return FR_ERROR;
1080 }
1081 init_menu(&priv->state, priv->menu, priv->mailbox, priv->sbar);
1082 if (priv->state.is_mailbox_list)
1083 menu_set_index(priv->menu, priv->last_selected_mailbox);
1084 return FR_ERROR;
1085}

1130int browser_function_dispatcher(struct MuttWindow *win, int op)
1131{

1135 return FR_ERROR;

1139 if (!priv)
1140 return FR_ERROR;

1142 int rc = FR_UNKNOWN;
1143 for (size_t i = 0; BrowserFunctions[i].op != OP_NULL; i++)
1144 {
1145 const struct BrowserFunction *fn = &BrowserFunctions[i];
1146 if (fn->op == op)
1147 {
1148 rc = fn->function(priv, op);

1150 }
1151 }
