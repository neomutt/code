161static void calc_header_width_padding(int idx, const char *header, bool calc_max)
162{
163 int width;
164
165 HeaderPadding[idx] = mutt_str_len(header);
166 width = mutt_strwidth(header);
167 if (calc_max && (MaxHeaderWidth < width))
168 MaxHeaderWidth = width;
169 HeaderPadding[idx] -= width;
170}

180static void init_header_padding(void)
181{
182 static bool done = false;
183
184 if (done)
185 return;
186 done = true;
187
188 for (int i = 0; i < HDR_ATTACH_TITLE; i++)
189 {
190 if (i == HDR_CRYPTINFO)
191 continue;
192 calc_header_width_padding(i, _(Prompts[i]), true);
193 }
194
195 /* Don't include "Sign as: " in the MaxHeaderWidth calculation. It
196 * doesn't show up by default, and so can make the indentation of
197 * the other fields look funny. */
198 calc_header_width_padding(HDR_CRYPTINFO, _(Prompts[HDR_CRYPTINFO]), false);
199
200 for (int i = 0; i < HDR_ATTACH_TITLE; i++)
201 {
202 HeaderPadding[i] += MaxHeaderWidth;
203 if (HeaderPadding[i] < 0)
204 HeaderPadding[i] = 0;
205 }
206}

217static int calc_address(struct AddressList *al, short cols, short *srows)
218{
219 struct ListHead slist = STAILQ_HEAD_INITIALIZER(slist);
220 mutt_addrlist_write_list(al, &slist);
221
222 short rows = 1;
223 int addr_len;
224 int width_left = cols;
225 struct ListNode *next = NULL;
226 struct ListNode *np = NULL;
227 STAILQ_FOREACH(np, &slist, entries)
228 {
229 next = STAILQ_NEXT(np, entries);
230 addr_len = mutt_strwidth(np->data);
231 if (next)
232 addr_len += 2; // ", "
233
234 try_again:
235 if (addr_len >= width_left)
236 {
237 if (width_left == cols)
238 break;
239
240 rows++;
241 width_left = cols;
242 goto try_again;
243 }
244
245 if (addr_len < width_left)
246 width_left -= addr_len;
247 }
248
249 mutt_list_free(&slist);
250
251 *srows = MIN(rows, MAX_ADDR_ROWS);
252 return *srows;
253}

262static int calc_security(struct Email *e, short *rows, const struct ConfigSubset *sub)
263{
264 if ((WithCrypto & (APPLICATION_PGP | APPLICATION_SMIME)) == 0)
265 *rows = 0; // Neither PGP nor SMIME are built into NeoMutt
266 else if ((e->security & (SEC_ENCRYPT | SEC_SIGN)) != 0)
267 *rows = 2; // 'Security:' and 'Sign as:'
268 else
269 *rows = 1; // Just 'Security:'
270
271#ifdef USE_AUTOCRYPT
272 const bool c_autocrypt = cs_subset_bool(sub, "autocrypt");
273 if (c_autocrypt)
274 *rows += 1;
275#endif
276
277 return *rows;
278}

304static int calc_envelope(struct MuttWindow *win, struct EnvelopeWindowData *wdata)
305{
306 int rows = 4; // 'From:', 'Subject:', 'Reply-To:', 'Fcc:'
307 struct Email *e = wdata->email;
308
309 struct Envelope *env = e->env;
310 const int cols = win->state.cols - MaxHeaderWidth;
311
312 if (wdata->is_news)
313 {
314 rows += 2; // 'Newsgroups:' and 'Followup-To:'
315 const bool c_x_comment_to = cs_subset_bool(wdata->sub, "x_comment_to");
316 if (c_x_comment_to)
317 rows++;
318 }
319 else
320 {
321 rows += calc_address(&env->to, cols, &wdata->to_rows);
322 rows += calc_address(&env->cc, cols, &wdata->cc_rows);
323 rows += calc_address(&env->bcc, cols, &wdata->bcc_rows);
324 }
325 rows += calc_security(e, &wdata->sec_rows, wdata->sub);
326 const bool c_compose_show_user_headers = cs_subset_bool(wdata->sub, "compose_show_user_headers");
327 if (c_compose_show_user_headers)
328 rows += calc_user_hdrs(&env->userhdrs);
329
330 return rows;
331}

440
441 if (((WithCrypto & APPLICATION_PGP) != 0) &&
442 (e->security & APPLICATION_PGP) && (e->security & SEC_SIGN))
443 {
444 draw_header(win, row++, HDR_CRYPTINFO);
445 const char *const c_pgp_sign_as = cs_subset_string(wdata->sub, "pgp_sign_as");

447 }
448
449 if (((WithCrypto & APPLICATION_SMIME) != 0) &&
450 (e->security & APPLICATION_SMIME) && (e->security & SEC_SIGN))

452 draw_header(win, row++, HDR_CRYPTINFO);

459 (e->security & SEC_ENCRYPT) && c_smime_encrypt_with)

502static int draw_envelope_addr(int field, struct AddressList *al,
503 struct MuttWindow *win, int row, size_t max_lines)
504{
505 draw_header(win, row, field);
506
507 struct ListHead list = STAILQ_HEAD_INITIALIZER(list);
508 int count = mutt_addrlist_count_recips(al);
509
510 int lines_used = 1;
511 int width_left = win->state.cols - MaxHeaderWidth;
512 char more[32] = { 0 };
513 int more_len = 0;
514
515 struct Buffer *buf = buf_pool_get();
516 bool in_group = false;
517 char *sep = NULL;
518 struct Address *addr = NULL;
519 TAILQ_FOREACH(addr, al, entries)
520 {
521 struct Address *next = TAILQ_NEXT(addr, entries);
522
523 if (addr->group)
524 {
525 in_group = true;
526 }
527
528 buf_reset(buf);
529 mutt_addr_write(buf, addr, true);
530 size_t addr_len = buf_len(buf);
531
532 sep = "";
533 if (!addr->group)
534 {
535 // group terminator
536 if (in_group && next && !next->mailbox && !next->personal)
537 {
538 sep = ";";
539 addr_len += 1;
540 in_group = false;
541 }
542 else if (next)
543 {
544 sep = ", ";
545 addr_len += 2;
546 }
547 }
548
549 count--;
550 try_again:
551 more_len = snprintf(more, sizeof(more),
552 ngettext("(+%d more)", "(+%d more)", count), count);
553 mutt_debug(LL_DEBUG3, "text: '%s' len: %d\n", more, more_len);
554
555 int reserve = ((count > 0) && (lines_used == max_lines)) ? more_len : 0;
556 mutt_debug(LL_DEBUG3, "processing: %s (al:%zu, wl:%d, r:%d, lu:%d)\n",
557 buf_string(buf), addr_len, width_left, reserve, lines_used);
558 if (addr_len >= (width_left - reserve))
559 {
560 mutt_debug(LL_DEBUG3, "not enough space\n");
561 if (lines_used == max_lines)
562 {
563 mutt_debug(LL_DEBUG3, "no more lines\n");
564 mutt_debug(LL_DEBUG3, "truncating: %s\n", buf_string(buf));
565 mutt_paddstr(win, width_left, buf_string(buf));
566 break;
567 }
568
569 if (width_left == (win->state.cols - MaxHeaderWidth))
570 {
571 mutt_debug(LL_DEBUG3, "couldn't print: %s\n", buf_string(buf));
572 mutt_paddstr(win, width_left, buf_string(buf));
573 break;
574 }
575
576 mutt_debug(LL_DEBUG3, "start a new line\n");
577 mutt_window_clrtoeol(win);
578 row++;
579 lines_used++;
580 width_left = win->state.cols - MaxHeaderWidth;
581 mutt_window_move(win, row, MaxHeaderWidth);
582 goto try_again;
583 }
584
585 if (addr_len < width_left)
586 {
587 mutt_debug(LL_DEBUG3, "space for: %s\n", buf_string(buf));
588 mutt_window_addstr(win, buf_string(buf));
589 mutt_window_addstr(win, sep);
590 width_left -= addr_len;
591 }
592 mutt_debug(LL_DEBUG3, "%d addresses remaining\n", count);
593 mutt_debug(LL_DEBUG3, "%zd lines remaining\n", max_lines - lines_used);
594 }
595 mutt_list_free(&list);
596 buf_pool_release(&buf);
597
598 if (count > 0)
599 {
600 mutt_window_move(win, row, win->state.cols - more_len);
601 mutt_curses_set_normal_backed_color_by_id(MT_COLOR_BOLD);
602 mutt_window_addstr(win, more);
603 mutt_curses_set_color_by_id(MT_COLOR_NORMAL);
604 mutt_debug(LL_DEBUG3, "%d more (len %d)\n", count, more_len);
605 }
606 else
607 {
608 mutt_window_clrtoeol(win);
609 }
610
611 for (int i = lines_used; i < max_lines; i++)
612 {
613 mutt_window_move(win, row + i, 0);
614 mutt_window_clrtoeol(win);
615 }
616
617 mutt_debug(LL_DEBUG3, "used %d lines\n", lines_used);
618 return lines_used;
619}

628static int draw_envelope_user_hdrs(struct MuttWindow *win,
629 struct EnvelopeWindowData *wdata, int row)
630{
631 const char *overflow_text = "...";
632 int rows_used = 0;
633
634 struct ListNode *first = STAILQ_FIRST(&wdata->email->env->userhdrs);
635 if (!first)
636 return rows_used;
637
638 /* Draw first entry on same line as prompt */
639 draw_header(win, row, HDR_CUSTOM_HEADERS);
640 mutt_paddstr(win,
641 win->state.cols - (HeaderPadding[HDR_CUSTOM_HEADERS] +
642 mutt_strwidth(_(Prompts[HDR_CUSTOM_HEADERS]))),
643 first->data);
644 rows_used++;
645
646 /* Draw any following entries on their own line */
647 struct ListNode *np = STAILQ_NEXT(first, entries);
648 if (!np)
649 return rows_used;
650
651 STAILQ_FOREACH_FROM(np, &wdata->email->env->userhdrs, entries)
652 {
653 if ((rows_used == (MAX_USER_HDR_ROWS - 1)) && STAILQ_NEXT(np, entries))
654 {
655 draw_header_content(win, row + rows_used, HDR_CUSTOM_HEADERS, overflow_text);
656 rows_used++;
657 break;
658 }
659 draw_header_content(win, row + rows_used, HDR_CUSTOM_HEADERS, np->data);
660 rows_used++;
661 }
662 return rows_used;
663}

755static int env_color_observer(struct NotifyCallback *nc)
756{

784 }

791static int env_config_observer(struct NotifyCallback *nc)
792{
793 if (nc->event_type != NT_CONFIG)
794 return 0;
795 if (!nc->global_data || !nc->event_data)
796 return -1;
797
798 struct EventConfig *ev_c = nc->event_data;
799 if (!ev_c->name)
800 return 0;
801
802 struct MuttWindow *win_env = nc->global_data;
803
804 switch (ev_c->name[0])
805 {
806 case 'a':
807 if (mutt_str_equal(ev_c->name, "autocrypt"))
808 break;
809 return 0;
810 case 'c':
811 if (mutt_str_equal(ev_c->name, "compose_show_user_headers") ||
812 mutt_str_equal(ev_c->name, "crypt_opportunistic_encrypt"))
813 {
814 break;
815 }
816 return 0;
817 case 'p':
818 if (mutt_str_equal(ev_c->name, "pgp_sign_as"))
819 break;
820 return 0;
821 case 's':
822 if (mutt_str_equal(ev_c->name, "smime_encrypt_with") ||
823 mutt_str_equal(ev_c->name, "smime_sign_as"))
824 {

868static int env_header_observer(struct NotifyCallback *nc)
869{

895 mutt_debug(LL_DEBUG5, "header done, request reflow\n");

908{
909 if (nc->event_type != NT_WINDOW)

911 if (!nc->global_data || !nc->event_data)
912 return -1;
913
914 struct MuttWindow *win_env = nc->global_data;
915 struct EventWindow *ev_w = nc->event_data;
916 if (ev_w->win != win_env)
917 return 0;

920 {
921 win_env->actions |= WA_RECALC;
