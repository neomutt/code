50bool regex_equal(const struct Regex *a, const struct Regex *b)
51{
52 if (!a && !b) /* both empty */
53 return true;
54 if (!a ^ !b) /* one is empty, but not the other */
55 return false;
56 if (a->pat_not != b->pat_not)
57 return false;
58
59 return mutt_str_equal(a->pattern, b->pattern);
60}

100struct Regex *regex_new(const char *str, uint32_t flags, struct Buffer *err)
101{
102 if (!str)
103 return NULL;
104
105 int rflags = 0;
106 struct Regex *reg = mutt_mem_calloc(1, sizeof(struct Regex));
107
108 reg->regex = mutt_mem_calloc(1, sizeof(regex_t));
109 reg->pattern = mutt_str_dup(str);
110
111 /* Should we use smart case matching? */
112 if (((flags & DT_REGEX_MATCH_CASE) == 0) && mutt_mb_is_lower(str))
113 rflags |= REG_ICASE;
114
115 if ((flags & DT_REGEX_NOSUB))
116 rflags |= REG_NOSUB;
117
118 /* Is a prefix of '!' allowed? */
119 if (((flags & DT_REGEX_ALLOW_NOT) != 0) && (str[0] == '!'))
120 {
121 reg->pat_not = true;
122 str++;
123 }
124
125 int rc = REG_COMP(reg->regex, str, rflags);
126 if (rc != 0)
127 {
128 if (err)
129 regerror(rc, reg->regex, err->data, err->dsize);
130 regex_free(&reg);
131 return NULL;
132 }
133
134 return reg;
135}

140static int regex_string_set(const struct ConfigSet *cs, void *var, struct ConfigDef *cdef,
141 const char *value, struct Buffer *err)
142{
143 /* Store empty regexes as NULL */
144 if (value && (value[0] == '\0'))
145 value = NULL;
146
147 struct Regex *r = NULL;
148
149 int rc = CSR_SUCCESS;
150
151 if (var)
152 {
153 struct Regex *curval = *(struct Regex **) var;
154 if (curval && mutt_str_equal(value, curval->pattern))
155 return CSR_SUCCESS | CSR_SUC_NO_CHANGE;
156
157 if (startup_only(cdef, err))
158 return CSR_ERR_INVALID | CSR_INV_VALIDATOR;
159
160 if (value)
161 {
162 r = regex_new(value, cdef->type, err);
163 if (!r)
164 return CSR_ERR_INVALID;
165 }
166
167 if (cdef->validator)
168 {
169 rc = cdef->validator(cs, cdef, (intptr_t) r, err);
170
171 if (CSR_RESULT(rc) != CSR_SUCCESS)
172 {
173 regex_free(&r);
174 return rc | CSR_INV_VALIDATOR;
175 }
176 }
177
178 regex_destroy(cs, var, cdef);
179
180 *(struct Regex **) var = r;
181
182 if (!r)
183 rc |= CSR_SUC_EMPTY;
184 }
185 else
186 {
187 if (cdef->type & DT_INITIAL_SET)
188 FREE(&cdef->initial);
189
190 cdef->type |= DT_INITIAL_SET;
191 cdef->initial = (intptr_t) mutt_str_dup(value);
192 }
193
194 return rc;

284static int regex_reset(const struct ConfigSet *cs, void *var,
285 const struct ConfigDef *cdef, struct Buffer *err)
286{
287 struct Regex *r = NULL;
288 const char *initial = (const char *) cdef->initial;
289
290 struct Regex *currx = *(struct Regex **) var;
291 const char *curval = currx ? currx->pattern : NULL;
292
293 int rc = CSR_SUCCESS;
294 if (!currx)
295 rc |= CSR_SUC_EMPTY;
296
297 if (mutt_str_equal(initial, curval))
298 return rc | CSR_SUC_NO_CHANGE;
299
300 if (startup_only(cdef, err))
301 return CSR_ERR_INVALID | CSR_INV_VALIDATOR;
302
303 if (initial)
304 {
305 r = regex_new(initial, cdef->type, err);
306 if (!r)
307 return CSR_ERR_CODE;
308 }
309
310 if (cdef->validator)
311 {
312 rc = cdef->validator(cs, cdef, (intptr_t) r, err);
313
314 if (CSR_RESULT(rc) != CSR_SUCCESS)
315 {
316 regex_destroy(cs, &r, cdef);
317 return rc | CSR_INV_VALIDATOR;
318 }
319 }
320
321 if (!r)
322 rc |= CSR_SUC_EMPTY;
323
324 regex_destroy(cs, var, cdef);
325
326 *(struct Regex **) var = r;
327 return rc;
328}
