165int parse_fkey(char *s)
166{
167 char *t = NULL;
168 int n = 0;
169
170 if ((s[0] != '<') || (mutt_tolower(s[1]) != 'f'))
171 return -1;
172
173 for (t = s + 2; *t && mutt_isdigit(*t); t++)
174 {
175 n *= 10;
176 n += *t - '0';
177 }
178
179 if (*t != '>')
180 return -1;
181 return n;
182}

192static int parse_keycode(const char *s)
193{
194 char *end_char = NULL;
195 long int result = strtol(s + 1, &end_char, 8);
196 /* allow trailing whitespace, eg. < 1001 > */
197 while (mutt_isspace(*end_char))
198 end_char++;
199 /* negative keycodes don't make sense, also detect overflow */
200 if ((*end_char != '>') || (result < 0) || (result == LONG_MAX))
201 {
202 return -1;
203 }
204
205 return result;
206}

215size_t parsekeys(const char *str, keycode_t *d, size_t max)
216{
217 int n;
218 size_t len = max;
219 char buf[128] = { 0 };
220 char c;
221 char *t = NULL;
222
223 mutt_str_copy(buf, str, sizeof(buf));
224 char *s = buf;
225
226 while (*s && len)
227 {
228 *d = '\0';
229 if ((*s == '<') && (t = strchr(s, '>')))
230 {
231 t++;
232 c = *t;
233 *t = '\0';
234
235 n = mutt_map_get_value(s, KeyNames);
236 if (n != -1)
237 {
238 s = t;
239 *d = n;
240 }
241 else if ((n = parse_fkey(s)) > 0)
242 {
243 s = t;
244 *d = KEY_F(n);
245 }
246 else if ((n = parse_keycode(s)) > 0)
247 {
248 s = t;
249 *d = n;
250 }
251
252 *t = c;
253 }
254
255 if (!*d)
256 {
257 *d = (unsigned char) *s;
258 s++;
259 }
260 d++;
261 len--;
262 }
263
264 return max - len;
265}

341void escape_macro(const char *macro, struct Buffer *buf)
342{
343 wchar_t wc = 0;
344 size_t k;
345 size_t len = mutt_str_len(macro);
346 mbstate_t mbstate1 = { 0 };
347 mbstate_t mbstate2 = { 0 };
348
349 for (; (len > 0) && (k = mbrtowc(&wc, macro, MB_LEN_MAX, &mbstate1)); macro += k, len -= k)
350 {
351 if ((k == ICONV_ILLEGAL_SEQ) || (k == ICONV_BUF_TOO_SMALL))
352 {
353 if (k == ICONV_ILLEGAL_SEQ)
354 memset(&mbstate1, 0, sizeof(mbstate1));
355 k = (k == ICONV_ILLEGAL_SEQ) ? 1 : len;
356 wc = ReplacementChar;
357 }
358
359 const int w = wcwidth(wc);
360 if (IsWPrint(wc) && (w >= 0))
361 {
362 char tmp[MB_LEN_MAX * 2] = { 0 };
363 if (wcrtomb(tmp, wc, &mbstate2) != ICONV_ILLEGAL_SEQ)
364 {
365 buf_addstr(buf, tmp);
366 }
367 }
368 else if ((wc < 0x20) || (wc == 0x7f))
369 {
370 if (wc == '\033') // Escape
371 buf_addstr(buf, "\\e");
372 else if (wc == '\n')
373 buf_addstr(buf, "\\n");
374 else if (wc == '\r')
375 buf_addstr(buf, "\\r");
376 else if (wc == '\t')
377 buf_addstr(buf, "\\t");
378 else
379 buf_add_printf(buf, "^%c", (char) ((wc + '@') & 0x7f));
380 }
381 else
382 {
383 buf_addch(buf, '?');
384 }
385 }
386}

474void km_keyname(int c, struct Buffer *buf)
475{
476 const char *name = mutt_map_get_name(c, KeyNames);
477 if (name)
478 {
479 buf_addstr(buf, name);
480 return;
481 }
482
483 if ((c < 256) && (c > -128) && iscntrl((unsigned char) c))
484 {
485 if (c < 0)
486 c += 256;
487
488 if (c < 128)
489 {
490 buf_addch(buf, '^');
491 buf_addch(buf, (c + '@') & 0x7f);
492 }
493 else
494 {
495 buf_add_printf(buf, "\\%d%d%d", c >> 6, (c >> 3) & 7, c & 7);
496 }
497 }
498 else if ((c >= KEY_F0) && (c < KEY_F(256))) /* this maximum is just a guess */
499 {
500 buf_add_printf(buf, "<F%d>", c - KEY_F0);
501 }
502 else if ((c < 256) && (c >= -128) && IsPrint(c))
503 {
504 buf_add_printf(buf, "%c", (unsigned char) c);
505 }
506 else
507 {
508 buf_add_printf(buf, "<%ho>", (unsigned short) c);
509 }
510}

633void gather_menu(enum MenuType menu, struct BindingInfoArray *bia_bind,
634 struct BindingInfoArray *bia_macro)
635{
636 struct Buffer *key_binding = buf_pool_get();
637 struct Buffer *macro = buf_pool_get();
638
639 struct Keymap *map = NULL;
640 STAILQ_FOREACH(map, &Keymaps[menu], entries)
641 {
642 struct BindingInfo bi = { 0 };
643
644 buf_reset(key_binding);
645 km_expand_key(map, key_binding);
646
647 if (map->op == OP_MACRO)
648 {
649 if (!bia_macro || (map->op == OP_NULL))
650 continue;
651
652 buf_reset(macro);
653 escape_macro(map->macro, macro);
654 bi.a[0] = buf_strdup(key_binding);
655 bi.a[1] = buf_strdup(macro);
656 bi.a[2] = map->desc;
657 ARRAY_ADD(bia_macro, bi);
658 }
659 else
660 {
661 if (!bia_bind)
662 continue;
663
664 if (map->op == OP_NULL)
665 {
666 bi.a[0] = buf_strdup(key_binding);
667 bi.a[1] = "noop";
668 ARRAY_ADD(bia_bind, bi);
669 continue;
670 }
671
672 bi.a[0] = buf_strdup(key_binding);
673 bi.a[1] = help_lookup_function(map->op, menu);
674 bi.a[2] = _(opcodes_get_description(map->op));
675 ARRAY_ADD(bia_bind, bi);
676 }
677 }
678
679 buf_pool_release(&key_binding);
680 buf_pool_release(&macro);
681}
