183int mutt_monitor_getch(void)
184{
185 /* ncurses has its own internal buffer, so before we perform a poll,
186 * we need to make sure there isn't a character waiting */
187 timeout(0);
188 int ch = getch();
189 timeout(1000); // 1 second
190 if (ch == ERR)
191 {
192 if (mutt_monitor_poll() != 0)
193 ch = ERR;
194 else
195 ch = getch();
196 }
197 return ch;
198}
199#endif /* USE_INOTIFY */

216struct KeyEvent mutt_getch(GetChFlags flags)
217{
218 static const struct KeyEvent event_abort = { 0, OP_ABORT };
219 static const struct KeyEvent event_repaint = { 0, OP_REPAINT };
220 static const struct KeyEvent event_timeout = { 0, OP_TIMEOUT };
221
222 if (!OptGui)
223 return event_abort;
224
225 struct KeyEvent *event_key = array_pop(&UngetKeyEvents);
226 if (event_key)
227 return *event_key;
228
229 if (!(flags & GETCH_IGNORE_MACRO))
230 {
231 event_key = array_pop(&MacroEvents);
232 if (event_key)
233 return *event_key;
234 }
235
236 int ch;
237 SigInt = false;
238 mutt_sig_allow_interrupt(true);
239 timeout(1000); // 1 second
240#ifdef USE_INOTIFY
241 ch = mutt_monitor_getch();
242#else
243 ch = getch();
244#endif
245 mutt_sig_allow_interrupt(false);
246
247 if (SigInt)
248 {
249 mutt_query_exit();
250 return event_abort;
251 }
252
253 if (ch == KEY_RESIZE)
254 {
255 timeout(0);
256 while ((ch = getch()) == KEY_RESIZE)
257 {
258 // do nothing
259 }
260 }
261
262 if (ch == ERR)
263 {
264 if (!isatty(STDIN_FILENO)) // terminal was lost
265 mutt_exit(1);
266
267 if (SigWinch)
268 {
269 SigWinch = false;
270 notify_send(NeoMutt->notify_resize, NT_RESIZE, 0, NULL);
271 return event_repaint;
272 }
273
274 notify_send(NeoMutt->notify_timeout, NT_TIMEOUT, 0, NULL);
275 return event_timeout;
276 }
277
278 if (ch == AbortKey)
279 return event_abort;
280
281 if (ch & 0x80)
282 {
283 const bool c_meta_key = cs_subset_bool(NeoMutt->sub, "meta_key");
284 if (c_meta_key)
285 {
286 /* send ALT-x as ESC-x */
287 ch &= ~0x80;
288 mutt_unget_ch(ch);
289 return (struct KeyEvent) { '\033', OP_NULL }; // Escape
290 }
291 }
292
293 return (struct KeyEvent) { ch, OP_NULL };
294}

321void generic_tokenize_push_string(char *s)
322{
323 char *pp = NULL;
324 char *p = s + mutt_str_len(s) - 1;
325 size_t l;
326 int i, op = OP_NULL;
327
328 while (p >= s)
329 {
330 /* if we see something like "<PageUp>", look to see if it is a real
331 * function name and return the corresponding value */
332 if (*p == '>')
333 {
334 for (pp = p - 1; pp >= s && *pp != '<'; pp--)
335 ; // do nothing
336
337 if (pp >= s)
338 {
339 i = parse_fkey(pp);
340 if (i > 0)
341 {
342 mutt_push_macro_event(KEY_F(i), 0);
343 p = pp - 1;
344 continue;
345 }
346
347 l = p - pp + 1;
348 for (i = 0; KeyNames[i].name; i++)
349 {
350 if (mutt_istrn_equal(pp, KeyNames[i].name, l))
351 break;
352 }
353 if (KeyNames[i].name)
354 {
355 /* found a match */
356 mutt_push_macro_event(KeyNames[i].value, 0);
357 p = pp - 1;
358 continue;
359 }
360
361 /* See if it is a valid command
362 * skip the '<' and the '>' when comparing */
363 struct Buffer *buf = buf_pool_get();
364 buf_strcpy_n(buf, pp + 1, l - 2);
365
366 for (enum MenuType j = 0; j < MENU_MAX; j++)
367 {
368 op = km_get_op_menu(j, buf_string(buf));
369 if (op != OP_NULL)
370 break;
371 }
372
373 buf_pool_release(&buf);
374
375 if (op != OP_NULL)
376 {
377 mutt_push_macro_event(0, op);
378 p = pp - 1;
379 continue;
380 }
381 }
382 }
383 mutt_push_macro_event((unsigned char) *p--, 0); /* independent 8 bits chars */
384 }
385}

395KeyGatherFlags gather_functions(const struct MenuDefinition *md, const keycode_t *keys,
396 int key_len, struct KeymapMatchArray *kma)
397{
398 if (!md || !keys || !kma)
399 return 0;
400
401 struct SubMenu **smp = NULL;
402 KeyGatherFlags flags = KEY_GATHER_NO_MATCH;
403
404 ARRAY_FOREACH(smp, &md->submenus)
405 {
406 struct Keymap *km = NULL;
407
408 STAILQ_FOREACH(km, &(*smp)->keymaps, entries)
409 {
410 bool match = true;
411
412 for (int i = 0; i < key_len; i++)
413 {
414 if (keys[i] != km->keys[i])
415 {
416 match = false;
417 break;
418 }
419 }
420
421 if (match)
422 {
423 KeyGatherFlags fmatch = KEY_GATHER_NO_MATCH;
424 if (km->len == key_len)
425 fmatch = KEY_GATHER_MATCH;
426 else
427 fmatch = KEY_GATHER_LONGER;
428
429 flags |= fmatch;
430
431 struct KeymapMatch kmatch = { md->id, fmatch, km };
432 ARRAY_ADD(kma, kmatch);
433 }
434 }
435 }
436
437 return flags;
438}

446struct KeyEvent km_dokey_event(enum MenuType mtype, GetChFlags flags)
447{
448 struct KeyEvent event = { 0, OP_NULL };
449 int pos = 0;
450 const struct MenuDefinition *md = NULL;
451 static bool dump = true;
452 keycode_t keys[MAX_SEQ] = { 0 };
453
454 ARRAY_FOREACH(md, &MenuDefs)
455 {
456 if (md->id == mtype)
457 break;
458 }
459
460 struct SubMenu **smp = NULL;
461
462 ARRAY_FOREACH(smp, &md->submenus)
463 {
464 struct SubMenu *sm = *smp;
465 struct Keymap *km = NULL;
466 int num_fn = 0;
467 int num_km = 0;
468
469 for (; sm->functions[num_fn].name; num_fn++)
470 ; // do nothing
471
472 STAILQ_FOREACH(km, &sm->keymaps, entries)
473 {
474 num_km++;
475 }
476
477 mutt_debug(LL_DEBUG1, "KEY: SubMenu: %d functions, %d keymaps\n", num_fn, num_km);
478
479 if (dump && !STAILQ_EMPTY(&sm->keymaps))
480 {
481 mutt_debug(LL_DEBUG1, "KEY: Keymaps:\n");
482
483 struct Buffer *buf = buf_pool_get();
484 STAILQ_FOREACH(km, &sm->keymaps, entries)
485 {
486 buf_reset(buf);
487 keymap_expand_key(km, buf);
488 mutt_debug(LL_DEBUG1, "KEY: \"%s\" -> %s (%d)\n",
489 buf_string(buf), opcodes_get_name(km->op), km->op);
490 }
491 buf_pool_release(&buf);
492 }
493 }
494 dump = false;
495
496 for (int n = 0; n < MaxKeyLoop; n++)
497 {
498 event = mutt_getch(flags);
499 mutt_debug(LL_DEBUG1, "KEY: \n");
500
501 // abort, timeout, repaint
502 if (event.op < OP_NULL)
503 {
504 mutt_debug(LL_DEBUG1, "KEY: getch() %s\n", opcodes_get_name(event.op));
505 return event;
506 }
507
508 mutt_debug(LL_DEBUG1, "KEY: getch() '%c'\n", isprint(event.ch) ? event.ch : '?');
509
510 keys[pos] = event.ch;
511 struct KeymapMatchArray kma = ARRAY_HEAD_INITIALIZER;
512 KeyGatherFlags kfg = gather_functions(md, keys, pos + 1, &kma);
513
514 mutt_debug(LL_DEBUG1, "KEY: flags = %x\n", kfg);
515
516 if (kfg == KEY_GATHER_NO_MATCH)
517 {
518 mutt_debug(LL_DEBUG1, "KEY: \033[1;31mFAIL1: ('%c', %s)\033[0m\n",
519 isprint(event.ch) ? event.ch : '?', opcodes_get_name(event.op));
520 return event;
521 }
522
523 if ((kfg & KEY_GATHER_MATCH) == KEY_GATHER_MATCH)
524 {
525 struct KeymapMatch *kmatch = NULL;
526
527 ARRAY_FOREACH(kmatch, &kma)
528 {
529 if (kmatch->flags == KEY_GATHER_MATCH)
530 {
531 struct Keymap *map = kmatch->keymap;
532
533 if (map->op != OP_MACRO)
534 {
535 mutt_debug(LL_DEBUG1, "KEY: \033[1;32mSUCCESS: ('%c', %s)\033[0m\n",
536 isprint(event.ch) ? event.ch : '?', opcodes_get_name(map->op));
537 ARRAY_FREE(&kma);
538 return (struct KeyEvent) { event.ch, map->op };
539 }
540
541 /* #GETCH_IGNORE_MACRO turns off processing the MacroEvents buffer
542 * in mutt_getch(). Generating new macro events during that time would
543 * result in undesired behavior once the option is turned off.
544 *
545 * Originally this returned -1, however that results in an unbuffered
546 * username or password prompt being aborted. Returning OP_NULL allows
547 * mw_get_field() to display the keybinding pressed instead.
548 *
549 * It may be unexpected for a macro's keybinding to be returned,
550 * but less so than aborting the prompt. */
551 if (flags & GETCH_IGNORE_MACRO)
552 {
553 ARRAY_FREE(&kma);
554 return (struct KeyEvent) { event.ch, OP_NULL };
555 }
556
557 generic_tokenize_push_string(map->macro);
558 pos = 0;
559 ARRAY_FREE(&kma);
560 break;
561 }
562 }
563 }
564 else
565 {
566 mutt_debug(LL_DEBUG1, "KEY: \033[1;33mLONGER: getch() '%c'\033[0m\n",
567 isprint(event.ch) ? event.ch : '?');
568 pos++;
569 }
570
571 ARRAY_FREE(&kma);
572 }
573
574 mutt_flushinp();
575 mutt_error(_("Macro loop detected"));
576 return (struct KeyEvent) { '\0', OP_ABORT };
577}
