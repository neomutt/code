171static int mutt_monitor_getch_timeout(int timeout_ms)
172{
173 /* ncurses has its own internal buffer, so before we perform a poll,
174 * we need to make sure there isn't a character waiting */
175 timeout(0);
176 int ch = getch();
177 timeout(timeout_ms);
178 if (ch == ERR)
179 {
180 if (mutt_monitor_poll() != 0)
181 ch = ERR;
182 else
183 ch = getch();
184 }
185 return ch;
186}
187#endif /* USE_INOTIFY */

195static struct KeyEvent mutt_getch_timeout(GetChFlags flags, int timeout_ms)
196{
197 static const struct KeyEvent event_abort = { 0, OP_ABORT, 0 };
198 static const struct KeyEvent event_repaint = { 0, OP_REPAINT, 0 };
199 static const struct KeyEvent event_timeout = { 0, OP_TIMEOUT, 0 };
200
201 if (!OptGui)
202 return event_abort;
203
204 struct KeyModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_KEY);
205
206 struct KeyEvent *event_key = array_pop(&mod_data->unget_key_events);
207 if (event_key)
208 return *event_key;
209
210 if (!(flags & GETCH_IGNORE_MACRO))
211 {
212 event_key = array_pop(&mod_data->macro_events);
213 if (event_key)
214 return *event_key;
215 }
216
217 int ch;
218 SigInt = false;
219 mutt_sig_allow_interrupt(true);
220 timeout(timeout_ms);
221#ifdef USE_INOTIFY
222 ch = mutt_monitor_getch_timeout(timeout_ms);
223#else
224 ch = getch();
225#endif
226 mutt_sig_allow_interrupt(false);
227
228 if (SigInt)
229 {
230 mutt_query_exit();
231 return event_abort;
232 }
233
234 if (ch == KEY_RESIZE)
235 {
236 timeout(0);
237 while ((ch = getch()) == KEY_RESIZE)
238 {
239 // do nothing
240 }
241 }
242
243 if (ch == ERR)
244 {
245 if (!isatty(STDIN_FILENO)) // terminal was lost
246 mutt_exit(1);
247
248 if (SigWinch)
249 {
250 SigWinch = false;
251 notify_send(NeoMutt->notify_resize, NT_RESIZE, 0, NULL);
252 return event_repaint;
253 }
254
255 notify_send(NeoMutt->notify_timeout, NT_TIMEOUT, 0, NULL);
256 return event_timeout;
257 }
258
259 if (ch == mod_data->abort_key)
260 return event_abort;
261
262 if (ch & 0x80)
263 {
264 const bool c_meta_key = cs_subset_bool(NeoMutt->sub, "meta_key");
265 if (c_meta_key)
266 {
267 /* send ALT-x as ESC-x */
268 ch &= ~0x80;
269 mutt_unget_ch(ch);
270 return (struct KeyEvent) { '\033', OP_NULL, 0 }; // Escape
271 }
272 }
273
274 return (struct KeyEvent) { ch, OP_NULL, 0 };
275}

325void generic_tokenize_push_string(char *s)
326{
327 struct KeyModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_KEY);
328 char *pp = NULL;
329 char *p = s + mutt_str_len(s) - 1;
330 size_t l;
331 int i, op = OP_NULL;
332
333 while (p >= s)
334 {
335 /* if we see something like "<PageUp>", look to see if it is a real
336 * function name and return the corresponding value */
337 if (*p == '>')
338 {
339 for (pp = p - 1; pp >= s && *pp != '<'; pp--)
340 ; // do nothing
341
342 if (pp >= s)
343 {
344 i = parse_fkey(pp);
345 if (i > 0)
346 {
347 mutt_push_macro_event(KEY_F(i), 0);
348 p = pp - 1;
349 continue;
350 }
351
352 l = p - pp + 1;
353 for (i = 0; mod_data->key_names[i].name; i++)
354 {
355 if (mutt_istrn_equal(pp, mod_data->key_names[i].name, l))
356 break;
357 }
358 if (mod_data->key_names[i].name)
359 {
360 /* found a match */
361 mutt_push_macro_event(mod_data->key_names[i].value, 0);
362 p = pp - 1;
363 continue;
364 }
365
366 /* See if it is a valid command
367 * skip the '<' and the '>' when comparing */
368 struct Buffer *buf = buf_pool_get();
369 buf_strcpy_n(buf, pp + 1, l - 2);
370
371 for (enum MenuType j = 0; j < MENU_MAX; j++)
372 {
373 op = km_get_op_menu(j, buf_string(buf));
374 if (op != OP_NULL)
375 break;
376 }
377
378 buf_pool_release(&buf);
379
380 if (op != OP_NULL)
381 {
382 mutt_push_macro_event(0, op);
383 p = pp - 1;
384 continue;
385 }
386 }
387 }
388 mutt_push_macro_event((unsigned char) *p--, 0); /* independent 8 bits chars */
389 }
390}

400KeyGatherFlags gather_functions(const struct MenuDefinition *md, const keycode_t *keys,
401 int key_len, struct KeymapMatchArray *kma)
402{
403 if (!md || !keys || !kma)
404 return 0;
405
406 struct SubMenu **smp = NULL;
407 KeyGatherFlags flags = KEY_GATHER_NO_MATCH;
408
409 ARRAY_FOREACH(smp, &md->submenus)
410 {
411 struct Keymap *km = NULL;
412
413 STAILQ_FOREACH(km, &(*smp)->keymaps, entries)
414 {
415 if (key_len > km->len)
416 continue;
417
418 bool match = true;
419
420 for (int i = 0; i < key_len; i++)
421 {
422 if (keys[i] != km->keys[i])
423 {
424 match = false;
425 break;
426 }
427 }
428
429 if (match)
430 {
431 KeyGatherFlags fmatch = KEY_GATHER_NO_MATCH;
432 if (km->len == key_len)
433 fmatch = KEY_GATHER_MATCH;
434 else
435 fmatch = KEY_GATHER_LONGER;
436
437 flags |= fmatch;
438
439 struct KeymapMatch kmatch = { md->id, fmatch, km };
440 ARRAY_ADD(kma, kmatch);
441 }
442 }
443 }
444
445 return flags;
446}

491struct KeyEvent km_dokey(const struct MenuDefinition *md, GetChFlags flags)
492{
493 struct KeyEvent event = { 0, OP_NULL, 0 };
494 enum DokeyState state = DKS_START;
495 int count = 0;
496 int count_digits = 0;
497 int key_len = 0;
498 struct Keymap *pending_exact = NULL;
499 bool feedback_active = false;
500 keycode_t keys[KEY_SEQ_MAX_LEN] = { 0 };
501
502 const short c_key_timeout_idle = cs_subset_number(NeoMutt->sub, "key_timeout_idle");
503 const short c_key_timeout_progress = cs_subset_number(NeoMutt->sub, "key_timeout_partial");
504
505 if (!md)
506 return event;
507
508 for (int n = 0; n < MaxKeyLoop; n++)
509 {
510 const int timeout_ms = (state == DKS_START) ? c_key_timeout_idle : c_key_timeout_progress;
511 event = mutt_getch_timeout(flags, timeout_ms);
512
513 // abort, timeout, repaint
514 if (event.op < OP_NULL)
515 {
516 if (event.op == OP_TIMEOUT)
517 {
518 if ((state == DKS_NEED_MORE) && pending_exact)
519 {
520 if (pending_exact->op != OP_MACRO)
521 {
522 key_progress_notify(md, count, keys, key_len, flags);
523 key_progress_notify(md, 0, NULL, 0, flags);
524 return (struct KeyEvent) { 0, pending_exact->op, count };
525 }
526
527 if (flags & GETCH_IGNORE_MACRO)
528 {
529 key_progress_notify(md, 0, NULL, 0, flags);
530 return (struct KeyEvent) { 0, OP_NULL, 0 };
531 }
532
533 generic_tokenize_push_string(pending_exact->macro);
534 state = DKS_START;
535 count = 0;
536 count_digits = 0;
537 key_len = 0;
538 pending_exact = NULL;
539 memset(keys, 0, sizeof(keys));
540 feedback_active = false;
541 key_progress_notify(md, 0, NULL, 0, flags);
542 continue;
543 }
544 }
545
546 if (feedback_active || (state != DKS_START))
547 {
548 key_progress_notify(md, 0, NULL, 0, flags);
549 }
550 mutt_debug(LL_DEBUG3, "KEY: getch() %s\n", opcodes_get_name(event.op));
551 return event;
552 }
553
554 // macro op pushed into queue (e.g. from `exec`)
555 if (event.op > OP_NULL)
556 {
557 if (feedback_active || (state != DKS_START))
558 {
559 key_progress_notify(md, 0, NULL, 0, flags);
560 }
561 return event;
562 }
563
564 mutt_debug(LL_DEBUG3, "KEY: getch() '%c'\n", isprint(event.ch) ? event.ch : '?');
565
566 // Check if digit keys have explicit bindings before using them as a counter
567 if (!(flags & GETCH_NO_COUNTER) && (state != DKS_NEED_MORE) &&
568 (event.ch >= '0') && (event.ch <= '9'))
569 {
570 keycode_t digit_key = event.ch;
571 struct KeymapMatchArray digit_kma = ARRAY_HEAD_INITIALIZER;
572 KeyGatherFlags digit_kgf = gather_functions(md, &digit_key, 1, &digit_kma);
573 bool digit_bound = (digit_kgf & KEY_GATHER_MATCH);
574 ARRAY_FREE(&digit_kma);
575
576 if (!digit_bound)
577 {
578 if (count_digits >= KEY_COUNT_MAX_DIGITS)
579 {
580 key_progress_notify(md, count, keys, key_len, flags);
581 key_progress_notify(md, 0, NULL, 0, flags);
582 return (struct KeyEvent) { event.ch, OP_NULL, 0 };
583 }
584
585 const int digit = event.ch - '0';
586 if ((count > (INT_MAX / 10)) ||
587 ((count == (INT_MAX / 10)) && (digit > (INT_MAX % 10))))
588 {
589 key_progress_notify(md, count, keys, key_len, flags);
590 key_progress_notify(md, 0, NULL, 0, flags);
591 return (struct KeyEvent) { event.ch, OP_NULL, 0 };
592 }
593
594 count = (count * 10) + digit;
595 count_digits++;
596 state = DKS_COUNTER;
597 feedback_active = true;
598 key_progress_notify(md, count, keys, key_len, flags);
599 continue;
600 }
601 }
602
603 if (key_len >= KEY_SEQ_MAX_LEN)
604 {
605 key_progress_notify(md, count, keys, key_len, flags);
606 key_progress_notify(md, 0, NULL, 0, flags);
607 return (struct KeyEvent) { event.ch, OP_NULL, 0 };
608 }
609
610 keys[key_len] = event.ch;
611 key_len++;
612
613 struct KeymapMatchArray kma = ARRAY_HEAD_INITIALIZER;
614 KeyGatherFlags kgf = gather_functions(md, keys, key_len, &kma);
615
616 mutt_debug(LL_DEBUG3, "KEY: flags = %x\n", kgf);
617
618 bool has_exact = false;
619 bool has_longer = false;
620 pending_exact = NULL;
621
622 struct KeymapMatch *kmatch = NULL;
623 ARRAY_FOREACH(kmatch, &kma)
624 {
625 if (kmatch->flags == KEY_GATHER_MATCH)
626 {
627 has_exact = true;
628 if (!pending_exact)
629 pending_exact = kmatch->keymap;
630 }
631 else if (kmatch->flags == KEY_GATHER_LONGER)
632 {
633 has_longer = true;
634 }
635 }
636
637 if (!has_exact && !has_longer)
638 {
639 key_progress_notify(md, count, keys, key_len, flags);
640 key_progress_notify(md, 0, NULL, 0, flags);
641 mutt_debug(LL_DEBUG3, "KEY: FAIL1: ('%c', %s)\n",
642 isprint(event.ch) ? event.ch : '?', opcodes_get_name(event.op));
643 ARRAY_FREE(&kma);
644 return event;
645 }
646
647 if (has_exact && has_longer)
648 {
649 state = DKS_NEED_MORE;
650 feedback_active = true;
651 key_progress_notify(md, count, keys, key_len, flags);
652 ARRAY_FREE(&kma);
653 continue;
654 }
655
656 if (!has_exact && has_longer)
657 {
658 state = DKS_NEED_MORE;
659 feedback_active = true;
660 key_progress_notify(md, count, keys, key_len, flags);
661 ARRAY_FREE(&kma);
662 continue;
663 }
664
665 if (has_exact && pending_exact)
666 {
667 struct Keymap *map = pending_exact;
668
669 if (map->op != OP_MACRO)
670 {
671 key_progress_notify(md, count, keys, key_len, flags);
672 key_progress_notify(md, 0, NULL, 0, flags);
673 mutt_debug(LL_DEBUG3, "KEY: SUCCESS: ('%c', %s)\n",
674 isprint(event.ch) ? event.ch : '?', opcodes_get_name(map->op));
675 ARRAY_FREE(&kma);
676 return (struct KeyEvent) { event.ch, map->op, count_digits > 0 ? count : 0 };
677 }
678
679 /* #GETCH_IGNORE_MACRO turns off processing the MacroEvents buffer
680 * in mutt_getch(). Generating new macro events during that time would
681 * result in undesired behavior once the option is turned off.
682 *
683 * Originally this returned -1, however that results in an unbuffered
684 * username or password prompt being aborted. Returning OP_NULL allows
685 * mw_get_field() to display the keybinding pressed instead.
686 *
687 * It may be unexpected for a macro's keybinding to be returned,
688 * but less so than aborting the prompt. */
689 if (flags & GETCH_IGNORE_MACRO)
690 {
691 key_progress_notify(md, 0, NULL, 0, flags);
692 ARRAY_FREE(&kma);
693 return (struct KeyEvent) { event.ch, OP_NULL, 0 };
694 }
695
696 generic_tokenize_push_string(map->macro);
697 state = DKS_START;
698 count = 0;
699 count_digits = 0;
700 key_len = 0;
701 pending_exact = NULL;
702 memset(keys, 0, sizeof(keys));
703 feedback_active = false;
704 key_progress_notify(md, 0, NULL, 0, flags);
705 ARRAY_FREE(&kma);
706 continue;
707 }
708
709 ARRAY_FREE(&kma);
710 }
711
712 key_progress_notify(md, 0, NULL, 0, flags);
713 mutt_flushinp();
714 mutt_error(_("Macro loop detected"));
715 return (struct KeyEvent) { '\0', OP_ABORT, 0 };
716}
