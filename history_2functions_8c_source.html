79int history_function_dispatcher(struct MuttWindow *win, int op)
80{
81 // The Dispatcher may be called on any Window in the Dialog
82 struct MuttWindow *dlg = dialog_find(win);
83 if (!dlg || !dlg->wdata)
84 return FR_ERROR;
85
86 struct Menu *menu = dlg->wdata;
87 struct HistoryData *hd = menu->mdata;
88
89 int rc = FR_UNKNOWN;
90 for (size_t i = 0; HistoryFunctions[i].op != OP_NULL; i++)
91 {
92 const struct HistoryFunction *fn = &HistoryFunctions[i];
93 if (fn->op == op)
94 {
95 rc = fn->function(hd, op);
96 break;
97 }
98 }
99
100 if (rc == FR_UNKNOWN) // Not our function
101 return rc;
102
103 const char *result = dispatcher_get_retval_name(rc);
104 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
105
106 return rc;
107}
