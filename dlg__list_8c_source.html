114static int list_make_entry(struct Menu *menu, int line, int max_cols, struct Buffer *buf)
115{
116 struct ListData *ld = menu->mdata;
117 if (!ld || (line < 0) || (line >= (int) countof(ListActions)))
118 return 0;
119
120 const char *name = _(ListActions[line].name);
121 const char *value = *get_action_value(&ld->headers, &ListActions[line]);
122
123 buf_strcpy(buf, name);
124 for (int i = mutt_strwidth(name); i < ld->label_width; i++)
125 buf_addch(buf, ' ');
126 buf_addstr(buf, ": ");
127 buf_addstr(buf, value ? value : "--");
128
129 return mutt_strwidth(buf_string(buf));

139static bool compose_list_action(struct ListData *ld, const struct ListAction *action)
140{
141 char *mailto = *get_action_value(&ld->headers, action);
142 if (!mailto)
143 {
144 mutt_error(_("No list action available for %s"), _(action->name));
145 return false;
146 }
147
148 if (url_check_scheme(mailto) != U_MAILTO)
149 {
150 mutt_error(_("List actions only support mailto: URIs. (Try a browser?)"));
151 return true;
152 }
153
154 struct Email *e = email_new();
155 e->env = mutt_env_new();
156
157 char *body = NULL;
158 if (!mutt_parse_mailto(e->env, &body, mailto))
159 {
160 mutt_error(_("Could not parse mailto: URI"));
161 FREE(&body);
162 email_free(&e);
163 return true;
164 }
165
166 e->body = mutt_body_new();
167 char ctype[] = "text/plain";
168 mutt_parse_content_type(ctype, e->body);
169 e->body->use_disp = false;
170 e->body->disposition = DISP_INLINE;
171
172 struct Buffer *tempfile = buf_pool_get();
173 buf_mktemp_draft(tempfile);
174 FILE *fp = mutt_file_fopen(buf_string(tempfile), "w+");
175 if (!fp)
176 {
177 mutt_perror("%s", buf_string(tempfile));
178 FREE(&body);
179 email_free(&e);
180 buf_pool_release(&tempfile);
181 return true;
182 }
183
184 if (body)
185 fprintf(fp, "%s\n", body);
186 mutt_file_fclose(&fp);
187 FREE(&body);
188
189 e->body->filename = buf_strdup(tempfile);
190 e->body->unlink = true;
191 buf_pool_release(&tempfile);
192
193 (void) mutt_send_message(SEND_DRAFT_FILE, e, NULL, ld->mailbox, NULL, NeoMutt->sub);
194 return true;
195}

218static int op_select_action(struct ListData *ld, const struct KeyEvent *event)
219{
220 const struct ListAction *action = NULL;
221
222 if (event->op == OP_GENERIC_SELECT_ENTRY)
223 {
224 const int index = menu_get_index(ld->menu);
225 if ((index < 0) || (index >= (int) countof(ListActions)))
226 return FR_NO_ACTION;
227 action = &ListActions[index];
228 }
229 else
230 {
231 for (size_t i = 0; i < countof(ListActions); i++)
232 {
233 if (ListActions[i].op == event->op)
234 {
235 action = &ListActions[i];
236 break;
237 }
238 }
239 if (!action)
240 return FR_NO_ACTION;
241 }
242
243 ld->done = compose_list_action(ld, action);
244 return FR_SUCCESS;
245}

274static int list_function_dispatcher(struct MuttWindow *win, const struct KeyEvent *event)
275{
276 struct MuttWindow *dlg = dialog_find(win);
277 if (!event || !dlg || !dlg->wdata)
278 {
279 dispatcher_flush_on_error(FR_ERROR);
280 return FR_ERROR;
281 }
282
283 struct Menu *menu = dlg->wdata;
284 struct ListData *ld = menu->mdata;
285 if (!ld)
286 {
287 dispatcher_flush_on_error(FR_ERROR);
288 return FR_ERROR;
289 }
290
291 int rc = FR_UNKNOWN;
292 for (size_t i = 0; ListFunctions[i].op != OP_NULL; i++)
293 {
294 if (ListFunctions[i].op == event->op)
295 {
296 rc = ListFunctions[i].function(ld, event);
297 break;
298 }
299 }
300
301 if (rc == FR_UNKNOWN)
302 return rc;
303
304 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(event->op),
305 event->op, NONULL(dispatcher_get_retval_name(rc)));
306 dispatcher_flush_on_error(rc);
307 return rc;
308}

315void dlg_list(struct Mailbox *m, struct Email *e)
316{
317 if (!m || !e)
318 return;
319
320 struct IndexModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_INDEX);
321 ASSERT(mod_data);
322
323 struct ListData *ld = MUTT_MEM_CALLOC(1, struct ListData);
324 ld->mailbox = m;
325
326 struct Message *msg = mx_msg_open(m, e);
327 if (!msg)
328 {
329 FREE(&ld);
330 return;
331 }
332
333 if (!mutt_file_seek(msg->fp, e->offset, SEEK_SET))
334 {
335 mutt_error(_("Unable to read mailing list headers"));
336 mx_msg_close(m, &msg);
337 FREE(&ld);
338 return;
339 }
340
341 mutt_rfc2369_read_headers(msg->fp, &ld->headers);
342 mx_msg_close(m, &msg);
343
344 for (size_t i = 0; i < countof(ListActions); i++)
345 ld->label_width = MAX(ld->label_width, mutt_strwidth(_(ListActions[i].name)));
346
347 struct SimpleDialogWindows sdw = simple_dialog_new(mod_data->menu_list,
348 WT_DLG_LIST, ListHelp);
349 struct Menu *menu = sdw.menu;
350 ld->menu = menu;
351 menu->max = countof(ListActions);
352 menu->make_entry = list_make_entry;
353 menu->mdata = ld;
354 menu->mdata_free = list_data_free;
355
356 sbar_set_title(sdw.sbar, _("Available mailing list actions"));
357
358 struct MuttWindow *old_focus = window_set_focus(menu->win);
359 // ---------------------------------------------------------------------------
360 // Event Loop
361 int op = OP_NULL;
362 struct KeyEvent event = { 0, OP_NULL };
363 do
364 {
365 menu_tagging_dispatcher(menu->win, &event);
366 window_redraw(NULL);
367
368 event = km_dokey(mod_data->menu_list, GETCH_NONE);
369 op = event.op;
370 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
371 if (op < 0)
372 continue;
373 if (op == OP_NULL)
374 {
375 km_error_key(mod_data->menu_list);
376 continue;
377 }
378 mutt_clear_error();
379
380 int rc = list_function_dispatcher(sdw.dlg, &event);
381 if (rc == FR_UNKNOWN)
382 rc = menu_function_dispatcher(menu->win, &event);
383 if (rc == FR_UNKNOWN)
384 rc = global_function_dispatcher(menu->win, &event);
385 } while (!ld->done);
386 // ---------------------------------------------------------------------------
387
388 window_set_focus(old_focus);
389 simple_dialog_free(&sdw.dlg);
390}
