82int ansi_color_seq_length(const char *str)
83{
84 if (!str || !*str)
85 return 0;
86
87 if ((str[0] != '\033') || (str[1] != '[')) // Escape
88 return 0;
89
90 int i = 2;
91 while ((str[i] != '\0') && (isdigit(str[i]) || (str[i] == ';')))
92 {
93 i++;
94 }
95
96 if (str[i] == 'm')
97 return i + 1;
98
99 return 0;
100}

112static int ansi_color_parse_single(const char *buf, struct AnsiColor *ansi, bool dry_run)
113{
114 int seq_len = ansi_color_seq_length(buf);
115 if (seq_len == 0)
116 return 0;
117
118 if (dry_run || !ansi)
119 return seq_len;
120
121 int pos = 2; // Skip '<esc>['
122
123 while (pos < seq_len)
124 {
125 if ((buf[pos] == '0') && isdigit(buf[pos + 1]))
126 {
127 pos++; // Skip the leading zero
128 }
129 else if ((buf[pos] == '0') && ansi_is_end_char(buf[pos + 1]))
130 {
131 ansi->fg = COLOR_DEFAULT;
132 ansi->bg = COLOR_DEFAULT;
133 ansi->attrs = 0;
134 ansi->attr_color = NULL;
135 pos += 2;
136 }
137 else if ((buf[pos] == '1') && ansi_is_end_char(buf[pos + 1]))
138 {
139 ansi->attrs |= A_BOLD;
140 pos += 2;
141 }
142 else if ((buf[pos] == '3') && ansi_is_end_char(buf[pos + 1]))
143 {
144 ansi->attrs |= A_ITALIC;
145 pos += 2;
146 }
147 else if (buf[pos] == '3')
148 {
149 // 30-37 basic fg
150 if ((buf[pos + 1] >= '0') && (buf[pos + 1] < '8') && ansi_is_end_char(buf[pos + 2]))
151 {
152 ansi->fg = buf[pos + 1] - '0';
153 pos += 3;
154 }
155 else if (buf[pos + 1] == '8')
156 {
157 if (mutt_str_startswith(buf + pos, "38;5;") && isdigit(buf[pos + 5]))
158 {
159 // 38;5;n palette fg
160 char *end = NULL;
161 int value = strtoul(buf + pos + 5, &end, 10);
162 if ((value >= 0) && (value < 256) && end && ansi_is_end_char(end[0]))
163 {
164 ansi->fg = value;
165 pos += end - &buf[pos];
166 }
167 else
168 {
169 pos += ansi_skip_sequence(buf + pos);
170 }
171 }

267static void ansi_color_list_add(struct AttrColorList *acl, struct AnsiColor *ansi)
268{
269 if (!acl || !ansi)
270 return;
271
272 if ((ansi->fg == COLOR_DEFAULT) && (ansi->bg == COLOR_DEFAULT))
273 {
274 switch (ansi->attrs)

277 return;
278 case A_BOLD:

288 }

296
