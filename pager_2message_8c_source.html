68static void process_protected_headers(struct Mailbox *m, struct Email *e)
69{
70 struct Envelope *prot_headers = NULL;
71
72 const bool c_crypt_protected_headers_read = cs_subset_bool(NeoMutt->sub, "crypt_protected_headers_read");
73#ifdef USE_AUTOCRYPT
74 const bool c_autocrypt = cs_subset_bool(NeoMutt->sub, "autocrypt");
75 if (!c_crypt_protected_headers_read && !c_autocrypt)
76 return;
77#else
78 if (!c_crypt_protected_headers_read)
79 return;
80#endif
81
82 /* Grab protected headers to update in the index */
83 if (e->security & SEC_SIGN)
84 {
85 /* Don't update on a bad signature.
86 *
87 * This is a simplification. It's possible the headers are in the
88 * encrypted part of a nested encrypt/signed. But properly handling that
89 * case would require more complexity in the decryption handlers, which
90 * I'm not sure is worth it. */
91 if (!(e->security & SEC_GOODSIGN))
92 return;
93
94 if (mutt_is_multipart_signed(e->body) && e->body->parts)
95 {
96 prot_headers = e->body->parts->mime_headers;
97 }
98 else if (((WithCrypto & APPLICATION_SMIME) != 0) && mutt_is_application_smime(e->body))
99 {
100 prot_headers = e->body->mime_headers;
101 }
102 }
103 if (!prot_headers && (e->security & SEC_ENCRYPT))
104 {
105 if (((WithCrypto & APPLICATION_PGP) != 0) &&
106 (mutt_is_valid_multipart_pgp_encrypted(e->body) ||
107 mutt_is_malformed_multipart_pgp_encrypted(e->body)))
108 {
109 prot_headers = e->body->mime_headers;
110 }
111 else if (((WithCrypto & APPLICATION_SMIME) != 0) && mutt_is_application_smime(e->body))
112 {
113 prot_headers = e->body->mime_headers;
114 }
115 }
116
117 /* Update protected headers in the index and header cache. */
118 if (c_crypt_protected_headers_read && prot_headers && prot_headers->subject &&
119 !mutt_str_equal(e->env->subject, prot_headers->subject))
120 {
121 if (m->subj_hash && e->env->real_subj)
122 mutt_hash_delete(m->subj_hash, e->env->real_subj, e);
123
124 mutt_env_set_subject(e->env, prot_headers->subject);
125 FREE(&e->env->disp_subj);
126
127 if (m->subj_hash)
128 mutt_hash_insert(m->subj_hash, e->env->real_subj, e);
129
130 mx_save_hcache(m, e);
131
132 /* Also persist back to the message headers if this is set */
133 const bool c_crypt_protected_headers_save = cs_subset_bool(NeoMutt->sub, "crypt_protected_headers_save");
134 if (c_crypt_protected_headers_save)
135 {
136 e->env->changed |= MUTT_ENV_CHANGED_SUBJECT;
137 e->changed = true;
138 m->changed = true;
139 }
140 }
141
142#ifdef USE_AUTOCRYPT
143 if (c_autocrypt && (e->security & SEC_ENCRYPT) && prot_headers && prot_headers->autocrypt_gossip)
144 {
145 mutt_autocrypt_process_gossip_header(e, prot_headers);
146 }
147#endif
148}

164static int email_to_file(struct Message *msg, struct Buffer *tempfile,
165 struct Mailbox *m, struct Email *e, const char *header,
166 int wrap_len, CopyMessageFlags *cmflags)
167{
168 int rc = 0;
169 pid_t filterpid = -1;
170
171 mutt_parse_mime_message(e, msg->fp);
172
173 char columns[16] = { 0 };
174 // win_pager might not be visible and have a size yet, so use win_index
175 snprintf(columns, sizeof(columns), "%d", wrap_len);
176 envlist_set(&NeoMutt->env, "COLUMNS", columns, true);
177
178 /* see if crypto is needed for this message. if so, we should exit curses */
179 if ((WithCrypto != 0) && e->security)
180 {
181 if (e->security & SEC_ENCRYPT)
182 {
183 if (e->security & APPLICATION_SMIME)
184 crypt_smime_getkeys(e->env);
185 if (!crypt_valid_passphrase(e->security))
186 goto cleanup;
187
188 *cmflags |= MUTT_CM_VERIFY;
189 }
190 else if (e->security & SEC_SIGN)
191 {
192 /* find out whether or not the verify signature */
193 /* L10N: Used for the $crypt_verify_sig prompt */
194 if (query_quadoption(_("Verify signature?"), NeoMutt->sub, "crypt_verify_sig") == MUTT_YES)
195 {
196 *cmflags |= MUTT_CM_VERIFY;
197 }
198 }
199 }
200
201 if (*cmflags & MUTT_CM_VERIFY || e->security & SEC_ENCRYPT)
202 {
203 if (e->security & APPLICATION_PGP)
204 {
205 if (!TAILQ_EMPTY(&e->env->from))
206 crypt_pgp_invoke_getkeys(TAILQ_FIRST(&e->env->from));
207
208 crypt_invoke_message(APPLICATION_PGP);
209 }
210
211 if (e->security & APPLICATION_SMIME)
212 crypt_invoke_message(APPLICATION_SMIME);
213 }
214
215 FILE *fp_filter_out = NULL;
216 buf_mktemp(tempfile);
217 FILE *fp_out = mutt_file_fopen(buf_string(tempfile), "w");
218 if (!fp_out)
219 {
220 mutt_error(_("Could not create temporary file"));
221 goto cleanup;
222 }
223
224 const char *const c_display_filter = cs_subset_string(NeoMutt->sub, "display_filter");
225 if (c_display_filter)
226 {
227 fp_filter_out = fp_out;
228 fp_out = NULL;
229 filterpid = filter_create_fd(c_display_filter, &fp_out, NULL, NULL, -1,
230 fileno(fp_filter_out), -1, NeoMutt->env);
231 if (filterpid < 0)
232 {
233 mutt_error(_("Can't create display filter"));
234 mutt_file_fclose(&fp_filter_out);
235 unlink(buf_string(tempfile));
236 goto cleanup;
237 }
238 }
239
240 if (header)
241 {
242 fputs(header, fp_out);
243 fputs("\n\n", fp_out);
244 }
245
246 const bool c_weed = cs_subset_bool(NeoMutt->sub, "weed");
247 CopyHeaderFlags chflags = (c_weed ? (CH_WEED | CH_REORDER) : CH_NO_FLAGS) |
248 CH_DECODE | CH_FROM | CH_DISPLAY;
249#ifdef USE_NOTMUCH
250 if (m->type == MUTT_NOTMUCH)
251 chflags |= CH_VIRTUAL;
252#endif
253 rc = mutt_copy_message(fp_out, e, msg, *cmflags, chflags, wrap_len);
254
255 if (((mutt_file_fclose(&fp_out) != 0) && (errno != EPIPE)) || (rc < 0))
256 {
257 mutt_error(_("Could not copy message"));
258 if (fp_filter_out)
259 {
260 filter_wait(filterpid);
261 mutt_file_fclose(&fp_filter_out);
262 }
263 mutt_file_unlink(buf_string(tempfile));
264 goto cleanup;
265 }
266
267 if (fp_filter_out && (filter_wait(filterpid) != 0))
268 mutt_any_key_to_continue(NULL);
269
270 mutt_file_fclose(&fp_filter_out); /* XXX - check result? */
271
272 if (WithCrypto)
273 {
274 /* update crypto information for this message */
275 e->security &= ~(SEC_GOODSIGN | SEC_BADSIGN);
276 e->security |= crypt_query(e->body);
277
278 /* Remove color cache for this message, in case there
279 * are color patterns for both ~g and ~V */
280 e->attr_color = NULL;
281
282 /* Process protected headers and autocrypt gossip headers */
283 process_protected_headers(m, e);
284 }
285
286cleanup:
287 envlist_unset(&NeoMutt->env, "COLUMNS");
288 return rc;
289}

360static void notify_crypto(struct Email *e, struct Message *msg, CopyMessageFlags cmflags)
361{
362 if ((WithCrypto != 0) && (e->security & APPLICATION_SMIME) && (cmflags & MUTT_CM_VERIFY))
363 {
364 if (e->security & SEC_GOODSIGN)
365 {
366 if (crypt_smime_verify_sender(e, msg) == 0)
367 mutt_message(_("S/MIME signature successfully verified"));
368 else
369 mutt_error(_("S/MIME certificate owner does not match sender"));
370 }
371 else if (e->security & SEC_PARTSIGN)
372 {
373 mutt_message(_("Warning: Part of this message has not been signed"));
374 }
375 else if (e->security & SEC_SIGN || e->security & SEC_BADSIGN)
376 {
377 mutt_error(_("S/MIME signature could NOT be verified"));
378 }
379 }
380
381 if ((WithCrypto != 0) && (e->security & APPLICATION_PGP) && (cmflags & MUTT_CM_VERIFY))
382 {
383 if (e->security & SEC_GOODSIGN)
384 mutt_message(_("PGP signature successfully verified"));
385 else if (e->security & SEC_PARTSIGN)
386 mutt_message(_("Warning: Part of this message has not been signed"));
387 else if (e->security & SEC_SIGN)
388 mutt_message(_("PGP signature could NOT be verified"));
389 }
390}

446int mutt_display_message(struct MuttWindow *win_index, struct IndexSharedData *shared)
447{
448 struct MuttWindow *dlg = dialog_find(win_index);
449 struct MuttWindow *win_pager = window_find_child(dlg, WT_CUSTOM);
450 struct MuttWindow *win_pbar = window_find_child(dlg, WT_STATUS_BAR);
451 struct Buffer *tempfile = buf_pool_get();
452 struct Message *msg = NULL;
453
454 squash_index_panel(shared->mailbox, win_index, win_pager);
455
456 int rc = PAGER_LOOP_QUIT;
457 do
458 {
459 msg = mx_msg_open(shared->mailbox, shared->email);
460 if (!msg)
461 break;
462
463 CopyMessageFlags cmflags = MUTT_CM_DECODE | MUTT_CM_DISPLAY | MUTT_CM_CHARCONV;
464
465 buf_reset(tempfile);
466 // win_pager might not be visible and have a size yet, so use win_index
467 rc = email_to_file(msg, tempfile, shared->mailbox, shared->email, NULL,
468 win_index->state.cols, &cmflags);
469 if (rc < 0)
470 break;
471
472 notify_crypto(shared->email, msg, cmflags);
473
474 /* Invoke the built-in pager */
475 struct PagerData pdata = { 0 };
476 struct PagerView pview = { &pdata };
477
478 pdata.fp = msg->fp;
479 pdata.fname = buf_string(tempfile);
480
481 pview.mode = PAGER_MODE_EMAIL;
482 pview.banner = NULL;
483 pview.flags = MUTT_PAGER_MESSAGE |
484 (shared->email->body->nowrap ? MUTT_PAGER_NOWRAP : 0);
485 pview.win_index = win_index;
486 pview.win_pbar = win_pbar;
487 pview.win_pager = win_pager;
488
489 rc = dlg_pager(&pview);
490 mx_msg_close(shared->mailbox, &msg);
491 } while (rc == PAGER_LOOP_RELOAD);
492
493 expand_index_panel(win_index, win_pager);
494
495 mx_msg_close(shared->mailbox, &msg);
496 buf_pool_release(&tempfile);
497 return rc;
498}
