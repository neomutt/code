101int km_config_observer(struct NotifyCallback *nc)
102{
103 if (nc->event_type != NT_CONFIG)
104 return 0;
105 if (!nc->event_data)
106 return -1;
107
108 struct EventConfig *ev_c = nc->event_data;
109
110 if (!mutt_str_equal(ev_c->name, "abort_key"))
111 return 0;
112
113 km_set_abort_key();
114 mutt_debug(LL_DEBUG5, "config done\n");
115 return 0;
116}

137void km_set_abort_key(void)
138{
139 keycode_t buf[4] = { 0 };
140 const char *const c_abort_key = cs_subset_string(NeoMutt->sub, "abort_key");
141
142 size_t len = parse_keys(c_abort_key, buf, countof(buf));
143 if (len == 0)
144 {
145 mutt_error(_("Abort key is not set, defaulting to Ctrl-G"));
146 AbortKey = ctrl('G');
147 return;
148 }
149
150 if (len > 1)
151 {
152 mutt_warning(_("Specified abort key sequence (%s) will be truncated to first key"),
153 c_abort_key);
154 }
155 AbortKey = buf[0];
156
157 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, km_config_observer, NULL);
158}
