61static void create_bindings(const struct MenuOpSeq *map, enum MenuType mtype)
62{
63 STAILQ_INIT(&Keymaps[mtype]);

65 for (int i = 0; map[i].op != OP_NULL; i++)
66 if (map[i].seq)

125void mutt_init_abort_key(void)
126{
127 keycode_t buf[2];
128 const char *const c_abort_key = cs_subset_string(NeoMutt->sub, "abort_key");
129 size_t len = parsekeys(c_abort_key, buf, countof(buf));
130 if (len == 0)
131 {
132 mutt_error(_("Abort key is not set, defaulting to Ctrl-G"));
133 AbortKey = ctrl('G');
134 return;
135 }
136 if (len > 1)
137 {
138 mutt_warning(_("Specified abort key sequence (%s) will be truncated to first key"),
139 c_abort_key);
140 }
141 AbortKey = buf[0];
142}

147int main_config_observer(struct NotifyCallback *nc)
148{
149 if (nc->event_type != NT_CONFIG)
150 return 0;
151 if (!nc->event_data)
152 return -1;
153
154 struct EventConfig *ev_c = nc->event_data;
155
156 if (!mutt_str_equal(ev_c->name, "abort_key"))
157 return 0;
158
159 mutt_init_abort_key();
160 mutt_debug(LL_DEBUG5, "config done\n");
161 return 0;
162}
