150int km_config_observer(struct NotifyCallback *nc)
151{
152 if (nc->event_type != NT_CONFIG)
153 return 0;
154 if (!nc->event_data)
155 return -1;
156
157 struct EventConfig *ev_c = nc->event_data;
158
159 if (!mutt_str_equal(ev_c->name, "abort_key"))
160 return 0;
161
162 struct KeyModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_KEY);
163 km_set_abort_key(&mod_data->abort_key);
164 mutt_debug(LL_DEBUG5, "config done\n");
165 return 0;
166}

205void km_cleanup(struct KeyModuleData *mod_data)

226}

234void km_set_abort_key(keycode_t *abort_key)
235{
236 keycode_t buf[4] = { 0 };
237 const char *const c_abort_key = cs_subset_string(NeoMutt->sub, "abort_key");
238
239 size_t len = parse_keys(c_abort_key, buf, countof(buf));
240 if (len == 0)
241 {
242 mutt_error(_("Abort key is not set, defaulting to Ctrl-G"));
243 *abort_key = ctrl('G');
244 return;
245 }
246
247 if (len > 1)
248 {
249 mutt_warning(_("Specified abort key sequence (%s) will be truncated to first key"),
250 c_abort_key);
251 }
252 *abort_key = buf[0];
253}
