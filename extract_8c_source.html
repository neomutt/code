49int parse_extract_token(struct Buffer *dest, struct Buffer *line, TokenFlags flags)
50{
51 if (!dest || !line)
52 return -1;
53
54 char ch;
55 char qc = '\0'; /* quote char */
56 char *pc = NULL;
57
58 buf_reset(dest);
59
60 SKIPWS(line->dptr);
61 while ((ch = *line->dptr))
62 {
63 if (qc == '\0')
64 {
65 if (mutt_isspace(ch) && !(flags & TOKEN_SPACE))
66 break;
67 if ((ch == '#') && !(flags & TOKEN_COMMENT))
68 break;
69 if ((ch == '+') && (flags & TOKEN_PLUS))
70 break;
71 if ((ch == '-') && (flags & TOKEN_MINUS))
72 break;
73 if ((ch == '=') && (flags & TOKEN_EQUAL))
74 break;
75 if ((ch == '?') && (flags & TOKEN_QUESTION))
76 break;
77 if ((ch == ';') && !(flags & TOKEN_SEMICOLON))
78 break;
79 if ((flags & TOKEN_PATTERN) && strchr("~%=!|", ch))
80 break;
81 }
82
83 line->dptr++;
84
85 if (ch == qc)
86 {
87 qc = 0; /* end of quote */
88 }
89 else if (!qc && ((ch == '\'') || (ch == '"')) && !(flags & TOKEN_QUOTE))
90 {
91 qc = ch;
92 }
93 else if ((ch == '\\') && (qc != '\''))
94 {
95 if (line->dptr[0] == '\0')
96 return -1; /* premature end of token */
97 switch (ch = *line->dptr++)
98 {
99 case 'c':
100 case 'C':
101 if (line->dptr[0] == '\0')
102 return -1; /* premature end of token */
103 buf_addch(dest, (mutt_toupper(line->dptr[0]) - '@') & 0x7f);
104 line->dptr++;
105 break;
106 case 'e':
107 buf_addch(dest, '\033'); // Escape
108 break;
109 case 'f':
110 buf_addch(dest, '\f');
111 break;
112 case 'n':
113 buf_addch(dest, '\n');
114 break;
115 case 'r':
116 buf_addch(dest, '\r');
117 break;
118 case 't':
119 buf_addch(dest, '\t');
120 break;
121 default:
122 if (mutt_isdigit(ch) && mutt_isdigit(line->dptr[0]) &&
123 mutt_isdigit(line->dptr[1]))
124 {
125 buf_addch(dest, (ch << 6) + (line->dptr[0] << 3) + line->dptr[1] - 3504);
126 line->dptr += 2;
127 }
128 else
129 {
130 buf_addch(dest, ch);
131 }
132 }
133 }
134 else if ((ch == '^') && (flags & TOKEN_CONDENSE))
135 {
136 if (line->dptr[0] == '\0')
137 return -1; /* premature end of token */
138 ch = *line->dptr++;
139 if (ch == '^')
140 {
141 buf_addch(dest, ch);
142 }
143 else if (ch == '[')
144 {
145 buf_addch(dest, '\033'); // Escape
146 }
147 else if (mutt_isalpha(ch))
148 {
149 buf_addch(dest, mutt_toupper(ch) - '@');
150 }
151 else
152 {
153 buf_addch(dest, '^');
154 buf_addch(dest, ch);
155 }
156 }
157 else if ((ch == '`') && (!qc || (qc == '"')))
158 {
159 FILE *fp = NULL;
160 pid_t pid;
161
162 pc = line->dptr;
163 do
164 {
165 pc = strpbrk(pc, "\\`");
166 if (pc)
167 {
168 /* skip any quoted chars */
169 if (*pc == '\\')
170 {
171 if (*(pc + 1))
172 pc += 2;
173 else
174 pc = NULL;
175 }
176 }
177 } while (pc && (pc[0] != '`'));
178 if (!pc)
179 {
180 mutt_debug(LL_DEBUG1, "mismatched backticks\n");
181 return -1;
182 }
183 struct Buffer *cmd = buf_pool_get();
184 *pc = '\0';
185 if (flags & TOKEN_BACKTICK_VARS)
186 {
187 /* recursively extract tokens to interpolate variables */
188 parse_extract_token(cmd, line,
189 TOKEN_QUOTE | TOKEN_SPACE | TOKEN_COMMENT |
190 TOKEN_SEMICOLON | TOKEN_NOSHELL);
191 }
192 else
193 {
194 buf_strcpy(cmd, line->dptr);
195 }
196 *pc = '`';
197 pid = filter_create(buf_string(cmd), NULL, &fp, NULL, NeoMutt->env);
198 if (pid < 0)
199 {
200 mutt_debug(LL_DEBUG1, "unable to fork command: %s\n", buf_string(cmd));
201 buf_pool_release(&cmd);
202 return -1;
203 }
204
205 line->dptr = pc + 1;
206
207 /* read line */
208 char *expn = NULL;
209 size_t expn_len = 0;
210 expn = mutt_file_read_line(expn, &expn_len, fp, NULL, MUTT_RL_NO_FLAGS);
211 mutt_file_fclose(&fp);
212 int rc = filter_wait(pid);
213 if (rc != 0)
214 {
215 mutt_debug(LL_DEBUG1, "backticks exited code %d for command: %s\n", rc,
216 buf_string(cmd));
217 }
218 buf_pool_release(&cmd);
219
220 /* if we got output, make a new string consisting of the shell output
221 * plus whatever else was left on the original line */
222 /* BUT: If this is inside a quoted string, directly add output to
223 * the token */
224 if (expn)
225 {
226 if (qc)
227 {
228 buf_addstr(dest, expn);
229 }
230 else
231 {
232 struct Buffer *copy = buf_pool_get();
233 buf_strcpy(copy, expn);
234 buf_addstr(copy, line->dptr);
235 buf_copy(line, copy);
236 buf_seek(line, 0);
237 buf_pool_release(&copy);
238 }
239 FREE(&expn);
240 }
241 }
242 else if ((ch == '$') && (!qc || (qc == '"')) &&
243 ((line->dptr[0] == '{') || mutt_isalpha(line->dptr[0])))
244 {
245 const char *env = NULL;
246 char *var = NULL;
247
248 if (line->dptr[0] == '{')
249 {
250 pc = strchr(line->dptr, '}');
251 if (pc)
252 {
253 var = mutt_strn_dup(line->dptr + 1, pc - (line->dptr + 1));
254 line->dptr = pc + 1;
255
256 if ((flags & TOKEN_NOSHELL))
257 {
258 buf_addch(dest, ch);
259 buf_addch(dest, '{');
260 buf_addstr(dest, var);
261 buf_addch(dest, '}');
262 FREE(&var);
263 }
264 }
265 }
266 else
267 {
268 for (pc = line->dptr; mutt_isalnum(*pc) || (pc[0] == '_'); pc++)
269 ; // do nothing
270
271 var = mutt_strn_dup(line->dptr, pc - line->dptr);
272 line->dptr = pc;
273 }
274 if (var)
275 {
276 struct Buffer *result = buf_pool_get();
277 int rc = cs_subset_str_string_get(NeoMutt->sub, var, result);
278
279 if (CSR_RESULT(rc) == CSR_SUCCESS)
280 {
281 buf_addstr(dest, buf_string(result));
282 }
283 else if (!(flags & TOKEN_NOSHELL) && (env = mutt_str_getenv(var)))
284 {
285 buf_addstr(dest, env);
286 }
287 else
288 {
289 buf_addch(dest, ch);
290 buf_addstr(dest, var);
291 }
292 FREE(&var);
293 buf_pool_release(&result);
294 }
295 }
296 else
297 {
298 buf_addch(dest, ch);
299 }
300 }
301
302 SKIPWS(line->dptr);
303 return 0;
304}
