343 int alias_complete(char *buf, size_t buflen)
344 {

346  char bestname[8192] = { 0 };
347  struct AliasMenuData *mdata = NULL;
348 
349  if (buf[0] != '\0')

352  {
353  if (np->name && (strncmp(np->name, buf, strlen(buf)) == 0))
354  {
355  if (bestname[0] == '\0') /* init */
356  {

358  MIN(mutt_str_strlen(np->name) + 1, sizeof(bestname)));
359  }
360  else
361  {
362  int i;
363  for (i = 0; np->name[i] && (np->name[i] == bestname[i]); i++)
364  ; // do nothing

366  bestname[i] = '\0';
367  }
368  }
369  }
370 
371  if (bestname[0] != '\0')
372  {
373  if (mutt_str_strcmp(bestname, buf) != 0)
374  {
375  /* we are adding something to the completion */
376  mutt_str_strfcpy(buf, bestname, mutt_str_strlen(bestname) + 1);
377  return 1;
378  }
379

381  mdata = menu_data_new();
382  TAILQ_FOREACH(np, &Aliases, entries)
383  {
384  if (np->name && (strncmp(np->name, buf, strlen(buf)) == 0))
385  {
386  menu_data_alias_add(mdata, np);
387  }
388  }
389  }
390  }
391 
392  if (!mdata)
393  {
394  mdata = menu_data_new();

396  {
397  menu_data_alias_add(mdata, np);
398  }
399  }

401 
402  bestname[0] = '\0';
403  alias_menu(bestname, sizeof(bestname), mdata);
404  if (bestname[0] != '\0')
405  mutt_str_strfcpy(buf, bestname, buflen);
