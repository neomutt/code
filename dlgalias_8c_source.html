183 static void dlg_select_alias(char *buf, size_t buflen, struct AliasMenuData *mdata)
184 {

190

193

211  ((C_SortAlias & SORT_MASK) == SORT_ADDRESS) ? alias_sort_address : alias_sort_name);
212  }

214  for (int i = 0; i < menu->max; i++)

216 
217  while (!done)

220  switch (op)

222  case OP_DELETE:
223  case OP_UNDELETE:

227  if (mdata->av[i]->is_tagged)
228  mdata->av[i]->is_deleted = (op == OP_DELETE);
229  menu->redraw |= REDRAW_INDEX;
230  }
231  else
232  {
233  mdata->av[menu->current]->is_deleted = (op == OP_DELETE);
234  menu->redraw |= REDRAW_CURRENT;
235  if (C_Resolve && (menu->current < menu->max - 1))
236  {
237  menu->current++;
238  menu->redraw |= REDRAW_INDEX;
239  }

251  }
252  }

255  {

257  {
258  mutt_addrlist_write(&mdata->av[i]->alias->addr, buf, buflen, true);
259  t = -1;
260  }
261  }
262 
263  if (t != -1)

265  mutt_addrlist_write(&mdata->av[t]->alias->addr, buf, buflen, true);
266  }
267

269  mutt_menu_pop_current(menu);
270  mutt_menu_free(&menu);

272 }
