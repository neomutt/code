53void mutt_check_rescore(struct Mailbox *m)
54{
55 const bool c_score = cs_subset_bool(NeoMutt->sub, "score");

58 const enum EmailSortType c_sort = cs_subset_sort(NeoMutt->sub, "sort");
59 const enum EmailSortType c_sort_aux = cs_subset_sort(NeoMutt->sub, "sort_aux");

80void mutt_score_message(struct Mailbox *m, struct Email *e, bool upd_mbox)
81{
82 struct Score *tmp = NULL;
83 struct PatternCache cache = { 0 };
84
85 e->score = 0; /* in case of re-scoring */
86 for (tmp = ScoreList; tmp; tmp = tmp->next)
87 {
88 if (mutt_pattern_exec(SLIST_FIRST(tmp->pat), MUTT_MATCH_FULL_ADDRESS, NULL, e, &cache) > 0)
89 {
90 if (tmp->exact || (tmp->val == 9999) || (tmp->val == -9999))
91 {
92 e->score = tmp->val;
93 break;
94 }
95 e->score += tmp->val;
96 }
97 }
98 if (e->score < 0)
99 e->score = 0;
100
101 const short c_score_threshold_delete = cs_subset_number(NeoMutt->sub, "score_threshold_delete");
102 const short c_score_threshold_flag = cs_subset_number(NeoMutt->sub, "score_threshold_flag");
103 const short c_score_threshold_read = cs_subset_number(NeoMutt->sub, "score_threshold_read");
104
105 if (e->score <= c_score_threshold_delete)
106 mutt_set_flag(m, e, MUTT_DELETE, true, upd_mbox);
107 if (e->score <= c_score_threshold_read)
108 mutt_set_flag(m, e, MUTT_READ, true, upd_mbox);
109 if (e->score >= c_score_threshold_flag)
110 mutt_set_flag(m, e, MUTT_FLAG, true, upd_mbox);
111}
