60enum CommandResult parse_dump(const struct Command *cmd, struct Buffer *line,
61 const struct ParseContext *pc, struct ParseError *pe)
62{
63 struct Buffer *err = pe->message;
64
65 struct Buffer *token = buf_pool_get();
66 int mtype = MENU_MAX;
67 enum CommandResult rc = MUTT_CMD_ERROR;
68
69 if (MoreArgs(line))
70 {
71 parse_extract_token(token, line, TOKEN_NO_FLAGS);
72
73 if (MoreArgs(line))
74 {
75 /* More arguments potentially means the user is using the
76 * ::command_t :bind command thus we delegate the task. */
77 goto done;
78 }
79
80 if (!mutt_str_equal(buf_string(token), "all"))
81 {
82 mtype = km_get_menu_id(buf_string(token));
83 if (mtype == -1)
84 {
85 // L10N: '%s' is the (misspelled) name of the menu, e.g. 'index' or 'pager'
86 buf_printf(err, _("%s: no such menu"), buf_string(token));
87 goto done;
88 }
89 }
90 }
91
92 dump_bind_macro(cmd, mtype, token, err);
93 rc = MUTT_CMD_SUCCESS;
94
95done:
96 buf_pool_release(&token);
97 return rc;
98}

114char *parse_keymap(const struct Command *cmd, enum MenuType *mtypes, struct Buffer *line,
115 int max_menus, int *num_menus, struct Buffer *err)
116{
117 struct Buffer *token = buf_pool_get();
118 int i = 0;
119 char *q = NULL;
120 char *result = NULL;
121
122 /* menu name */
123 parse_extract_token(token, line, TOKEN_NO_FLAGS);
124 char *p = token->data;
125 if (MoreArgs(line))
126 {
127 while (i < max_menus)
128 {
129 q = strchr(p, ',');
130 if (q)
131 *q = '\0';
132
133 int val = km_get_menu_id(p);
134 if (val == -1)
135 {
136 buf_printf(err, _("%s: no such menu"), p);
137 goto done;
138 }
139 mtypes[i] = val;
140 i++;
141 if (q)
142 p = q + 1;
143 else
144 break;
145 }
146 *num_menus = i;
147 /* key sequence */
148 parse_extract_token(token, line, TOKEN_NO_FLAGS);
149
150 if (buf_at(token, 0) == '\0')
151 {
152 buf_printf(err, _("%s: null key sequence"), cmd->name);
153 }
154 else if (MoreArgs(line))
155 {
156 result = buf_strdup(token);
157 goto done;
158 }
159 }
160 else
161 {
162 buf_printf(err, _("%s: too few arguments"), cmd->name);
163 }
164done:
165 buf_pool_release(&token);
166 return result;
167}

177void parse_menu(bool *menus, const char *s, struct Buffer *err)
178{
179 char *menu_names_dup = mutt_str_dup(s);
180 char *marker = menu_names_dup;
181 char *menu_name = NULL;
182
183 while ((menu_name = mutt_str_sep(&marker, ",")))
184 {
185 int value = km_get_menu_id(menu_name);
186 if (value == -1)
187 {
188 buf_printf(err, _("%s: no such menu"), menu_name);
189 break;
190 }
191 else
192 {
193 menus[value] = true;
194 }
195 }
196
197 FREE(&menu_names_dup);

243enum CommandResult parse_bind(const struct Command *cmd, struct Buffer *line,
244 const struct ParseContext *pc, struct ParseError *pe)
245{
246 struct Buffer *err = pe->message;
247
248 if (StartupComplete)
249 {
250 // Save and restore the offset in `line` because parse_dump() might change it
251 char *dptr = line->dptr;
252 if (parse_dump(cmd, line, pc, pe) == MUTT_CMD_SUCCESS)
253 {
254 return MUTT_CMD_SUCCESS;
255 }
256 if (!buf_is_empty(err))
257 {
258 return MUTT_CMD_ERROR;
259 }
260 line->dptr = dptr;
261 }
262
263 struct Buffer *token = buf_pool_get();
264 struct Buffer *keystr = NULL;
265
266 enum MenuType mtypes[MENU_MAX];
267 int num_menus = 0;
268 enum CommandResult rc = MUTT_CMD_ERROR;
269
270 char *key = parse_keymap(cmd, mtypes, line, countof(mtypes), &num_menus, err);
271 if (!key)
272 goto done;
273
274 /* function to execute */
275 parse_extract_token(token, line, TOKEN_NO_FLAGS);
276 if (MoreArgs(line))
277 {
278 buf_printf(err, _("%s: too many arguments"), cmd->name);
279 goto done;
280 }
281
282 rc = MUTT_CMD_SUCCESS;
283
284 if (mutt_istr_equal("noop", buf_string(token)))
285 {
286 keystr = buf_pool_get();
287 for (int i = 0; i < num_menus; i++)
288 {
289 struct MenuDefinition *md = NULL;
290 ARRAY_FOREACH(md, &MenuDefs)
291 {
292 if (md->id == mtypes[i])
293 break;
294 }
295
296 km_bind(md, key, OP_NULL, NULL, NULL, NULL); /* the 'unbind' command */
297
298 buf_reset(keystr);
299 keymap_expand_string(key, keystr);
300 const char *mname = km_get_menu_name(mtypes[i]);
301 mutt_debug(LL_NOTIFY, "NT_BINDING_DELETE: %s %s\n", mname, buf_string(keystr));
302
303 int op = km_get_op_menu(mtypes[i], buf_string(token));
304
305 struct EventBinding ev_b = { mtypes[i], key, op };
306 notify_send(NeoMutt->notify, NT_BINDING, NT_BINDING_DELETE, &ev_b);
307 }
308 }
309 else
310 {
311 keystr = buf_pool_get();
312 for (int i = 0; i < num_menus; i++)
313 {
314 struct MenuDefinition *md = NULL;
315 ARRAY_FOREACH(md, &MenuDefs)
316 {
317 if (md->id == mtypes[i])
318 break;
319 }
320
321 int op = OP_NULL;
322 struct SubMenu **ptr = NULL;
323 ARRAY_FOREACH(ptr, &md->submenus)
324 {
325 struct SubMenu *sm = *ptr;
326 const struct MenuFuncOp *mfo = NULL;
327 for (int j = 0; sm->functions[j].name; j++)
328 {
329 mfo = &sm->functions[j];
330 if (mutt_str_equal(buf_string(token), mfo->name))
331 {
332 op = mfo->op;
333 break;
334 }
335 }
336 }
337
338 rc = km_bind(md, key, op, NULL, NULL, err);
339 if (rc == MUTT_CMD_SUCCESS)
340 {
341 buf_reset(keystr);
342 keymap_expand_string(key, keystr);
343 const char *mname = km_get_menu_name(mtypes[i]);
344 mutt_debug(LL_NOTIFY, "NT_BINDING_NEW: %s %s\n", mname, buf_string(keystr));
345
346 struct EventBinding ev_b = { mtypes[i], key, op };
347 notify_send(NeoMutt->notify, NT_BINDING, NT_BINDING_ADD, &ev_b);
348 }
349 }
350 }
351
352done:
353 FREE(&key);
354 buf_pool_release(&keystr);
355 buf_pool_release(&token);
356 return rc;
357}

467bool parse_unbind_args(const struct Command *cmd, struct Buffer *line,
468 struct Buffer *err, struct ParseUnbind *args)
469{
470 if (!cmd || !line || !err || !args)
471 return false;
472
473 if (!MoreArgs(line))
474 {
475 buf_printf(err, _("%s: too few arguments"), cmd->name);
476 return false;
477 }
478
479 struct Buffer *token = buf_pool_get();
480 bool rc = false;
481
482 parse_extract_token(token, line, TOKEN_NO_FLAGS);
483 if (mutt_str_equal(buf_string(token), "*"))
484 {
485 args->all_menus = true;
486
487 if (MoreArgs(line))
488 {
489 // `unbind * key`
490 // `unmacro * key`
491 parse_extract_token(token, line, TOKEN_NO_FLAGS);
492 args->key = buf_strdup(token);
493 }
494 else
495 {
496 // `unbind *`
497 // `unmacro *`
498 args->all_keys = true;
499 }
500 }
501 else
502 {
503 parse_menu(args->menus, buf_string(token), err);
504 if (!buf_is_empty(err))
505 goto done;
506
507 if (MoreArgs(line))
508 {
509 parse_extract_token(token, line, TOKEN_NO_FLAGS);
510 if (mutt_str_equal(buf_string(token), "*"))
511 {
512 // `unbind menu *`
513 // `unmacro menu *`
514 args->all_keys = true;
515 }
516 else
517 {
518 // `unbind menu key`
519 // `unmacro menu key`
520 args->key = buf_strdup(token);
521 }
522 }
523 else
524 {
525 // `unbind menu`
526 // `unmacro menu`
527 args->all_keys = true;
528 }
529 }
530
531 if (MoreArgs(line))
532 {
533 buf_printf(err, _("%s: too many arguments"), cmd->name);
534 goto done;
535 }
536
537 rc = true;
538
539done:
540 buf_pool_release(&token);
541 return rc;
542}

555bool parse_unbind_exec(const struct Command *cmd, struct ParseUnbind *args, struct Buffer *err)
556{
557 if (!cmd || !args || !err)
558 return false;
559
560 keycode_t key_bytes[MAX_SEQ] = { 0 };
561 int key_len = 0;
562 if (args->key)
563 {
564 key_len = parse_keys(args->key, key_bytes, MAX_SEQ);
565 if (key_len == 0)
566 return false;
567 }
568
569 bool success = false;
570 struct Buffer *keystr = buf_pool_get();
571 enum NotifyBinding nb = (cmd->id == CMD_UNBIND) ? NT_BINDING_DELETE : NT_MACRO_DELETE;
572 struct MenuDefinition *md = NULL;
573 ARRAY_FOREACH(md, &MenuDefs)
574 {
575 if (!args->all_menus && !args->menus[md->id])
576 continue;
577
578 struct SubMenu **smp = ARRAY_GET(&md->submenus, 0);
579 if (!smp || !*smp)
580 continue;
581
582 struct SubMenu *sm = *smp;
583
584 struct Keymap *km = NULL;
585 struct Keymap *km_tmp = NULL;
586 STAILQ_FOREACH_SAFE(km, &sm->keymaps, entries, km_tmp)
587 {
588 if (args->all_keys ||
589 ((key_len == km->len) && (memcmp(km->keys, key_bytes, km->len) == 0)))
590 {
591 STAILQ_REMOVE(&sm->keymaps, km, Keymap, entries);
592 keymap_free(&km);
593 success = true;
594
595 if (!args->all_keys && !args->all_menus)
596 {
597 buf_reset(keystr);
598 keymap_expand_string(args->key, keystr);
599 mutt_debug(LL_NOTIFY, "%s: %s %s\n", notify_binding_name(nb),
600 md->name, buf_string(keystr));
601
602 struct EventBinding ev_b = { md->id, args->key, OP_NULL };
603 notify_send(NeoMutt->notify, NT_BINDING, nb, &ev_b);
604 }
605 }
606 }
607
608 if (args->all_keys && !args->all_menus && success)
609 {
610 buf_reset(keystr);
611 keymap_expand_string(args->key, keystr);
612 mutt_debug(LL_NOTIFY, "%s: %s %s\n", notify_binding_name(nb), md->name,
613 buf_string(keystr));
614
615 struct EventBinding ev_b = { md->id, args->key, OP_NULL };
616 notify_send(NeoMutt->notify, NT_BINDING, nb, &ev_b);
617
618 // restore some bindings for this menu
619 set_default_bindings(md->id);
620 }
621 }
622
623 if (args->all_menus && success)
624 {
625 buf_reset(keystr);
626 keymap_expand_string(args->key, keystr);
627 mutt_debug(LL_NOTIFY, "%s: ALL %s\n", notify_binding_name(nb), buf_string(keystr));
628
629 struct EventBinding ev_b = { MENU_MAX, args->key, OP_NULL };
630 notify_send(NeoMutt->notify, NT_BINDING, nb, &ev_b);
631
632 // restore some bindings for all menus
633 set_default_bindings(MENU_MAX);
634 }
635
636 buf_pool_release(&keystr);
637 return true;
638}

778enum CommandResult parse_exec(const struct Command *cmd, struct Buffer *line,
779 const struct ParseContext *pc, struct ParseError *pe)
780{
781 struct Buffer *err = pe->message;
782
783 if (!MoreArgs(line))
784 {
785 buf_printf(err, _("%s: too few arguments"), cmd->name);
786 return MUTT_CMD_WARNING;
787 }
788
789 struct Buffer *token = buf_pool_get();
790 enum CommandResult rc = MUTT_CMD_ERROR;
791
792 int ops[128];
793 int nops = 0;
794 char *function = NULL;
795
796 do
797 {
798 parse_extract_token(token, line, TOKEN_NO_FLAGS);
799 function = token->data;
800
801 const enum MenuType mtype = menu_get_current_type();
802 ops[nops] = km_get_op_menu(mtype, function);
803 if (ops[nops] == OP_NULL)
804 {
805 mutt_flushinp();
806 mutt_error(_("%s: no such function"), function);
807 goto done;
808 }
809 nops++;
810 } while (MoreArgs(line) && nops < countof(ops));
811
812 while (nops)
813 mutt_push_macro_event(0, ops[--nops]);
814
815 rc = MUTT_CMD_SUCCESS;
816
817done:
818 buf_pool_release(&token);
819 return rc;
820}
