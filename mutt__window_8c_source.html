159 static int mutt_dlg_rootwin_observer(struct NotifyCallback *nc)
160 {
161  if (!nc->event_data || !nc->global_data)
162  return -1;
163  if (nc->event_type != NT_CONFIG)
164  return 0;
165 
166  struct EventConfig *ec = nc->event_data;
167  struct MuttWindow *root_win = nc->global_data;
168 
169  if (mutt_str_strcmp(ec->name, "help") == 0)
170  {
171  MuttHelpWindow->state.visible = C_Help;
172  goto reflow;
173  }
174 
175  if (mutt_str_strcmp(ec->name, "status_on_top") == 0)
176  {
177  struct MuttWindow *first = TAILQ_FIRST(&root_win->children);
178  if (!first)
179  return -1;
180 
181  if ((C_StatusOnTop && (first->type == WT_HELP_BAR)) ||
182  (!C_StatusOnTop && (first->type != WT_HELP_BAR)))
183  {
184  // Swap the HelpLine and the Dialogs Container
185  struct MuttWindow *next = TAILQ_NEXT(first, entries);
186  if (!next)
187  return -1;
188  TAILQ_REMOVE(&root_win->children, next, entries);
189  TAILQ_INSERT_HEAD(&root_win->children, next, entries);
190  }
191  }
192 
193 reflow:
194  mutt_window_reflow(root_win);
195  return 0;
196 }
