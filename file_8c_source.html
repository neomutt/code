298
299 if (!oldpath || !newpath)
300 return -1;

303 return -1;

309 }

322 if (symlink(buf_string(&abs_oldpath), newpath) == -1)

355
356 if (!src || !target)
357 return -1;
358
359 if (link(src, target) != 0)
360 {
361 link_errno = errno;
362
363 /* It is historically documented that link can return -1 if NFS
364 * dies after creating the link. In that case, we are supposed
365 * to use stat to check if the link was created.
366 *
367 * Derek Martin notes that some implementations of link() follow a
368 * source symlink. It might be more correct to use stat() on src.
369 * I am not doing so to minimize changes in behavior: the function
370 * used lstat() further below for 20 years without issue, and I
371 * believe was never intended to be used on a src symlink. */
372 if ((lstat(src, &st_src) == 0) && (lstat(target, &st_target) == 0) &&
373 (stat_equal(&st_src, &st_target) == 0))

376 src, target, strerror(errno), errno);

383 * us it's a cross-filesystem linking attempt.
384 *
385 * However, the Coda rename call is allegedly safe to use.
386 *
387 * With other file systems, rename should just fail when
388 * the files reside on different file systems, so it's safe
389 * to try it here. */
390 mutt_debug(LL_DEBUG1, "link (%s, %s) failed: %s (%d)\n", src, target,
391 strerror(errno), errno);

393 /* FUSE may return ENOSYS. VFAT may return EPERM. FreeBSD's
394 * msdosfs may return EOPNOTSUPP. ENOTSUP can also appear. */
395 if ((errno == EXDEV) || (errno == ENOSYS) || errno == EPERM
396#ifdef ENOTSUP
397 || errno == ENOTSUP
398#endif
399#ifdef EOPNOTSUPP
400 || errno == EOPNOTSUPP
401#endif
402 )
403 {
404 mutt_debug(LL_DEBUG1, "trying rename\n");

408 strerror(errno), errno);

410 }
411 mutt_debug(LL_DEBUG1, "rename succeeded\n");

671void mutt_file_sanitize_filename(char *path, bool slash)
672{
673 if (!path)
674 return;
675
676 size_t size = strlen(path);
677
678 wchar_t c;
679 mbstate_t mbstate = { 0 };
680 for (size_t consumed; size && (consumed = mbrtowc(&c, path, size, &mbstate));
681 size -= consumed, path += consumed)
682 {
683 switch (consumed)
684 {
685 case ICONV_ILLEGAL_SEQ:
686 mbstate = (mbstate_t){ 0 };
687 consumed = 1;
688 memset(path, '_', consumed);
689 break;
690
691 case ICONV_BUF_TOO_SMALL:
692 consumed = size;
693 memset(path, '_', consumed);
694 break;
695
696 default:
697 if ((slash && (c == L'/')) || ((c <= 0x7F) && !strchr(FilenameSafeChars, c)))
698 {
699 memset(path, '_', consumed);
700 }
701 break;
702 }
703 }

767char *mutt_file_read_line(char *line, size_t *size, FILE *fp, int *line_num, ReadLineFlags flags)
768{
769 if (!size || !fp)
770 return NULL;
771
772 size_t offset = 0;
773 char *ch = NULL;
774
775 if (!line)
776 {
777 *size = 256;
778 line = mutt_mem_malloc(*size);
779 }
780
781 while (true)
782 {
783 if (!fgets(line + offset, *size - offset, fp))
784 {
785 FREE(&line);
786 return NULL;
787 }
788 ch = strchr(line + offset, '\n');
789 if (ch)
790 {
791 if (line_num)
792 (*line_num)++;
793 if (flags & MUTT_RL_EOL)
794 return line;
795 *ch = '\0';
796 if ((ch > line) && (*(ch - 1) == '\r'))
797 *--ch = '\0';
798 if (!(flags & MUTT_RL_CONT) || (ch == line) || (*(ch - 1) != '\\'))
799 return line;
800 offset = ch - line - 1;
801 }
802 else
803 {
804 int c;
805 c = getc(fp); /* This is kind of a hack. We want to know if the
806 char at the current point in the input stream is EOF.
807 feof() will only tell us if we've already hit EOF, not
808 if the next character is EOF. So, we need to read in
809 the next character and manually check if it is EOF. */
810 if (c == EOF)
811 {
812 /* The last line of fp isn't \n terminated */
813 if (line_num)
814 (*line_num)++;
815 return line;
816 }
817 else
818 {
819 ungetc(c, fp); /* undo our damage */
820 /* There wasn't room for the line -- increase "line" */
821 offset = *size - 1; /* overwrite the terminating 0 */
822 *size += 256;
823 mutt_mem_realloc(&line, *size);
824 }
825 }
826 }
827}

1209int mutt_file_lock(int fd, bool excl, bool timeout)
1210{
1211 struct stat st = { 0 }, prev_sb = { 0 };
1212 int count = 0;
1213 int attempt = 0;
1214
1215 struct flock lck = { 0 };
1216 lck.l_type = excl ? F_WRLCK : F_RDLCK;
1217 lck.l_whence = SEEK_SET;
1218

1220 {
1221 mutt_debug(LL_DEBUG1, "fcntl errno %d\n", errno);
1222 if ((errno != EAGAIN) && (errno != EACCES))
1223 {
1224 mutt_perror("fcntl");
1225 return -1;
1226 }
1227

1229 st.st_size = 0;
1230
1231 if (count == 0)
1232 prev_sb = st;
1233
1234 /* only unlock file if it is unchanged */
1235 if ((prev_sb.st_size == st.st_size) && (++count >= (timeout ? MAX_LOCK_ATTEMPTS : 0)))
1236 {
1237 if (timeout)
1238 mutt_error(_("Timeout exceeded while attempting fcntl lock"));
1239 return -1;
1240 }
1241
1242 prev_sb = st;
1243
