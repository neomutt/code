66int source_rc(const char *rcfile_path, struct Buffer *err)
67{
68 int lineno = 0, rc = 0, warnings = 0;
69 enum CommandResult line_rc;
70 struct Buffer *linebuf = NULL;
71 char *line = NULL;
72 char *currentline = NULL;
73 char rcfile[PATH_MAX + 1] = { 0 };
74 size_t linelen = 0;
75 pid_t pid;
76
77 mutt_str_copy(rcfile, rcfile_path, sizeof(rcfile));
78
79 size_t rcfilelen = mutt_str_len(rcfile);
80 if (rcfilelen == 0)
81 return -1;
82
83 bool ispipe = rcfile[rcfilelen - 1] == '|';
84
85 if (!ispipe)
86 {
87 struct ListNode *np = STAILQ_FIRST(&MuttrcStack);
88 if (!mutt_path_to_absolute(rcfile, np ? NONULL(np->data) : ""))
89 {
90 mutt_error(_("Error: Can't build path of '%s'"), rcfile_path);
91 return -1;
92 }
93
94 STAILQ_FOREACH(np, &MuttrcStack, entries)
95 {
96 if (mutt_str_equal(np->data, rcfile))
97 {
98 break;
99 }
100 }
101 if (np)
102 {
103 mutt_error(_("Error: Cyclic sourcing of configuration file '%s'"), rcfile);
104 return -1;
105 }
106
107 mutt_list_insert_head(&MuttrcStack, mutt_str_dup(rcfile));
108 }
109
110 mutt_debug(LL_DEBUG2, "Reading configuration file '%s'\n", rcfile);
111
112 FILE *fp = mutt_open_read(rcfile, &pid);
113 if (!fp)
114 {
115 buf_printf(err, "%s: %s", rcfile, strerror(errno));
116 return -1;
117 }
118
119 linebuf = buf_pool_get();
120
121 const char *const c_config_charset = cs_subset_string(NeoMutt->sub, "config_charset");
122 const char *const c_charset = cc_charset();
123 while ((line = mutt_file_read_line(line, &linelen, fp, &lineno, MUTT_RL_CONT)) != NULL)
124 {
125 const bool conv = c_config_charset && c_charset;
126 if (conv)
127 {
128 currentline = mutt_str_dup(line);
129 if (!currentline)
130 continue;
131 mutt_ch_convert_string(&currentline, c_config_charset, c_charset, MUTT_ICONV_NO_FLAGS);
132 }
133 else
134 {
135 currentline = line;
136 }
137
138 buf_strcpy(linebuf, currentline);
139
140 buf_reset(err);
141 line_rc = parse_rc_line(linebuf, err);
142 if (line_rc == MUTT_CMD_ERROR)
143 {
144 mutt_error("%s:%d: %s", rcfile, lineno, buf_string(err));
145 if (--rc < -MAX_ERRS)
146 {
147 if (conv)
148 FREE(&currentline);
149 break;
150 }
151 }
152 else if (line_rc == MUTT_CMD_WARNING)
153 {
154 /* Warning */
155 mutt_warning("%s:%d: %s", rcfile, lineno, buf_string(err));
156 warnings++;
157 }
158 else if (line_rc == MUTT_CMD_FINISH)
159 {
160 if (conv)
161 FREE(&currentline);
162 break; /* Found "finish" command */
163 }
164 else
165 {
166 if (rc < 0)
167 rc = -1;
168 }
169 if (conv)
170 FREE(&currentline);
171 }
172
173 FREE(&line);
174 mutt_file_fclose(&fp);
175 if (pid != -1)
176 filter_wait(pid);
177
178 if (rc)
179 {
180 /* the neomuttrc source keyword */
181 buf_reset(err);
182 buf_printf(err, (rc >= -MAX_ERRS) ? _("source: errors in %s") : _("source: reading aborted due to too many errors in %s"),
183 rcfile);
184 rc = -1;
185 }
186 else
187 {
188 /* Don't alias errors with warnings */
189 if (warnings > 0)
190 {
191 buf_printf(err, ngettext("source: %d warning in %s", "source: %d warnings in %s", warnings),
192 warnings, rcfile);
193 rc = -2;
194 }
195 }
196
197 if (!ispipe && !STAILQ_EMPTY(&MuttrcStack))
198 {
199 struct ListNode *np = STAILQ_FIRST(&MuttrcStack);
200 STAILQ_REMOVE_HEAD(&MuttrcStack, entries);
201 FREE(&np->data);
202 FREE(&np);
203 }
204
205 buf_pool_release(&linebuf);
206 return rc;
207}

215enum CommandResult parse_source(const struct Command *cmd, struct Buffer *line,
216 struct Buffer *err)
217{
218 if (!MoreArgs(line))
219 {
220 buf_printf(err, _("%s: too few arguments"), cmd->name);
221 return MUTT_CMD_WARNING;
222 }
223
224 struct Buffer *token = buf_pool_get();
225 struct Buffer *path = buf_pool_get();
226 enum CommandResult rc = MUTT_CMD_ERROR;
227
228 do
229 {
230 if (parse_extract_token(token, line, TOKEN_BACKTICK_VARS) != 0)
231 {
232 buf_printf(err, _("source: error at %s"), line->dptr);
233 goto done;
234 }
235 buf_copy(path, token);
236 expand_path(path, false);
237
238 if (source_rc(buf_string(path), err) < 0)
239 {
240 buf_printf(err, _("source: file %s could not be sourced"), buf_string(path));
241 goto done;
242 }
243
244 } while (MoreArgs(line));
245
246 rc = MUTT_CMD_SUCCESS;
247
248done:
249 buf_pool_release(&path);
250 buf_pool_release(&token);
251 return rc;
252}
