81static int op_delete(struct PostponeData *pd, int op)
82{
83 struct Menu *menu = pd->menu;
84 struct MailboxView *mv = pd->mailbox_view;
85 struct Mailbox *m = mv->mailbox;
86
87 const int index = menu_get_index(menu);
88 /* should deleted draft messages be saved in the trash folder? */
89 mutt_set_flag(m, m->emails[index], MUTT_DELETE, (op == OP_DELETE), true);
90 PostCount = m->msg_count - m->msg_deleted;
91 const bool c_resolve = cs_subset_bool(NeoMutt->sub, "resolve");
92 if (c_resolve && (index < (menu->max - 1)))
93 {
94 menu_set_index(menu, index + 1);
95 if (index >= (menu->top + menu->page_len))
96 {
97 menu->top = index;
98 menu_queue_redraw(menu, MENU_REDRAW_INDEX);
99 }
100 }
101 else
102 {
103 menu_queue_redraw(menu, MENU_REDRAW_CURRENT);
104 }
105
106 return FR_SUCCESS;
107}

134static int op_search(struct PostponeData *pd, int op)
135{
136 SearchFlags flags = SEARCH_NO_FLAGS;
137 switch (op)
138 {
139 case OP_SEARCH:
140 flags |= SEARCH_PROMPT;
141 pd->search_state->reverse = false;
142 break;
143 case OP_SEARCH_REVERSE:
144 flags |= SEARCH_PROMPT;
145 pd->search_state->reverse = true;
146 break;
147 case OP_SEARCH_NEXT:
148 break;
149 case OP_SEARCH_OPPOSITE:
150 flags |= SEARCH_OPPOSITE;
151 break;
152 }
153
154 int index = menu_get_index(pd->menu);
155 struct MailboxView *mv = pd->mailbox_view;
156 index = mutt_search_command(mv, pd->menu, index, pd->search_state, flags);
157 if (index != -1)
158 menu_set_index(pd->menu, index);
159
160 return FR_SUCCESS;
161}

185int postpone_function_dispatcher(struct MuttWindow *win, int op)
186{
187 // The Dispatcher may be called on any Window in the Dialog
188 struct MuttWindow *dlg = dialog_find(win);
189 if (!dlg || !dlg->wdata)
190 return FR_ERROR;
191
192 struct Menu *menu = dlg->wdata;
193 struct PostponeData *pd = menu->mdata;
194
195 int rc = FR_UNKNOWN;
196 for (size_t i = 0; PostponeFunctions[i].op != OP_NULL; i++)
197 {
198 const struct PostponeFunction *fn = &PostponeFunctions[i];
199 if (fn->op == op)
200 {
201 rc = fn->function(pd, op);
202 break;
203 }
204 }
205
206 if (rc == FR_UNKNOWN) // Not our function
207 return rc;
208
209 const char *result = dispatcher_get_retval_name(rc);
210 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
211
212 return rc;
213}
