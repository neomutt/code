63static void autocrypt_compose_menu(struct Email *e, const struct ConfigSubset *sub)
64{
65 /* L10N: The compose menu autocrypt prompt.
66 (e)ncrypt enables encryption via autocrypt.
67 (c)lear sets cleartext.
68 (a)utomatic defers to the recommendation. */
69 const char *prompt = _("Autocrypt: (e)ncrypt, (c)lear, (a)utomatic?");
70
71 e->security |= APPLICATION_PGP;
72
73 /* L10N: The letter corresponding to the compose menu autocrypt prompt
74 (e)ncrypt, (c)lear, (a)utomatic */
75 const char *letters = _("eca");
76
77 int choice = mw_multi_choice(prompt, letters);
78 switch (choice)
79 {
80 case 1:
81 e->security |= (SEC_AUTOCRYPT | SEC_AUTOCRYPT_OVERRIDE);
82 e->security &= ~(SEC_ENCRYPT | SEC_SIGN | SEC_OPPENCRYPT | SEC_INLINE);
83 break;
84 case 2:
85 e->security &= ~SEC_AUTOCRYPT;
86 e->security |= SEC_AUTOCRYPT_OVERRIDE;
87 break;
88 case 3:
89 {
90 e->security &= ~SEC_AUTOCRYPT_OVERRIDE;
91 const bool c_crypt_opportunistic_encrypt = cs_subset_bool(sub, "crypt_opportunistic_encrypt");
92 if (c_crypt_opportunistic_encrypt)
93 e->security |= SEC_OPPENCRYPT;
94 break;
95 }
96 }
97}

106static bool edit_address_list(enum HeaderField field, struct AddressList *al)
107{
108 struct Buffer *old_list = buf_pool_get();
109 struct Buffer *new_list = buf_pool_get();
110
111 /* need to be large for alias expansion */
112 buf_alloc(old_list, 8192);
113 buf_alloc(new_list, 8192);
114
115 mutt_addrlist_to_local(al);
116 mutt_addrlist_write(al, new_list, false);
117 buf_fix_dptr(new_list);
118 if (!buf_is_empty(new_list))
119 buf_addstr(new_list, ", ");
120
121 buf_copy(old_list, new_list);
122 if (mw_get_field(_(Prompts[field]), new_list, MUTT_COMP_NO_FLAGS, HC_ALIAS,
123 &CompleteAliasOps, NULL) == 0)
124 {
125 mutt_addrlist_clear(al);
126 mutt_addrlist_parse2(al, buf_string(new_list));
127 mutt_expand_aliases(al);
128 }
129
130 char *err = NULL;
131 if (mutt_addrlist_to_intl(al, &err) != 0)
132 {
133 mutt_error(_("Bad IDN: '%s'"), err);
134 mutt_refresh();
135 FREE(&err);
136 }
137
138 const bool rc = !mutt_str_equal(buf_string(new_list), buf_string(old_list));
139 buf_pool_release(&old_list);
140 buf_pool_release(&new_list);
141 return rc;
142}

148void update_crypt_info(struct EnvelopeWindowData *wdata)
149{
150 struct Email *e = wdata->email;
151
152 const bool c_crypt_opportunistic_encrypt = cs_subset_bool(wdata->sub, "crypt_opportunistic_encrypt");
153 if (c_crypt_opportunistic_encrypt)
154 crypt_opportunistic_encrypt(e);
155
156#ifdef USE_AUTOCRYPT
157 const bool c_autocrypt = cs_subset_bool(wdata->sub, "autocrypt");
158 if (c_autocrypt)
159 {
160 wdata->autocrypt_rec = mutt_autocrypt_ui_recommendation(e, NULL);
161
162 /* Anything that enables SEC_ENCRYPT or SEC_SIGN, or turns on SMIME
163 * overrides autocrypt, be it oppenc or the user having turned on
164 * those flags manually. */
165 if (e->security & (SEC_ENCRYPT | SEC_SIGN | APPLICATION_SMIME))
166 {
167 e->security &= ~(SEC_AUTOCRYPT | SEC_AUTOCRYPT_OVERRIDE);
168 }
169 else
170 {
171 if (!(e->security & SEC_AUTOCRYPT_OVERRIDE))
172 {
173 if (wdata->autocrypt_rec == AUTOCRYPT_REC_YES)
174 {
175 e->security |= (SEC_AUTOCRYPT | APPLICATION_PGP);
176 e->security &= ~(SEC_INLINE | APPLICATION_SMIME);
177 }
178 else
179 {
180 e->security &= ~SEC_AUTOCRYPT;
181 }
182 }
183 }
184 }
185#endif
186}

318static int op_compose_pgp_menu(struct EnvelopeWindowData *wdata, int op)
319{
320 const SecurityFlags old_flags = wdata->email->security;
321 if (!(WithCrypto & APPLICATION_PGP))
322 return FR_NOT_IMPL;
323 if (!crypt_has_module_backend(APPLICATION_PGP))
324 {
325 mutt_error(_("No PGP backend configured"));
326 return FR_ERROR;
327 }
328 if (((WithCrypto & APPLICATION_SMIME) != 0) && (wdata->email->security & APPLICATION_SMIME))
329 {
330 if (wdata->email->security & (SEC_ENCRYPT | SEC_SIGN))
331 {
332 if (query_yesorno(_("S/MIME already selected. Clear and continue?"), MUTT_YES) != MUTT_YES)
333 {
334 mutt_clear_error();
335 return FR_NO_ACTION;
336 }
337 wdata->email->security &= ~(SEC_ENCRYPT | SEC_SIGN);
338 }
339 wdata->email->security &= ~APPLICATION_SMIME;
340 wdata->email->security |= APPLICATION_PGP;
341 update_crypt_info(wdata);
342 }
343 wdata->email->security = crypt_pgp_send_menu(wdata->email);
344 update_crypt_info(wdata);
345 if (wdata->email->security == old_flags)
346 return FR_NO_ACTION;
347
348 mutt_message_hook(NULL, wdata->email, MUTT_SEND2_HOOK);
349 notify_send(wdata->email->notify, NT_EMAIL, NT_EMAIL_CHANGE, NULL);
350 return FR_SUCCESS;
351}

530int env_function_dispatcher(struct MuttWindow *win, int op)
531{
532 if (!win || !win->wdata)
533 return FR_UNKNOWN;
534
535 int rc = FR_UNKNOWN;
536 for (size_t i = 0; EnvelopeFunctions[i].op != OP_NULL; i++)
537 {
538 const struct EnvelopeFunction *fn = &EnvelopeFunctions[i];
539 if (fn->op == op)
540 {
541 struct EnvelopeWindowData *wdata = win->wdata;
542 rc = fn->function(wdata, op);
543 break;
544 }
545 }
546
547 if (rc == FR_UNKNOWN) // Not our function
548 return rc;
549
550 const char *result = dispatcher_get_retval_name(rc);
551 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
552
553 return rc;
554}
