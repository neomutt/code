101static void cid_save_attachment(struct Body *b, struct CidMapList *cid_map_list)
102{
103 if (!b || !cid_map_list)
104 return;
105
106 char *id = b->content_id;
107 if (!id)
108 return;
109
110 struct Buffer *tempfile = buf_pool_get();
111 struct Buffer *cid = buf_pool_get();
112 bool has_tempfile = false;
113 FILE *fp = NULL;
114
115 mutt_debug(LL_DEBUG2, "attachment found with \"Content-ID: %s\"\n", id);
116 /* get filename */
117 char *fname = mutt_str_dup(b->filename);
118 if (b->aptr)
119 fp = b->aptr->fp;
120 mutt_file_sanitize_filename(fname, fp ? true : false);
121 mailcap_expand_filename("%s", fname, tempfile);
122 FREE(&fname);
123
124 /* save attachment */
125 if (mutt_save_attachment(fp, b, buf_string(tempfile), 0, NULL) == -1)
126 goto bail;
127 has_tempfile = true;
128 mutt_debug(LL_DEBUG2, "attachment with \"Content-ID: %s\" saved to file \"%s\"\n",
129 id, buf_string(tempfile));
130
131 /* add Content-ID to filename mapping to list */
132 buf_printf(cid, "cid:%s", id);
133 struct CidMap *cid_map = cid_map_new(buf_string(cid), buf_string(tempfile));
134 STAILQ_INSERT_TAIL(cid_map_list, cid_map, entries);
135
136bail:
137
138 if ((fp && !buf_is_empty(tempfile)) || has_tempfile)
139 mutt_add_temp_attachment(buf_string(tempfile));
140 buf_pool_release(&tempfile);
141 buf_pool_release(&cid);
142}

168void cid_to_filename(struct Buffer *filename, const struct CidMapList *cid_map_list)
169{
170 if (!filename || !cid_map_list)
171 return;
172
173 FILE *fp_out = NULL;
174 char *pbuf = NULL;
175 char *searchbuf = NULL;
176 char *buf = NULL;
177 char *cid = NULL;
178 size_t blen = 0;
179 struct CidMap *cid_map = NULL;
180
181 struct Buffer *tempfile = buf_pool_get();
182 struct Buffer *tmpbuf = buf_pool_get();
183
184 FILE *fp_in = mutt_file_fopen(buf_string(filename), "r");
185 if (!fp_in)
186 goto bail;
187
188 /* ensure tempfile has the same file extension as filename otherwise an
189 * HTML file may be opened as plain text by the viewer */
190 const char *suffix = buf_rfind(filename, ".");
191 if (suffix && *(suffix++))
192 buf_mktemp_pfx_sfx(tempfile, "neomutt", suffix);
193 else
194 buf_mktemp(tempfile);
195 fp_out = mutt_file_fopen(buf_string(tempfile), "w+");
196 if (!fp_out)
197 goto bail;
198
199 /* Read in lines from filename into buf */
200 while ((buf = mutt_file_read_line(buf, &blen, fp_in, NULL, MUTT_RL_NO_FLAGS)) != NULL)
201 {
202 if (mutt_str_len(buf) == 0)
203 {
204 fputs(buf, fp_out);
205 continue;
206 }
207
208 /* copy buf to searchbuf because we need to edit multiple times */
209 searchbuf = mutt_str_dup(buf);
210 buf_reset(tmpbuf);
211
212 /* loop through Content-ID to filename mappings and do search and replace */
213 STAILQ_FOREACH(cid_map, cid_map_list, entries)
214 {
215 pbuf = searchbuf;
216 while ((cid = strstr(pbuf, cid_map->cid)) != NULL)
217 {
218 buf_addstr_n(tmpbuf, pbuf, cid - pbuf);
219 buf_addstr(tmpbuf, cid_map->fname);
220 pbuf = cid + mutt_str_len(cid_map->cid);
221 mutt_debug(LL_DEBUG2, "replaced \"%s\" with \"%s\" in file \"%s\"\n",
222 cid_map->cid, cid_map->fname, buf_string(filename));
223 }
224 buf_addstr(tmpbuf, pbuf);
225 FREE(&searchbuf);
226 searchbuf = buf_strdup(tmpbuf);
227 buf_reset(tmpbuf);
228 }
229
230 /* write edited line to output file */
231 fputs(searchbuf, fp_out);
232 fputs("\n", fp_out);
233 FREE(&searchbuf);
234 }
235
236 mutt_file_set_mtime(buf_string(filename), buf_string(tempfile));
237
238 /* add filename to TempAtachmentsList so it doesn't get left lying around */
239 mutt_add_temp_attachment(buf_string(filename));
240 /* update filename to point to new file */
241 buf_copy(filename, tempfile);
242
243bail:
244 FREE(&buf);
245 mutt_file_fclose(&fp_in);
246 mutt_file_fclose(&fp_out);
247 buf_pool_release(&tempfile);
248 buf_pool_release(&tmpbuf);
249}
