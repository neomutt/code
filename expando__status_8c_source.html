235static void index_mailbox_path(const struct ExpandoNode *node, void *data,
236 MuttFormatFlags flags, struct Buffer *buf)
237{
238 const struct MenuStatusLineData *msld = data;
239 const struct IndexSharedData *shared = msld->shared;
240 const struct Mailbox *m = shared->mailbox;
241
242 if (m && m->compress_info && (m->realpath[0] != '\0'))
243 {
244 buf_strcpy(buf, m->realpath);
245 pretty_mailbox(buf);
246 }
247 else if (m && (m->type == MUTT_NOTMUCH) && m->name)
248 {
249 buf_strcpy(buf, m->name);
250 }
251 else if (m && !buf_is_empty(&m->pathbuf))
252 {
253 buf_strcpy(buf, mailbox_path(m));
254 pretty_mailbox(buf);
255 }
256 else
257 {
258 buf_strcpy(buf, _("(no mailbox)"));
259 }
260}

343static void index_readonly(const struct ExpandoNode *node, void *data,
344 MuttFormatFlags flags, struct Buffer *buf)
345{
346 const struct MbTable *c_status_chars = cs_subset_mbtable(NeoMutt->sub, "status_chars");
347 if (!c_status_chars || !c_status_chars->len)
348 return;
349
350 const struct MenuStatusLineData *msld = data;
351 const struct IndexSharedData *shared = msld->shared;
352 const struct Mailbox *m = shared->mailbox;
353
354 int i = STATUS_CHAR_UNCHANGED;
355
356 if (m)
357 {
358 if (shared->attach_msg)
359 i = STATUS_CHAR_ATTACH;
360 else if (m->readonly || m->dontwrite)
361 i = STATUS_CHAR_READ_ONLY;
362 else if (m->changed || ((m->type != MUTT_IMAP) && m->msg_deleted)) /* deleted doesn't necessarily mean changed in IMAP */
363 i = STATUS_CHAR_NEED_RESYNC;
364 else
365 i = STATUS_CHAR_UNCHANGED;
366 }
367
368 if (i >= c_status_chars->len)
369 buf_addstr(buf, c_status_chars->chars[STATUS_CHAR_UNCHANGED]);
370 else
371 buf_addstr(buf, c_status_chars->chars[i]);
372}

426static void menu_percentage(const struct ExpandoNode *node, void *data,
427 MuttFormatFlags flags, struct Buffer *buf)
428{
429 const struct MenuStatusLineData *msld = data;
430 const struct Menu *menu = msld->menu;
431 if (!menu)
432 return;
433
434 char tmp[128] = { 0 };
435
436 char *cp = NULL;
437 if (menu->top + menu->page_len >= menu->max)
438 {
439 cp = menu->top ?
440 /* L10N: Status bar message: the end of the list emails is visible in the index */
441 _("end") :
442 /* L10N: Status bar message: all the emails are visible in the index */
443 _("all");
444 }
445 else
446 {
447 int count = (100 * (menu->top + menu->page_len)) / menu->max;
448 /* L10N: Status bar, percentage of way through index.
449 `%d` is the number, `%%` is the percent symbol.
450 They may be reordered, or space inserted, if you wish. */
451 snprintf(tmp, sizeof(tmp), _("%d%%"), count);
452 cp = tmp;
453 }
454
455 buf_strcpy(buf, cp);
456}
