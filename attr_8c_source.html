302color_t color_xterm256_to_24bit(const color_t color)
303{
304 static const color_t basic[] = {
305 0x000000, 0x800000, 0x008000, 0x808000, 0x000080, 0x800080,
306 0x008080, 0xc0c0c0, 0x808080, 0xff0000, 0x00ff00, 0xffff00,
307 0x0000ff, 0xff00ff, 0x00ffff, 0xffffff,
308 };
309
310 assert(color < 256);
311
312 if (color < 0)
313 return color;
314
315 if (color < 16)
316 {
317 color_debug(LL_DEBUG5, "Converted color 0-15: %d\n", color);
318 /* The first 16 colours are the "usual" terminal colours */
319 return basic[color];
320 }
321
322 if (color < 232)
323 {
324 /* The Color palette is divided in 6x6x6 colours, i.e. each R, G, B channel
325 * has six values:
326 *
327 * value: 1 2 3 4 5 6
328 * color: 0x00 0x5f 0x87 0xaf 0xd7 0xff
329 *
330 * The steps between the values is 0x28 = 40, the EXCEPT for the first one
331 * where it is 0x5f = 95.
332 *
333 * If we express the xterm color number minus 16 to base 6, i.e.
334 *
335 * color - 16 = (vr * 36) + (vg * 6) + (vb * 1)
336 *
337 * with vr, vg, vb integers between 0 and 5, then vr, vg, vb is the channel
338 * value for red, green, and blue, respectively.
339 */
340
341 color_t normalised_color = color - 16;
342 color_t vr = (normalised_color % 216) / 36; /* 216 = 6*6*6 */
343 color_t vg = (normalised_color % 36) / 6;
344 color_t vb = (normalised_color % 6) / 1;
345
346 /* First step is wider than the other ones, so add the difference if needed */
347 color_t r = vr * 0x28 + ((vr > 0) ? (0x5f - 0x40) : 0);
348 color_t g = vg * 0x28 + ((vg > 0) ? (0x5f - 0x40) : 0);
349 color_t b = vb * 0x28 + ((vb > 0) ? (0x5f - 0x40) : 0);
350
351 color_t rgb = (r << 16) + (g << 8) + (b << 0);
352 color_debug(LL_DEBUG5, "Converted xterm color %d to RGB #%x:\n", color, rgb);
353 return rgb;
354 }
355
356 /* Grey scale starts at 0x08 and adds 0xa = 10 in very step ending in 0xee.
357 * There are a total of 6*4 = 24 grey colors in total. */
358 color_t steps = color - 232;
359 color_t grey = (steps * 0x0a) + 0x08;
360 color_t rgb = (grey << 16) + (grey << 8) + (grey << 0);
361 color_debug(LL_DEBUG5, "Converted xterm color %d to RGB #%x:\n", color, rgb);
362 return rgb;
363}
