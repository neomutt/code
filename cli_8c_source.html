47static bool parse_args(const uint8_t *data, size_t size, int *argc, char ***argv)
48{
49 // Limit input size to prevent excessive memory usage
50 if (size > 4096)
51 return false;
52
53 // Create a null-terminated copy of the input
54 char *input = MUTT_MEM_MALLOC(size + 1, char);
55 memcpy(input, data, size);
56 input[size] = '\0';
57
58 // Count arguments (words separated by spaces/control chars)
59 int count = 1; // Start with 1 for argv[0] = "neomutt"
60 bool in_word = false;
61
62 for (size_t i = 0; i < size; i++)
63 {
64 unsigned char c = (unsigned char) input[i];
65 // Treat control characters and space as separators
66 if (iscntrl(c) || (c == ' '))
67 {
68 input[i] = '\0';
69 in_word = false;
70 }
71 else if (!in_word)
72 {
73 count++;
74 in_word = true;
75 }
76 }
77
78 // Allocate argv array
79 char **args = MUTT_MEM_MALLOC(count + 1, char *);
80 args[0] = mutt_str_dup("neomutt");
81
82 // Fill in arguments
83 int idx = 1;
84 in_word = false;
85 char *word_start = NULL;
86
87 for (size_t i = 0; i < size; i++)
88 {
89 if (input[i] == '\0')
90 {
91 if (in_word && word_start)
92 {
93 args[idx++] = mutt_str_dup(word_start);
94 word_start = NULL;
95 }
96 in_word = false;
97 }
98 else if (!in_word)
99 {
100 word_start = &input[i];
101 in_word = true;
102 }
103 }
104
105 // Handle last word if input doesn't end with separator
106 if (in_word && word_start)
107 {
108 args[idx++] = mutt_str_dup(word_start);
109 }
110
111 args[idx] = NULL;
112 FREE(&input);
113
114 *argc = idx;
115 *argv = args;
116 return true;
