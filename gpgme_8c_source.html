190 }

195 mutt_debug(LL_DEBUG1, "Generated key with id %s\n", buf_string(keyid));
196
197 /* Get gpgme_key_t to create the secondary key and export keydata */

199 if (err)

212 if (export_keydata(ctx, primary_key, keydata))

216 rc = 0;

218cleanup:
