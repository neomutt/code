52static bool calc_divider(struct SidebarWindowData *wdata)
53{
54 enum DivType type = SB_DIV_USER;
55 bool changed = false;
56 const char *const c_sidebar_divider_char = cs_subset_string(NeoMutt->sub, "sidebar_divider_char");
57
58 // Calculate the width of the delimiter in screen cells
59 int width = mutt_strwidth(c_sidebar_divider_char);
60 if (width < 1)
61 {
62 type = SB_DIV_ASCII;
63 goto done;
64 }
65
66 const bool c_ascii_chars = cs_subset_bool(NeoMutt->sub, "ascii_chars");
67 if (c_ascii_chars || !CharsetIsUtf8)
68 {
69 const size_t len = mutt_str_len(c_sidebar_divider_char);
70 for (size_t i = 0; i < len; i++)
71 {
72 if (c_sidebar_divider_char[i] & ~0x7F) // high-bit is set
73 {
74 type = SB_DIV_ASCII;
75 width = 1;
76 break;
77 }
78 }
79 }
80
81done:
82 changed = (width != wdata->divider_width);
83
84 wdata->divider_type = type;
85 wdata->divider_width = width;
86
87 return changed;
88}

103 mutt_window_remove_child(dlg, pager_panel);

115 struct MuttWindow *win_sidebar = mutt_window_new(WT_SIDEBAR, MUTT_WIN_ORIENT_HORIZONTAL,

119 win_sidebar->state.visible = c_sidebar_visible && (c_sidebar_width > 0);

123 win_sidebar->wdata_free = sb_wdata_free;

125 calc_divider(win_sidebar->wdata);
126
127 win_sidebar->recalc = sb_recalc;
128 win_sidebar->repaint = sb_repaint;
129

131 if (c_sidebar_on_right)
132 {
133 mutt_window_add_child(dlg, cont_right);

138 mutt_window_add_child(dlg, win_sidebar);
139 mutt_window_add_child(dlg, cont_right);

143

269static int sb_config_observer(struct NotifyCallback *nc)
270{
271 if (nc->event_type != NT_CONFIG)
272 return 0;
273 if (!nc->global_data || !nc->event_data)
274 return -1;
275
276 struct EventConfig *ev_c = nc->event_data;
277
278 if (!mutt_strn_equal(ev_c->name, "sidebar_", 8) &&
279 !mutt_str_equal(ev_c->name, "ascii_chars") &&
280 !mutt_str_equal(ev_c->name, "folder") && !mutt_str_equal(ev_c->name, "spool_file"))
281 {
282 return 0;
283 }
284
285 if (mutt_str_equal(ev_c->name, "sidebar_next_new_wrap"))
286 return 0; // Affects the behaviour, but not the display
287
288 mutt_debug(LL_DEBUG5, "config: %s\n", ev_c->name);
289
290 struct MuttWindow *win = nc->global_data;
291
292 if (mutt_str_equal(ev_c->name, "sidebar_visible"))
293 {
294 const bool c_sidebar_visible = cs_subset_bool(NeoMutt->sub, "sidebar_visible");
295 window_set_visible(win, c_sidebar_visible);
296 window_reflow(win->parent);
297 mutt_debug(LL_DEBUG5, "config done, request WA_REFLOW\n");
298 return 0;
299 }
300
301 if (mutt_str_equal(ev_c->name, "sidebar_width"))
302 {
303 const short c_sidebar_width = cs_subset_number(NeoMutt->sub, "sidebar_width");
304 win->req_cols = c_sidebar_width;
305 window_reflow(win->parent);
306 mutt_debug(LL_DEBUG5, "config done, request WA_REFLOW\n");
307 return 0;
308 }
309
310 if (mutt_str_equal(ev_c->name, "spool_file"))
311 {
312 win->actions |= WA_REPAINT;
313 mutt_debug(LL_DEBUG5, "config done, request WA_REPAINT\n");
314 return 0;
315 }
316
317 if (mutt_str_equal(ev_c->name, "sidebar_on_right"))
318 {
319 struct MuttWindow *parent = win->parent;
320 struct MuttWindow *first = TAILQ_FIRST(&parent->children);
321 const bool c_sidebar_on_right = cs_subset_bool(NeoMutt->sub, "sidebar_on_right");
322
323 if ((c_sidebar_on_right && (first == win)) || (!c_sidebar_on_right && (first != win)))
324 {
325 // Swap the Sidebar and the Container of the Index/Pager
326 TAILQ_REMOVE(&parent->children, first, entries);
327 TAILQ_INSERT_TAIL(&parent->children, first, entries);
328 }
329
330 window_reflow(win->parent);
331 mutt_debug(LL_DEBUG5, "config done, request WA_REFLOW\n");
332 return 0;
333 }
334
335 if (mutt_str_equal(ev_c->name, "ascii_chars") ||
336 mutt_str_equal(ev_c->name, "sidebar_divider_char"))
337 {
338 struct SidebarWindowData *wdata = sb_wdata_get(win);
339 calc_divider(wdata);
340 win->actions |= WA_RECALC;
341 mutt_debug(LL_DEBUG5, "config done, request WA_RECALC\n");
342 return 0;
343 }
344
345 // All the remaining config changes...
346 win->actions |= WA_RECALC;
347 mutt_debug(LL_DEBUG5, "config done, request WA_RECALC\n");
348 return 0;
349}

497 sb_win_remove_observers(ev_w->win);
