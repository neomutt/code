118static int pgp_sort_address(const void *a, const void *b, void *sdata)
119{
120 struct PgpUid const *s = *(struct PgpUid const *const *) a;
121 struct PgpUid const *t = *(struct PgpUid const *const *) b;
122 const bool sort_reverse = *(bool *) sdata;
123

125 if (rc != 0)
126 goto done;
127
128 rc = mutt_istr_cmp(pgp_fpr_or_lkeyid(s->parent), pgp_fpr_or_lkeyid(t->parent));

130done:

218static char *pgp_key_abilities(KeyFlags flags)
219{
220 static char buf[3];
221
222 if (!(flags & KEYFLAG_CANENCRYPT))
223 buf[0] = '-';
224 else if (flags & KEYFLAG_PREFER_SIGNING)
225 buf[0] = '.';
226 else
227 buf[0] = 'e';
228
229 if (!(flags & KEYFLAG_CANSIGN))
230 buf[1] = '-';
231 else if (flags & KEYFLAG_PREFER_ENCRYPTION)
232 buf[1] = '.';
233 else
234 buf[1] = 's';
235
236 buf[2] = '\0';
237
238 return buf;
239}

282static const char *pgp_entry_format_str(char *buf, size_t buflen, size_t col, int cols,
283 char op, const char *src, const char *prec,
284 const char *if_str, const char *else_str,
285 intptr_t data, MuttFormatFlags flags)
286{
287 char fmt[128] = { 0 };
288 bool optional = (flags & MUTT_FORMAT_OPTIONAL);
289
290 struct PgpEntry *entry = (struct PgpEntry *) data;
291 struct PgpUid *uid = entry->uid;
292 struct PgpKeyInfo *key = uid->parent;
293 struct PgpKeyInfo *pkey = pgp_principal_key(key);
294
295 if (isupper((unsigned char) op))
296 key = pkey;
297
298 KeyFlags kflags = key->flags | (pkey->flags & KEYFLAG_RESTRICTIONS) | uid->flags;
299
300 switch (tolower(op))
301 {
302 case 'a':
303 if (!optional)
304 {
305 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
306 snprintf(buf, buflen, fmt, key->algorithm);
307 }
308 break;
309 case 'c':
310 if (!optional)
311 {
312 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
313 snprintf(buf, buflen, fmt, pgp_key_abilities(kflags));
314 }
315 else if (!(kflags & KEYFLAG_ABILITIES))
316 {
317 optional = false;
318 }
319 break;
320 case 'f':
321 if (!optional)
322 {
323 snprintf(fmt, sizeof(fmt), "%%%sc", prec);
324 snprintf(buf, buflen, fmt, pgp_flags(kflags));
325 }
326 else if (!(kflags & KEYFLAG_RESTRICTIONS))
327 {
328 optional = false;
329 }
330 break;
331 case 'k':
332 if (!optional)
333 {
334 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
335 snprintf(buf, buflen, fmt, pgp_this_keyid(key));
336 }
337 break;
338 case 'l':
339 if (!optional)
340 {
341 snprintf(fmt, sizeof(fmt), "%%%sd", prec);
342 snprintf(buf, buflen, fmt, key->keylen);
343 }
344 break;
345 case 'n':
346 if (!optional)
347 {
348 snprintf(fmt, sizeof(fmt), "%%%sd", prec);
349 snprintf(buf, buflen, fmt, entry->num);
350 }
351 break;
352 case 't':
353 if (!optional)
354 {
355 snprintf(fmt, sizeof(fmt), "%%%sc", prec);
356 snprintf(buf, buflen, fmt, TrustFlags[uid->trust & 0x03]);
357 }
358 else if (!(uid->trust & 0x03))
359 {
360 /* undefined trust */
361 optional = false;
362 }
363 break;
364 case 'u':
365 if (!optional)
366 {
367 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
368 snprintf(buf, buflen, fmt, NONULL(uid->addr));
369 }
370 break;
371 case '[':
372 {
373 char buf2[128];
374 bool use_c_locale = false;
375 size_t len;
376
377 char *p = buf;
378
379 const char *cp = src;
380 if (*cp == '!')
381 {
382 use_c_locale = true;
383 cp++;
384 }
385
386 len = buflen - 1;
387 while ((len > 0) && (*cp != ']'))
388 {
389 if (*cp == '%')
390 {
391 cp++;
392 if (len >= 2)
393 {
394 *p++ = '%';
395 *p++ = *cp;
396 len -= 2;
397 }
398 else
399 {
400 break; /* not enough space */
401 }
402 cp++;
403 }
404 else
405 {
406 *p++ = *cp++;
407 len--;
408 }
409 }
410 *p = '\0';
411
412 if (use_c_locale)
413 {
414 mutt_date_localtime_format_locale(buf2, sizeof(buf2), buf,
415 key->gen_time, NeoMutt->time_c_locale);
416 }
417 else
418 {
419 mutt_date_localtime_format(buf2, sizeof(buf2), buf, key->gen_time);
420 }
421
422 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
423 snprintf(buf, buflen, fmt, buf2);
424 if (len > 0)
425 src = cp + 1;
426 break;
427 }
428 default:
429 *buf = '\0';
430 }
431
432 if (optional)
433 {
434 mutt_expando_format(buf, buflen, col, cols, if_str, pgp_entry_format_str,
435 data, MUTT_FORMAT_NO_FLAGS);
436 }
437 else if (flags & MUTT_FORMAT_OPTIONAL)
438 {
439 mutt_expando_format(buf, buflen, col, cols, else_str, pgp_entry_format_str,
440 data, MUTT_FORMAT_NO_FLAGS);
441 }
442
443 /* We return the format string, unchanged */
444 return src;
445}

480{

541{

545 struct PgpUid *a = NULL;
546 bool unusable = false;
547 int keymax = 0;
548
549 const bool c_pgp_show_unusable = cs_subset_bool(NeoMutt->sub, "pgp_show_unusable");
550 int i = 0;
551 for (struct PgpKeyInfo *kp = keys; kp; kp = kp->next)
552 {
553 if (!c_pgp_show_unusable && (kp->flags & KEYFLAG_CANTUSE))
554 {
555 unusable = true;
556 continue;
557 }
558
559 for (a = kp->address; a; a = a->next)
560 {
561 if (!c_pgp_show_unusable && (a->flags & KEYFLAG_CANTUSE))
562 {
563 unusable = true;
564 continue;
565 }
566
567 if (i == keymax)
568 {
569 keymax += 5;
570 mutt_mem_realloc(&key_table, sizeof(struct PgpUid *) * keymax);
571 }
572
573 key_table[i++] = a;
574 }
575 }
576
577 if ((i == 0) && unusable)
578 {
579 mutt_error(_("All matching keys are expired, revoked, or disabled"));
580 return NULL;
581 }
582
583 sort_t f = NULL;
584 short c_pgp_sort_keys = cs_subset_sort(NeoMutt->sub, "pgp_sort_keys");
585 switch (c_pgp_sort_keys & SORT_MASK)
586 {
587 case SORT_ADDRESS:
588 f = pgp_sort_address;
589 break;
590 case SORT_DATE:
591 f = pgp_sort_date;
592 break;
593 case SORT_KEYID:
594 f = pgp_sort_keyid;
595 break;
596 case SORT_TRUST:
597 default:
598 f = pgp_sort_trust;
599 break;
600 }
601
602 if (key_table)
603 {
604 bool sort_reverse = c_pgp_sort_keys & SORT_REVERSE;
605 mutt_qsort_r(key_table, i, sizeof(struct PgpUid *), f, &sort_reverse);
606 }
607
608 struct MuttWindow *dlg = simple_dialog_new(MENU_PGP, WT_DLG_PGP, PgpHelp);
609
610 menu = dlg->wdata;
611 menu->max = i;
612 menu->make_entry = pgp_make_entry;
613 menu->mdata = key_table;
614 menu->mdata_free = pgp_key_table_free;
615
616 struct PgpData pd = { false, menu, key_table, NULL };
617 dlg->wdata = &pd;
618
619 // NT_COLOR is handled by the SimpleDialog
620 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, pgp_key_config_observer, menu);
621 notify_observer_add(menu->win->notify, NT_WINDOW, pgp_key_window_observer, menu->win);
622
623 if (p)
624 snprintf(buf, sizeof(buf), _("PGP keys matching <%s>"), buf_string(p->mailbox));
625 else
626 snprintf(buf, sizeof(buf), _("PGP keys matching \"%s\""), s);
627
628 struct MuttWindow *sbar = window_find_child(dlg, WT_STATUS_BAR);
629 sbar_set_title(sbar, buf);
630
631 mutt_clear_error();
632
633 struct MuttWindow *old_focus = window_set_focus(menu->win);
634 // ---------------------------------------------------------------------------
635 // Event Loop
636 int op = OP_NULL;
637 do
638 {
639 menu_tagging_dispatcher(menu->win, op);
640 window_redraw(NULL);
641
642 op = km_dokey(MENU_PGP, GETCH_NO_FLAGS);
643 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
644 if (op < 0)
645 continue;
646 if (op == OP_NULL)
647 {
648 km_error_key(MENU_PGP);
649 continue;
650 }
651 mutt_clear_error();
652
653 int rc = pgp_function_dispatcher(dlg, op);
654
655 if (rc == FR_UNKNOWN)
656 rc = menu_function_dispatcher(menu->win, op);
657 if (rc == FR_UNKNOWN)
658 rc = global_function_dispatcher(NULL, op);
659 } while (!pd.done);
660 // ---------------------------------------------------------------------------
661
662 window_set_focus(old_focus);
663 simple_dialog_free(&dlg);
664 return pd.key;
665}
