105static int pgp_make_entry(struct Menu *menu, int line, int max_cols, struct Buffer *buf)
106{
107 struct PgpData *pd = menu->mdata;
108 struct PgpUid **puid = ARRAY_GET(pd->key_table, line);
109 if (!*puid)
110 return 0;
111
112 struct PgpEntry entry = { line + 1, *puid };
113
114 const bool c_arrow_cursor = cs_subset_bool(menu->sub, "arrow_cursor");
115 if (c_arrow_cursor)
116 {
117 const char *const c_arrow_string = cs_subset_string(menu->sub, "arrow_string");
118 if (max_cols > 0)
119 max_cols -= (mutt_strwidth(c_arrow_string) + 1);
120 }
121
122 const struct Expando *c_pgp_entry_format = cs_subset_expando(NeoMutt->sub, "pgp_entry_format");
123 return expando_filter(c_pgp_entry_format, PgpEntryRenderCallbacks, &entry,
124 MUTT_FORMAT_ARROWCURSOR, max_cols, NeoMutt->env, buf);
125}

130static int pgp_key_config_observer(struct NotifyCallback *nc)
131{
132 if (nc->event_type != NT_CONFIG)
133 return 0;
134 if (!nc->global_data || !nc->event_data)
135 return -1;
136
137 struct EventConfig *ev_c = nc->event_data;
138
139 if (!mutt_str_equal(ev_c->name, "pgp_entry_format") &&
140 !mutt_str_equal(ev_c->name, "pgp_key_sort"))
141 {
142 return 0;
143 }
144
145 struct Menu *menu = nc->global_data;
146 menu_queue_redraw(menu, MENU_REDRAW_FULL);
147 mutt_debug(LL_DEBUG5, "config done, request WA_RECALC, MENU_REDRAW_FULL\n");
148
149 return 0;
150}

159static int pgp_key_window_observer(struct NotifyCallback *nc)
160{
161 if (nc->event_type != NT_WINDOW)
162 return 0;
163 if (!nc->global_data || !nc->event_data)
164 return -1;
165 if (nc->event_subtype != NT_WINDOW_DELETE)
166 return 0;
167
168 struct MuttWindow *win_menu = nc->global_data;
169 struct EventWindow *ev_w = nc->event_data;
170 if (ev_w->win != win_menu)
171 return 0;
172
173 struct Menu *menu = win_menu->wdata;
174
175 notify_observer_remove(NeoMutt->sub->notify, pgp_key_config_observer, menu);
176 notify_observer_remove(win_menu->notify, pgp_key_window_observer, win_menu);
177
178 mutt_debug(LL_DEBUG5, "window delete done\n");
179 return 0;
180}

191struct PgpKeyInfo *dlg_pgp(struct PgpKeyInfo *keys, struct Address *p, const char *s)
192{
193 struct Menu *menu = NULL;
194 char buf[1024] = { 0 };
195 bool unusable = false;
196 struct PgpUidArray pua = ARRAY_HEAD_INITIALIZER;
197
198 const bool c_pgp_show_unusable = cs_subset_bool(NeoMutt->sub, "pgp_show_unusable");
199 for (struct PgpKeyInfo *kp = keys; kp; kp = kp->next)
200 {
201 if (!c_pgp_show_unusable && (kp->flags & KEYFLAG_CANTUSE))
202 {
203 unusable = true;
204 continue;
205 }
206
207 for (struct PgpUid *a = kp->address; a; a = a->next)
208 {
209 if (!c_pgp_show_unusable && (a->flags & KEYFLAG_CANTUSE))
210 {
211 unusable = true;
212 continue;
213 }
214
215 ARRAY_ADD(&pua, a);
216 }
217 }
218
219 if ((ARRAY_SIZE(&pua) == 0) && unusable)
220 {
221 mutt_error(_("All matching keys are expired, revoked, or disabled"));
222 return NULL;
223 }
224
225 pgp_sort_keys(&pua);
226
227 struct SimpleDialogWindows sdw = simple_dialog_new(MENU_PGP, WT_DLG_PGP, PgpHelp);
228 menu = sdw.menu;
229 struct PgpData pd = { false, menu, &pua, NULL };
230
231 menu->max = ARRAY_SIZE(&pua);
232 menu->make_entry = pgp_make_entry;
233 menu->mdata = &pd;
234 menu->mdata_free = NULL; // Menu doesn't own the data
235
236 // NT_COLOR is handled by the SimpleDialog
237 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, pgp_key_config_observer, menu);
238 notify_observer_add(menu->win->notify, NT_WINDOW, pgp_key_window_observer, menu->win);
239
240 if (p)
241 snprintf(buf, sizeof(buf), _("PGP keys matching <%s>"), buf_string(p->mailbox));
242 else
243 snprintf(buf, sizeof(buf), _("PGP keys matching \"%s\""), s);
244
245 sbar_set_title(sdw.sbar, buf);
246
247 mutt_clear_error();
248
249 struct MuttWindow *old_focus = window_set_focus(menu->win);
250 // ---------------------------------------------------------------------------
251 // Event Loop
252 int op = OP_NULL;
253 struct KeyEvent event = { 0, OP_NULL };
254 do
255 {
256 menu_tagging_dispatcher(menu->win, &event);
257 window_redraw(NULL);
258
259 event = km_dokey(MENU_PGP, GETCH_NO_FLAGS);
260 op = event.op;
261 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
262 if (op < 0)
263 continue;
264 if (op == OP_NULL)
265 {
266 km_error_key(MENU_PGP);
267 continue;
268 }
269 mutt_clear_error();
270
271 int rc = pgp_function_dispatcher(sdw.dlg, &event);
272
273 if (rc == FR_UNKNOWN)
274 rc = menu_function_dispatcher(menu->win, &event);
275 if (rc == FR_UNKNOWN)
276 rc = global_function_dispatcher(menu->win, &event);
277 } while (!pd.done);
278 // ---------------------------------------------------------------------------
279
280 ARRAY_FREE(&pua);
281 window_set_focus(old_focus);
282 simple_dialog_free(&sdw.dlg);
283 return pd.key;
284}
