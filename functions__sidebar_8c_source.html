250static int sb_screen_row_index(struct SidebarWindowData *wdata, int wanted_row)
251{
252 if ((wdata->top_index < 0) || (wanted_row < 0))
253 return -1;
254
255 const int page_size = wdata->win->state.rows;
256 int row = 0;
257 int last = -1;
258
259 struct SbEntry **sbep = NULL;
260 ARRAY_FOREACH_FROM(sbep, &wdata->entries, wdata->top_index)
261 {
262 if (row >= page_size)
263 break;
264 if ((*sbep)->is_hidden)
265 continue;
266
267 last = ARRAY_FOREACH_IDX_sbep;
268 if (row == wanted_row)
269 return last;
270 row++;
271 }
272
273 return last; // clamp to the last visible entry on the page
274}

364static int sb_scroll_view(struct SidebarFunctionData *fdata, int delta)
365{
366 struct SidebarWindowData *wdata = fdata->wdata;
367 if (!mutt_window_is_visible(wdata->win))
368 return FR_NO_ACTION;
369
370 if (ARRAY_EMPTY(&wdata->entries) || (wdata->hil_index < 0) || (wdata->top_index < 0))
371 return FR_NO_ACTION;
372
373 const int page_size = wdata->win->state.rows;
374
375 int new_top;
376 if (delta < 0)
377 new_top = sb_index_visible_before(wdata, wdata->top_index, -delta);
378 else
379 new_top = sb_index_visible_after(wdata, wdata->top_index, delta);
380
381 /* Tie the last entry to the bottom of the screen (don't scroll into space) */
382 const int last = sb_last_visible(wdata);
383 if (last >= 0)
384 {
385 const int max_top = sb_index_visible_before(wdata, last, page_size - 1);
386 if (new_top > max_top)
387 new_top = max_top;
388 }
389
390 if (new_top == wdata->top_index)
391 return FR_NO_ACTION;
392
393 wdata->top_index = new_top;
394
395 /* Drag the selection into the visible range */
396 const int first = sb_screen_row_index(wdata, 0);
397 const int bottom = sb_screen_row_index(wdata, page_size - 1);
398 if ((first >= 0) && (wdata->hil_index < first))
399 wdata->hil_index = first;
400 else if ((bottom >= 0) && (wdata->hil_index > bottom))
401 wdata->hil_index = bottom;
402
403 wdata->win->actions |= WA_RECALC;
404 return FR_SUCCESS;
405}

438static int op_sidebar_select_entry_by_number(struct SidebarFunctionData *fdata, const struct KeyEvent *event)
439{
440 struct SidebarWindowData *wdata = fdata->wdata;
441 if (!mutt_window_is_visible(wdata->win))
442 return FR_NO_ACTION;
443
444 if (ARRAY_EMPTY(&wdata->entries) || (wdata->hil_index < 0))
445 {
446 mutt_error(_("There are no mailboxes"));
447 return FR_ERROR;
448 }
449
450 const int count = ARRAY_SIZE(&wdata->entries);
451 struct Buffer *buf = buf_pool_get();
452 int rc = FR_ERROR;
453
454 int num = event->count;
455 if (num == 0)
456 {
457 if ((mw_get_field(_("Jump to: "), buf, MUTT_COMP_NONE, HC_OTHER, NULL, NULL) != 0) ||
458 buf_is_empty(buf))
459 {
460 rc = FR_NO_ACTION;
461 goto done;
462 }
463
464 if (!mutt_str_atoi_full(buf_string(buf), &num))
465 {
466 mutt_error(_("Invalid index number"));
467 goto done;
468 }
469 }
470
471 if ((num < 1) || (num > count))
472 {
473 mutt_error(_("Invalid index number"));
474 goto done;
475 }
476
477 if ((num - 1) == wdata->hil_index)
478 {
479 rc = FR_NO_ACTION;
480 goto done;
481 }
482
483 wdata->hil_index = num - 1; // entry numbers are 1-based
484 wdata->win->actions |= WA_RECALC;
485 rc = op_sidebar_open(fdata, event);
486
487done:
488 buf_pool_release(&buf);
489 return rc;
490}

907int sb_function_dispatcher(struct MuttWindow *win, const struct KeyEvent *event)
908{
909 if (!event || !win || !win->wdata)
910 return FR_UNKNOWN;
911
912 const int op = event->op;
913
914 struct SidebarModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_SIDEBAR);
915
916 struct SidebarFunctionData fdata = {
917 .n = NeoMutt,
918 .mod_data = mod_data,
919 .wdata = win->wdata,
920 };
921
922 int rc = FR_UNKNOWN;
923 for (size_t i = 0; SidebarFunctions[i].op != OP_NULL; i++)
924 {
925 const struct SidebarFunction *fn = &SidebarFunctions[i];
926 if (fn->op == op)
927 {
928 rc = fn->function(&fdata, event);
929 break;
930 }
931 }
932
933 if (rc == FR_UNKNOWN) // Not our function
934 return rc;
935
936 const char *result = dispatcher_get_retval_name(rc);
937 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
938
939 dispatcher_flush_on_error(rc);
940 return rc;
941}
