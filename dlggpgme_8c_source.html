381 static const char *parse_dn_part(struct DnArray *array, const char *str)
382 {
383  const char *s = NULL, *s1 = NULL;
384  size_t n;
385  char *p = NULL;
386 
387  /* parse attribute type */
388  for (s = str + 1; (s[0] != '\0') && (s[0] != '='); s++)
389  ; // do nothing
390 
391  if (s[0] == '\0')
392  return NULL; /* error */
393  n = s - str;
394  if (n == 0)
395  return NULL; /* empty key */
396  array->key = mutt_mem_malloc(n + 1);
397  p = array->key;
398  memcpy(p, str, n); /* fixme: trim trailing spaces */
399  p[n] = 0;
400  str = s + 1;
401 
402  if (*str == '#')
403  { /* hexstring */
404  str++;
405  for (s = str; isxdigit(*s); s++)
406  s++;
407  n = s - str;
408  if ((n == 0) || (n & 1))
409  return NULL; /* empty or odd number of digits */
410  n /= 2;
411  p = mutt_mem_malloc(n + 1);
412  array->value = (char *) p;
413  for (s1 = str; n; s1 += 2, n--)
414  sscanf(s1, "%2hhx", (unsigned char *) p++);
415  *p = '\0';
416  }
417  else
418  { /* regular v3 quoted string */
419  for (n = 0, s = str; *s; s++)
420  {
421  if (*s == '\\')
422  { /* pair */
423  s++;
424  if ((*s == ',') || (*s == '=') || (*s == '+') || (*s == '<') || (*s == '>') ||
425  (*s == '#') || (*s == ';') || (*s == '\\') || (*s == '\"') || (*s == ' '))
426  {
427  n++;
428  }
429  else if (isxdigit(s[0]) && isxdigit(s[1]))
430  {
431  s++;
432  n++;
433  }
434  else
435  return NULL; /* invalid escape sequence */
436  }
437  else if (*s == '\"')
438  return NULL; /* invalid encoding */
439  else if ((*s == ',') || (*s == '=') || (*s == '+') || (*s == '<') ||
440  (*s == '>') || (*s == '#') || (*s == ';'))
441  {
442  break;
443  }
444  else
445  n++;
446  }
447 
448  p = mutt_mem_malloc(n + 1);
449  array->value = (char *) p;
450  for (s = str; n; s++, n--)
451  {
452  if (*s == '\\')
453  {
454  s++;
455  if (isxdigit(*s))
456  {
457  sscanf(s, "%2hhx", (unsigned char *) p++);
458  s++;
459  }
460  else
461  *p++ = *s;
462  }
463  else
464  *p++ = *s;
465  }
466  *p = '\0';
467  }
468  return s;
469 }
