113static char *crypt_key_abilities(KeyFlags flags)
114{
115 static char buf[3];
116
117 if (!(flags & KEYFLAG_CANENCRYPT))
118 buf[0] = '-';
119 else if (flags & KEYFLAG_PREFER_SIGNING)
120 buf[0] = '.';
121 else
122 buf[0] = 'e';
123
124 if (!(flags & KEYFLAG_CANSIGN))
125 buf[1] = '-';
126 else if (flags & KEYFLAG_PREFER_ENCRYPTION)
127 buf[1] = '.';
128 else
129 buf[1] = 's';
130
131 buf[2] = '\0';
132
133 return buf;
134}

173void pgp_entry_gpgme_date(const struct ExpandoNode *node, void *data,
174 MuttFormatFlags flags, struct Buffer *buf)
175{
176#ifdef HAVE_PKG_GPGME
177 const struct CryptEntry *entry = data;
178 const struct CryptKeyInfo *key = entry->key;
179
180 const char *text = node->text;
181 bool use_c_locale = false;
182 if (*text == '!')
183 {
184 use_c_locale = true;
185 text++;
186 }
187
188 struct tm tm = { 0 };
189 if (key->kobj->subkeys && (key->kobj->subkeys->timestamp > 0))
190 {
191 tm = mutt_date_localtime(key->kobj->subkeys->timestamp);
192 }
193 else
194 {
195 tm = mutt_date_localtime(0); // Default to 1970-01-01
196 }
197
198 char tmp[128] = { 0 };
199 if (use_c_locale)
200 {
201 strftime_l(tmp, sizeof(tmp), text, &tm, NeoMutt->time_c_locale);
202 }
203 else
204 {
205 strftime(tmp, sizeof(tmp), text, &tm);
206 }
207
208 buf_strcpy(buf, tmp);
209#endif

429static int gpgme_key_config_observer(struct NotifyCallback *nc)

492struct CryptKeyInfo *dlg_gpgme(struct CryptKeyInfo *keys, struct Address *p,
493 const char *s, unsigned int app, bool *forced_valid)
494{
495 /* build the key table */
496 struct CryptKeyInfoArray ckia = ARRAY_HEAD_INITIALIZER;
497 const bool c_pgp_show_unusable = cs_subset_bool(NeoMutt->sub, "pgp_show_unusable");
498 bool unusable = false;
499 for (struct CryptKeyInfo *k = keys; k; k = k->next)
500 {
501 if (!c_pgp_show_unusable && (k->flags & KEYFLAG_CANTUSE))
502 {
503 unusable = true;
504 continue;
505 }
506
507 ARRAY_ADD(&ckia, k);
508 }
509
510 if ((ARRAY_SIZE(&ckia) == 0) && unusable)
511 {
512 mutt_error(_("All matching keys are marked expired/revoked"));
513 return NULL;
514 }
515
516 gpgme_sort_keys(&ckia);
517
518 enum MenuType menu_to_use = MENU_GENERIC;
519 if (app & APPLICATION_PGP)
520 menu_to_use = MENU_KEY_SELECT_PGP;
521 else if (app & APPLICATION_SMIME)
522 menu_to_use = MENU_KEY_SELECT_SMIME;
523
524 struct SimpleDialogWindows sdw = simple_dialog_new(menu_to_use, WT_DLG_GPGME, GpgmeHelp);
525
526 struct Menu *menu = sdw.menu;
527 struct GpgmeData gd = { false, menu, &ckia, NULL, forced_valid };
528
529 menu->max = ARRAY_SIZE(&ckia);
530 menu->make_entry = crypt_make_entry;
531 menu->mdata = &gd;
532 menu->mdata_free = NULL; // Menu doesn't own the data
533
534 // NT_COLOR is handled by the SimpleDialog
535 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, gpgme_key_config_observer, menu);
536 notify_observer_add(menu->win->notify, NT_WINDOW, gpgme_key_window_observer, menu->win);
537
538 const char *ts = NULL;
539
540 if ((app & APPLICATION_PGP) && (app & APPLICATION_SMIME))
541 ts = _("PGP and S/MIME keys matching");
542 else if ((app & APPLICATION_PGP))
543 ts = _("PGP keys matching");
544 else if ((app & APPLICATION_SMIME))
545 ts = _("S/MIME keys matching");
546 else
547 ts = _("keys matching");
548
549 char buf[1024] = { 0 };
550 if (p)
551 {
552 /* L10N: 1$s is one of the previous four entries.
553 %2$s is an address.
554 e.g. "S/MIME keys matching <john.doe@example.com>" */
555 snprintf(buf, sizeof(buf), _("%s <%s>"), ts, buf_string(p->mailbox));
556 }
557 else
558 {
559 /* L10N: e.g. 'S/MIME keys matching "John Doe".' */
560 snprintf(buf, sizeof(buf), _("%s \"%s\""), ts, s);
561 }
562
563 sbar_set_title(sdw.sbar, buf);
564
565 mutt_clear_error();
566
567 struct MuttWindow *old_focus = window_set_focus(menu->win);
568 // ---------------------------------------------------------------------------
569 // Event Loop
570 int op = OP_NULL;
571 do
572 {
573 menu_tagging_dispatcher(menu->win, op);
574 window_redraw(NULL);
575
576 op = km_dokey(menu_to_use, GETCH_NO_FLAGS);
577 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
578 if (op < 0)
579 continue;
580 if (op == OP_NULL)
581 {
582 km_error_key(menu_to_use);
583 continue;
584 }
585 mutt_clear_error();
586
587 int rc = gpgme_function_dispatcher(sdw.dlg, op);
588
589 if (rc == FR_UNKNOWN)
590 rc = menu_function_dispatcher(menu->win, op);
591 if (rc == FR_UNKNOWN)
592 rc = global_function_dispatcher(NULL, op);
593 } while (!gd.done);
594 // ---------------------------------------------------------------------------
595
596 ARRAY_FREE(&ckia);
597 window_set_focus(old_focus);
598 simple_dialog_free(&sdw.dlg);
599 return gd.key;
600}
