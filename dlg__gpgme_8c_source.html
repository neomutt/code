319static const char *crypt_format_str(char *buf, size_t buflen, size_t col, int cols,
320 char op, const char *src, const char *prec,
321 const char *if_str, const char *else_str,
322 intptr_t data, MuttFormatFlags flags)
323{
324 char fmt[128] = { 0 };
325 bool optional = (flags & MUTT_FORMAT_OPTIONAL);
326
327 struct CryptEntry *entry = (struct CryptEntry *) data;
328 struct CryptKeyInfo *key = entry->key;
329
330 KeyFlags kflags = (key->flags);
331
332 switch (tolower(op))
333 {
334 case 'a':
335 if (!optional)
336 {
337 const char *s = NULL;
338 snprintf(fmt, sizeof(fmt), "%%%s.3s", prec);
339 if (key->kobj->subkeys)
340 s = gpgme_pubkey_algo_name(key->kobj->subkeys->pubkey_algo);
341 else
342 s = "?";
343 snprintf(buf, buflen, fmt, s);
344 }
345 break;
346
347 case 'c':
348 if (!optional)
349 {
350 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
351 snprintf(buf, buflen, fmt, crypt_key_abilities(kflags));
352 }
353 else if (!(kflags & KEYFLAG_ABILITIES))
354 {
355 optional = false;
356 }
357 break;
358
359 case 'f':
360 if (!optional)
361 {
362 snprintf(fmt, sizeof(fmt), "%%%sc", prec);
363 snprintf(buf, buflen, fmt, crypt_flags(kflags));
364 }
365 else if (!(kflags & KEYFLAG_RESTRICTIONS))
366 {
367 optional = false;
368 }
369 break;
370
371 case 'i':
372 if (!optional)
373 {
374 /* fixme: we need a way to distinguish between main and subkeys.
375 * Store the idx in entry? */
376 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
377 snprintf(buf, buflen, fmt, crypt_fpr_or_lkeyid(key));
378 }
379 break;
380
381 case 'k':
382 if (!optional)
383 {
384 /* fixme: we need a way to distinguish between main and subkeys.
385 * Store the idx in entry? */
386 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
387 snprintf(buf, buflen, fmt, crypt_keyid(key));
388 }
389 break;
390
391 case 'l':
392 if (!optional)
393 {
394 snprintf(fmt, sizeof(fmt), "%%%slu", prec);
395 unsigned long val;
396 if (key->kobj->subkeys)
397 val = key->kobj->subkeys->length;
398 else
399 val = 0;
400 snprintf(buf, buflen, fmt, val);
401 }
402 break;
403
404 case 'n':
405 if (!optional)
406 {
407 snprintf(fmt, sizeof(fmt), "%%%sd", prec);
408 snprintf(buf, buflen, fmt, entry->num);
409 }
410 break;
411
412 case 'p':
413 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
414 snprintf(buf, buflen, fmt, gpgme_get_protocol_name(key->kobj->protocol));
415 break;
416
417 case 't':
418 {
419 char *s = NULL;
420 if ((kflags & KEYFLAG_ISX509))
421 {
422 s = "x";
423 }
424 else
425 {
426 switch (key->validity)
427 {
428 case GPGME_VALIDITY_FULL:
429 s = "f";
430 break;
431 case GPGME_VALIDITY_MARGINAL:
432 s = "m";
433 break;
434 case GPGME_VALIDITY_NEVER:
435 s = "n";
436 break;
437 case GPGME_VALIDITY_ULTIMATE:
438 s = "u";
439 break;
440 case GPGME_VALIDITY_UNDEFINED:
441 s = "q";
442 break;
443 case GPGME_VALIDITY_UNKNOWN:
444 default:
445 s = "?";
446 break;
447 }
448 }
449 snprintf(fmt, sizeof(fmt), "%%%sc", prec);
450 snprintf(buf, buflen, fmt, *s);
451 break;
452 }
453
454 case 'u':
455 if (!optional)
456 {
457 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
458 snprintf(buf, buflen, fmt, key->uid);
459 }
460 break;
461
462 case '[':
463 {
464 char buf2[128] = { 0 };
465 bool use_c_locale = false;
466 struct tm tm = { 0 };
467
468 char *p = buf;
469
470 const char *cp = src;
471 if (*cp == '!')
472 {
473 use_c_locale = true;
474 cp++;
475 }
476
477 size_t len = buflen - 1;
478 while ((len > 0) && (*cp != ']'))
479 {
480 if (*cp == '%')
481 {
482 cp++;
483 if (len >= 2)
484 {
485 *p++ = '%';
486 *p++ = *cp;
487 len -= 2;
488 }
489 else
490 {
491 break; /* not enough space */
492 }
493 cp++;
494 }
495 else
496 {
497 *p++ = *cp++;
498 len--;
499 }
500 }
501 *p = '\0';
502
503 if (key->kobj->subkeys && (key->kobj->subkeys->timestamp > 0))
504 tm = mutt_date_localtime(key->kobj->subkeys->timestamp);
505 else
506 tm = mutt_date_localtime(0); // Default to 1970-01-01
507
508 if (use_c_locale)
509 strftime_l(buf2, sizeof(buf2), buf, &tm, NeoMutt->time_c_locale);
510 else
511 strftime(buf2, sizeof(buf2), buf, &tm);
512
513 snprintf(fmt, sizeof(fmt), "%%%ss", prec);
514 snprintf(buf, buflen, fmt, buf2);
515 if (len > 0)
516 src = cp + 1;
517 break;
518 }
519
520 default:
521 *buf = '\0';
522 }
523
524 if (optional)
525 {
526 mutt_expando_format(buf, buflen, col, cols, if_str, crypt_format_str, data,
527 MUTT_FORMAT_NO_FLAGS);
528 }
529 else if (flags & MUTT_FORMAT_OPTIONAL)
530 {
531 mutt_expando_format(buf, buflen, col, cols, else_str, crypt_format_str,
532 data, MUTT_FORMAT_NO_FLAGS);
533 }
534
535 /* We return the format string, unchanged */
536 return src;
537}

633struct CryptKeyInfo *dlg_gpgme(struct CryptKeyInfo *keys, struct Address *p,
634 const char *s, unsigned int app, bool *forced_valid)
635{
636 int keymax;
637 int i;
638 sort_t f = NULL;
639 enum MenuType menu_to_use = MENU_GENERIC;
640 bool unusable = false;
641
642 /* build the key table */
643 keymax = 0;
644 i = 0;
645 struct CryptKeyInfo **key_table = NULL;
646 const bool c_pgp_show_unusable = cs_subset_bool(NeoMutt->sub, "pgp_show_unusable");
647 for (struct CryptKeyInfo *k = keys; k; k = k->next)
648 {
649 if (!c_pgp_show_unusable && (k->flags & KEYFLAG_CANTUSE))
650 {
651 unusable = true;
652 continue;
653 }
654
655 if (i == keymax)
656 {
657 keymax += 20;
658 mutt_mem_realloc(&key_table, sizeof(struct CryptKeyInfo *) * keymax);
659 }
660
661 key_table[i++] = k;
662 }
663
664 if (!i && unusable)
665 {
666 mutt_error(_("All matching keys are marked expired/revoked"));
667 return NULL;
668 }
669
670 const short c_pgp_sort_keys = cs_subset_sort(NeoMutt->sub, "pgp_sort_keys");
671 switch (c_pgp_sort_keys & SORT_MASK)
672 {
673 case SORT_ADDRESS:
674 f = crypt_sort_address;
675 break;
676 case SORT_DATE:
677 f = crypt_sort_date;
678 break;
679 case SORT_KEYID:
680 f = crypt_sort_keyid;
681 break;
682 case SORT_TRUST:
683 default:
684 f = crypt_sort_trust;
685 break;
686 }
687
688 if (key_table)
689 {
690 bool sort_reverse = c_pgp_sort_keys & SORT_REVERSE;
691 mutt_qsort_r(key_table, i, sizeof(struct CryptKeyInfo *), f, &sort_reverse);
692 }
693
694 if (app & APPLICATION_PGP)
695 menu_to_use = MENU_KEY_SELECT_PGP;
696 else if (app & APPLICATION_SMIME)
697 menu_to_use = MENU_KEY_SELECT_SMIME;
698
699 struct MuttWindow *dlg = simple_dialog_new(menu_to_use, WT_DLG_GPGME, GpgmeHelp);
700
701 struct Menu *menu = dlg->wdata;
702 menu->max = i;
703 menu->make_entry = crypt_make_entry;
704 menu->mdata = key_table;
705 menu->mdata_free = gpgme_key_table_free;
706
707 struct GpgmeData gd = { false, menu, key_table, NULL, forced_valid };
708 dlg->wdata = &gd;
709
710 // NT_COLOR is handled by the SimpleDialog
711 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, gpgme_key_config_observer, menu);
712 notify_observer_add(menu->win->notify, NT_WINDOW, gpgme_key_window_observer, menu->win);
713
714 const char *ts = NULL;
715
716 if ((app & APPLICATION_PGP) && (app & APPLICATION_SMIME))
717 ts = _("PGP and S/MIME keys matching");
718 else if ((app & APPLICATION_PGP))
719 ts = _("PGP keys matching");
720 else if ((app & APPLICATION_SMIME))
721 ts = _("S/MIME keys matching");
722 else
723 ts = _("keys matching");
724
725 char buf[1024] = { 0 };
726 if (p)
727 {
728 /* L10N: 1$s is one of the previous four entries.
729 %2$s is an address.
730 e.g. "S/MIME keys matching <john.doe@example.com>" */
731 snprintf(buf, sizeof(buf), _("%s <%s>"), ts, buf_string(p->mailbox));
732 }
733 else
734 {
735 /* L10N: e.g. 'S/MIME keys matching "John Doe".' */
736 snprintf(buf, sizeof(buf), _("%s \"%s\""), ts, s);
737 }
738
739 struct MuttWindow *sbar = window_find_child(dlg, WT_STATUS_BAR);
740 sbar_set_title(sbar, buf);
741
742 mutt_clear_error();
743
744 struct MuttWindow *old_focus = window_set_focus(menu->win);
745 // ---------------------------------------------------------------------------
746 // Event Loop
747 int op = OP_NULL;
748 do
749 {
750 menu_tagging_dispatcher(menu->win, op);
751 window_redraw(NULL);
752
753 op = km_dokey(menu_to_use, GETCH_NO_FLAGS);
754 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
755 if (op < 0)
756 continue;
757 if (op == OP_NULL)
758 {
759 km_error_key(menu_to_use);
760 continue;
761 }
762 mutt_clear_error();
763
764 int rc = gpgme_function_dispatcher(dlg, op);
765
766 if (rc == FR_UNKNOWN)
767 rc = menu_function_dispatcher(menu->win, op);
768 if (rc == FR_UNKNOWN)
769 rc = global_function_dispatcher(NULL, op);
770 } while (!gd.done);
771 // ---------------------------------------------------------------------------
772
773 window_set_focus(old_focus);
774 simple_dialog_free(&dlg);
775 return gd.key;
776}
