51void log_bind(const struct MenuDefinition *md, const char *keystr, struct Keymap *km)
52{
53 const char *fn_name = NULL;
54
55 struct SubMenu **smp = NULL;

63 mutt_debug(LL_DEBUG1, " bind %-8s <%s>\n", keystr, fn_name);

130void dump_submenu_bindings(const struct MenuDefinition *md, const struct SubMenu *sm, bool brief)
131{
132 if (STAILQ_EMPTY(&sm->keymaps))
133 {
134 mutt_debug(LL_DEBUG1, " [NONE]\n");
135 return;
136 }
137
138 struct Buffer *binding = buf_pool_get();
139 struct Buffer *esc_key = buf_pool_get();
140
141 int count = 0;
142 struct Keymap *km = NULL;
143 STAILQ_FOREACH(km, &sm->keymaps, entries)
144 {
145 if (brief && (count++ > 2))
146 {
147 mutt_debug(LL_DEBUG1, " ...\n");
148 break;
149 }
150
151 buf_reset(binding);
152 keymap_expand_key(km, binding);
153
154 buf_reset(esc_key);
155 escape_string(esc_key, buf_string(binding));
156
157 if (km->op == OP_MACRO)
158 log_macro(buf_string(esc_key), km);
159 else
160 log_bind(md, buf_string(esc_key), km);
161 }
162
163 buf_pool_release(&binding);
164 buf_pool_release(&esc_key);
165}

199void dump_menus(struct SubMenuIdArray *smia)
200{
201 struct Buffer *buf = buf_pool_get();
202 struct MenuDefinition *md = NULL;
203
204 mutt_debug(LL_DEBUG1, "Menus:\n");
205 ARRAY_FOREACH(md, &MenuDefs)
206 {
207 buf_printf(buf, " \"%s\" - %s (%d) - SubMenu IDs: ", md->name,
208 name_menu_type(md->id), md->id);
209
210 struct SubMenu **smp = NULL;
211 ARRAY_FOREACH(smp, &md->submenus)
212 {
213 struct SubMenuId *smi = NULL;
214 struct SubMenu *sm = *smp;
215 int id = -1;
216
217 ARRAY_FOREACH(smi, smia)
218 {
219 if (smi->sm == sm)
220 {
221 id = smi->id;
222 break;
223 }
224 }
225
226 const char *name = sm->parent ? sm->parent->name : "UNKNOWN";
227 buf_add_printf(buf, "%s (%d)", name, id);
228
229 if (ARRAY_FOREACH_IDX_smp < (ARRAY_SIZE(&md->submenus) - 1))
230 {
231 buf_add_printf(buf, ", ");
232 }
233 }
234 mutt_debug(LL_DEBUG1, "%s\n", buf_string(buf));
235 }
236
237 buf_pool_release(&buf);
238}
