1077 int imap_cmd_step(struct ImapAccountData *adata)
1078 {
1079  if (!adata)
1080  return -1;
1081 
1082  size_t len = 0;
1083  int c;
1084  int rc;
1085  int stillrunning = 0;
1086  struct ImapCommand *cmd = NULL;
1087 
1088  if (adata->status == IMAP_FATAL)
1089  {
1090  cmd_handle_fatal(adata);
1091  return IMAP_RES_BAD;
1092  }
1093 
1094  /* read into buffer, expanding buffer as necessary until we have a full
1095  * line */
1096  do
1097  {
1098  if (len == adata->blen)
1099  {
1100  mutt_mem_realloc(&adata->buf, adata->blen + IMAP_CMD_BUFSIZE);
1101  adata->blen = adata->blen + IMAP_CMD_BUFSIZE;
1102  mutt_debug(LL_DEBUG3, "grew buffer to %lu bytes\n", adata->blen);
1103  }
1104 
1105  /* back up over '\0' */
1106  if (len)
1107  len--;
1108  c = mutt_socket_readln_d(adata->buf + len, adata->blen - len, adata->conn, MUTT_SOCK_LOG_FULL);
1109  if (c <= 0)
1110  {
1111  mutt_debug(LL_DEBUG1, "Error reading server response\n");
1112  cmd_handle_fatal(adata);
1113  return IMAP_RES_BAD;
1114  }
1115 
1116  len += c;
1117  }
1118  /* if we've read all the way to the end of the buffer, we haven't read a
1119  * full line (mutt_socket_readln strips the \r, so we always have at least
1120  * one character free when we've read a full line) */
1121  while (len == adata->blen);
1122 
1123  /* don't let one large string make cmd->buf hog memory forever */
1124  if ((adata->blen > IMAP_CMD_BUFSIZE) && (len <= IMAP_CMD_BUFSIZE))
1125  {
1126  mutt_mem_realloc(&adata->buf, IMAP_CMD_BUFSIZE);
1127  adata->blen = IMAP_CMD_BUFSIZE;
1128  mutt_debug(LL_DEBUG3, "shrank buffer to %lu bytes\n", adata->blen);
1129  }
1130 
1131  adata->lastread = mutt_date_epoch();
1132 
1133  /* handle untagged messages. The caller still gets its shot afterwards. */
1134  if ((mutt_str_startswith(adata->buf, "* ", CASE_MATCH) ||
1135  mutt_str_startswith(imap_next_word(adata->buf), "OK [", CASE_MATCH)) &&
1136  cmd_handle_untagged(adata))
1137  {
1138  return IMAP_RES_BAD;
1139  }
1140 
1141  /* server demands a continuation response from us */
1142  if (adata->buf[0] == '+')
1143  return IMAP_RES_RESPOND;
1144 
1145  /* Look for tagged command completions.
1146  *
1147  * Some response handlers can end up recursively calling
1148  * imap_cmd_step() and end up handling all tagged command
1149  * completions.
1150  * (e.g. FETCH->set_flag->set_header_color->~h pattern match.)
1151  *
1152  * Other callers don't even create an adata->cmds entry.
1153  *
1154  * For both these cases, we default to returning OK */
1155  rc = IMAP_RES_OK;
1156  c = adata->lastcmd;
1157  do
1158  {
1159  cmd = &adata->cmds[c];
1160  if (cmd->state == IMAP_RES_NEW)
1161  {
1162  if (mutt_str_startswith(adata->buf, cmd->seq, CASE_MATCH))
1163  {
1164  if (!stillrunning)
1165  {
1166  /* first command in queue has finished - move queue pointer up */
1167  adata->lastcmd = (adata->lastcmd + 1) % adata->cmdslots;
1168  }
1169  cmd->state = cmd_status(adata->buf);
1170  rc = cmd->state;
1171  if (cmd->state == IMAP_RES_NO || cmd->state == IMAP_RES_BAD)
1172  {
1173  mutt_message(_("IMAP command failed: %s"), adata->buf);
1174  }
1175  }
1176  else
1177  stillrunning++;
1178  }
1179 
1180  c = (c + 1) % adata->cmdslots;
1181  } while (c != adata->nextcmd);
1182 
1183  if (stillrunning)
1184  rc = IMAP_RES_CONTINUE;
1185  else
1186  {
1187  mutt_debug(LL_DEBUG3, "IMAP queue drained\n");
1188  imap_cmd_finish(adata);
1189  }
1190 
1191  return rc;
1192 }
