325static void print_compile_options(const struct CompileOptions *co, FILE *fp)
326{
327 if (!co || !fp)
328 return;
329
330 size_t used = 2;
331 bool tty = isatty(fileno(fp));
332
333 fprintf(fp, " ");
334 for (int i = 0; co[i].name; i++)
335 {
336 const size_t len = strlen(co[i].name) + 2; /* +/- and a space */
337 if ((used + len) > SCREEN_WIDTH)
338 {
339 used = 2;
340 fprintf(fp, "\n ");
341 }
342 used += len;
343 const char *fmt = "?%s ";
344 switch (co[i].enabled)
345 {
346 case 0: // Disabled
347 if (tty)
348 fmt = "\033[1;31m-%s\033[0m "; // Escape, red
349 else
350 fmt = "-%s ";
351 break;
352 case 1: // Enabled
353 if (tty)
354 fmt = "\033[1;32m+%s\033[0m "; // Escape, green
355 else
356 fmt = "+%s ";
357 break;
358 case 2: // Devel only
359 if (tty)
360 fmt = "\033[1;36m%s\033[0m "; // Escape, cyan
361 else
362 fmt = "%s ";
363 break;
364 }
365 fprintf(fp, fmt, co[i].name);
366 }
367 fprintf(fp, "\n");
368}

377static char *rstrip_in_place(char *s)
378{
379 if (!s)
380 return NULL;
381
382 char *p = &s[strlen(s)];
383 if (p == s)
384 return s;
385 p--;
386 while ((p >= s) && ((*p == '\n') || (*p == '\r')))
387 *p-- = '\0';
388 return s;
389}

399bool print_version(FILE *fp)
400{
401 if (!fp)
402 return false;
403
404 struct utsname uts = { 0 };
405 bool tty = isatty(fileno(fp));
406 const char *fmt = "%s\n";
407
408 if (tty)
409 fmt = "\033[1;36m%s\033[0m\n"; // Escape, cyan
410
411 fprintf(fp, fmt, mutt_make_version());
412 fprintf(fp, "%s\n", _(Notice));
413
414 uname(&uts);
415
416#ifdef SCO
417 fprintf(fp, "System: SCO %s", uts.release);
418#else
419 fprintf(fp, "System: %s %s", uts.sysname, uts.release);
420#endif
421
422 fprintf(fp, " (%s)", uts.machine);
423
424#ifdef NCURSES_VERSION
425 fprintf(fp, "\nncurses: %s (compiled with %s.%d)", curses_version(),
426 NCURSES_VERSION, NCURSES_VERSION_PATCH);
427#else
428 fprintf(fp, "\nncurses: %s", curses_version());
429#endif
430
431#ifdef _LIBICONV_VERSION
432 fprintf(fp, "\nlibiconv: %d.%d", _LIBICONV_VERSION >> 8, _LIBICONV_VERSION & 0xff);
433#endif
434
435#ifdef HAVE_LIBIDN
436 fprintf(fp, "\n%s", mutt_idna_print_version());
437#endif
438
439#ifdef CRYPT_BACKEND_GPGME
440 fprintf(fp, "\nGPGME: %s", mutt_gpgme_print_version());
441#endif
442
443#ifdef USE_SSL_OPENSSL
444#ifdef LIBRESSL_VERSION_TEXT
445 fprintf(fp, "\nLibreSSL: %s", LIBRESSL_VERSION_TEXT);
446#endif
447#ifdef OPENSSL_VERSION_TEXT
448 fprintf(fp, "\nOpenSSL: %s", OPENSSL_VERSION_TEXT);
449#endif
450#endif
451
452#ifdef USE_SSL_GNUTLS
453 fprintf(fp, "\nGnuTLS: %s", GNUTLS_VERSION);
454#endif
455
456#ifdef HAVE_NOTMUCH
457 fprintf(fp, "\nlibnotmuch: %d.%d.%d", LIBNOTMUCH_MAJOR_VERSION,
458 LIBNOTMUCH_MINOR_VERSION, LIBNOTMUCH_MICRO_VERSION);
459#endif
460
461#ifdef HAVE_PCRE2

465 fprintf(fp, "\nPCRE2: %s", version);

468
469#ifdef USE_HCACHE

476 FREE(&backends);
477#endif
478#endif
479
480 rstrip_in_place((char *) configure_options);
481 fprintf(fp, "\n\nConfigure options: %s\n", (char *) configure_options);
482
483 rstrip_in_place((char *) cc_cflags);
484 fprintf(fp, "\nCompilation CFLAGS: %s\n", (char *) cc_cflags);
485
486 fprintf(fp, "\n%s\n", _("Compile options:"));
487 print_compile_options(CompOpts, fp);
488
489 if (DebugOpts[0].name)
490 {
491 fprintf(fp, "\n%s\n", _("Devel options:"));
492 print_compile_options(DebugOpts, fp);
493 }
494
495 fprintf(fp, "\n");
496#ifdef DOMAIN
497 fprintf(fp, "DOMAIN=\"%s\"\n", DOMAIN);
498#endif
499#ifdef ISPELL
500 fprintf(fp, "ISPELL=\"%s\"\n", ISPELL);
501#endif
502 fprintf(fp, "MAILPATH=\"%s\"\n", MAILPATH);
503#ifdef MIXMASTER
504 fprintf(fp, "MIXMASTER=\"%s\"\n", MIXMASTER);
505#endif
506 fprintf(fp, "PKGDATADIR=\"%s\"\n", PKGDATADIR);
507 fprintf(fp, "SENDMAIL=\"%s\"\n", SENDMAIL);
508 fprintf(fp, "SYSCONFDIR=\"%s\"\n", SYSCONFDIR);
509
510 fprintf(fp, "\n");
511 fputs(_(ReachingUs), fp);
512
513 fflush(fp);
514 return !ferror(fp);
515}

551bool feature_enabled(const char *name)
552{

556 {
