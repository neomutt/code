72static MixCapFlags mix_get_caps(const char *capstr)
73{
74 MixCapFlags caps = MIX_CAP_NO_FLAGS;
75
76 while (*capstr)
77 {
78 switch (*capstr)
79 {
80 case 'C':
81 caps |= MIX_CAP_COMPRESS;
82 break;
83
84 case 'M':
85 caps |= MIX_CAP_MIDDLEMAN;
86 break;
87
88 case 'N':
89 {
90 switch (*++capstr)
91 {
92 case 'm':
93 caps |= MIX_CAP_NEWSMAIL;
94 break;
95
96 case 'p':
97 caps |= MIX_CAP_NEWSPOST;
98 break;
99 }
100 }
101 }
102
103 if (*capstr)
104 capstr++;
105 }
106
107 return caps;
108}
