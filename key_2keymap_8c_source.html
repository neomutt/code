162struct Keymap *keymap_compare(struct Keymap *km1, struct Keymap *km2, size_t *pos)
163{
164 *pos = 0;
165
166 while ((*pos < km1->len) && (*pos < km2->len))
167 {
168 if (km1->keys[*pos] < km2->keys[*pos])
169 return km2;
170
171 if (km1->keys[*pos] > km2->keys[*pos])
172 return km1;
173
174 *pos = *pos + 1;
175 }
176
177 return NULL;
178}

185void keymap_get_name(int c, struct Buffer *buf)
186{
187 const char *name = mutt_map_get_name(c, KeyNames);
188 if (name)
189 {
190 buf_addstr(buf, name);
191 return;
192 }
193
194 if ((c < 256) && (c > -128) && iscntrl((unsigned char) c))
195 {
196 if (c < 0)
197 c += 256;
198
199 if (c < 128)
200 {
201 buf_addch(buf, '^');
202 buf_addch(buf, (c + '@') & 0x7f);
203 }
204 else
205 {
206 buf_add_printf(buf, "\\%d%d%d", c >> 6, (c >> 3) & 7, c & 7);
207 }
208 }
209 else if ((c >= KEY_F0) && (c < KEY_F(256))) /* this maximum is just a guess */
210 {
211 buf_add_printf(buf, "<F%d>", c - KEY_F0);
212 }
213 else if ((c < 256) && (c >= -128) && IsPrint(c))
214 {
215 buf_add_printf(buf, "%c", (unsigned char) c);
216 }
217 else
218 {
219 buf_add_printf(buf, "<%ho>", (unsigned short) c);
220 }
221}

266int parse_fkey(char *str)
267{
268 char *t = NULL;
269 int n = 0;
270
271 if ((str[0] != '<') || (mutt_tolower(str[1]) != 'f'))
272 return -1;
273
274 for (t = str + 2; *t && mutt_isdigit(*t); t++)
275 {
276 n *= 10;
277 n += *t - '0';
278 }
279
280 if (*t != '>')
281 return -1;
282 return n;
283}

293int parse_keycode(const char *str)
294{
295 char *end_char = NULL;
296 long int result = strtol(str + 1, &end_char, 8);
297
298 /* allow trailing whitespace, eg. < 1001 > */
299 while (mutt_isspace(*end_char))
300 end_char++;
301
302 /* negative keycodes don't make sense, also detect overflow */
303 if ((*end_char != '>') || (result < 0) || (result == LONG_MAX))
304 {
305 return -1;
306 }
307
308 return result;
309}

318size_t parse_keys(const char *str, keycode_t *d, size_t max)
319{
320 int n;
321 size_t len = max;
322 char buf[128] = { 0 };
323 char c;
324 char *t = NULL;
325
326 mutt_str_copy(buf, str, sizeof(buf));
327 char *s = buf;
328
329 while (*s && len)
330 {
331 *d = '\0';
332 if ((*s == '<') && (t = strchr(s, '>')))
333 {
334 t++;
335 c = *t;
336 *t = '\0';
337
338 n = mutt_map_get_value(s, KeyNames);
339 if (n != -1)
340 {
341 s = t;
342 *d = n;
343 }
344 else if ((n = parse_fkey(s)) > 0)
345 {
346 s = t;
347 *d = KEY_F(n);
348 }
349 else if ((n = parse_keycode(s)) > 0)
350 {
351 s = t;
352 *d = n;
353 }
354
355 *t = c;
356 }
357
358 if (!*d)
359 {
360 *d = (unsigned char) *s;
361 s++;
362 }
363 d++;
364 len--;
365 }
366
367 return max - len;
368}
