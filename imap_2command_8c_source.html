113static struct ImapCommand *cmd_new(struct ImapAccountData *adata)
114{
115 struct ImapCommand *cmd = NULL;
116
117 if (cmd_queue_full(adata))
118 {
119 mutt_debug(LL_DEBUG3, "IMAP command queue full\n");
120 return NULL;
121 }
122
123 cmd = adata->cmds + adata->nextcmd;
124 adata->nextcmd = (adata->nextcmd + 1) % adata->cmdslots;
125
126 snprintf(cmd->seq, sizeof(cmd->seq), "%c%04u", adata->seqid, adata->seqno++);
127 if (adata->seqno > 9999)
128 adata->seqno = 0;
129
130 cmd->state = IMAP_RES_NEW;
131
132 return cmd;
133}

145static int cmd_queue(struct ImapAccountData *adata, const char *cmdstr, ImapCmdFlags flags)
146{
147 if (cmd_queue_full(adata))
148 {
149 mutt_debug(LL_DEBUG3, "Draining IMAP command pipeline\n");
150
151 const int rc = imap_exec(adata, NULL, flags & IMAP_CMD_POLL);
152
153 if (rc == IMAP_EXEC_ERROR)
154 return IMAP_RES_BAD;
155 }
156
157 struct ImapCommand *cmd = cmd_new(adata);
158 if (!cmd)
159 return IMAP_RES_BAD;
160
161 if (buf_add_printf(&adata->cmdbuf, "%s %s\r\n", cmd->seq, cmdstr) < 0)
162 return IMAP_RES_BAD;
163
164 return 0;

207static int cmd_start(struct ImapAccountData *adata, const char *cmdstr, ImapCmdFlags flags)
208{
209 int rc;
210
211 if (adata->status == IMAP_FATAL)
212 {
213 cmd_handle_fatal(adata);
214 return -1;
215 }
216
217 if (cmdstr && ((rc = cmd_queue(adata, cmdstr, flags)) < 0))
218 return rc;
219
220 if (flags & IMAP_CMD_QUEUE)
221 return 0;
222
223 if (buf_is_empty(&adata->cmdbuf))
224 return IMAP_RES_BAD;
225
226 rc = mutt_socket_send_d(adata->conn, adata->cmdbuf.data,
227 (flags & IMAP_CMD_PASS) ? IMAP_LOG_PASS : IMAP_LOG_CMD);
228 buf_reset(&adata->cmdbuf);
229
230 /* unidle when command queue is flushed */
231 if (adata->state == IMAP_IDLE)
232 adata->state = IMAP_SELECTED;
233
234 return (rc < 0) ? IMAP_RES_BAD : 0;
235}

263static void cmd_parse_expunge(struct ImapAccountData *adata, const char *s)
264{
265 unsigned int exp_msn = 0;
266 struct Email *e = NULL;
267
268 mutt_debug(LL_DEBUG2, "Handling EXPUNGE\n");
269
270 struct ImapMboxData *mdata = adata->mailbox->mdata;
271 if (!mdata)
272 return;
273
274 if (!mutt_str_atoui(s, &exp_msn) || (exp_msn < 1) ||
275 (exp_msn > imap_msn_highest(&mdata->msn)))
276 {
277 return;
278 }
279
280 e = imap_msn_get(&mdata->msn, exp_msn - 1);
281 if (e)
282 {
283 /* imap_expunge_mailbox() will rewrite e->index.
284 * It needs to resort using EMAIL_SORT_UNSORTED anyway, so setting to INT_MAX
285 * makes the code simpler and possibly more efficient. */
286 e->index = INT_MAX;
287
288 struct ImapEmailData *edata = imap_edata_get(e);
289 if (edata)
290 edata->msn = 0;
291 }
292
293 /* decrement seqno of those above. */
294 const size_t max_msn = imap_msn_highest(&mdata->msn);
295 for (unsigned int cur = exp_msn; cur < max_msn; cur++)
296 {
297 e = imap_msn_get(&mdata->msn, cur);
298 if (e)
299 {
300 struct ImapEmailData *edata = imap_edata_get(e);
301 if (edata)
302 edata->msn--;
303 }
304 imap_msn_set(&mdata->msn, cur - 1, e);
305 }
306 imap_msn_shrink(&mdata->msn, 1);
307
308 mdata->reopen |= IMAP_EXPUNGE_PENDING;
309}

319static void cmd_parse_vanished(struct ImapAccountData *adata, char *s)
320{
321 bool earlier = false;
322 int rc;
323 unsigned int uid = 0;
324
325 struct ImapMboxData *mdata = adata->mailbox->mdata;
326 if (!mdata)
327 return;
328
329 mutt_debug(LL_DEBUG2, "Handling VANISHED\n");
330
331 if (mutt_istr_startswith(s, "(EARLIER)"))
332 {
333 /* The RFC says we should not decrement msns with the VANISHED EARLIER tag.
334 * My experimentation says that's crap. */
335 earlier = true;
336 s = imap_next_word(s);
337 }
338
339 char *end_of_seqset = s;
340 while (*end_of_seqset)
341 {
342 if (!strchr("0123456789:,", *end_of_seqset))
343 *end_of_seqset = '\0';
344 else
345 end_of_seqset++;
346 }
347
348 struct SeqsetIterator *iter = mutt_seqset_iterator_new(s);
349 if (!iter)
350 {
351 mutt_debug(LL_DEBUG2, "VANISHED: empty seqset [%s]?\n", s);
352 return;
353 }
354
355 while ((rc = mutt_seqset_iterator_next(iter, &uid)) == 0)
356 {
357 struct Email *e = mutt_hash_int_find(mdata->uid_hash, uid);
358 if (!e)
359 continue;
360
361 unsigned int exp_msn = imap_edata_get(e)->msn;
362
363 /* imap_expunge_mailbox() will rewrite e->index.
364 * It needs to resort using EMAIL_SORT_UNSORTED anyway, so setting to INT_MAX
365 * makes the code simpler and possibly more efficient. */
366 e->index = INT_MAX;
367 imap_edata_get(e)->msn = 0;
368
369 if ((exp_msn < 1) || (exp_msn > imap_msn_highest(&mdata->msn)))
370 {
371 mutt_debug(LL_DEBUG1, "VANISHED: msn for UID %u is incorrect\n", uid);
372 continue;
373 }
374 if (imap_msn_get(&mdata->msn, exp_msn - 1) != e)
375 {
376 mutt_debug(LL_DEBUG1, "VANISHED: msn_index for UID %u is incorrect\n", uid);
377 continue;
378 }
379
380 imap_msn_remove(&mdata->msn, exp_msn - 1);
381
382 if (!earlier)
383 {
384 /* decrement seqno of those above. */
385 const size_t max_msn = imap_msn_highest(&mdata->msn);
386 for (unsigned int cur = exp_msn; cur < max_msn; cur++)
387 {
388 e = imap_msn_get(&mdata->msn, cur);
389 if (e)
390 imap_edata_get(e)->msn--;
391 imap_msn_set(&mdata->msn, cur - 1, e);
392 }
393
394 imap_msn_shrink(&mdata->msn, 1);
395 }
396 }
397
398 if (rc < 0)
399 mutt_debug(LL_DEBUG1, "VANISHED: illegal seqset %s\n", s);
400
401 mdata->reopen |= IMAP_EXPUNGE_PENDING;
402
403 mutt_seqset_iterator_free(&iter);
404}

415static void cmd_parse_fetch(struct ImapAccountData *adata, char *s)
416{
417 unsigned int msn, uid;
418 struct Email *e = NULL;
419 char *flags = NULL;
420 int uid_checked = 0;
421 bool server_changes = false;
422
423 struct ImapMboxData *mdata = imap_mdata_get(adata->mailbox);
424 if (!mdata)
425 return;
426
427 mutt_debug(LL_DEBUG3, "Handling FETCH\n");
428
429 if (!mutt_str_atoui(s, &msn))
430 {
431 mutt_debug(LL_DEBUG3, "Skipping FETCH response - illegal MSN\n");
432 return;
433 }
434
435 if ((msn < 1) || (msn > imap_msn_highest(&mdata->msn)))
436 {
437 mutt_debug(LL_DEBUG3, "Skipping FETCH response - MSN %u out of range\n", msn);
438 return;
439 }
440
441 e = imap_msn_get(&mdata->msn, msn - 1);
442 if (!e || !e->active)
443 {
444 mutt_debug(LL_DEBUG3, "Skipping FETCH response - MSN %u not in msn_index\n", msn);
445 return;
446 }
447
448 struct ImapEmailData *edata = imap_edata_get(e);
449 if (!edata)
450 {
451 mutt_debug(LL_DEBUG3, "Skipping FETCH response - MSN %u missing edata\n", msn);
452 return;
453 }
454 /* skip FETCH */
455 s = imap_next_word(s);
456 s = imap_next_word(s);
457
458 if (*s != '(')
459 {
460 mutt_debug(LL_DEBUG1, "Malformed FETCH response\n");
461 return;
462 }
463 s++;
464
465 while (*s)
466 {
467 SKIPWS(s);
468 size_t plen = mutt_istr_startswith(s, "FLAGS");
469 if (plen != 0)
470 {
471 flags = s;
472 if (uid_checked)
473 break;
474
475 s += plen;
476 SKIPWS(s);
477 if (*s != '(')
478 {
479 mutt_debug(LL_DEBUG1, "bogus FLAGS response: %s\n", s);
480 return;
481 }
482 s++;
483 while (*s && (*s != ')'))
484 s++;
485 if (*s == ')')
486 {
487 s++;
488 }
489 else
490 {
491 mutt_debug(LL_DEBUG1, "Unterminated FLAGS response: %s\n", s);
492 return;
493 }
494 }
495 else if ((plen = mutt_istr_startswith(s, "UID")))
496 {
497 s += plen;
498 SKIPWS(s);
499 if (!mutt_str_atoui(s, &uid))
500 {
501 mutt_debug(LL_DEBUG1, "Illegal UID. Skipping update\n");
502 return;
503 }
504 if (uid != edata->uid)
505 {
506 mutt_debug(LL_DEBUG1, "UID vs MSN mismatch. Skipping update\n");
507 return;
508 }
509 uid_checked = 1;
510 if (flags)
511 break;
512 s = imap_next_word(s);
513 }
514 else if ((plen = mutt_istr_startswith(s, "MODSEQ")))
515 {
516 s += plen;
517 SKIPWS(s);
518 if (*s != '(')
519 {
520 mutt_debug(LL_DEBUG1, "bogus MODSEQ response: %s\n", s);
521 return;
522 }
523 s++;
524 while (*s && (*s != ')'))
525 s++;
526 if (*s == ')')
527 {
528 s++;
529 }
530 else
531 {
532 mutt_debug(LL_DEBUG1, "Unterminated MODSEQ response: %s\n", s);
533 return;
534 }
535 }
536 else if (*s == ')')
537 {
538 break; /* end of request */
539 }
540 else if (*s)
541 {
542 mutt_debug(LL_DEBUG2, "Only handle FLAGS updates\n");
543 break;
544 }
545 }
546
547 if (flags)
548 {
549 imap_set_flags(adata->mailbox, e, flags, &server_changes);
550 if (server_changes)
551 {
552 /* If server flags could conflict with NeoMutt's flags, reopen the mailbox. */
553 if (e->changed)
554 mdata->reopen |= IMAP_EXPUNGE_PENDING;
555 else
556 mdata->check_status |= IMAP_FLAGS_PENDING;
557 }
558 }
559}

567{

599static void cmd_parse_list(struct ImapAccountData *adata, char *s)
600{
601 struct ImapList *list = NULL;
602 struct ImapList lb = { 0 };
603 unsigned int litlen;
604
605 if (adata->cmdresult)
606 list = adata->cmdresult;
607 else
608 list = &lb;
609
610 memset(list, 0, sizeof(struct ImapList));
611
612 /* flags */
613 s = imap_next_word(s);
614 if (*s != '(')
615 {
616 mutt_debug(LL_DEBUG1, "Bad LIST response\n");
617 return;
618 }
619 s++;
620 while (*s)
621 {
622 if (mutt_istr_startswith(s, "\\NoSelect"))
623 list->noselect = true;
624 else if (mutt_istr_startswith(s, "\\NonExistent")) /* rfc5258 */
625 list->noselect = true;
626 else if (mutt_istr_startswith(s, "\\NoInferiors"))
627 list->noinferiors = true;
628 else if (mutt_istr_startswith(s, "\\HasNoChildren")) /* rfc5258*/
629 list->noinferiors = true;
630
631 s = imap_next_word(s);
632 if (*(s - 2) == ')')
633 break;
634 }
635
636 /* Delimiter */
637 if (!mutt_istr_startswith(s, "NIL"))
638 {
639 char delimbuf[5] = { 0 }; // worst case: "\\"\0
640 snprintf(delimbuf, sizeof(delimbuf), "%s", s);
641 imap_unquote_string(delimbuf);
642 list->delim = delimbuf[0];
643 }
644
645 /* Name */
646 s = imap_next_word(s);
647 /* Notes often responds with literals here. We need a real tokenizer. */
648 if (imap_get_literal_count(s, &litlen) == 0)
649 {
650 if (imap_cmd_step(adata) != IMAP_RES_CONTINUE)
651 {
652 adata->status = IMAP_FATAL;
653 return;
654 }
655
656 if (strlen(adata->buf) < litlen)
657 {
658 mutt_debug(LL_DEBUG1, "Error parsing LIST mailbox\n");
659 return;
660 }
661
662 list->name = adata->buf;
663 s = list->name + litlen;
664 if (s[0] != '\0')
665 {
666 s[0] = '\0';
667 s++;
668 SKIPWS(s);
669 }
670 }
671 else
672 {
673 list->name = s;
674 /* Exclude rfc5258 RECURSIVEMATCH CHILDINFO suffix */
675 s = imap_next_word(s);
676 if (s[0] != '\0')
677 s[-1] = '\0';
678 imap_unmunge_mbox_name(adata->unicode, list->name);
679 }
680
681 if (list->name[0] == '\0')
682 {
683 adata->delim = list->delim;
684 mutt_debug(LL_DEBUG3, "Root delimiter: %c\n", adata->delim);
685 }
686}

835static void cmd_parse_status(struct ImapAccountData *adata, char *s)
836{
837 unsigned int litlen = 0;
838
839 char *mailbox = imap_next_word(s);
840
841 /* We need a real tokenizer. */
842 if (imap_get_literal_count(mailbox, &litlen) == 0)
843 {
844 if (imap_cmd_step(adata) != IMAP_RES_CONTINUE)
845 {
846 adata->status = IMAP_FATAL;
847 return;
848 }
849
850 if (strlen(adata->buf) < litlen)
851 {
852 mutt_debug(LL_DEBUG1, "Error parsing STATUS mailbox\n");
853 return;
854 }
855
856 mailbox = adata->buf;
857 s = mailbox + litlen;
858 s[0] = '\0';
859 s++;
860 SKIPWS(s);
861 }
862 else
863 {
864 s = imap_next_word(mailbox);
865 s[-1] = '\0';
866 imap_unmunge_mbox_name(adata->unicode, mailbox);
867 }
868
869 struct Mailbox *m = find_mailbox(adata, mailbox);
870 struct ImapMboxData *mdata = imap_mdata_get(m);
871 if (!mdata)
872 {
873 mutt_debug(LL_DEBUG3, "Received status for an unexpected mailbox: %s\n", mailbox);
874 return;
875 }
876 uint32_t olduv = mdata->uidvalidity;
877 unsigned int oldun = mdata->uid_next;
878
879 if (*s++ != '(')
880 {
881 mutt_debug(LL_DEBUG1, "Error parsing STATUS\n");
882 return;
883 }
884 while ((s[0] != '\0') && (s[0] != ')'))
885 {
886 char *value = imap_next_word(s);
887
888 errno = 0;
889 const unsigned long ulcount = strtoul(value, &value, 10);
890 const bool truncated = ((errno == ERANGE) && (ulcount == ULONG_MAX)) ||
891 ((unsigned int) ulcount != ulcount);
892 const unsigned int count = (unsigned int) ulcount;
893
894 // we accept truncating a larger value only for UIDVALIDITY, to accommodate
895 // IMAP servers that use 64-bits for it. This seems to be what Thunderbird
896 // is also doing, see #3830
897 if (mutt_str_startswith(s, "UIDVALIDITY"))
898 {
899 if (truncated)
900 {
901 mutt_debug(LL_DEBUG1,
902 "UIDVALIDITY [%lu] exceeds 32 bits, "
903 "truncated to [%u]\n",

1154int imap_cmd_step(struct ImapAccountData *adata)
1155{
1156 if (!adata)
1157 return -1;
1158
1159 size_t len = 0;
1160 int c;
1161 int rc;
1162 int stillrunning = 0;
1163 struct ImapCommand *cmd = NULL;
1164
1165 if (adata->status == IMAP_FATAL)
1166 {
1167 cmd_handle_fatal(adata);
1168 return IMAP_RES_BAD;
1169 }
1170
1171 /* read into buffer, expanding buffer as necessary until we have a full
1172 * line */
1173 do
1174 {
1175 if (len == adata->blen)
1176 {
1177 MUTT_MEM_REALLOC(&adata->buf, adata->blen + IMAP_CMD_BUFSIZE, char);
1178 adata->blen = adata->blen + IMAP_CMD_BUFSIZE;
1179 mutt_debug(LL_DEBUG3, "grew buffer to %zu bytes\n", adata->blen);
1180 }
1181
1182 /* back up over '\0' */
1183 if (len)
1184 len--;
1185 c = mutt_socket_readln_d(adata->buf + len, adata->blen - len, adata->conn, MUTT_SOCK_LOG_FULL);
1186 if (c <= 0)
1187 {
1188 mutt_debug(LL_DEBUG1, "Error reading server response\n");
1189 cmd_handle_fatal(adata);
1190 return IMAP_RES_BAD;
1191 }
1192
1193 len += c;
1194 }
1195 /* if we've read all the way to the end of the buffer, we haven't read a
1196 * full line (mutt_socket_readln strips the \r, so we always have at least
1197 * one character free when we've read a full line) */
1198 while (len == adata->blen);
1199
1200 /* don't let one large string make cmd->buf hog memory forever */
1201 if ((adata->blen > IMAP_CMD_BUFSIZE) && (len <= IMAP_CMD_BUFSIZE))
1202 {
1203 MUTT_MEM_REALLOC(&adata->buf, IMAP_CMD_BUFSIZE, char);
1204 adata->blen = IMAP_CMD_BUFSIZE;
1205 mutt_debug(LL_DEBUG3, "shrank buffer to %zu bytes\n", adata->blen);
1206 }
1207
1208 adata->lastread = mutt_date_now();
1209
1210 /* handle untagged messages. The caller still gets its shot afterwards. */
1211 if ((mutt_str_startswith(adata->buf, "* ") ||
1212 mutt_str_startswith(imap_next_word(adata->buf), "OK [")) &&
1213 cmd_handle_untagged(adata))
1214 {
1215 return IMAP_RES_BAD;
1216 }
1217
1218 /* server demands a continuation response from us */
1219 if (adata->buf[0] == '+')
1220 return IMAP_RES_RESPOND;
1221
1222 /* Look for tagged command completions.
1223 *
1224 * Some response handlers can end up recursively calling
1225 * imap_cmd_step() and end up handling all tagged command
1226 * completions.
1227 * (e.g. FETCH->set_flag->set_header_color->~h pattern match.)
1228 *
1229 * Other callers don't even create an adata->cmds entry.
1230 *
1231 * For both these cases, we default to returning OK */
1232 rc = IMAP_RES_OK;
1233 c = adata->lastcmd;
1234 do
1235 {
1236 cmd = &adata->cmds[c];
1237 if (cmd->state == IMAP_RES_NEW)
1238 {
1239 if (mutt_str_startswith(adata->buf, cmd->seq))
1240 {
1241 if (!stillrunning)
1242 {
1243 /* first command in queue has finished - move queue pointer up */
1244 adata->lastcmd = (adata->lastcmd + 1) % adata->cmdslots;
1245 }
1246 cmd->state = cmd_status(adata->buf);
1247 rc = cmd->state;
1248 if (cmd->state == IMAP_RES_NO || cmd->state == IMAP_RES_BAD)
1249 {
1250 mutt_message(_("IMAP command failed: %s"), adata->buf);
1251 }
1252 }
1253 else
1254 {
1255 stillrunning++;
1256 }
1257 }
1258
1259 c = (c + 1) % adata->cmdslots;
1260 } while (c != adata->nextcmd);
1261
1262 if (stillrunning)
1263 {
1264 rc = IMAP_RES_CONTINUE;
1265 }
1266 else
1267 {
1268 mutt_debug(LL_DEBUG3, "IMAP queue drained\n");
1269 imap_cmd_finish(adata);
1270 }
1271
1272 return rc;
1273}

1329int imap_exec(struct ImapAccountData *adata, const char *cmdstr, ImapCmdFlags flags)
1330{
1331 if (!adata)
1332 return IMAP_EXEC_ERROR;
1333
1334 if (flags & IMAP_CMD_SINGLE)
1335 {
1336 // Process any existing commands
1337 if (adata->nextcmd != adata->lastcmd)
1338 imap_exec(adata, NULL, IMAP_CMD_POLL);
1339 }
1340
1341 int rc = cmd_start(adata, cmdstr, flags);
1342 if (rc < 0)
1343 {
1344 cmd_handle_fatal(adata);
1345 return IMAP_EXEC_FATAL;
1346 }
1347
1348 if (flags & IMAP_CMD_QUEUE)
1349 return IMAP_EXEC_SUCCESS;
1350
1351 const short c_imap_poll_timeout = cs_subset_number(NeoMutt->sub, "imap_poll_timeout");
1352 if ((flags & IMAP_CMD_POLL) && (c_imap_poll_timeout > 0) &&

1354 {
1355 mutt_error(_("Connection to %s timed out"), adata->conn->account.host);
1356 cmd_handle_fatal(adata);
1357 return IMAP_EXEC_FATAL;
1358 }
1359
1360 /* Allow interruptions, particularly useful if there are network problems. */
1361 mutt_sig_allow_interrupt(true);
1362 do
1363 {
1364 rc = imap_cmd_step(adata);
1365 // The queue is empty, so the single command has been processed
1366 if ((flags & IMAP_CMD_SINGLE) && (adata->nextcmd == adata->lastcmd))
1367 break;
1368 } while (rc == IMAP_RES_CONTINUE);
1369 mutt_sig_allow_interrupt(false);
1370
1371 if (rc == IMAP_RES_NO)
1372 return IMAP_EXEC_ERROR;
1373 if (rc != IMAP_RES_OK)
1374 {
1375 if (adata->status != IMAP_FATAL)
1376 return IMAP_EXEC_ERROR;
1377
1378 mutt_debug(LL_DEBUG1, "command failed: %s\n", adata->buf);
1379 return IMAP_EXEC_FATAL;
1380 }
1381
1382 return IMAP_EXEC_SUCCESS;

1395void imap_cmd_finish(struct ImapAccountData *adata)
1396{
1397 if (!adata)
1398 return;
1399
1400 if (adata->status == IMAP_FATAL)
1401 {
1402 adata->closing = false;
1403 cmd_handle_fatal(adata);
1404 return;
1405 }
1406
1407 if (!(adata->state >= IMAP_SELECTED) || (adata->mailbox && adata->closing))
1408 {
1409 adata->closing = false;
1410 return;
1411 }
1412
1413 adata->closing = false;
1414
1415 struct ImapMboxData *mdata = imap_mdata_get(adata->mailbox);
1416
1417 if (mdata && mdata->reopen & IMAP_REOPEN_ALLOW)
1418 {
1419 // First remove expunged emails from the msn_index
1420 if (mdata->reopen & IMAP_EXPUNGE_PENDING)
1421 {
1422 mutt_debug(LL_DEBUG2, "Expunging mailbox\n");
1423 imap_expunge_mailbox(adata->mailbox, true);
1424 /* Detect whether we've gotten unexpected EXPUNGE messages */
1425 if (!(mdata->reopen & IMAP_EXPUNGE_EXPECTED))
1426 mdata->check_status |= IMAP_EXPUNGE_PENDING;
1427 mdata->reopen &= ~(IMAP_EXPUNGE_PENDING | IMAP_EXPUNGE_EXPECTED);
1428 }
1429
1430 // Then add new emails to it
1431 if (mdata->reopen & IMAP_NEWMAIL_PENDING)
1432 {
1433 const size_t max_msn = imap_msn_highest(&mdata->msn);
1434 if (mdata->new_mail_count > max_msn)
1435 {
1436 if (!(mdata->reopen & IMAP_EXPUNGE_PENDING))
1437 mdata->check_status |= IMAP_NEWMAIL_PENDING;
1438
1439 mutt_debug(LL_DEBUG2, "Fetching new mails from %zd to %u\n",
1440 max_msn + 1, mdata->new_mail_count);
1441 imap_read_headers(adata->mailbox, max_msn + 1, mdata->new_mail_count, false);
1442 }
1443 }
1444
1445 // And to finish inform about MUTT_REOPEN if needed
1446 if (mdata->reopen & IMAP_EXPUNGE_PENDING && !(mdata->reopen & IMAP_EXPUNGE_EXPECTED))
1447 mdata->check_status |= IMAP_EXPUNGE_PENDING;
1448
1449 if (mdata->reopen & IMAP_EXPUNGE_PENDING)
1450 mdata->reopen &= ~(IMAP_EXPUNGE_PENDING | IMAP_EXPUNGE_EXPECTED);
1451 }
1452
1453 adata->status = 0;
1454}
