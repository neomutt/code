120static bool count_body_parts_check(struct ListHead *checklist, struct Body *b, bool dflt)
121{
122 /* If list is null, use default behavior. */
123 if (!checklist || STAILQ_EMPTY(checklist))
124 {
125 return false;
126 }
127
128 struct AttachMatch *a = NULL;
129 struct ListNode *np = NULL;
130 STAILQ_FOREACH(np, checklist, entries)
131 {
132 a = (struct AttachMatch *) np->data;
133 mutt_debug(LL_DEBUG3, "%s %d/%s ?? %s/%s [%d]... ", dflt ? "[OK] " : "[EXCL] ",
134 b->type, b->subtype ? b->subtype : "*", a->major, a->minor, a->major_int);
135 if (((a->major_int == TYPE_ANY) || (a->major_int == b->type)) &&
136 (!b->subtype || (regexec(&a->minor_regex, b->subtype, 0, NULL, 0) == 0)))
137 {
138 mutt_debug(LL_DEBUG3, "yes\n");
139 return true;
140 }
141 else
142 {
143 mutt_debug(LL_DEBUG3, "no\n");
144 }
145 }

155static int count_body_parts(struct Body *b)
156{
157 if (!b)
158 return 0;
159
160 int count = 0;
161
162 for (struct Body *bp = b; bp; bp = bp->next)
163 {
164 /* Initial disposition is to count and not to recurse this part. */
165 bool shallcount = true; /* default */
166 bool shallrecurse = false;
167
168 mutt_debug(LL_DEBUG5, "desc=\"%s\"; fn=\"%s\", type=\"%d/%s\"\n",
169 bp->description ? bp->description : ("none"),
170 bp->filename ? bp->filename :
171 bp->d_filename ? bp->d_filename :
172 "(none)",
173 bp->type, bp->subtype ? bp->subtype : "*");
174
175 if (bp->type == TYPE_MESSAGE)
176 {
177 shallrecurse = true;
178
179 /* If it's an external body pointer, don't recurse it. */
180 if (mutt_istr_equal(bp->subtype, "external-body"))
181 shallrecurse = false;
182 }
183 else if (bp->type == TYPE_MULTIPART)
184 {
185 /* Always recurse multiparts, except multipart/alternative. */
186 shallrecurse = true;
187 if (mutt_istr_equal(bp->subtype, "alternative"))
188 {
189 const bool c_count_alternatives = cs_subset_bool(NeoMutt->sub, "count_alternatives");
190 shallrecurse = c_count_alternatives;
191 }
192 }
193
194 if ((bp->disposition == DISP_INLINE) && (bp->type != TYPE_MULTIPART) &&
195 (bp->type != TYPE_MESSAGE) && (bp == b))
196 {
197 shallcount = false; /* ignore fundamental inlines */
198 }
199
200 /* If this body isn't scheduled for enumeration already, don't bother
201 * profiling it further. */
202 if (shallcount)
203 {
204 /* Turn off shallcount if message type is not in ok list,
205 * or if it is in except list. Check is done separately for
206 * inlines vs. attachments. */
207
208 if (bp->disposition == DISP_ATTACH)
209 {
210 if (!count_body_parts_check(&AttachAllow, bp, true))
211 shallcount = false; /* attach not allowed */
212 if (count_body_parts_check(&AttachExclude, bp, false))
213 shallcount = false; /* attach excluded */
214 }
215 else
216 {
217 if (!count_body_parts_check(&InlineAllow, bp, true))
218 shallcount = false; /* inline not allowed */
219 if (count_body_parts_check(&InlineExclude, bp, false))
220 shallcount = false; /* excluded */
221 }
222 }
223
224 if (shallcount)
225 count++;
226 bp->attach_qualifies = shallcount;
227
228 mutt_debug(LL_DEBUG3, "%p shallcount = %d\n", (void *) bp, shallcount);
229
230 if (shallrecurse)
231 {
232 mutt_debug(LL_DEBUG3, "%p pre count = %d\n", (void *) bp, count);
233 bp->attach_count = count_body_parts(bp->parts);
234 count += bp->attach_count;
235 mutt_debug(LL_DEBUG3, "%p post count = %d\n", (void *) bp, count);
236 }
237 }
238
239 mutt_debug(LL_DEBUG3, "return %d\n", (count < 0) ? 0 : count);
240 return (count < 0) ? 0 : count;
241}

249int mutt_count_body_parts(struct Email *e, FILE *fp)
250{
251 if (!e)
252 return 0;
253
254 bool keep_parts = false;
255
256 if (e->attach_valid)
257 return e->attach_total;
258
259 if (e->body->parts)
260 keep_parts = true;
261 else
262 mutt_parse_mime_message(e, fp);
263
264 if (!STAILQ_EMPTY(&AttachAllow) || !STAILQ_EMPTY(&AttachExclude) ||
265 !STAILQ_EMPTY(&InlineAllow) || !STAILQ_EMPTY(&InlineExclude))
266 {
267 e->attach_total = count_body_parts(e->body);
268 }
269 else
270 {
271 e->attach_total = 0;
272 }
273
274 e->attach_valid = true;
275
276 if (!keep_parts)
277 mutt_body_free(&e->body->parts);
278
279 return e->attach_total;
280}

311static enum CommandResult parse_attach_list(const struct Command *cmd, struct Buffer *line,
312 struct ListHead *head, struct Buffer *err)
313{
314 struct AttachMatch *a = NULL;
315 char *p = NULL;
316 char *tmpminor = NULL;
317 size_t len;
318 struct Buffer *token = buf_pool_get();
319 enum CommandResult rc = MUTT_CMD_ERROR;
320
321 do
322 {
323 parse_extract_token(token, line, TOKEN_NO_FLAGS);
324
325 if (buf_is_empty(token))
326 continue;
327
328 a = attachmatch_new();
329
330 /* some cheap hacks that I expect to remove */
331 if (mutt_istr_equal(token->data, "any"))
332 a->major = mutt_str_dup("*/.*");
333 else if (mutt_istr_equal(token->data, "none"))
334 a->major = mutt_str_dup("cheap_hack/this_should_never_match");
335 else
336 a->major = buf_strdup(token);
337
338 p = strchr(a->major, '/');
339 if (p)
340 {
341 *p = '\0';
342 p++;
343 a->minor = p;
344 }
345 else
346 {
347 a->minor = "unknown";
348 }
349
350 len = strlen(a->minor);
351 tmpminor = MUTT_MEM_MALLOC(len + 3, char);
352 strcpy(&tmpminor[1], a->minor);
353 tmpminor[0] = '^';
354 tmpminor[len + 1] = '$';
355 tmpminor[len + 2] = '\0';
356
357 a->major_int = mutt_check_mime_type(a->major);
358 int rc_regex = REG_COMP(&a->minor_regex, tmpminor, REG_ICASE);
359
360 FREE(&tmpminor);
361
362 if (rc_regex != 0)
363 {
364 regerror(rc_regex, &a->minor_regex, err->data, err->dsize);
365 FREE(&a->major);
366 FREE(&a);
367 goto done;
368 }
369
370 mutt_debug(LL_DEBUG3, "added %s/%s [%d]\n", a->major, a->minor, a->major_int);
371
372 mutt_list_insert_tail(head, (char *) a);
373 } while (MoreArgs(line));
374
375 if (!a)
376 goto done;
377
378 mutt_debug(LL_NOTIFY, "NT_ATTACH_ADD: %s/%s\n", a->major, a->minor);
379 notify_send(AttachmentsNotify, NT_ATTACH, NT_ATTACH_ADD, NULL);
380
381 rc = MUTT_CMD_SUCCESS;
382
383done:
384 buf_pool_release(&token);
385 return rc;
386}

442 STAILQ_REMOVE(head, np, ListNode, entries);
443 FREE(&np->data);
444 FREE(&np);
445 }
446 }
447
448 } while (MoreArgs(line));
449
450 FREE(&tmp);
451
452 notify_send(AttachmentsNotify, NT_ATTACH, NT_ATTACH_DELETE, NULL);
