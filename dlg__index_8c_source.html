165void collapse_all(struct MailboxView *mv, struct Menu *menu, int toggle)
166{
167 if (!mv || !mv->mailbox || (mv->mailbox->msg_count == 0) || !menu)
168 return;
169
170 struct Email *e_cur = mutt_get_virt_email(mv->mailbox, menu_get_index(menu));
171 if (!e_cur)
172 return;
173
174 int final;
175
176 /* Figure out what the current message would be after folding / unfolding,
177 * so that we can restore the cursor in a sane way afterwards. */
178 if (e_cur->collapsed && toggle)
179 final = mutt_uncollapse_thread(e_cur);
180 else if (mutt_thread_can_collapse(e_cur))
181 final = mutt_collapse_thread(e_cur);
182 else
183 final = e_cur->vnum;
184
185 if (final == -1)
186 return;
187
188 struct Email *base = mutt_get_virt_email(mv->mailbox, final);
189 if (!base)
190 return;
191
192 /* Iterate all threads, perform collapse/uncollapse as needed */
193 mv->collapsed = toggle ? !mv->collapsed : true;
194 mutt_thread_collapse(mv->threads, mv->collapsed);
195
196 /* Restore the cursor */
197 mutt_set_vnum(mv->mailbox);
198 menu->max = mv->mailbox->vcount;
199 for (int i = 0; i < mv->mailbox->vcount; i++)
200 {
201 struct Email *e = mutt_get_virt_email(mv->mailbox, i);
202 if (!e)
203 break;
204 if (e->index == base->index)
205 {
206 menu_set_index(menu, i);
207 break;
208 }
209 }
210
211 menu_queue_redraw(menu, MENU_REDRAW_INDEX);
212}

309int find_first_message(struct MailboxView *mv)
310{
311 if (!mv)
312 return 0;
313
314 struct Mailbox *m = mv->mailbox;
315 if (!m || (m->msg_count == 0))
316 return 0;
317
318 int old = -1;
319 for (int i = 0; i < m->vcount; i++)
320 {
321 struct Email *e = mutt_get_virt_email(m, i);
322 if (!e)
323 continue;
324 if (!e->read && !e->deleted)
325 {
326 if (!e->old)
327 return i;
328 if (old == -1)
329 old = i;
330 }
331 }
332 if (old != -1)
333 return old;
334
335 /* If `$use_threads` is not threaded and `$sort` is reverse, the latest
336 * message is first. Otherwise, the latest message is first if exactly
337 * one of `$use_threads` and `$sort` are reverse.
338 */
339 enum EmailSortType c_sort = cs_subset_sort(m->sub, "sort");
340 if ((c_sort & SORT_MASK) == EMAIL_SORT_THREADS)
341 c_sort = cs_subset_sort(m->sub, "sort_aux");
342 bool reverse = false;
343 switch (mutt_thread_style())
344 {
345 case UT_FLAT:
346 reverse = c_sort & SORT_REVERSE;
347 break;
348 case UT_THREADS:
349 reverse = c_sort & SORT_REVERSE;
350 break;
351 case UT_REVERSE:
352 reverse = !(c_sort & SORT_REVERSE);
353 break;
354 default:
355 ASSERT(false);
356 }
357
358 if (reverse || (m->vcount == 0))
359 return 0;
360
361 return m->vcount - 1;
362}

411static void update_index_threaded(struct MailboxView *mv, enum MxStatus check, int oldcount)
412{
413 struct Email **save_new = NULL;
414 const bool lmt = mview_has_limit(mv);
415
416 struct Mailbox *m = mv->mailbox;
417 int num_new = MAX(0, m->msg_count - oldcount);
418
419 const bool c_uncollapse_new = cs_subset_bool(m->sub, "uncollapse_new");
420 /* save the list of new messages */
421 if ((check != MX_STATUS_REOPENED) && (oldcount > 0) &&
422 (lmt || c_uncollapse_new) && (num_new > 0))
423 {
424 save_new = MUTT_MEM_MALLOC(num_new, struct Email *);
425 for (int i = oldcount; i < m->msg_count; i++)
426 save_new[i - oldcount] = m->emails[i];
427 }
428
429 /* Sort first to thread the new messages, because some patterns
430 * require the threading information.
431 *
432 * If the mailbox was reopened, need to rethread from scratch. */
433 mutt_sort_headers(mv, (check == MX_STATUS_REOPENED));
434
435 if (lmt)
436 {
437 for (int i = 0; i < m->msg_count; i++)
438 {
439 struct Email *e = m->emails[i];
440
441 if ((e->limit_visited && e->visible) ||
442 mutt_pattern_exec(SLIST_FIRST(mv->limit_pattern),
443 MUTT_MATCH_FULL_ADDRESS, m, e, NULL))
444 {
445 /* vnum will get properly set by mutt_set_vnum(), which
446 * is called by mutt_sort_headers() just below. */
447 e->vnum = 1;
448 e->visible = true;
449 }
450 else
451 {
452 e->vnum = -1;
453 e->visible = false;
454 }
455
456 // mark email as visited so we don't re-apply the pattern next time
457 e->limit_visited = true;
458 }
459 /* Need a second sort to set virtual numbers and redraw the tree */
460 mutt_sort_headers(mv, false);
461 }
462
463 /* uncollapse threads with new mail */
464 if (c_uncollapse_new)
465 {
466 if (check == MX_STATUS_REOPENED)
467 {
468 mv->collapsed = false;
469 mutt_thread_collapse(mv->threads, mv->collapsed);
470 mutt_set_vnum(m);
471 }
472 else if (oldcount > 0)
473 {
474 for (int j = 0; j < num_new; j++)
475 {
476 if (save_new[j]->visible)
477 {
478 mutt_uncollapse_thread(save_new[j]);
479 }
480 }
481 mutt_set_vnum(m);
482 }
483 }
484
485 FREE(&save_new);
486}

493static void update_index_unthreaded(struct MailboxView *mv, enum MxStatus check)
494{
495 /* We are in a limited view. Check if the new message(s) satisfy
496 * the limit criteria. If they do, set their virtual msgno so that
497 * they will be visible in the limited view */
498 if (mview_has_limit(mv))
499 {
500 int padding = mx_msg_padding_size(mv->mailbox);
501 mv->mailbox->vcount = mv->vsize = 0;
502 for (int i = 0; i < mv->mailbox->msg_count; i++)
503 {
504 struct Email *e = mv->mailbox->emails[i];
505 if (!e)
506 break;
507
508 if ((e->limit_visited && e->visible) ||
509 mutt_pattern_exec(SLIST_FIRST(mv->limit_pattern),
510 MUTT_MATCH_FULL_ADDRESS, mv->mailbox, e, NULL))
511 {
512 ASSERT(mv->mailbox->vcount < mv->mailbox->msg_count);
513 e->vnum = mv->mailbox->vcount;
514 mv->mailbox->v2r[mv->mailbox->vcount] = i;
515 e->visible = true;
516 mv->mailbox->vcount++;
517 struct Body *b = e->body;
518 mv->vsize += b->length + b->offset - b->hdr_offset + padding;
519 }
520 else
521 {
522 e->visible = false;
523 }
524
525 // mark email as visited so we don't re-apply the pattern next time
526 e->limit_visited = true;
527 }
528 }
529
530 /* if the mailbox was reopened, need to rethread from scratch */
531 mutt_sort_headers(mv, (check == MX_STATUS_REOPENED));
532}

542void update_index(struct Menu *menu, struct MailboxView *mv, enum MxStatus check,
543 int oldcount, const struct IndexSharedData *shared)
544{

546 return;
547
548 struct Mailbox *m = mv->mailbox;
549 if (mutt_using_threads())
550 update_index_threaded(mv, check, oldcount);
551 else
552 update_index_unthreaded(mv, check);
553
554 menu->max = m->vcount;
555 const int old_index = menu_get_index(menu);
556 int index = -1;
557 if (oldcount)
558 {
559 /* restore the current message to the message it was pointing to */

570 }
571 }
572

574 {
575 index = (old_index < m->vcount) ? old_index : find_first_message(mv);
576 }
577 menu_set_index(menu, index);

837 if (!m || (line < 0) || (line >= m->email_max))
838 return 0;
839

841 if (!e)
842 return 0;

845 struct MuttThread *tmp = NULL;
846
847 const enum UseThreads c_threads = mutt_thread_style();
848 if ((c_threads > UT_FLAT) && e->tree && e->thread)
849 {
850 flags |= MUTT_FORMAT_TREE; /* display the thread tree */
851 if (e->display_subject)
852 {
853 flags |= MUTT_FORMAT_FORCESUBJ;
854 }
855 else
856 {
857 const bool reverse = c_threads == UT_REVERSE;
858 int edgemsgno;
859 if (reverse)
860 {

862 edgemsgno = m->v2r[menu->max - 1];
863 else
864 edgemsgno = m->v2r[menu->top + menu->page_len - 1];
865 }
866 else
867 {
868 edgemsgno = m->v2r[menu->top];
869 }
870
871 for (tmp = e->thread->parent; tmp; tmp = tmp->parent)
872 {
873 if (!tmp->message)
874 continue;
875
876 /* if no ancestor is visible on current screen, provisionally force
877 * subject... */
878 if (reverse ? (tmp->message->msgno > edgemsgno) : (tmp->message->msgno < edgemsgno))
879 {
880 flags |= MUTT_FORMAT_FORCESUBJ;
881 break;
882 }
883 else if (tmp->message->vnum >= 0)
884 {
885 break;
886 }
887 }
888 if (flags & MUTT_FORMAT_FORCESUBJ)
889 {
890 for (tmp = e->thread->prev; tmp; tmp = tmp->prev)
891 {
892 if (!tmp->message)
893 continue;
894
895 /* ...but if a previous sibling is available, don't force it */
896 if (reverse ? (tmp->message->msgno > edgemsgno) : (tmp->message->msgno < edgemsgno))
897 {
898 break;
899 }
900 else if (tmp->message->vnum >= 0)
901 {
902 flags &= ~MUTT_FORMAT_FORCESUBJ;
903 break;
904 }
905 }
906 }
907 }
908 }

959void mutt_draw_statusline(struct MuttWindow *win, int max_cols, const char *buf, size_t buflen)
960{
961 if (!buf || !stdscr)
962 return;
963
964 size_t i = 0;
965 size_t offset = 0;
966 bool found = false;
967 size_t chunks = 0;
968 size_t len = 0;
969
973 struct StatusSyntax
974 {
975 const struct AttrColor *attr_color;
976 int first;
977 int last;
978 } *syntax = NULL;
979
980 const struct AttrColor *ac_base = merged_color_overlay(simple_color_get(MT_COLOR_NORMAL),
981 simple_color_get(MT_COLOR_STATUS));
982 do
983 {
984 struct RegexColor *cl = NULL;
985 found = false;
986
987 if (!buf[offset])
988 break;
989
990 /* loop through each "color status regex" */
991 STAILQ_FOREACH(cl, regex_colors_get_list(MT_COLOR_STATUS), entries)
992 {
993 regmatch_t pmatch[cl->match + 1];
994 memset(pmatch, 0, (cl->match + 1) * sizeof(regmatch_t));
995
996 if (regexec(&cl->regex, buf + offset, cl->match + 1, pmatch, 0) != 0)
997 continue; /* regex doesn't match the status bar */
998
999 int first = pmatch[cl->match].rm_so + offset;
1000 int last = pmatch[cl->match].rm_eo + offset;
1001
1002 if (first == last)
1003 continue; /* ignore an empty regex */
1004
1005 if (!found)
1006 {
1007 chunks++;
1008 MUTT_MEM_REALLOC(&syntax, chunks, struct StatusSyntax);
1009 }
1010
1011 i = chunks - 1;
1012 if (!found || (first < syntax[i].first) ||
1013 ((first == syntax[i].first) && (last > syntax[i].last)))
1014 {
1015 const struct AttrColor *ac_merge = merged_color_overlay(ac_base, &cl->attr_color);
1016
1017 syntax[i].attr_color = ac_merge;
1018 syntax[i].first = first;
1019 syntax[i].last = last;
1020 }
1021 found = true;
1022 }
1023
1024 if (syntax)
1025 {
1026 offset = syntax[i].last;
1027 }
1028 } while (found);
1029
1030 /* Only 'len' bytes will fit into 'max_cols' screen columns */
1031 len = mutt_wstr_trunc(buf, buflen, max_cols, NULL);
1032
1033 offset = 0;
1034
1035 if ((chunks > 0) && (syntax[0].first > 0))
1036 {
1037 /* Text before the first highlight */
1038 mutt_window_addnstr(win, buf, MIN(len, syntax[0].first));
1039 mutt_curses_set_color(ac_base);
1040 if (len <= syntax[0].first)
1041 goto dsl_finish; /* no more room */
1042
1043 offset = syntax[0].first;
1044 }
1045
1046 for (i = 0; i < chunks; i++)
1047 {
1048 /* Highlighted text */
1049 mutt_curses_set_color(syntax[i].attr_color);
1050 mutt_window_addnstr(win, buf + offset, MIN(len, syntax[i].last) - offset);
1051 if (len <= syntax[i].last)
1052 goto dsl_finish; /* no more room */
1053
1054 size_t next;
1055 if ((i + 1) == chunks)
1056 {
1057 next = len;
1058 }
1059 else
1060 {
1061 next = MIN(len, syntax[i + 1].first);
1062 }
1063
1064 mutt_curses_set_color(ac_base);
1065 offset = syntax[i].last;
1066 mutt_window_addnstr(win, buf + offset, next - offset);
1067
1068 offset = next;
1069 if (offset >= len)
1070 goto dsl_finish; /* no more room */
1071 }
1072
1073 mutt_curses_set_color(ac_base);
1074 if (offset < len)
1075 {
1076 /* Text after the last highlight */
1077 mutt_window_addnstr(win, buf + offset, len - offset);
1078 }
1079
1080 int width = mutt_strwidth(buf);
1081 if (width < max_cols)
1082 {
1083 /* Pad the rest of the line with whitespace */
1084 mutt_paddstr(win, max_cols - width, "");
1085 }
1086dsl_finish:
1087 FREE(&syntax);
1088}
