135static int buf_vaprintf(struct Buffer *buf, const char *fmt, va_list ap)
136{
137 if (!buf || !fmt)
138 return 0; /* LCOV_EXCL_LINE */
139
140 buf_alloc(buf, 128);
141
142 size_t doff = buf->dptr - buf->data;
143 size_t blen = buf->dsize - doff;
144
145 va_list ap_retry;
146 va_copy(ap_retry, ap);
147
148 int len = vsnprintf(buf->dptr, blen, fmt, ap);
149 if (len >= (int) blen)
150 {
151 buf_alloc(buf, buf->dsize + len - blen + 1);
152 len = vsnprintf(buf->dptr, len + 1, fmt, ap_retry);
153 }
154 if (len > 0)
155 buf->dptr += len;
156
157 va_end(ap_retry);
158
159 return len;
160}

265size_t buf_insert(struct Buffer *buf, size_t offset, const char *s)
266{
267 if (!buf || !s || (*s == '\0'))
268 {
269 return -1;
270 }
271
272 const size_t slen = mutt_str_len(s);
273 const size_t curlen = buf_len(buf);
274 buf_alloc(buf, curlen + slen + 1);
275
276 if (offset > curlen)
277 {
278 for (size_t i = curlen; i < offset; i++)
279 {
280 buf_addch(buf, ' ');
281 }
282 buf_addstr(buf, s);
283 }
284 else
285 {
286 memmove(buf->data + offset + slen, buf->data + offset, curlen - offset);
287 memcpy(buf->data + offset, s, slen);
288 buf->data[curlen + slen] = '\0';
289 buf->dptr = buf->data + curlen + slen;
290 }
291
292 return buf_len(buf) - curlen;
293}

344void buf_alloc(struct Buffer *buf, size_t new_size)
345{
346 if (!buf)
347 return;
348
349 if (buf->data && (new_size <= buf->dsize))
350 {
351 // Extra sanity-checking
352 if (!buf->dptr || (buf->dptr < buf->data) || (buf->dptr > (buf->data + buf->dsize)))
353 buf->dptr = buf->data; // LCOV_EXCL_LINE
354 return;
355 }
356
357 const size_t dsize = buf_step_size(new_size + 1);
358 if (dsize <= new_size)
359 {
360 // LCOV_EXCL_START
361 mutt_error(_("Out of memory"));
362 mutt_exit(1);
363 // LCOV_EXCL_STOP
364 }
365
366 const bool was_empty = (buf->dptr == NULL);
367 const size_t offset = (buf->dptr && buf->data) ? (buf->dptr - buf->data) : 0;
368
369 buf->dsize = dsize;
370
371 MUTT_MEM_REALLOC(&buf->data, buf->dsize, char);
372 buf->dptr = buf->data + offset;
373
374 // Ensures that initially NULL buf->data is properly terminated
375 if (was_empty)
376 {
377 *buf->dptr = '\0';
378 }
379}

767/*
768 * buf_inline_replace - Replace part of a string
769 * @param buf Buffer to modify
770 * @param pos Starting position of string to overwrite
771 * @param len Length of string to overwrite
772 * @param str Replacement string
773 *
774 * String (`11XXXOOOOOO`, 2, 3, `YYYY`) becomes `11YYYY000000`
775 */
776void buf_inline_replace(struct Buffer *buf, size_t pos, size_t len, const char *str)
777{
778 if (!buf || !buf->data || !str)
779 return;
780
781 size_t slen = buf_len(buf);
782 if (pos > slen)
783 return;
784 if (len > slen - pos)
785 len = slen - pos;
786
787 size_t rlen = mutt_str_len(str);
788 size_t new_size = slen - len + rlen + 1;
789 if (new_size > buf->dsize)
790 buf_alloc(buf, new_size);
791
792 memmove(buf->data + pos + rlen, buf->data + pos + len, slen - pos - len + 1);
793 memcpy(buf->data + pos, str, rlen);
794
795 buf_fix_dptr(buf);
796}

807const char *buf_rfind(const struct Buffer *buf, const char *str)
808{
809 if (buf_is_empty(buf) || !str || (*str == '\0'))
810 return NULL;
811
812 size_t len = strlen(str);
813 size_t slen = buf_len(buf);
814 if (len > slen)
815 return NULL;
816
817 const char *end = buf->data + slen - len;
818
819 for (const char *p = end; p >= buf->data; p--)
820 {
821 for (size_t i = 0; i < len; i++)
822 {
823 if (p[i] != str[i])
824 goto next;
825 }
826 return p;
827
828 next:;
829 }
830 return NULL;
