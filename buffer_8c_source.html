111size_t buf_addstr_n(struct Buffer *buf, const char *s, size_t len)
112{
113 if (!buf || !s)
114 return 0;
115
116 if (len > (SIZE_MAX - BufferStepSize))
117 {
118 // LCOV_EXCL_START
119 mutt_error(_("Out of memory"));
120 mutt_exit(1);
121 // LCOV_EXCL_STOP
122 }
123
124 if (!buf->data || !buf->dptr || ((buf->dptr + len + 1) > (buf->data + buf->dsize)))
125 buf_alloc(buf, buf->dsize + MAX(BufferStepSize, len + 1));
126
127 memcpy(buf->dptr, s, len);
128 buf->dptr += len;
129 *(buf->dptr) = '\0';

141static int buf_vaprintf(struct Buffer *buf, const char *fmt, va_list ap)
142{
143 if (!buf || !fmt)
144 return 0; /* LCOV_EXCL_LINE */
145
146 buf_alloc(buf, 128);
147
148 int doff = buf->dptr - buf->data;
149 int blen = buf->dsize - doff;
150
151 va_list ap_retry;
152 va_copy(ap_retry, ap);
153
154 int len = vsnprintf(buf->dptr, blen, fmt, ap);
155 if (len >= blen)
156 {
157 buf_alloc(buf, buf->dsize + len - blen + 1);
158 len = vsnprintf(buf->dptr, len + 1, fmt, ap_retry);
159 }
160 if (len > 0)
161 buf->dptr += len;
162
163 va_end(ap_retry);
164
165 return len;
166}

271size_t buf_insert(struct Buffer *buf, size_t offset, const char *s)
272{
273 if (!buf || !s || (*s == '\0'))
274 {
275 return -1;
276 }
277
278 const size_t slen = mutt_str_len(s);
279 const size_t curlen = buf_len(buf);
280 buf_alloc(buf, curlen + slen + 1);
281
282 if (offset > curlen)
283 {
284 for (size_t i = curlen; i < offset; ++i)
285 {
286 buf_addch(buf, ' ');
287 }
288 buf_addstr(buf, s);
289 }
290 else
291 {
292 memmove(buf->data + offset + slen, buf->data + offset, curlen - offset);
293 memcpy(buf->data + offset, s, slen);
294 buf->data[curlen + slen] = '\0';
295 buf->dptr = buf->data + curlen + slen;
296 }
297
298 return buf_len(buf) - curlen;
299}

352void buf_alloc(struct Buffer *buf, size_t new_size)
353{
354 if (!buf)
355 return;
356
357 if (buf->data && (new_size <= buf->dsize))
358 {
359 // Extra sanity-checking
360 if (!buf->dptr || (buf->dptr < buf->data) || (buf->dptr > (buf->data + buf->dsize)))
361 buf->dptr = buf->data; // LCOV_EXCL_LINE
362 return;
363 }
364
365 if (new_size > (SIZE_MAX - BufferStepSize))
366 {
367 // LCOV_EXCL_START
368 mutt_error(_("Out of memory"));
369 mutt_exit(1);
370 // LCOV_EXCL_STOP
371 }
372
373 const bool was_empty = (buf->dptr == NULL);
374 const size_t offset = (buf->dptr && buf->data) ? (buf->dptr - buf->data) : 0;
375
376 buf->dsize = ROUND_UP(new_size + 1, BufferStepSize);
377
378 mutt_mem_realloc(&buf->data, buf->dsize);
379 buf->dptr = buf->data + offset;
380
381 // Ensures that initially NULL buf->data is properly terminated
382 if (was_empty)
383 {
384 *buf->dptr = '\0';
385 }
386}
