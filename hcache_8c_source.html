252 static void hcache_per_folder(struct Buffer *hcpath, const char *path,
253  const char *folder, hcache_namer_t namer)
254 {
255  struct stat sb;
256 
257  int plen = mutt_str_len(path);
258  int rc = stat(path, &sb);
259  bool slash = (path[plen - 1] == '/');
260 
261  if (((rc == 0) && !S_ISDIR(sb.st_mode)) || ((rc == -1) && !slash))
262  {
263  /* An existing file or a non-existing path not ending with a slash */
264  mutt_encode_path(hcpath, path);
265  return;
266  }
267 
268  /* We have a directory - no matter whether it exists, or not */
269  struct Buffer *hcfile = mutt_buffer_pool_get();
270  if (namer)
271  {
272  namer(folder, hcfile);
273  mutt_buffer_concat_path(hcpath, path, mutt_b2s(hcfile));
274  }
275  else
276  {
277  unsigned char m[16]; /* binary md5sum */
278  struct Buffer *name = mutt_buffer_pool_get();
279 #ifdef USE_HCACHE_COMPRESSION
280  const char *cm = C_HeaderCacheCompressMethod;
281  mutt_buffer_printf(name, "%s|%s%s", hcache_get_ops()->name, folder, cm ? cm : "");
282 #else
283  mutt_buffer_printf(name, "%s|%s", hcache_get_ops()->name, folder);
284 #endif
285  mutt_md5(mutt_b2s(name), m);
286  mutt_buffer_reset(name);
287  mutt_md5_toascii(m, name->data);
288  mutt_buffer_printf(hcpath, "%s%s%s", path, slash ? "" : "/", mutt_b2s(name));
289  mutt_buffer_pool_release(&name);
290  }
291 
292  mutt_encode_path(hcpath, mutt_b2s(hcpath));
293  create_hcache_dir(mutt_b2s(hcpath));
294  mutt_buffer_pool_release(&hcfile);

319  if (!ops)
320  return NULL;
321

323 
324  /* Calculate the current hcache version from dynamic configuration */
325  if (hcachever == 0x0)
326  {
327  union
328  {
329  unsigned char charval[16];
330  unsigned int intval;
331  } digest;

333 
334  hcachever = HCACHEVER;
335 
336  mutt_md5_init_ctx(&md5ctx);
337 
338  /* Seed with the compiled-in header structure hash */
339  mutt_md5_process_bytes(&hcachever, sizeof(hcachever), &md5ctx);
340

343  STAILQ_FOREACH(sp, &SpamList, entries)
344  {
345  mutt_md5_process(sp->regex->pattern, &md5ctx);
346  mutt_md5_process(sp->templ, &md5ctx);
347  }
348 
349  /* Mix in user's nospam list */
350  struct RegexNode *np = NULL;
351  STAILQ_FOREACH(np, &NoSpamList, entries)
352  {
353  mutt_md5_process(np->regex->pattern, &md5ctx);
354  }
355 
356  /* Get a hash and take its bytes as an (unsigned int) hash version */
357  mutt_md5_finish_ctx(&md5ctx, digest.charval);
358  hcachever = digest.intval;
359  }

361 #ifdef USE_HCACHE_COMPRESSION
362  if (C_HeaderCacheCompressMethod)
363  {
364  const struct ComprOps *cops = compr_get_ops();
365 
366  hc->cctx = cops->open(C_HeaderCacheCompressLevel);
367  if (!hc->cctx)
368  {
369  FREE(&hc);
370  return NULL;
371  }
372 
373  /* remember the buffer of database backend */
374  mutt_debug(LL_DEBUG3, "Header cache will use %s compression\n", cops->name);
375  }
376 #endif
377 
378  hc->folder = get_foldername(folder);
379  hc->crc = hcachever;
380 
381  if (!path || (path[0] == '\0'))
382  {
383  FREE(&hc->folder);
384  FREE(&hc);
385  return NULL;
386  }

395  if (unlink(mutt_b2s(hcpath)) == 0)

397  hc->ctx = ops->open(mutt_b2s(hcpath));
398  if (!hc->ctx)
399  {
400  FREE(&hc->folder);
401  FREE(&hc);
402  }
403  }
404  }
405 
406  mutt_buffer_pool_release(&hcpath);
407  return hc;
