97static int utilwin_repaint(struct MuttWindow *win)
98{
99 struct UtilWinData *priv = win->wdata;
100
101 mutt_curses_set_color_by_id(MT_COLOR_NORMAL);
102
103 // Blank any extra rows above the bottom line
104 for (int r = 0; r < (win->state.rows - 1); r++)
105 {
106 mutt_window_move(win, r, 0);
107 mutt_window_clrtoeol(win);
108 }
109
110 int row = win->state.rows - 1;
111 mutt_window_move(win, row, 0);
112
113 if (priv->text)
114 mutt_window_addstr(win, priv->text);
115
116 mutt_window_clrtoeol(win);
117
118 mutt_debug(LL_DEBUG5, "utilwin repaint done\n");
119 return 0;
120}

142static int utilwin_key_observer(struct NotifyCallback *nc)
143{
144 if (nc->event_type != NT_KEY)
145 return 0;
146 if (nc->event_subtype != NT_KEY_PROGRESS)
147 return 0;
148 if (!nc->global_data || !nc->event_data)
149 return -1;
150
151 struct MuttWindow *win = nc->global_data;
152 struct EventKeyProgress *ev_k = nc->event_data;
153
154 // Nothing to display — clear/hide the window
155 if ((ev_k->count == 0) && (ev_k->key_len == 0))
156 {
157 utilwin_set_text(win, NULL);
158 window_redraw(NULL);
159 return 0;
160 }
161
162 struct Buffer *buf = buf_pool_get();
163
164 if (ev_k->count > 0)
165 buf_add_printf(buf, "%d", ev_k->count);
166
167 for (int i = 0; i < ev_k->key_len; i++)
168 {
169 keymap_get_name(ev_k->keys[i], buf);
170 }
171
172 utilwin_set_text(win, buf_string(buf));
173 buf_pool_release(&buf);
174 window_redraw(NULL);
175
176 return 0;
177}

182static int utilwin_window_observer(struct NotifyCallback *nc)
183{
184 if (nc->event_type != NT_WINDOW)
185 return 0;
186 if (!nc->global_data || !nc->event_data)
187 return -1;
188
189 struct MuttWindow *win = nc->global_data;
190 struct EventWindow *ev_w = nc->event_data;
191 if (ev_w->win != win)
192 return 0;
193
194 if (nc->event_subtype == NT_WINDOW_STATE)
195 {
196 win->actions |= WA_REPAINT;
197 mutt_debug(LL_DEBUG5, "window state done, request WA_REPAINT\n");
198 }
199 else if (nc->event_subtype == NT_WINDOW_DELETE)
200 {
201 notify_observer_remove(NeoMutt->notify, utilwin_key_observer, win);
202 notify_observer_remove(win->notify, utilwin_window_observer, win);
203 mutt_debug(LL_DEBUG5, "window delete done\n");
204 }
205
206 return 0;
207}

216{
217 struct MuttWindow *win = mutt_window_new(WT_CUSTOM, MUTT_WIN_ORIENT_VERTICAL,

228

230 if (mod_data)

232
233 notify_observer_add(NeoMutt->notify, NT_KEY, utilwin_key_observer, win);
234 notify_observer_add(win->notify, NT_WINDOW, utilwin_window_observer, win);
235
236 return win;
237}

248void utilwin_set_text(struct MuttWindow *win, const char *text)
249{
250 if (!win)
251 {
252 struct GuiModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_GUI);
253 if (mod_data)
254 win = mod_data->utility_window;
255 }
256 if (!win)
257 return;
258
259 struct UtilWinData *priv = win->wdata;
260
261 if (!text || (*text == '\0'))
262 {
263 // Clear the old on-screen contents before hiding the window.
264 // Hidden windows are skipped by repaint, so they can't erase themselves.
265 mutt_window_clear(win);
266 FREE(&priv->text);
267 window_set_visible(win, false);
268 mutt_window_reflow(NULL);
269 return;
270 }
271
272 mutt_str_replace(&priv->text, text);
273
274 window_set_visible(win, true);
275 win->actions |= WA_RECALC;
276 mutt_window_reflow(NULL);
277}
