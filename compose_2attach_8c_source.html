85unsigned long cum_attachs_size(struct ConfigSubset *sub, struct ComposeAttachData *adata)
86{
87 if (!adata || !adata->actx)
88 return 0;
89
90 size_t s = 0;
91 struct Content *info = NULL;
92 struct Body *b = NULL;
93 struct AttachCtx *actx = adata->actx;
94 struct AttachPtr **idx = actx->idx;
95
96 for (unsigned short i = 0; i < actx->idxlen; i++)
97 {
98 b = idx[i]->body;
99
100 if (!b->content)
101 b->content = mutt_get_content_info(b->filename, b, sub);
102
103 info = b->content;
104 if (info)
105 {
106 switch (b->encoding)
107 {
108 case ENC_QUOTED_PRINTABLE:
109 s += 3 * (info->lobin + info->hibin) + info->ascii + info->crlf;
110 break;
111 case ENC_BASE64:
112 s += (4 * (info->lobin + info->hibin + info->ascii + info->crlf)) / 3;
113 break;
114 default:
115 s += info->lobin + info->hibin + info->ascii + info->crlf;
116 break;
117 }
118 }
119 }
120
121 return s;
122}

168static int attach_window_observer(struct NotifyCallback *nc)
169{
170 if (nc->event_type != NT_WINDOW)
171 return 0;
172 if (!nc->global_data || !nc->event_data)
173 return -1;
174
175 struct MuttWindow *win_attach = nc->global_data;
176 struct EventWindow *ev_w = nc->event_data;
177 if (ev_w->win != win_attach)
178 return 0;
179
180 if (nc->event_subtype == NT_WINDOW_STATE)
181 {
182 win_attach->actions |= WA_RECALC;
183 mutt_debug(LL_DEBUG5, "window state done, request WA_RECALC\n");
184 }
185 else if (nc->event_subtype == NT_WINDOW_DELETE)
186 {
187 struct Menu *menu = win_attach->wdata;
188 struct ComposeAttachData *adata = menu->mdata;
189 struct AttachCtx *actx = adata->actx;
190 notify_observer_remove(actx->email->notify, attach_email_observer, win_attach);
191 notify_observer_remove(NeoMutt->sub->notify, attach_config_observer, win_attach);
192 notify_observer_remove(win_attach->notify, attach_window_observer, win_attach);
193 mutt_debug(LL_DEBUG5, "window delete done\n");
194 }
195
196 return 0;
197}

243static int attach_recalc(struct MuttWindow *win)
244{
245 struct Menu *menu = win->wdata;
246 struct ComposeAttachData *adata = menu->mdata;
247
248 const int cur_rows = win->state.rows;
249 const int new_rows = adata->actx->idxlen;
250
251 if (new_rows != cur_rows)
252 {
253 win->req_rows = new_rows;
254 mutt_window_reflow(win->parent);
255 menu_adjust(menu);
256 }
257
258 win->actions |= WA_REPAINT;
259 mutt_debug(LL_DEBUG5, "recalc done, request WA_REPAINT\n");
260 return 0;
261}

268struct MuttWindow *attach_new(struct MuttWindow *parent, struct ComposeSharedData *shared)
269{
270 struct MuttWindow *win_attach = menu_window_new(MENU_COMPOSE, NeoMutt->sub);
271
272 struct ComposeAttachData *adata = attach_data_new(shared->email);
273
274 shared->adata = adata;
275
276 // NT_COLOR is handled by the Menu Window
277 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, attach_config_observer, win_attach);

283 menu->win = win_attach;

286 menu->tag = compose_attach_tag;
287 menu->mdata = adata;
288 menu->mdata_free = attach_data_free;
289 adata->menu = menu;

291 return win_attach;
