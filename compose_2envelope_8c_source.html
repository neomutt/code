85 static int calc_address(struct AddressList *al, struct ListHead *slist,
86  short cols, short *srows)
87 {
88  mutt_list_free(slist);
89  mutt_addrlist_write_list(al, slist);
90 
91  int rows = 1;
92  int addr_len;
93  int width_left = cols;
94  struct ListNode *next = NULL;
95  struct ListNode *np = NULL;
96  STAILQ_FOREACH(np, slist, entries)
97  {
98  next = STAILQ_NEXT(np, entries);
99  addr_len = mutt_strwidth(np->data);
100  if (next)
101  addr_len += 2; // ", "
102 
103  try_again:
104  if (addr_len >= width_left)
105  {
106  if (width_left == cols)
107  break;
108 
109  rows++;
110  width_left = cols;
111  goto try_again;
112  }
113 
114  if (addr_len < width_left)
115  width_left -= addr_len;
116  }
117 
118  *srows = MIN(rows, MAX_ADDR_ROWS);
119  return *srows;
120 }

417 static int draw_envelope_addr(int field, struct AddressList *al,
418  struct MuttWindow *win, int row, size_t max_lines)
419 {
420  draw_header(win, row, field);
421 
422  struct ListHead list = STAILQ_HEAD_INITIALIZER(list);
423  int count = mutt_addrlist_write_list(al, &list);
424 
425  int lines_used = 1;
426  int width_left = win->state.cols - MaxHeaderWidth;
427  char more[32];
428  int more_len = 0;
429 
430  char *sep = NULL;
431  struct ListNode *next = NULL;
432  struct ListNode *np = NULL;
433  STAILQ_FOREACH(np, &list, entries)
434  {
435  next = STAILQ_NEXT(np, entries);
436  int addr_len = mutt_strwidth(np->data);
437  if (next)
438  {
439  sep = ", ";
440  addr_len += 2;
441  }
442  else
443  {
444  sep = "";
445  }
446 
447  count--;
448  try_again:
449  more_len = snprintf(more, sizeof(more),
450  ngettext("(+%d more)", "(+%d more)", count), count);
451  mutt_debug(LL_DEBUG3, "text: '%s' len: %d\n", more, more_len);
452 
453  int reserve = ((count > 0) && (lines_used == max_lines)) ? more_len : 0;
454  mutt_debug(LL_DEBUG3, "processing: %s (al:%ld, wl:%d, r:%d, lu:%ld)\n",
455  np->data, addr_len, width_left, reserve, lines_used);
456  if (addr_len >= (width_left - reserve))
457  {
458  mutt_debug(LL_DEBUG3, "not enough space\n");
459  if (lines_used == max_lines)
460  {
461  mutt_debug(LL_DEBUG3, "no more lines\n");
462  mutt_debug(LL_DEBUG3, "truncating: %s\n", np->data);
463  mutt_paddstr(win, width_left, np->data);
464  break;
465  }
466 
467  if (width_left == (win->state.cols - MaxHeaderWidth))
468  {
469  mutt_debug(LL_DEBUG3, "couldn't print: %s\n", np->data);
470  mutt_paddstr(win, width_left, np->data);
471  break;
472  }
473 
474  mutt_debug(LL_DEBUG3, "start a new line\n");
475  mutt_window_clrtoeol(win);
476  row++;
477  lines_used++;
478  width_left = win->state.cols - MaxHeaderWidth;
479  mutt_window_move(win, MaxHeaderWidth, row);
480  goto try_again;
481  }
482 
483  if (addr_len < width_left)
484  {
485  mutt_debug(LL_DEBUG3, "space for: %s\n", np->data);
486  mutt_window_addstr(win, np->data);
487  mutt_window_addstr(win, sep);
488  width_left -= addr_len;
489  }
490  mutt_debug(LL_DEBUG3, "%ld addresses remaining\n", count);
491  mutt_debug(LL_DEBUG3, "%ld lines remaining\n", max_lines - lines_used);
492  }
493  mutt_list_free(&list);
494 
495  if (count > 0)
496  {
497  mutt_window_move(win, win->state.cols - more_len, row);
498  mutt_curses_set_color(MT_COLOR_BOLD);
499  mutt_window_addstr(win, more);
500  mutt_curses_set_color(MT_COLOR_NORMAL);
501  mutt_debug(LL_DEBUG3, "%ld more (len %d)\n", count, more_len);
502  }
503  else
504  {
505  mutt_window_clrtoeol(win);
506  }
507 
508  for (int i = lines_used; i < max_lines; i++)
509  {
510  mutt_window_move(win, 0, row + i);
511  mutt_window_clrtoeol(win);
512  }
513 
514  mutt_debug(LL_DEBUG3, "used %d lines\n", lines_used);
515  return lines_used;
516 }
