80static int simple_config_observer(struct NotifyCallback *nc)
81{
82 if (nc->event_type != NT_CONFIG)
83 return 0;
84 if (!nc->global_data || !nc->event_data)
85 return -1;
86
87 struct EventConfig *ev_c = nc->event_data;
88 if (!mutt_str_equal(ev_c->name, "status_on_top"))
89 return 0;
90
91 struct MuttWindow *dlg = nc->global_data;
92 window_status_on_top(dlg, NeoMutt->sub);
93 mutt_debug(LL_DEBUG5, "config done\n");
94 return 0;
95}

104static int simple_window_observer(struct NotifyCallback *nc)
105{
106 if (nc->event_type != NT_WINDOW)
107 return 0;
108 if (!nc->global_data || !nc->event_data)
109 return -1;
110 if (nc->event_subtype != NT_WINDOW_DELETE)
111 return 0;
112
113 struct MuttWindow *dlg = nc->global_data;
114 struct EventWindow *ev_w = nc->event_data;
115 if (ev_w->win != dlg)
116 return 0;
117
118 notify_observer_remove(NeoMutt->sub->notify, simple_config_observer, dlg);
119 notify_observer_remove(dlg->notify, simple_window_observer, dlg);
120
121 mutt_debug(LL_DEBUG5, "window delete done\n");
122 return 0;
123}

143

146 if (c_status_on_top)
147 {
148 mutt_window_add_child(dlg, win_sbar);
149 mutt_window_add_child(dlg, win_menu);
150 }
151 else
152 {
153 mutt_window_add_child(dlg, win_menu);
154 mutt_window_add_child(dlg, win_sbar);
155 }
156
157 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, simple_config_observer, dlg);
158 notify_observer_add(dlg->notify, NT_WINDOW, simple_window_observer, dlg);
159 dialog_push(dlg);
160
161 return (struct SimpleDialogWindows) { dlg, win_sbar, win_menu->wdata };
