138 static const char *query_format_str(char *buf, size_t buflen, size_t col, int cols,
139  char op, const char *src, const char *prec,
140  const char *if_str, const char *else_str,
141  intptr_t data, MuttFormatFlags flags)
142 {
143  struct AliasView *av = (struct AliasView *) data;
144  struct Alias *alias = av->alias;
145  char fmt[128];
146  char tmp[256] = { 0 };
147  bool optional = (flags & MUTT_FORMAT_OPTIONAL);
148 
149  switch (op)
150  {
151  case 'a':
152  tmp[0] = '<';
153  mutt_addrlist_write(&alias->addr, tmp + 1, sizeof(tmp) - 1, true);
154  const size_t len = strlen(tmp);
155  if (len < (sizeof(tmp) - 1))
156  {
157  tmp[len] = '>';
158  tmp[len + 1] = '\0';
159  }
160  mutt_format_s(buf, buflen, prec, tmp);
161  break;
162  case 'c':
163  snprintf(fmt, sizeof(fmt), "%%%sd", prec);
164  snprintf(buf, buflen, fmt, av->num + 1);
165  break;
166  case 'e':
167  if (!optional)
168  mutt_format_s(buf, buflen, prec, NONULL(alias->comment));
169  else if (!alias->comment || (*alias->comment == '\0'))
170  optional = false;
171  break;
172  case 'n':
173  mutt_format_s(buf, buflen, prec, NONULL(alias->name));
174  break;
175  case 't':
176  snprintf(fmt, sizeof(fmt), "%%%sc", prec);
177  snprintf(buf, buflen, fmt, av->is_tagged ? '*' : ' ');
178  break;
179  default:
180  snprintf(fmt, sizeof(fmt), "%%%sc", prec);
181  snprintf(buf, buflen, fmt, op);
182  break;
183  }
184 
185  if (optional)
186  {
187  mutt_expando_format(buf, buflen, col, cols, if_str, query_format_str, data,
188  MUTT_FORMAT_NO_FLAGS);
189  }
190  else if (flags & MUTT_FORMAT_OPTIONAL)
191  {
192  mutt_expando_format(buf, buflen, col, cols, else_str, query_format_str,
193  data, MUTT_FORMAT_NO_FLAGS);
194  }
195 
196  return src;
197 }

233 static int query_run(char *s, bool verbose, struct AliasList *al)
234 {
235  FILE *fp = NULL;
236  char *buf = NULL;
237  size_t buflen;
238  char *msg = NULL;
239  size_t msglen = 0;
240  char *p = NULL;
241  struct Buffer *cmd = mutt_buffer_pool_get();
242 
243  mutt_buffer_file_expand_fmt_quote(cmd, C_QueryCommand, s);
244 
245  pid_t pid = filter_create(mutt_b2s(cmd), NULL, &fp, NULL);
246  if (pid < 0)
247  {
248  mutt_debug(LL_DEBUG1, "unable to fork command: %s\n", mutt_b2s(cmd));
249  mutt_buffer_pool_release(&cmd);
250  return -1;
251  }
252  mutt_buffer_pool_release(&cmd);
253 
254  if (verbose)
255  mutt_message(_("Waiting for response..."));
256 
257  /* The query protocol first reads one NL-terminated line. If an error
258  * occurs, this is assumed to be an error message. Otherwise it's ignored. */
259  msg = mutt_file_read_line(msg, &msglen, fp, NULL, 0);
260  while ((buf = mutt_file_read_line(buf, &buflen, fp, NULL, 0)))
261  {
262  p = strtok(buf, "\t\n");
263  if (p)
264  {
265  struct Alias *alias = alias_new();
266 
267  mutt_addrlist_parse(&alias->addr, p);
268  p = strtok(NULL, "\t\n");
269  if (p)
270  {
271  alias->name = mutt_str_dup(p);
272  p = strtok(NULL, "\t\n");
273  if (p)
274  alias->comment = mutt_str_dup(p);
275  }
276  TAILQ_INSERT_TAIL(al, alias, entries);
277  }
278  }
279  FREE(&buf);
280  mutt_file_fclose(&fp);
281  if (filter_wait(pid))
282  {
283  mutt_debug(LL_DEBUG1, "Error: %s\n", msg);
284  if (verbose)
285  mutt_error("%s", msg);
286  }
287  else
288  {
289  if (verbose)
290  mutt_message("%s", msg);
291  }
292  FREE(&msg);
293 
294  return 0;
295 }
