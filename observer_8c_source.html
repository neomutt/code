50 static bool calc_divider(struct SidebarWindowData *wdata)
51 {
52  enum DivType type = SB_DIV_USER;
53 
54  // Calculate the width of the delimiter in screen cells
55  int width = mutt_strwidth(C_SidebarDividerChar);
56 
57  if (C_AsciiChars)
58  {
59  if (width < 1) // empty or bad
60  {
61  type = SB_DIV_ASCII;
62  width = 1;
63  }
64  else
65  {
66  for (size_t i = 0; i < width; i++)
67  {
68  if (C_SidebarDividerChar[i] & ~0x7F) // high-bit is set
69  {
70  type = SB_DIV_ASCII;
71  width = 1;
72  break;
73  }
74  }
75  }
76  }
77  else
78  {
79  if (width < 1) // empty or bad
80  {
81  type = SB_DIV_UTF8;
82  width = 1;
83  }
84  }
85 
86  const bool changed = (width != wdata->divider_width);
87 
88  wdata->divider_type = type;
89  wdata->divider_width = width;
90 
91  return changed;
92 }
