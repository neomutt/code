137static void maildir_check_dir(struct Mailbox *m, const char *dir_name,
138 bool check_new, bool check_stats)
139{
140 DIR *dir = NULL;
141 struct dirent *de = NULL;
142 char *p = NULL;
143 struct stat st = { 0 };
144
145 struct Buffer *path = buf_pool_get();
146 struct Buffer *msgpath = buf_pool_get();
147 buf_printf(path, "%s/%s", mailbox_path(m), dir_name);
148
149 /* when $mail_check_recent is set, if the new/ directory hasn't been modified since
150 * the user last exited the mailbox, then we know there is no recent mail. */
151 const bool c_mail_check_recent = cs_subset_bool(NeoMutt->sub, "mail_check_recent");
152 if (check_new && c_mail_check_recent)
153 {
154 if ((stat(buf_string(path), &st) == 0) &&
155 (mutt_file_stat_timespec_compare(&st, MUTT_STAT_MTIME, &m->last_visited) < 0))
156 {
157 check_new = false;
158 }
159 }
160
161 if (!(check_new || check_stats))
162 goto cleanup;
163
164 dir = mutt_file_opendir(buf_string(path), MUTT_OPENDIR_CREATE);
165 if (!dir)
166 {
167 m->type = MUTT_UNKNOWN;
168 goto cleanup;
169 }
170
171 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
172
173 char delimiter_version[8] = { 0 };
174 snprintf(delimiter_version, sizeof(delimiter_version), "%c2,", c_maildir_field_delimiter);
175 while ((de = readdir(dir)))
176 {
177 if (*de->d_name == '.')
178 continue;
179
180 p = strstr(de->d_name, delimiter_version);
181 if (p && strchr(p + 3, 'T'))
182 continue;
183
184 if (check_stats)
185 {
186 m->msg_count++;
187 if (p && strchr(p + 3, 'F'))
188 m->msg_flagged++;
189 }
190 if (!p || !strchr(p + 3, 'S'))
191 {
192 if (check_stats)
193 m->msg_unread++;
194 if (check_new)
195 {
196 if (c_mail_check_recent)
197 {
198 buf_printf(msgpath, "%s/%s", buf_string(path), de->d_name);
199 /* ensure this message was received since leaving this m */
200 if ((stat(buf_string(msgpath), &st) == 0) &&
201 (mutt_file_stat_timespec_compare(&st, MUTT_STAT_CTIME, &m->last_visited) <= 0))
202 {
203 continue;
204 }
205 }
206 m->has_new = true;
207 if (check_stats)
208 {
209 m->msg_new++;
210 }
211 else
212 {
213 break;
214 }
215 }
216 }
217 }
218
219 closedir(dir);
220
221cleanup:
222 buf_pool_release(&path);
223 buf_pool_release(&msgpath);
224}

240void maildir_gen_flags(char *dest, size_t destlen, struct Email *e)
241{
242 *dest = '\0';
243
244 const char *flags = NULL;
245
246 struct MaildirEmailData *edata = maildir_edata_get(e);
247 if (edata)
248 flags = edata->maildir_flags;
249
250 /* The maildir specification requires that all files in the cur
251 * subdirectory have the :unique string appended, regardless of whether
252 * or not there are any flags. If .old is set, we know that this message
253 * will end up in the cur directory, so we include it in the following
254 * test even though there is no associated flag. */
255
256 if (e->flagged || e->replied || e->read || e->deleted || e->old || flags)
257 {
258 char tmp[1024] = { 0 };
259 snprintf(tmp, sizeof(tmp), "%s%s%s%s%s", e->flagged ? "F" : "", e->replied ? "R" : "",
260 e->read ? "S" : "", e->deleted ? "T" : "", NONULL(flags));
261 if (flags)
262 mutt_qsort_r(tmp, strlen(tmp), 1, maildir_sort_flags, NULL);
263
264 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
265 snprintf(dest, destlen, "%c2,%s", c_maildir_field_delimiter, tmp);
266 }

292static int maildir_commit_message(struct Mailbox *m, struct Message *msg, struct Email *e)
293{
294 char subdir[4] = { 0 };
295 char suffix[16] = { 0 };
296 int rc = 0;
297
298 if (mutt_file_fsync_close(&msg->fp))
299 {
300 mutt_perror(_("Could not flush message to disk"));
301 return -1;
302 }
303
304 /* extract the subdir */
305 char *s = strrchr(msg->path, '/') + 1;
306 mutt_str_copy(subdir, s, 4);
307
308 /* extract the flags */
309 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
310 s = strchr(s, c_maildir_field_delimiter);
311 if (s)
312 mutt_str_copy(suffix, s, sizeof(suffix));
313 else
314 suffix[0] = '\0';
315
316 /* construct a new file name. */
317 struct Buffer *path = buf_pool_get();
318 struct Buffer *full = buf_pool_get();
319 while (true)
320 {
321 buf_printf(path, "%s/%lld.R%" PRIu64 ".%s%s", subdir, (long long) mutt_date_now(),
322 mutt_rand64(), NONULL(ShortHostname), suffix);
323 buf_printf(full, "%s/%s", mailbox_path(m), buf_string(path));
324
325 mutt_debug(LL_DEBUG2, "renaming %s to %s\n", msg->path, buf_string(full));
326
327 if (mutt_file_safe_rename(msg->path, buf_string(full)) == 0)
328 {
329 /* Adjust the mtime on the file to match the time at which this
330 * message was received. Currently this is only set when copying
331 * messages between mailboxes, so we test to ensure that it is
332 * actually set. */
333 if (msg->received != 0)
334 {
335 struct utimbuf ut = { 0 };
336 int rc_utime;
337
338 ut.actime = msg->received;
339 ut.modtime = msg->received;
340 do
341 {
342 rc_utime = utime(buf_string(full), &ut);
343 } while ((rc_utime == -1) && (errno == EINTR));
344 if (rc_utime == -1)
345 {
346 mutt_perror(_("maildir_commit_message(): unable to set time on file"));
347 rc = -1;
348 goto cleanup;
349 }
350 }
351
352#ifdef USE_NOTMUCH
353 if (m->type == MUTT_NOTMUCH)
354 nm_update_filename(m, e->path, buf_string(full), e);
355#endif
356 if (e)
357 mutt_str_replace(&e->path, buf_string(path));
358 mutt_str_replace(&msg->committed_path, buf_string(full));
359 FREE(&msg->path);
360
361 goto cleanup;
362 }
363 else if (errno != EEXIST)
364 {
365 mutt_perror("%s", mailbox_path(m));
366 rc = -1;
367 goto cleanup;
368 }
369 }
370
371cleanup:
372 buf_pool_release(&path);
373 buf_pool_release(&full);
374
375 return rc;
376}

438static int maildir_sync_message(struct Mailbox *m, struct Email *e)
439{
440 if (!m || !e)
441 return -1;
442
443 struct Buffer *newpath = NULL;
444 struct Buffer *partpath = NULL;
445 struct Buffer *fullpath = NULL;
446 struct Buffer *oldpath = NULL;
447 char suffix[16] = { 0 };
448 int rc = 0;
449
450 if (e->attach_del || e->env->changed)
451 {
452 /* when doing attachment deletion/rethreading, fall back to the maildir case. */
453 if (maildir_rewrite_message(m, e) != 0)
454 return -1;
455 e->env->changed = false;
456 }
457 else
458 {
459 /* we just have to rename the file. */
460
461 char *p = strrchr(e->path, '/');
462 if (!p)
463 {
464 mutt_debug(LL_DEBUG1, "%s: unable to find subdir!\n", e->path);
465 return -1;
466 }
467 p++;
468 newpath = buf_pool_get();
469 partpath = buf_pool_get();
470 fullpath = buf_pool_get();
471 oldpath = buf_pool_get();
472
473 buf_strcpy(newpath, p);
474
475 /* kill the previous flags */
476 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
477 p = strchr(newpath->data, c_maildir_field_delimiter);
478 if (p)
479 {
480 *p = '\0';
481 newpath->dptr = p; /* fix buffer up, just to be safe */
482 }
483
484 maildir_gen_flags(suffix, sizeof(suffix), e);
485
486 buf_printf(partpath, "%s/%s%s", (e->read || e->old) ? "cur" : "new",
487 buf_string(newpath), suffix);
488 buf_printf(fullpath, "%s/%s", mailbox_path(m), buf_string(partpath));
489 buf_printf(oldpath, "%s/%s", mailbox_path(m), e->path);
490
491 if (mutt_str_equal(buf_string(fullpath), buf_string(oldpath)))
492 {
493 /* message hasn't really changed */
494 goto cleanup;
495 }
496
497 /* record that the message is possibly marked as trashed on disk */
498 e->trash = e->deleted;
499
500 struct stat st = { 0 };
501 if (stat(buf_string(oldpath), &st) == -1)
502 {
503 mutt_debug(LL_DEBUG1, "File already removed (just continuing)");
504 goto cleanup;
505 }
506
507 if (rename(buf_string(oldpath), buf_string(fullpath)) != 0)
508 {
509 mutt_perror("rename");
510 rc = -1;
511 goto cleanup;
512 }
513 mutt_str_replace(&e->path, buf_string(partpath));
514 }
515
516cleanup:
517 buf_pool_release(&newpath);
518 buf_pool_release(&partpath);
519 buf_pool_release(&fullpath);
520 buf_pool_release(&oldpath);
521
522 return rc;
523}

698#endif

703#ifdef USE_HCACHE

924
925 const char c_maildir_field_delimiter = *cc_maildir_field_delimiter();
926 char *p = strrchr(path, c_maildir_field_delimiter);
927 if (p && mutt_str_startswith(p + 1, "2,"))
928 {
929 p += 3;
930
931 mutt_str_replace(&edata->maildir_flags, p);
932 q = edata->maildir_flags;
933
934 while (*p)
935 {
936 switch (*p)
937 {
938 case 'F': // Flagged
939 e->flagged = true;
940 break;
941
942 case 'R': // Replied
943 e->replied = true;
944 break;
945
946 case 'S': // Seen
947 e->read = true;
948 break;
949
950 case 'T': // Trashed
951 {
952 const bool c_flag_safe = cs_subset_bool(NeoMutt->sub, "flag_safe");
953 if (!e->flagged || !c_flag_safe)
954 {
955 e->trash = true;
956 e->deleted = true;
957 }
958 break;
959 }
960
961 default:
962 *q++ = *p;
963 break;
964 }
965 p++;
966 }
967 }
968
969 if (q == edata->maildir_flags)
970 FREE(&edata->maildir_flags);
971 else if (q)
972 *q = '\0';
973}

986bool maildir_parse_stream(FILE *fp, const char *fname, bool is_old, struct Email *e)
987{
988 if (!fp || !fname || !e)
989 return false;

992 if (size == 0)

996

1004
1005 /* maildir stores its flags in the filename, so ignore the
1006 * flags in the header of the message */
1007 e->old = is_old;
1008 maildir_parse_flags(e, fname);
1009
1010 return e;

1261bool maildir_update_flags(struct Mailbox *m, struct Email *e_old, struct Email *e_new)
1262{
1263 if (!m)
1264 return false;
1265
1266 /* save the global state here so we can reset it at the
1267 * end of list block if required. */
1268 bool context_changed = m->changed;
1269
1270 /* user didn't modify this message. alter the flags to match the
1271 * current state on disk. This may not actually do
1272 * anything. mutt_set_flag() will just ignore the call if the status
1273 * bits are already properly set, but it is still faster not to pass
1274 * through it */
1275 if (e_old->flagged != e_new->flagged)
1276 mutt_set_flag(m, e_old, MUTT_FLAG, e_new->flagged, true);
1277 if (e_old->replied != e_new->replied)
1278 mutt_set_flag(m, e_old, MUTT_REPLIED, e_new->replied, true);
1279 if (e_old->read != e_new->read)
1280 mutt_set_flag(m, e_old, MUTT_READ, e_new->read, true);
1281 if (e_old->old != e_new->old)
1282 mutt_set_flag(m, e_old, MUTT_OLD, e_new->old, true);
1283
1284 /* mutt_set_flag() will set this, but we don't need to
1285 * sync the changes we made because we just updated the
1286 * context to match the current on-disk state of the
1287 * message. */
1288 bool header_changed = e_old->changed;
1289 e_old->changed = false;
1290
1291 /* if the mailbox was not modified before we made these
1292 * changes, unset the changed flag since nothing needs to
1293 * be synchronized. */
1294 if (!context_changed)
1295 m->changed = false;
1296
1297 return header_changed;
1298}

1312static enum MxStatus maildir_check(struct Mailbox *m)
1313{
1314 struct stat st_new = { 0 }; /* status of the "new" subdirectory */
1315 struct stat st_cur = { 0 }; /* status of the "cur" subdirectory */
1316 int changed = MMC_NO_DIRS; /* which subdirectories have changed */
1317 bool occult = false; /* messages were removed from the mailbox */
1318 int num_new = 0; /* number of new messages added to the mailbox */
1319 bool flags_changed = false; /* message flags were changed in the mailbox */
1320 struct HashTable *hash_names = NULL; // Hash Table: "base-filename" -> MdEmail
1321 struct MaildirMboxData *mdata = maildir_mdata_get(m);
1322
1323 /* XXX seems like this check belongs in mx_mbox_check() rather than here. */
1324 const bool c_check_new = cs_subset_bool(NeoMutt->sub, "check_new");
1325 if (!c_check_new)
1326 return MX_STATUS_OK;
1327
1328 struct Buffer *buf = buf_pool_get();
1329 buf_printf(buf, "%s/new", mailbox_path(m));
1330 if (stat(buf_string(buf), &st_new) == -1)
1331 {
1332 buf_pool_release(&buf);
1333 return MX_STATUS_ERROR;
1334 }
1335
1336 buf_printf(buf, "%s/cur", mailbox_path(m));
1337 if (stat(buf_string(buf), &st_cur) == -1)
1338 {
1339 buf_pool_release(&buf);
1340 return MX_STATUS_ERROR;
1341 }
1342
1343 /* determine which subdirectories need to be scanned */
1344 if (mutt_file_stat_timespec_compare(&st_new, MUTT_STAT_MTIME, &mdata->mtime) > 0)
1345 changed = MMC_NEW_DIR;
1346 if (mutt_file_stat_timespec_compare(&st_cur, MUTT_STAT_MTIME, &mdata->mtime_cur) > 0)
1347 changed |= MMC_CUR_DIR;
1348
1349 if (changed == MMC_NO_DIRS)
1350 {
1351 buf_pool_release(&buf);
1352 return MX_STATUS_OK; /* nothing to do */
1353 }
1354
1355 /* Update the modification times on the mailbox.
1356 *
1357 * The monitor code notices changes in the open mailbox too quickly.
1358 * In practice, this sometimes leads to all the new messages not being
1359 * noticed during the SAME group of mtime stat updates. To work around
1360 * the problem, don't update the stat times for a monitor caused check. */
1361#ifdef USE_INOTIFY
1362 if (MonitorContextChanged)
1363 {
1364 MonitorContextChanged = false;
1365 }
1366 else
1367#endif
1368 {
1369 mutt_file_get_stat_timespec(&mdata->mtime_cur, &st_cur, MUTT_STAT_MTIME);
1370 mutt_file_get_stat_timespec(&mdata->mtime, &st_new, MUTT_STAT_MTIME);
1371 }
1372
1373 /* do a fast scan of just the filenames in
1374 * the subdirectories that have changed. */
1375 struct MdEmailArray mda = ARRAY_HEAD_INITIALIZER;
1376 if (changed & MMC_NEW_DIR)
1377 maildir_parse_dir(m, &mda, "new", NULL);
1378 if (changed & MMC_CUR_DIR)
1379 maildir_parse_dir(m, &mda, "cur", NULL);
1380
1381 /* we create a hash table keyed off the canonical (sans flags) filename
1382 * of each message we scanned. This is used in the loop over the
1383 * existing messages below to do some correlation. */
1384 hash_names = mutt_hash_new(ARRAY_SIZE(&mda), MUTT_HASH_NO_FLAGS);
1385
1386 struct MdEmail *md = NULL;
1387 struct MdEmail **mdp = NULL;
1388 ARRAY_FOREACH(mdp, &mda)
1389 {
1390 md = *mdp;
1391 maildir_canon_filename(buf, md->email->path);
1392 md->canon_fname = buf_strdup(buf);
1393 mutt_hash_insert(hash_names, md->canon_fname, md);
1394 }
1395
1396 /* check for modifications and adjust flags */
1397 for (int i = 0; i < m->msg_count; i++)
1398 {
1399 struct Email *e = m->emails[i];
1400 if (!e)
1401 break;
1402
1403 maildir_canon_filename(buf, e->path);
1404 md = mutt_hash_find(hash_names, buf_string(buf));
1405 if (md && md->email)
1406 {
1407 /* message already exists, merge flags */
1408
1409 /* check to see if the message has moved to a different
1410 * subdirectory. If so, update the associated filename. */
1411 if (!mutt_str_equal(e->path, md->email->path))
1412 mutt_str_replace(&e->path, md->email->path);
1413
1414 /* if the user hasn't modified the flags on this message, update
1415 * the flags we just detected. */
1416 if (!e->changed)
1417 if (maildir_update_flags(m, e, md->email))
1418 flags_changed = true;
1419
1420 if (e->deleted == e->trash)
1421 {
1422 if (e->deleted != md->email->deleted)
1423 {
1424 e->deleted = md->email->deleted;
1425 flags_changed = true;
1426 }
1427 }
1428 e->trash = md->email->trash;
1429
1430 /* this is a duplicate of an existing email, so remove it */
1431 email_free(&md->email);
1432 }
1433 /* This message was not in the list of messages we just scanned.
1434 * Check to see if we have enough information to know if the
1435 * message has disappeared out from underneath us. */
1436 else if (((changed & MMC_NEW_DIR) && mutt_strn_equal(e->path, "new/", 4)) ||
1437 ((changed & MMC_CUR_DIR) && mutt_strn_equal(e->path, "cur/", 4)))
1438 {
1439 /* This message disappeared, so we need to simulate a "reopen"
1440 * event. We know it disappeared because we just scanned the
1441 * subdirectory it used to reside in. */
1442 occult = true;
1443 e->deleted = true;
1444 e->purge = true;
1445 }
1446 else
1447 {
1448 /* This message resides in a subdirectory which was not
1449 * modified, so we assume that it is still present and
1450 * unchanged. */
1451 }
1452 }
1453
1454 /* destroy the file name hash */
1455 mutt_hash_free(&hash_names);
1456
1457 /* If we didn't just get new mail, update the tables. */
1458 if (occult)
1459 mailbox_changed(m, NT_MAILBOX_RESORT);
1460
1461 /* do any delayed parsing we need to do. */
1462 maildir_delayed_parsing(m, &mda, NULL);
1463
1464 /* Incorporate new messages */
1465 num_new = maildir_move_to_mailbox(m, &mda);
1466 maildirarray_clear(&mda);
1467
1468 if (num_new > 0)
1469 {
1470 mailbox_changed(m, NT_MAILBOX_INVALID);
1471 m->changed = true;
1472 }
1473
1474 buf_pool_release(&buf);
1475
1476 ARRAY_FREE(&mda);
1477 if (occult)
1478 return MX_STATUS_REOPENED;
1479 if (num_new > 0)
1480 return MX_STATUS_NEW_MAIL;
1481 if (flags_changed)
1482 return MX_STATUS_FLAGS;
1483 return MX_STATUS_OK;
1484}
