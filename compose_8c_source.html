801  mutt_debug(LL_DEBUG3, "processing: %s (al:%ld, wl:%d, r:%d, lu:%ld)\n",
802  np->data, addr_len, width_left, reserve, lines_used);

804  {

808  mutt_debug(LL_DEBUG3, "no more lines\n");

811  break;
812  }
813

815  {
816  mutt_debug(LL_DEBUG3, "couldn't print: %s\n", np->data);
817  mutt_paddstr(width_left, np->data);
818  break;
819  }
820

822  mutt_window_clrtoeol(win);
823  row++;
824  lines_used++;
825  width_left = win->state.cols - MaxHeaderWidth;
826  mutt_window_move(win, MaxHeaderWidth, row);
827  goto try_again;
828  }
829 
830  if (addr_len < width_left)
831  {
832  mutt_debug(LL_DEBUG3, "space for: %s\n", np->data);
833  mutt_window_addstr(np->data);
834  mutt_window_addstr(sep);
835  width_left -= addr_len;
836  }
837  mutt_debug(LL_DEBUG3, "%ld addresses remaining\n", count);
838  mutt_debug(LL_DEBUG3, "%ld lines remaining\n", max_lines - lines_used);
839  }
840  mutt_list_free(&list);
841 
842  if (count > 0)
843  {
844  mutt_window_move(win, win->state.cols - more_len, row);
845  mutt_curses_set_color(MT_COLOR_BOLD);
846  mutt_window_addstr(more);
847  mutt_curses_set_color(MT_COLOR_NORMAL);
848  mutt_debug(LL_DEBUG3, "%ld more (len %d)\n", count, more_len);
849  }
850  else
851  {
852  mutt_window_clrtoeol(win);
853  }
854 
855  for (int i = lines_used; i < max_lines; i++)
856  {
857  mutt_window_move(win, 0, row + i);
858  mutt_window_clrtoeol(win);
859  }
860 
861  mutt_debug(LL_DEBUG3, "used %d lines\n", lines_used);
862  return lines_used;

955 static int delete_attachment(struct AttachCtx *actx, int x)
956 {

958  int rindex = actx->v2r[x];

972  break;

974  }

980  * body->email->body.
981  *
982  * Other mutt_send_message() message constructors are careful to free
983  * any body->parts, removing depth:
984  * - mutt_prepare_template() used by postponed, resent, and draft files
985  * - mutt_copy_body() used by the recvattach menu and $forward_attachments.
986  *
987  * I believe it is safe to completely remove the "body->parts =
988  * NULL" statement. But for safety, am doing so only for the case
989  * it must be avoided: message attachments.
990  */
991  if (!idx[rindex]->body->email)
992  idx[rindex]->body->parts = NULL;
993  mutt_body_free(&(idx[rindex]->body));
994  FREE(&idx[rindex]->tree);
995  FREE(&idx[rindex]);
996  for (; rindex < actx->idxlen - 1; rindex++)
997  idx[rindex] = idx[rindex + 1];
998  idx[actx->idxlen - 1] = NULL;
999  actx->idxlen--;
1000 
1001  return 0;
