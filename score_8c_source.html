67void mutt_check_rescore(struct Mailbox *m)
68{
69 const bool c_score = cs_subset_bool(NeoMutt->sub, "score");
70 if (OptNeedRescore && c_score)
71 {
72 const enum EmailSortType c_sort = cs_subset_sort(NeoMutt->sub, "sort");
73 const enum EmailSortType c_sort_aux = cs_subset_sort(NeoMutt->sub, "sort_aux");
74 if (((c_sort & SORT_MASK) == EMAIL_SORT_SCORE) ||
75 ((c_sort_aux & SORT_MASK) == EMAIL_SORT_SCORE))
76 {
77 OptNeedResort = true;
78 if (mutt_using_threads())
79 OptSortSubthreads = true;
80 }
81
82 mutt_debug(LL_NOTIFY, "NT_SCORE: %p\n", (void *) m);
83 notify_send(m->notify, NT_SCORE, 0, NULL);
84 }
85 OptNeedRescore = false;
86}

91enum CommandResult parse_score(struct Buffer *buf, struct Buffer *s,
92 intptr_t data, struct Buffer *err)
93{
94 struct Score *ptr = NULL, *last = NULL;
95 char *pattern = NULL, *pc = NULL;
96
97 parse_extract_token(buf, s, TOKEN_NO_FLAGS);
98 if (!MoreArgs(s))
99 {
100 buf_printf(err, _("%s: too few arguments"), "score");
101 return MUTT_CMD_WARNING;
102 }
103 pattern = buf_strdup(buf);
104 parse_extract_token(buf, s, TOKEN_NO_FLAGS);
105 if (MoreArgs(s))
106 {
107 FREE(&pattern);
108 buf_printf(err, _("%s: too many arguments"), "score");
109 return MUTT_CMD_WARNING;
110 }
111
112 /* look for an existing entry and update the value, else add it to the end
113 * of the list */
114 for (ptr = ScoreList, last = NULL; ptr; last = ptr, ptr = ptr->next)
115 if (mutt_str_equal(pattern, ptr->str))
116 break;
117
118 if (ptr)
119 {
120 /* 'buf' arg was cleared and 'pattern' holds the only reference;
121 * as here 'ptr' != NULL -> update the value only in which case
122 * ptr->str already has the string, so pattern should be freed. */
123 FREE(&pattern);
124 }
125 else
126 {
127 struct MailboxView *mv_cur = get_current_mailbox_view();
128 struct PatternList *pat = mutt_pattern_comp(mv_cur, pattern, MUTT_PC_NO_FLAGS, err);
129 if (!pat)
130 {
131 FREE(&pattern);
132 return MUTT_CMD_ERROR;
133 }
134 ptr = MUTT_MEM_CALLOC(1, struct Score);
135 if (last)
136 last->next = ptr;
137 else
138 ScoreList = ptr;
139 ptr->pat = pat;
140 ptr->str = pattern;
141 }
142 pc = buf->data;
143 if (*pc == '=')
144 {
145 ptr->exact = true;
146 pc++;
147 }
148 if (!mutt_str_atoi_full(pc, &ptr->val))
149 {
150 FREE(&pattern);
151 buf_strcpy(err, _("Error: score: invalid number"));
152 return MUTT_CMD_ERROR;
153 }
154 OptNeedRescore = true;
155 return MUTT_CMD_SUCCESS;
156}

164void mutt_score_message(struct Mailbox *m, struct Email *e, bool upd_mbox)
165{
166 struct Score *tmp = NULL;
167 struct PatternCache cache = { 0 };
168
169 e->score = 0; /* in case of re-scoring */
170 for (tmp = ScoreList; tmp; tmp = tmp->next)
171 {
172 if (mutt_pattern_exec(SLIST_FIRST(tmp->pat), MUTT_MATCH_FULL_ADDRESS, NULL, e, &cache) > 0)
173 {
174 if (tmp->exact || (tmp->val == 9999) || (tmp->val == -9999))
175 {
176 e->score = tmp->val;
177 break;
178 }
179 e->score += tmp->val;
180 }
181 }
182 if (e->score < 0)
183 e->score = 0;
184
185 const short c_score_threshold_delete = cs_subset_number(NeoMutt->sub, "score_threshold_delete");
186 const short c_score_threshold_flag = cs_subset_number(NeoMutt->sub, "score_threshold_flag");
187 const short c_score_threshold_read = cs_subset_number(NeoMutt->sub, "score_threshold_read");
188
189 if (e->score <= c_score_threshold_delete)
190 mutt_set_flag(m, e, MUTT_DELETE, true, upd_mbox);
191 if (e->score <= c_score_threshold_read)
192 mutt_set_flag(m, e, MUTT_READ, true, upd_mbox);
193 if (e->score >= c_score_threshold_flag)
194 mutt_set_flag(m, e, MUTT_FLAG, true, upd_mbox);
195}

200enum CommandResult parse_unscore(struct Buffer *buf, struct Buffer *s,
201 intptr_t data, struct Buffer *err)
202{
203 struct Score *tmp = NULL, *last = NULL;
204
205 while (MoreArgs(s))
206 {
207 parse_extract_token(buf, s, TOKEN_NO_FLAGS);
208 if (mutt_str_equal("*", buf->data))
209 {
210 for (tmp = ScoreList; tmp;)
211 {
212 last = tmp;
213 tmp = tmp->next;
214 mutt_pattern_free(&last->pat);
215 FREE(&last);
216 }
217 ScoreList = NULL;
218 }
219 else
220 {
221 for (tmp = ScoreList; tmp; last = tmp, tmp = tmp->next)
222 {
223 if (mutt_str_equal(buf->data, tmp->str))
224 {
225 if (last)
226 last->next = tmp->next;
227 else
228 ScoreList = tmp->next;
229 mutt_pattern_free(&tmp->pat);
230 FREE(&tmp);
231 /* there should only be one score per pattern, so we can stop here */
232 break;
233 }
234 }
235 }
236 }
237 OptNeedRescore = true;
238 return MUTT_CMD_SUCCESS;
239}
