130static void notify_dump_window_state(struct NotifyCallback *nc)
131{
132 struct EventWindow *ev_w = nc->event_data;
133 const struct MuttWindow *win = ev_w->win;
134 WindowNotifyFlags flags = ev_w->flags;
135
136 struct Buffer *buf = buf_pool_get();
137
138 buf_add_printf(buf, "[%s] ", mutt_window_win_name(win));
139
140 if (flags & WN_VISIBLE)
141 buf_addstr(buf, "visible ");
142 if (flags & WN_HIDDEN)
143 buf_addstr(buf, "hidden ");
144
145 if (flags & WN_MOVED)
146 {
147 buf_add_printf(buf, "moved (C%d,R%d)->(C%d,R%d) ", win->old.col_offset,
148 win->old.row_offset, win->state.col_offset, win->state.row_offset);
149 }
150
151 if (flags & WN_TALLER)
152 buf_add_printf(buf, "taller [%d->%d] ", win->old.rows, win->state.rows);
153 if (flags & WN_SHORTER)
154 buf_add_printf(buf, "shorter [%d->%d] ", win->old.rows, win->state.rows);
155 if (flags & WN_WIDER)
156 buf_add_printf(buf, "wider [%d->%d] ", win->old.cols, win->state.cols);
157 if (flags & WN_NARROWER)
158 buf_add_printf(buf, "narrower [%d->%d] ", win->old.cols, win->state.cols);
159
160 mutt_debug(LL_DEBUG1, " Window: %s\n", buf_string(buf));
161
162 buf_pool_release(&buf);
163}

165static void notify_dump_window_focus(struct NotifyCallback *nc)
166{
167 struct EventWindow *ev_w = nc->event_data;
168 struct MuttWindow *win = ev_w->win;
169
170 struct Buffer *buf = buf_pool_get();
171
172 buf_addstr(buf, "Focus: ");
173
174 if (win)
175 {
176 struct MuttWindow *dlg = dialog_find(win);
177 if (dlg && (dlg != win))
178 buf_add_printf(buf, "%s:", mutt_window_win_name(dlg));
179
180 buf_add_printf(buf, "%s ", mutt_window_win_name(win));
181
182 buf_add_printf(buf, "(C%d,R%d) [%dx%d]", win->state.col_offset,
183 win->state.row_offset, win->state.cols, win->state.rows);
184 }
185 else
186 {
187 buf_addstr(buf, "NONE");
188 }
189
190 mutt_debug(LL_DEBUG1, " Window: %s\n", buf_string(buf));
191
192 buf_pool_release(&buf);
193}

195int debug_all_observer(struct NotifyCallback *nc)
196{
197 mutt_debug(LL_DEBUG1, "\033[1;31mNotification:\033[0m %s\n",
198 name_notify_type(nc->event_type));
199
200 switch (nc->event_type)
201 {
202 case NT_ACCOUNT:
203 notify_dump_account(nc);
204 break;
205 case NT_COLOR:
206 notify_dump_color(nc);
207 break;
208 case NT_COMMAND:
209 notify_dump_command(nc);
210 break;
211 case NT_CONFIG:
212 notify_dump_config(nc);
213 break;
214 case NT_MVIEW:
215 notify_dump_mview(nc);
216 break;
217 case NT_EMAIL:
218 notify_dump_email(nc);
219 break;
220 case NT_GLOBAL:
221 notify_dump_global(nc);
222 break;
223 case NT_MAILBOX:
224 notify_dump_mailbox(nc);
225 break;
226 case NT_RESIZE:
227 case NT_TIMEOUT:
228 break; // no other data
229 case NT_WINDOW:
230 if (nc->event_subtype == NT_WINDOW_STATE)
231 notify_dump_window_state(nc);
232 else if (nc->event_subtype == NT_WINDOW_FOCUS)
233 notify_dump_window_focus(nc);
234 break;
235 default:
236 mutt_debug(LL_DEBUG1, " Event Type: %d\n", nc->event_type);
237 mutt_debug(LL_DEBUG1, " Event Sub-type: %d\n", nc->event_subtype);
238 mutt_debug(LL_DEBUG1, " Event Data: %p\n", nc->event_data);
239 break;
240 }
241
242 mutt_debug(LL_DEBUG1, " Global Data: %p\n", nc->global_data);
243
244 mutt_debug(LL_DEBUG5, "debug done\n");
245 return 0;
246}
