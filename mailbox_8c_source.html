90void mailbox_free(struct Mailbox **ptr)
91{
92 if (!ptr || !*ptr)
93 return;
94
95 struct Mailbox *m = *ptr;
96
97 const bool do_free = (m->opened == 0) && !m->visible;
98
99 mutt_debug(LL_DEBUG3, "%sfreeing %s mailbox %s with refcount %d\n",
100 do_free ? "" : "not ", m->visible ? "visible" : "invisible",
101 buf_string(&m->pathbuf), m->opened);
102
103 if (!do_free)
104 {
105 return;
106 }
107
108 mutt_debug(LL_NOTIFY, "NT_MAILBOX_DELETE: %s %p\n",
109 mailbox_get_type_name(m->type), (void *) m);
110 struct EventMailbox ev_m = { m };
111 notify_send(m->notify, NT_MAILBOX, NT_MAILBOX_DELETE, &ev_m);
112
113 mutt_debug(LL_NOTIFY, "NT_EMAIL_DELETE_ALL\n");
114 struct EventEmail ev_e = { 0, NULL };
115 notify_send(m->notify, NT_EMAIL, NT_EMAIL_DELETE_ALL, &ev_e);
116
117 for (size_t i = 0; i < m->email_max; i++)
118 email_free(&m->emails[i]);
119
120 if (m->mdata_free && m->mdata)
121 m->mdata_free(&m->mdata);
122
123 buf_dealloc(&m->pathbuf);
124 cs_subset_free(&m->sub);
125 FREE(&m->name);
126 FREE(&m->realpath);
127 FREE(&m->emails);
128 FREE(&m->v2r);
129 notify_free(&m->notify);
130 mailbox_gc_run();
131
132 /* The NT_MAILBOX_DELETE notification might already have caused *ptr to be NULL,
133 * so call free() on the m pointer */
134 *ptr = NULL;
135 FREE(&m);
136}

143struct Mailbox *mailbox_find(const char *path)
144{
145 if (!path)
146 return NULL;
147
148 struct stat st = { 0 };
149 struct stat st_tmp = { 0 };
150
151 if (stat(path, &st) != 0)
152 return NULL;
153
154 struct MailboxList ml = STAILQ_HEAD_INITIALIZER(ml);
155 neomutt_mailboxlist_get_all(&ml, NeoMutt, MUTT_MAILBOX_ANY);
156 struct MailboxNode *np = NULL;
157 struct Mailbox *m = NULL;
158 STAILQ_FOREACH(np, &ml, entries)
159 {
160 if ((stat(mailbox_path(np->mailbox), &st_tmp) == 0) &&
161 (st.st_dev == st_tmp.st_dev) && (st.st_ino == st_tmp.st_ino))
162 {
163 m = np->mailbox;
164 break;
165 }
166 }
167 neomutt_mailboxlist_clear(&ml);
168
169 return m;
170}

291void mailbox_gc_add(struct Email *e)
292{
293 assert(e);
294 if (GC.idx == mutt_array_size(GC.arr))
295 {
296 mailbox_gc_run();
297 }
298 GC.arr[GC.idx] = e;
299 GC.idx++;
300}
301
305void mailbox_gc_run(void)
306{
307 for (size_t i = 0; i < GC.idx; i++)
308 {
309 email_free(&GC.arr[i]);
310 }
311 GC.idx = 0;
312}
