197void browser_add_folder(const struct Menu *menu, struct BrowserState *state,
198 const char *name, const char *desc,
199 const struct stat *st, struct Mailbox *m, void *data)
200{
201 if ((!menu || state->is_mailbox_list) && m && !m->visible)
202 {
203 return;
204 }
205
206 struct FolderFile ff = { 0 };
207
208 if (st)
209 {
210 ff.mode = st->st_mode;
211 ff.mtime = st->st_mtime;
212 ff.size = st->st_size;
213 ff.gid = st->st_gid;
214 ff.uid = st->st_uid;
215 ff.nlink = st->st_nlink;
216 ff.local = true;
217 }
218 else
219 {
220 ff.local = false;
221 }
222
223 if (m)
224 {
225 ff.has_mailbox = true;
226 ff.gen = m->gen;
227 ff.has_new_mail = m->has_new;
228 ff.msg_count = m->msg_count;
229 ff.msg_unread = m->msg_unread;
230 ff.notify_user = m->notify_user;
231 ff.poll_new_mail = m->poll_new_mail;
232 }
233
234 ff.name = mutt_str_dup(name);
235 ff.desc = mutt_str_dup(desc ? desc : name);
236 ff.imap = false;
237 if (OptNews)
238 ff.nd = data;
239
240 ARRAY_ADD(&state->entry, ff);
241}

264int examine_directory(struct Mailbox *m, struct Menu *menu, struct BrowserState *state,
265 const char *dirname, const char *prefix)
266{
267 int rc = -1;
268 struct Buffer *buf = buf_pool_get();
269 if (OptNews)
270 {
271 struct NntpAccountData *adata = CurrentNewsSrv;
272
273 init_state(state);
274
275 const struct Regex *c_mask = cs_subset_regex(NeoMutt->sub, "mask");
276 for (unsigned int i = 0; i < adata->groups_num; i++)
277 {
278 struct NntpMboxData *mdata = adata->groups_list[i];
279 if (!mdata)
280 continue;
281 if (prefix && *prefix && !mutt_str_startswith(mdata->group, prefix))
282 continue;
283 if (!mutt_regex_match(c_mask, mdata->group))
284 {
285 continue;
286 }
287 browser_add_folder(menu, state, mdata->group, NULL, NULL, NULL, mdata);
288 }
289 }
290 else
291 {
292 struct stat st = { 0 };
293 DIR *dir = NULL;
294 struct dirent *de = NULL;
295
296 while (stat(dirname, &st) == -1)
297 {
298 if (errno == ENOENT)
299 {
300 /* The last used directory is deleted, try to use the parent dir. */
301 char *c = strrchr(dirname, '/');
302
303 if (c && (c > dirname))
304 {
305 *c = '\0';
306 continue;
307 }
308 }
309 mutt_perror("%s", dirname);
310 goto ed_out;
311 }
312
313 if (!S_ISDIR(st.st_mode))
314 {
315 mutt_error(_("%s is not a directory"), dirname);
316 goto ed_out;
317 }
318
319 if (m)
320 mutt_mailbox_check(m, MUTT_MAILBOX_CHECK_NO_FLAGS);
321
322 dir = mutt_file_opendir(dirname, MUTT_OPENDIR_NONE);
323 if (!dir)
324 {
325 mutt_perror("%s", dirname);
326 goto ed_out;
327 }
328
329 init_state(state);
330
331 struct MailboxArray ma = neomutt_mailboxes_get(NeoMutt, MUTT_MAILBOX_ANY);
332
333 const struct Regex *c_mask = cs_subset_regex(NeoMutt->sub, "mask");
334 while ((de = readdir(dir)))
335 {
336 if (mutt_str_equal(de->d_name, "."))
337 continue; /* we don't need . */
338
339 if (prefix && *prefix && !mutt_str_startswith(de->d_name, prefix))
340 {
341 continue;
342 }
343 if (!mutt_regex_match(c_mask, de->d_name))
344 {
345 continue;
346 }
347
348 buf_concat_path(buf, dirname, de->d_name);
349 if (lstat(buf_string(buf), &st) == -1)
350 continue;
351
352 /* No size for directories or symlinks */
353 if (S_ISDIR(st.st_mode) || S_ISLNK(st.st_mode))
354 st.st_size = 0;
355 else if (!S_ISREG(st.st_mode))
356 continue;
357
358 struct Mailbox **mp = NULL;
359 ARRAY_FOREACH(mp, &ma)
360 {
361 if (mutt_str_equal(buf_string(buf), mailbox_path(*mp)))
362 break;
363 }
364
365 struct Mailbox *m_match = mp ? *mp : NULL;
366 if (m_match && m && m->poll_new_mail && mutt_str_equal(m_match->realpath, m->realpath))
367 {
368 m_match->msg_count = m->msg_count;
369 m_match->msg_unread = m->msg_unread;
370 }
371 browser_add_folder(menu, state, de->d_name, NULL, &st, m_match, NULL);
372 }
373 ARRAY_FREE(&ma); // Clean up the ARRAY, but not the Mailboxes
374 closedir(dir);
375 }
376 browser_sort(state);
377 rc = 0;
378ed_out:
379 buf_pool_release(&buf);
380 return rc;
381}

391int examine_mailboxes(struct Mailbox *m, struct Menu *menu, struct BrowserState *state)
392{
393 struct stat st = { 0 };
394 struct Buffer *md = NULL;
395 struct Buffer *mailbox = NULL;
396
397 if (OptNews)
398 {
399 struct NntpAccountData *adata = CurrentNewsSrv;
400
401 init_state(state);
402
403 const bool c_show_only_unread = cs_subset_bool(NeoMutt->sub, "show_only_unread");
404 for (unsigned int i = 0; i < adata->groups_num; i++)
405 {
406 struct NntpMboxData *mdata = adata->groups_list[i];
407 if (mdata && (mdata->has_new_mail ||
408 (mdata->subscribed && (mdata->unread || !c_show_only_unread))))
409 {
410 browser_add_folder(menu, state, mdata->group, NULL, NULL, NULL, mdata);
411 }
412 }
413 }
414 else
415 {
416 init_state(state);
417
418 if (ARRAY_EMPTY(&NeoMutt->accounts))
419 return -1;
420
421 mailbox = buf_pool_get();
422 md = buf_pool_get();
423
424 mutt_mailbox_check(m, MUTT_MAILBOX_CHECK_NO_FLAGS);
425
426 struct MailboxArray ma = neomutt_mailboxes_get(NeoMutt, MUTT_MAILBOX_ANY);
427 const bool c_browser_abbreviate_mailboxes = cs_subset_bool(NeoMutt->sub, "browser_abbreviate_mailboxes");
428
429 struct Mailbox **mp = NULL;
430 ARRAY_FOREACH(mp, &ma)
431 {
432 struct Mailbox *m_match = *mp;
433
434 if (m && m->poll_new_mail && mutt_str_equal(m_match->realpath, m->realpath))
435 {
436 m_match->msg_count = m->msg_count;
437 m_match->msg_unread = m->msg_unread;
438 }
439
440 buf_strcpy(mailbox, mailbox_path(m_match));
441 if (c_browser_abbreviate_mailboxes)
442 buf_pretty_mailbox(mailbox);
443
444 switch (m_match->type)
445 {
446 case MUTT_IMAP:
447 case MUTT_POP:
448 browser_add_folder(menu, state, buf_string(mailbox), m_match->name,
449 NULL, m_match, NULL);
450 continue;
451 case MUTT_NOTMUCH:
452 case MUTT_NNTP:
453 browser_add_folder(menu, state, mailbox_path(m_match), m_match->name,
454 NULL, m_match, NULL);
455 continue;
456 default: /* Continue */
457 break;
458 }
459
460 if (lstat(mailbox_path(m_match), &st) == -1)
461 continue;
462
463 if ((!S_ISREG(st.st_mode)) && (!S_ISDIR(st.st_mode)) && (!S_ISLNK(st.st_mode)))
464 continue;
465
466 if (m_match->type == MUTT_MAILDIR)
467 {
468 struct stat st2 = { 0 };
469
470 buf_printf(md, "%s/new", mailbox_path(m_match));
471 if (stat(buf_string(md), &st) < 0)
472 st.st_mtime = 0;
473 buf_printf(md, "%s/cur", mailbox_path(m_match));
474 if (stat(buf_string(md), &st2) < 0)
475 st2.st_mtime = 0;
476 if (st2.st_mtime > st.st_mtime)
477 st.st_mtime = st2.st_mtime;
478 }
479
480 browser_add_folder(menu, state, buf_string(mailbox), m_match->name, &st, m_match, NULL);
481 }
482 ARRAY_FREE(&ma); // Clean up the ARRAY, but not the Mailboxes
483 }
484 browser_sort(state);
485
486 buf_pool_release(&mailbox);
487 buf_pool_release(&md);
488 return 0;
489}

582void init_menu(struct BrowserState *state, struct Menu *menu, struct Mailbox *m,
583 struct MuttWindow *sbar)
584{
585 char title[256] = { 0 };
586 menu->max = ARRAY_SIZE(&state->entry);
587
588 int index = menu_get_index(menu);
589 if (index >= menu->max)
590 menu_set_index(menu, menu->max - 1);
591 if (index < 0)
592 menu_set_index(menu, 0);
593 if (menu->top > index)
594 menu->top = 0;
595
596 menu->num_tagged = 0;
597
598 if (OptNews)
599 {
600 if (state->is_mailbox_list)
601 {
602 snprintf(title, sizeof(title), _("Subscribed newsgroups"));
603 }
604 else
605 {
606 snprintf(title, sizeof(title), _("Newsgroups on server [%s]"),
607 CurrentNewsSrv->conn->account.host);
608 }
609 }
610 else
611 {
612 if (state->is_mailbox_list)
613 {
614 snprintf(title, sizeof(title), _("Mailboxes [%d]"),
615 mutt_mailbox_check(m, MUTT_MAILBOX_CHECK_NO_FLAGS));
616 }
617 else
618 {
619 struct Buffer *path = buf_pool_get();
620 buf_copy(path, &LastDir);
621 buf_pretty_mailbox(path);
622 const struct Regex *c_mask = cs_subset_regex(NeoMutt->sub, "mask");
623 const bool c_imap_list_subscribed = cs_subset_bool(NeoMutt->sub, "imap_list_subscribed");
624 if (state->imap_browse && c_imap_list_subscribed)
625 {
626 snprintf(title, sizeof(title), _("Subscribed [%s], File mask: %s"),
627 buf_string(path), NONULL(c_mask ? c_mask->pattern : NULL));
628 }
629 else
630 {
631 snprintf(title, sizeof(title), _("Directory [%s], File mask: %s"),
632 buf_string(path), NONULL(c_mask ? c_mask->pattern : NULL));
633 }
634 buf_pool_release(&path);
635 }
636 }
637 sbar_set_title(sbar, title);
638
639 /* Browser tracking feature.
640 * The goal is to highlight the good directory if LastDir is the parent dir
641 * of LastDirBackup (this occurs mostly when one hit "../"). It should also work
642 * properly when the user is in examine_mailboxes-mode. */
643 if (mutt_str_startswith(buf_string(&LastDirBackup), buf_string(&LastDir)))
644 {
645 char target_dir[PATH_MAX] = { 0 };
646
647 /* Check what kind of dir LastDirBackup is. */
648 if (imap_path_probe(buf_string(&LastDirBackup), NULL) == MUTT_IMAP)
649 {
650 mutt_str_copy(target_dir, buf_string(&LastDirBackup), sizeof(target_dir));
651 imap_clean_path(target_dir, sizeof(target_dir));
652 }
653 else
654 {
655 mutt_str_copy(target_dir, strrchr(buf_string(&LastDirBackup), '/') + 1,
656 sizeof(target_dir));
657 }
658
659 /* If we get here, it means that LastDir is the parent directory of
660 * LastDirBackup. I.e., we're returning from a subdirectory, and we want
661 * to position the cursor on the directory we're returning from. */
662 bool matched = false;
663 struct FolderFile *ff = NULL;
664 ARRAY_FOREACH(ff, &state->entry)
665 {
666 if (mutt_str_equal(ff->name, target_dir))
667 {
668 menu_set_index(menu, ARRAY_FOREACH_IDX_ff);
669 matched = true;
670 break;
671 }
672 }
673 if (!matched)
674 browser_highlight_default(state, menu);
675 }
676 else
677 {
678 browser_highlight_default(state, menu);
679 }
680
681 menu_queue_redraw(menu, MENU_REDRAW_FULL);
682}

687static int file_tag(struct Menu *menu, int sel, int act)
688{
689 struct BrowserPrivateData *priv = menu->mdata;
690 struct BrowserEntryArray *entry = &priv->state.entry;
691 struct FolderFile *ff = ARRAY_GET(entry, sel);
692 if (S_ISDIR(ff->mode) ||
693 (S_ISLNK(ff->mode) && link_is_dir(buf_string(&LastDir), ff->name)))
694 {
695 mutt_error(_("Can't attach a directory"));
696 return 0;
697 }
698
699 bool ot = ff->tagged;
700 ff->tagged = ((act >= 0) ? act : !ff->tagged);
701
702 return ff->tagged - ot;
703}

708static int browser_config_observer(struct NotifyCallback *nc)
709{
710 if (nc->event_type != NT_CONFIG)
711 return 0;
712 if (!nc->global_data || !nc->event_data)
713 return -1;
714
715 struct EventConfig *ev_c = nc->event_data;
716
717 struct BrowserPrivateData *priv = nc->global_data;
718 struct Menu *menu = priv->menu;
719
720 if (mutt_str_equal(ev_c->name, "browser_sort_dirs_first"))
721 {
722 struct BrowserState *state = &priv->state;
723 browser_sort(state);
724 browser_highlight_default(state, menu);
725 }
726 else if (!mutt_str_equal(ev_c->name, "browser_abbreviate_mailboxes") &&
727 !mutt_str_equal(ev_c->name, "browser_sort") &&
728 !mutt_str_equal(ev_c->name, "date_format") &&
729 !mutt_str_equal(ev_c->name, "folder") &&
730 !mutt_str_equal(ev_c->name, "folder_format") &&
731 !mutt_str_equal(ev_c->name, "group_index_format") &&

733 {
734 return 0;
735 }
736
737 menu_queue_redraw(menu, MENU_REDRAW_FULL);
738 mutt_debug(LL_DEBUG5, "config done, request WA_RECALC, MENU_REDRAW_FULL\n");

748static int browser_mailbox_observer(struct NotifyCallback *nc)
749{
750 if (nc->event_type != NT_MAILBOX)

752 if (nc->event_subtype == NT_MAILBOX_DELETE)
753 return 0;
754 if (!nc->global_data || !nc->event_data)

756

761 {

763 struct Mailbox *m = ev_m->mailbox;

765 ARRAY_FOREACH(ff, &state->entry)
766 {
767 if (ff->gen != m->gen)
768 continue;
769
770 ff->has_new_mail = m->has_new;
771 ff->msg_count = m->msg_count;
772 ff->msg_unread = m->msg_unread;
773 ff->notify_user = m->notify_user;
774 ff->poll_new_mail = m->poll_new_mail;
775 mutt_str_replace(&ff->desc, m->name);
776 break;
777 }
778 }
779
780 menu_queue_redraw(priv->menu, MENU_REDRAW_FULL);
781 mutt_debug(LL_DEBUG5, "mailbox done, request WA_RECALC, MENU_REDRAW_FULL\n");
782
783 return 0;
784}

850void dlg_browser(struct Buffer *file, SelectFileFlags flags, struct Mailbox *m,
851 char ***files, int *numfiles)
852{
853 struct BrowserPrivateData *priv = browser_private_data_new();
854 priv->file = file;
855 priv->mailbox = m;
856 priv->files = files;
857 priv->numfiles = numfiles;
858 priv->multiple = (flags & MUTT_SEL_MULTI);
859 priv->folder = (flags & MUTT_SEL_FOLDER);
860 priv->state.is_mailbox_list = (flags & MUTT_SEL_MAILBOX) && priv->folder;
861 priv->last_selected_mailbox = -1;
862
863 init_lastdir();
864
865 if (OptNews)
866 {
867 if (buf_is_empty(file))
868 {
869 struct NntpAccountData *adata = CurrentNewsSrv;
870
871 /* default state for news reader mode is browse subscribed newsgroups */
872 priv->state.is_mailbox_list = false;
873 for (size_t i = 0; i < adata->groups_num; i++)
874 {
875 struct NntpMboxData *mdata = adata->groups_list[i];
876 if (mdata && mdata->subscribed)
877 {
878 priv->state.is_mailbox_list = true;
879 break;
880 }
881 }
882 }
883 else
884 {
885 buf_copy(priv->prefix, file);
886 }
887 }
888 else if (!buf_is_empty(file))
889 {
890 buf_expand_path(file);
891 if (imap_path_probe(buf_string(file), NULL) == MUTT_IMAP)
892 {
893 init_state(&priv->state);
894 priv->state.imap_browse = true;
895 if (imap_browse(buf_string(file), &priv->state) == 0)
896 {
897 buf_strcpy(&LastDir, priv->state.folder);
898 browser_sort(&priv->state);
899 }
900 }
901 else
902 {
903 int i = buf_len(file);
904 i--;
905 for (; (i > 0) && ((buf_string(file))[i] != '/'); i--)
906 {
907 ; // do nothing
908 }
909
910 if (i > 0)
911 {
912 if ((buf_string(file))[0] == '/')
913 {
914 buf_strcpy_n(&LastDir, buf_string(file), i);
915 }
916 else
917 {
918 mutt_path_getcwd(&LastDir);
919 buf_addch(&LastDir, '/');
920 buf_addstr_n(&LastDir, buf_string(file), i);
921 }
922 }
923 else
924 {
925 if ((buf_string(file))[0] == '/')
926 buf_strcpy(&LastDir, "/");
927 else
928 mutt_path_getcwd(&LastDir);
929 }
930
931 if ((i <= 0) && (buf_string(file)[0] != '/'))
932 buf_copy(priv->prefix, file);
933 else
934 buf_strcpy(priv->prefix, buf_string(file) + i + 1);
935 priv->kill_prefix = true;
936 }
937 }
938 else
939 {
940 if (priv->folder)
941 {
942 /* Whether we use the tracking feature of the browser depends
943 * on which sort method we chose to use. This variable is defined
944 * only to help readability of the code. */
945 bool browser_track = false;
946
947 const enum EmailSortType c_browser_sort = cs_subset_sort(NeoMutt->sub, "browser_sort");
948 switch (c_browser_sort & SORT_MASK)
949 {
950 case BROWSER_SORT_ALPHA:
951 case BROWSER_SORT_DESC:
952 case BROWSER_SORT_UNSORTED:
953 browser_track = true;
954 break;
955 }
956
957 /* We use mutt_browser_select_dir to initialize the two
958 * variables (LastDir, LastDirBackup) at the appropriate
959 * values.
960 *
961 * We do it only when LastDir is not set (first pass there)
962 * or when CurrentFolder and LastDirBackup are not the same.
963 * This code is executed only when we list files, not when
964 * we press up/down keys to navigate in a displayed list.
965 *
966 * We only do this when CurrentFolder has been set (ie, not
967 * when listing folders on startup with "neomutt -y").
968 *
969 * This tracker is only used when browser_track is true,
970 * meaning only with sort methods SUBJECT/DESC for now. */
971 if (CurrentFolder)
972 {
973 if (buf_is_empty(&LastDir))
974 {
975 /* If browsing in "local"-mode, than we chose to define LastDir to
976 * MailDir */
977 switch (mx_path_probe(CurrentFolder))
978 {
979 case MUTT_IMAP:
980 case MUTT_MAILDIR:
981 case MUTT_MBOX:
982 case MUTT_MH:
983 case MUTT_MMDF:
984 {
985 const char *const c_folder = cs_subset_string(NeoMutt->sub, "folder");
986 const char *const c_spool_file = cs_subset_string(NeoMutt->sub, "spool_file");
987 if (c_folder)
988 buf_strcpy(&LastDir, c_folder);
989 else if (c_spool_file)
990 mutt_browser_select_dir(c_spool_file);
991 break;
992 }
993 default:
994 mutt_browser_select_dir(CurrentFolder);
995 break;
996 }
997 }
998 else if (!mutt_str_equal(CurrentFolder, buf_string(&LastDirBackup)))
999 {
1000 mutt_browser_select_dir(CurrentFolder);
1001 }
1002 }
1003
1004 /* When browser tracking feature is disabled, clear LastDirBackup */
1005 if (!browser_track)
1006 buf_reset(&LastDirBackup);
1007 }
1008 else
1009 {
1010 mutt_path_getcwd(&LastDir);
1011 }
1012
1013 if (!priv->state.is_mailbox_list &&
1014 (imap_path_probe(buf_string(&LastDir), NULL) == MUTT_IMAP))
1015 {
1016 init_state(&priv->state);
1017 priv->state.imap_browse = true;
1018 imap_browse(buf_string(&LastDir), &priv->state);
1019 browser_sort(&priv->state);
1020 }
1021 else
1022 {
1023 size_t i = buf_len(&LastDir);
1024 while ((i > 0) && (buf_string(&LastDir)[--i] == '/'))
1025 LastDir.data[i] = '\0';
1026 buf_fix_dptr(&LastDir);
1027 if (buf_is_empty(&LastDir))
1028 mutt_path_getcwd(&LastDir);
1029 }
1030 }
1031
1032 buf_reset(file);
1033
1034 const struct Mapping *help_data = NULL;
1035
1036 if (OptNews)
1037 help_data = FolderNewsHelp;
1038 else
1039 help_data = FolderHelp;
1040
1041 struct SimpleDialogWindows sdw = simple_dialog_new(MENU_BROWSER, WT_DLG_BROWSER, help_data);
1042
1043 struct Menu *menu = sdw.menu;
1044 menu->make_entry = folder_make_entry;
1045 menu->search = select_file_search;
1046 menu->mdata = priv;
1047
1048 priv->menu = menu;
1049 if (priv->multiple)
1050 priv->menu->tag = file_tag;
1051
1052 priv->sbar = sdw.sbar;
1053
1054 struct MuttWindow *win_menu = priv->menu->win;
1055
1056 // NT_COLOR is handled by the SimpleDialog
1057 notify_observer_add(NeoMutt->sub->notify, NT_CONFIG, browser_config_observer, priv);
1058 notify_observer_add(win_menu->notify, NT_WINDOW, browser_window_observer, priv);
1059 notify_observer_add(NeoMutt->notify, NT_MAILBOX, browser_mailbox_observer, priv);
1060
1061 struct MuttWindow *old_focus = window_set_focus(priv->menu->win);
1062
1063 if (priv->state.is_mailbox_list)
1064 {
1065 examine_mailboxes(m, NULL, &priv->state);
1066 }
1067 else if (!priv->state.imap_browse)
1068 {
1069 // examine_directory() calls browser_add_folder() which needs the menu
1070 if (examine_directory(m, priv->menu, &priv->state, buf_string(&LastDir),
1071 buf_string(priv->prefix)) == -1)
1072 {
1073 goto bail;
1074 }
1075 }
1076
1077 init_menu(&priv->state, priv->menu, m, priv->sbar);
1078
1079 // ---------------------------------------------------------------------------
1080 // Event Loop
1081 int op = OP_NULL;
1082 do
1083 {
1084 menu_tagging_dispatcher(priv->menu->win, op);
1085 window_redraw(NULL);
1086
1087 op = km_dokey(MENU_BROWSER, GETCH_NO_FLAGS);
1088 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
1089 if (op < 0)
1090 continue;
1091 if (op == OP_NULL)
1092 {
1093 km_error_key(MENU_BROWSER);
1094 continue;
1095 }
1096 mutt_clear_error();
1097
1098 int rc = browser_function_dispatcher(sdw.dlg, op);
1099
1100 if (rc == FR_UNKNOWN)
1101 rc = menu_function_dispatcher(menu->win, op);
1102 if (rc == FR_UNKNOWN)
1103 rc = global_function_dispatcher(menu->win, op);
1104 } while (!priv->done);
1105 // ---------------------------------------------------------------------------
1106
1107bail:
1108 window_set_focus(old_focus);
1109 simple_dialog_free(&sdw.dlg);
1110 browser_private_data_free(&priv);
1111}
