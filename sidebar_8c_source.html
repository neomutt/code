331 static int cb_qsort_sbe(const void *a, const void *b)
332 {
333  const struct SbEntry *sbe1 = *(struct SbEntry const *const *) a;
334  const struct SbEntry *sbe2 = *(struct SbEntry const *const *) b;
335  const struct Mailbox *m1 = sbe1->mailbox;
336  const struct Mailbox *m2 = sbe2->mailbox;
337 
338  int rc = 0;
339 
340  switch ((C_SidebarSortMethod & SORT_MASK))
341  {
342  case SORT_COUNT:
343  if (m2->msg_count == m1->msg_count)
344  rc = mutt_str_strcoll(mailbox_path(m1), mailbox_path(m2));
345  else
346  rc = (m2->msg_count - m1->msg_count);
347  break;
348  case SORT_UNREAD:
349  if (m2->msg_unread == m1->msg_unread)
350  rc = mutt_str_strcoll(mailbox_path(m1), mailbox_path(m2));
351  else
352  rc = (m2->msg_unread - m1->msg_unread);
353  break;
354  case SORT_DESC:
355  rc = mutt_str_strcmp(m1->name, m2->name);
356  break;
357  case SORT_FLAGGED:
358  if (m2->msg_flagged == m1->msg_flagged)
359  rc = mutt_str_strcoll(mailbox_path(m1), mailbox_path(m2));
360  else
361  rc = (m2->msg_flagged - m1->msg_flagged);
362  break;
363  case SORT_PATH:
364  {
365  rc = mutt_inbox_cmp(mailbox_path(m1), mailbox_path(m2));
366  if (rc == 0)
367  rc = mutt_str_strcoll(mailbox_path(m1), mailbox_path(m2));
368  break;
369  }
370  }
371 
372  if (C_SidebarSortMethod & SORT_REVERSE)
373  rc = -rc;
374 
375  return rc;
376 }

692 static bool prepare_sidebar(int page_size)
693 {
694  if (!EntryCount || (page_size <= 0))
695  return false;
696 
697  const struct SbEntry *opn_entry = (OpnIndex >= 0) ? Entries[OpnIndex] : NULL;
698  const struct SbEntry *hil_entry = (HilIndex >= 0) ? Entries[HilIndex] : NULL;
699 
700  update_entries_visibility();
701  sort_entries();
702 
703  for (int i = 0; i < EntryCount; i++)
704  {
705  if (opn_entry == Entries[i])
706  OpnIndex = i;
707  if (hil_entry == Entries[i])
708  HilIndex = i;
709  }
710 
711  if ((HilIndex < 0) || Entries[HilIndex]->is_hidden || (C_SidebarSortMethod != PreviousSort))
712  {
713  if (OpnIndex >= 0)
714  HilIndex = OpnIndex;
715  else
716  {
717  HilIndex = 0;
718  if (Entries[HilIndex]->is_hidden)
719  select_next();
720  }
721  }
722 
723  /* Set the Top and Bottom to frame the HilIndex in groups of page_size */
724 
725  /* If C_SidebarNewMailOnly or C_SidebarNonEmptyMailboxOnly is set, some entries
726  * may be hidden so we need to scan for the framing interval */
727  if (C_SidebarNewMailOnly || C_SidebarNonEmptyMailboxOnly)
728  {
729  TopIndex = -1;
730  BotIndex = -1;
731  while (BotIndex < HilIndex)
732  {
733  TopIndex = BotIndex + 1;
734  int page_entries = 0;
735  while (page_entries < page_size)
736  {
737  BotIndex++;
738  if (BotIndex >= EntryCount)
739  break;
740  if (!Entries[BotIndex]->is_hidden)
741  page_entries++;
742  }
743  }
744  }
745  /* Otherwise we can just calculate the interval */
746  else
747  {
748  TopIndex = (HilIndex / page_size) * page_size;
749  BotIndex = TopIndex + page_size - 1;
750  }
751 
752  if (BotIndex > (EntryCount - 1))
753  BotIndex = EntryCount - 1;
754 
755  PreviousSort = C_SidebarSortMethod;
756  return true;
757 }

814  delim_len = 1;
815  break;
816  }

819  }

827

829  {

831 
832  switch (altchar)
833  {
834  case SB_DIV_USER:
835  mutt_window_addstr(NONULL(C_SidebarDividerChar));
836  break;
837  case SB_DIV_ASCII:
838  mutt_window_addch('|');
839  break;
840  case SB_DIV_UTF8:
841  mutt_window_addch(ACS_VLINE);
842  break;
843  }
844  }
845 
846  return delim_len;
