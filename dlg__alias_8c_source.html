123static const char *alias_format_str(char *buf, size_t buflen, size_t col, int cols,
124 char op, const char *src, const char *prec,
125 const char *if_str, const char *else_str,
126 intptr_t data, MuttFormatFlags flags)
127{
128 char tmp[1024];
129 struct AliasView *av = (struct AliasView *) data;
130 struct Alias *alias = av->alias;
131
132 switch (op)
133 {
134 case 'a':
135 mutt_format_s(buf, buflen, prec, alias->name);
136 break;
137 case 'c':
138 mutt_format_s(buf, buflen, prec, alias->comment);
139 break;
140 case 'f':
141 snprintf(tmp, sizeof(tmp), "%%%ss", prec);
142 snprintf(buf, buflen, tmp, av->is_deleted ? "D" : " ");
143 break;
144 case 'n':
145 snprintf(tmp, sizeof(tmp), "%%%sd", prec);
146 snprintf(buf, buflen, tmp, av->num + 1);
147 break;
148 case 'r':
149 {
150 struct Buffer *tmpbuf = buf_pool_get();
151 mutt_addrlist_write(&alias->addr, tmpbuf, true);
152 mutt_str_copy(tmp, buf_string(tmpbuf), sizeof(tmp));
153 buf_pool_release(&tmpbuf);
154 mutt_format_s(buf, buflen, prec, tmp);
155 break;
156 }
157 case 't':
158 buf[0] = av->is_tagged ? '*' : ' ';
159 buf[1] = '\0';
160 break;
161 case 'Y':
162 {
163 struct Buffer *tags = buf_pool_get();
164 alias_tags_to_buffer(&av->alias->tags, tags);
165 mutt_format_s(buf, buflen, prec, buf_string(tags));
166 buf_pool_release(&tags);
167 break;
168 }
169 }
170
171 return src;
172}

210static int alias_alias_observer(struct NotifyCallback *nc)
211{
212 if (nc->event_type != NT_ALIAS)
213 return 0;
214 if (!nc->global_data || !nc->event_data)
215 return -1;
216
217 struct EventAlias *ev_a = nc->event_data;
218 struct Menu *menu = nc->global_data;
219 struct AliasMenuData *mdata = menu->mdata;
220 struct Alias *alias = ev_a->alias;
221
222 if (nc->event_subtype == NT_ALIAS_ADD)
223 {
224 alias_array_alias_add(&mdata->ava, alias);
225
226 if (alias_array_count_visible(&mdata->ava) != ARRAY_SIZE(&mdata->ava))
227 {
228 mutt_pattern_alias_func(NULL, mdata, menu);
229 }
230 }
231 else if (nc->event_subtype == NT_ALIAS_DELETE)
232 {
233 alias_array_alias_delete(&mdata->ava, alias);
234
235 int vcount = alias_array_count_visible(&mdata->ava);
236 int index = menu_get_index(menu);
237 if ((index > (vcount - 1)) && (index > 0))
238 menu_set_index(menu, index - 1);
239 }
240
241 alias_array_sort(&mdata->ava, mdata->sub);
242
243 menu->max = alias_array_count_visible(&mdata->ava);
244 menu_queue_redraw(menu, MENU_REDRAW_FULL);
245 mutt_debug(LL_DEBUG5, "alias done, request WA_RECALC, MENU_REDRAW_FULL\n");
246
247 return 0;
248}

394int alias_complete(struct Buffer *buf, struct ConfigSubset *sub)
395{
396 struct Alias *np = NULL;
397 char bestname[8192] = { 0 };
398
399 struct AliasMenuData mdata = { ARRAY_HEAD_INITIALIZER, NULL, sub };
400 mdata.limit = buf_strdup(buf);
401 mdata.search_state = search_state_new();
402
403 if (buf_at(buf, 0) != '\0')
404 {
405 TAILQ_FOREACH(np, &Aliases, entries)
406 {
407 if (np->name && mutt_strn_equal(np->name, buf_string(buf), buf_len(buf)))
408 {
409 if (bestname[0] == '\0') /* init */
410 {
411 mutt_str_copy(bestname, np->name,
412 MIN(mutt_str_len(np->name) + 1, sizeof(bestname)));
413 }
414 else
415 {
416 int i;
417 for (i = 0; np->name[i] && (np->name[i] == bestname[i]); i++)
418 ; // do nothing
419
420 bestname[i] = '\0';
421 }
422 }
423 }
424
425 if (bestname[0] == '\0')
426 {
427 // Create a View Array of all the Aliases
428 FREE(&mdata.limit);
429 TAILQ_FOREACH(np, &Aliases, entries)
430 {
431 alias_array_alias_add(&mdata.ava, np);
432 }
433 }
434 else
435 {
436 /* fake the pattern for menu title */
437 char *mtitle = NULL;
438 mutt_str_asprintf(&mtitle, "~f ^%s", buf_string(buf));
439 FREE(&mdata.limit);
440 mdata.limit = mtitle;
441
442 if (!mutt_str_equal(bestname, buf_string(buf)))
443 {
444 /* we are adding something to the completion */
445 buf_strcpy_n(buf, bestname, mutt_str_len(bestname) + 1);
446 FREE(&mdata.limit);
447 search_state_free(&mdata.search_state);
448 return 1;
449 }
450
451 /* build alias list and show it */
452 TAILQ_FOREACH(np, &Aliases, entries)
453 {
454 int aasize = alias_array_alias_add(&mdata.ava, np);
455
456 struct AliasView *av = ARRAY_GET(&mdata.ava, aasize - 1);
457
458 if (np->name && !mutt_strn_equal(np->name, buf_string(buf), buf_len(buf)))
459 {
460 av->is_visible = false;
461 }
462 }
463 }
464 }
465
466 if (ARRAY_EMPTY(&mdata.ava))
467 {
468 TAILQ_FOREACH(np, &Aliases, entries)
469 {
470 alias_array_alias_add(&mdata.ava, np);
471 }
472
473 mutt_pattern_alias_func(NULL, &mdata, NULL);
474 }
475
476 if (!dlg_alias(NULL, &mdata))
477 goto done;
478
479 buf_reset(buf);
480
481 // Extract the selected aliases
482 struct Buffer *tmpbuf = buf_pool_get();
483 struct AliasView *avp = NULL;
484 ARRAY_FOREACH(avp, &mdata.ava)
485 {
486 if (!avp->is_tagged)
487 continue;
488
489 mutt_addrlist_write(&avp->alias->addr, tmpbuf, true);
490 buf_addstr(tmpbuf, ", ");
491 }
492 buf_copy(buf, tmpbuf);
493 buf_pool_release(&tmpbuf);
494
495done:
496 // Process any deleted aliases
497 ARRAY_FOREACH(avp, &mdata.ava)
498 {
499 if (!avp->is_deleted)
500 continue;
501
502 TAILQ_REMOVE(&Aliases, avp->alias, entries);
503 alias_free(&avp->alias);
504 }
505
506 ARRAY_FREE(&mdata.ava);
507 FREE(&mdata.limit);
508 FREE(&mdata.title);
509 search_state_free(&mdata.search_state);
510
511 return 0;

520{

531
532 if (!dlg_alias(NULL, &mdata))

535 // Prepare the "To:" field of a new email

549 mutt_addrlist_clear(&al_copy);

551 }
552
553 mutt_send_message(SEND_REVIEW_TO, e, NULL, m, NULL, sub);
554
555done:
556 // Process any deleted aliases
557 ARRAY_FOREACH(avp, &mdata.ava)
558 {
559 if (avp->is_deleted)

561 TAILQ_REMOVE(&Aliases, avp->alias, entries);
562 alias_free(&avp->alias);

564 }
565
