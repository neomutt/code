166enum CommandResult mutt_parse_hook(struct Buffer *buf, struct Buffer *s,
167 intptr_t data, struct Buffer *err)
168{
169 struct Hook *hook = NULL;
170 int rc = MUTT_CMD_ERROR;
171 bool pat_not = false;
172 bool use_regex = true;
173 regex_t *rx = NULL;
174 struct PatternList *pat = NULL;
175 const bool folder_or_mbox = (data & (MUTT_FOLDER_HOOK | MUTT_MBOX_HOOK));
176
177 struct Buffer *cmd = buf_pool_get();
178 struct Buffer *pattern = buf_pool_get();
179
180 if (~data & MUTT_GLOBAL_HOOK) /* NOT a global hook */
181 {
182 if (*s->dptr == '!')
183 {
184 s->dptr++;
185 SKIPWS(s->dptr);
186 pat_not = true;
187 }
188
189 parse_extract_token(pattern, s, TOKEN_NO_FLAGS);
190 if (folder_or_mbox && mutt_str_equal(buf_string(pattern), "-noregex"))
191 {
192 use_regex = false;
193 if (!MoreArgs(s))
194 {
195 buf_printf(err, _("%s: too few arguments"), buf->data);
196 rc = MUTT_CMD_WARNING;
197 goto cleanup;
198 }
199 parse_extract_token(pattern, s, TOKEN_NO_FLAGS);
200 }
201
202 if (!MoreArgs(s))
203 {
204 buf_printf(err, _("%s: too few arguments"), buf->data);
205 rc = MUTT_CMD_WARNING;
206 goto cleanup;
207 }
208 }
209
210 parse_extract_token(cmd, s,
211 (data & (MUTT_FOLDER_HOOK | MUTT_SEND_HOOK | MUTT_SEND2_HOOK |
212 MUTT_ACCOUNT_HOOK | MUTT_REPLY_HOOK)) ?
213 TOKEN_SPACE :
214 TOKEN_NO_FLAGS);
215
216 if (buf_is_empty(cmd))
217 {
218 buf_printf(err, _("%s: too few arguments"), buf->data);
219 rc = MUTT_CMD_WARNING;
220 goto cleanup;
221 }
222
223 if (MoreArgs(s))
224 {
225 buf_printf(err, _("%s: too many arguments"), buf->data);
226 rc = MUTT_CMD_WARNING;
227 goto cleanup;
228 }
229
230 const char *const c_default_hook = cs_subset_string(NeoMutt->sub, "default_hook");
231 if (folder_or_mbox)
232 {
233 /* Accidentally using the ^ mailbox shortcut in the .neomuttrc is a
234 * common mistake */
235 if ((buf_at(pattern, 0) == '^') && !CurrentFolder)
236 {
237 buf_strcpy(err, _("current mailbox shortcut '^' is unset"));
238 goto cleanup;
239 }
240
241 struct Buffer *tmp = buf_pool_get();
242 buf_copy(tmp, pattern);
243 buf_expand_path_regex(tmp, use_regex);
244
245 /* Check for other mailbox shortcuts that expand to the empty string.
246 * This is likely a mistake too */
247 if (buf_is_empty(tmp) && !buf_is_empty(pattern))
248 {
249 buf_strcpy(err, _("mailbox shortcut expanded to empty regex"));
250 buf_pool_release(&tmp);
251 goto cleanup;
252 }
253
254 if (use_regex)
255 {
256 buf_copy(pattern, tmp);
257 }
258 else
259 {
260 mutt_file_sanitize_regex(pattern, buf_string(tmp));
261 }
262 buf_pool_release(&tmp);
263 }
264 else if (data & (MUTT_APPEND_HOOK | MUTT_OPEN_HOOK | MUTT_CLOSE_HOOK))
265 {
266 if (mutt_comp_valid_command(buf_string(cmd)) == 0)
267 {
268 buf_strcpy(err, _("badly formatted command string"));
269 goto cleanup;
270 }
271 }
272 else if (c_default_hook && (~data & MUTT_GLOBAL_HOOK) &&
273 !(data & (MUTT_ACCOUNT_HOOK)) && (!WithCrypto || !(data & MUTT_CRYPT_HOOK)))
274 {
275 /* At this stage only these hooks remain:
276 * fcc-, fcc-save-, index-format-, message-, reply-, save-, send- and send2-hook
277 * If given a plain string, or regex, we expand it using $default_hook. */
278 mutt_check_simple(pattern, c_default_hook);
279 }
280
281 if (data & (MUTT_MBOX_HOOK | MUTT_SAVE_HOOK | MUTT_FCC_HOOK))
282 {
283 buf_expand_path(cmd);
284 }
285
286 /* check to make sure that a matching hook doesn't already exist */
287 TAILQ_FOREACH(hook, &Hooks, entries)
288 {
289 if (data & MUTT_GLOBAL_HOOK)
290 {
291 /* Ignore duplicate global hooks */
292 if (mutt_str_equal(hook->command, buf_string(cmd)))
293 {
294 rc = MUTT_CMD_SUCCESS;
295 goto cleanup;
296 }
297 }
298 else if ((hook->type == data) && (hook->regex.pat_not == pat_not) &&
299 mutt_str_equal(buf_string(pattern), hook->regex.pattern))
300 {
301 if (data & (MUTT_FOLDER_HOOK | MUTT_SEND_HOOK | MUTT_SEND2_HOOK | MUTT_MESSAGE_HOOK |
302 MUTT_ACCOUNT_HOOK | MUTT_REPLY_HOOK | MUTT_CRYPT_HOOK |
303 MUTT_TIMEOUT_HOOK | MUTT_STARTUP_HOOK | MUTT_SHUTDOWN_HOOK))
304 {
305 /* these hooks allow multiple commands with the same
306 * pattern, so if we've already seen this pattern/command pair, just
307 * ignore it instead of creating a duplicate */
308 if (mutt_str_equal(hook->command, buf_string(cmd)))
309 {
310 rc = MUTT_CMD_SUCCESS;
311 goto cleanup;
312 }
313 }
314 else
315 {
316 /* other hooks only allow one command per pattern, so update the
317 * entry with the new command. this currently does not change the
318 * order of execution of the hooks, which i think is desirable since
319 * a common action to perform is to change the default (.) entry
320 * based upon some other information. */
321 FREE(&hook->command);
322 hook->command = buf_strdup(cmd);
323 FREE(&hook->source_file);
324 hook->source_file = mutt_get_sourced_cwd();
325
326 if (data & (MUTT_IDXFMTHOOK | MUTT_MBOX_HOOK | MUTT_SAVE_HOOK | MUTT_FCC_HOOK))
327 {
328 expando_free(&hook->expando);
329 hook->expando = expando_parse(buf_string(cmd), IndexFormatDef, err);
330 }
331
332 rc = MUTT_CMD_SUCCESS;
333 goto cleanup;
334 }
335 }
336 }
337
338 if (data & (MUTT_SEND_HOOK | MUTT_SEND2_HOOK | MUTT_SAVE_HOOK |
339 MUTT_FCC_HOOK | MUTT_MESSAGE_HOOK | MUTT_REPLY_HOOK))
340 {
341 PatternCompFlags comp_flags;
342
343 if (data & (MUTT_SEND2_HOOK))
344 comp_flags = MUTT_PC_SEND_MODE_SEARCH;
345 else if (data & (MUTT_SEND_HOOK | MUTT_FCC_HOOK))
346 comp_flags = MUTT_PC_NO_FLAGS;
347 else
348 comp_flags = MUTT_PC_FULL_MSG;
349
350 struct MailboxView *mv_cur = get_current_mailbox_view();
351 pat = mutt_pattern_comp(mv_cur, buf_string(pattern), comp_flags, err);
352 if (!pat)
353 goto cleanup;
354 }
355 else if (~data & MUTT_GLOBAL_HOOK) /* NOT a global hook */
356 {
357 /* Hooks not allowing full patterns: Check syntax of regex */
358 rx = MUTT_MEM_CALLOC(1, regex_t);
359 int rc2 = REG_COMP(rx, buf_string(pattern), ((data & MUTT_CRYPT_HOOK) ? REG_ICASE : 0));
360 if (rc2 != 0)
361 {
362 regerror(rc2, rx, err->data, err->dsize);
363 FREE(&rx);
364 goto cleanup;
365 }
366 }
367
368 struct Expando *exp = NULL;
369 if (data & (MUTT_IDXFMTHOOK | MUTT_MBOX_HOOK | MUTT_SAVE_HOOK | MUTT_FCC_HOOK))
370 exp = expando_parse(buf_string(cmd), IndexFormatDef, err);
371
372 hook = hook_new();
373 hook->type = data;
374 hook->command = buf_strdup(cmd);
375 hook->source_file = mutt_get_sourced_cwd();
376 hook->pattern = pat;
377 hook->regex.pattern = buf_strdup(pattern);
378 hook->regex.regex = rx;
379 hook->regex.pat_not = pat_not;
380 hook->expando = exp;
381
382 TAILQ_INSERT_TAIL(&Hooks, hook, entries);
383 rc = MUTT_CMD_SUCCESS;
384
385cleanup:
386 buf_pool_release(&cmd);
387 buf_pool_release(&pattern);
388 return rc;
389}

441static enum CommandResult mutt_parse_idxfmt_hook(struct Buffer *buf, struct Buffer *s,
442 intptr_t data, struct Buffer *err)
443{
444 enum CommandResult rc = MUTT_CMD_ERROR;
445 bool pat_not = false;
446
447 struct Buffer *name = buf_pool_get();
448 struct Buffer *pattern = buf_pool_get();
449 struct Buffer *fmtstring = buf_pool_get();
450 struct Expando *exp = NULL;
451
452 if (!IdxFmtHooks)
453 {
454 IdxFmtHooks = mutt_hash_new(30, MUTT_HASH_STRDUP_KEYS);
455 mutt_hash_set_destructor(IdxFmtHooks, idxfmt_hashelem_free, 0);
456 }
457
458 if (!MoreArgs(s))
459 {
460 buf_printf(err, _("%s: too few arguments"), buf->data);
461 goto out;
462 }
463 parse_extract_token(name, s, TOKEN_NO_FLAGS);
464 struct HookList *hl = mutt_hash_find(IdxFmtHooks, buf_string(name));
465
466 if (*s->dptr == '!')
467 {
468 s->dptr++;
469 SKIPWS(s->dptr);
470 pat_not = true;
471 }
472 parse_extract_token(pattern, s, TOKEN_NO_FLAGS);
473
474 if (!MoreArgs(s))
475 {
476 buf_printf(err, _("%s: too few arguments"), buf->data);
477 goto out;
478 }
479 parse_extract_token(fmtstring, s, TOKEN_NO_FLAGS);
480
481 exp = expando_parse(buf_string(fmtstring), IndexFormatDef, err);
482 if (!exp)
483 goto out;
484
485 if (MoreArgs(s))
486 {
487 buf_printf(err, _("%s: too many arguments"), buf->data);
488 goto out;
489 }
490
491 const char *const c_default_hook = cs_subset_string(NeoMutt->sub, "default_hook");
492 if (c_default_hook)
493 mutt_check_simple(pattern, c_default_hook);
494
495 /* check to make sure that a matching hook doesn't already exist */
496 struct Hook *hook = NULL;
497 if (hl)
498 {
499 TAILQ_FOREACH(hook, hl, entries)
500 {
501 if ((hook->regex.pat_not == pat_not) &&
502 mutt_str_equal(buf_string(pattern), hook->regex.pattern))
503 {
504 expando_free(&hook->expando);
505 hook->expando = exp;
506 exp = NULL;
507 rc = MUTT_CMD_SUCCESS;
508 goto out;
509 }
510 }
511 }
512
513 /* MUTT_PC_PATTERN_DYNAMIC sets so that date ranges are regenerated during
514 * matching. This of course is slower, but index-format-hook is commonly
515 * used for date ranges, and they need to be evaluated relative to "now", not
516 * the hook compilation time. */
517 struct MailboxView *mv_cur = get_current_mailbox_view();
518 struct PatternList *pat = mutt_pattern_comp(mv_cur, buf_string(pattern),
519 MUTT_PC_FULL_MSG | MUTT_PC_PATTERN_DYNAMIC,
520 err);
521 if (!pat)
522 goto out;
523
524 hook = hook_new();
525 hook->type = MUTT_IDXFMTHOOK;
526 hook->command = NULL;
527 hook->source_file = mutt_get_sourced_cwd();
528 hook->pattern = pat;
529 hook->regex.pattern = buf_strdup(pattern);
530 hook->regex.regex = NULL;
531 hook->regex.pat_not = pat_not;
532 hook->expando = exp;
533 exp = NULL;
534
535 if (!hl)
536 {
537 hl = MUTT_MEM_CALLOC(1, struct HookList);
538 TAILQ_INIT(hl);
539 mutt_hash_insert(IdxFmtHooks, buf_string(name), hl);
540 }
541
542 TAILQ_INSERT_TAIL(hl, hook, entries);
543 rc = MUTT_CMD_SUCCESS;
544
545out:
546 buf_pool_release(&name);
547 buf_pool_release(&pattern);
548 buf_pool_release(&fmtstring);
549 expando_free(&exp);
550
551 return rc;
552}

579static enum CommandResult mutt_parse_unhook(struct Buffer *buf, struct Buffer *s,
580 intptr_t data, struct Buffer *err)
581{
582 while (MoreArgs(s))
583 {
584 parse_extract_token(buf, s, TOKEN_NO_FLAGS);
585 if (mutt_str_equal("*", buf->data))
586 {
587 if (CurrentHookType != TOKEN_NO_FLAGS)
588 {
589 buf_addstr(err, _("unhook: Can't do unhook * from within a hook"));
590 return MUTT_CMD_WARNING;
591 }
592 mutt_delete_hooks(MUTT_HOOK_NO_FLAGS);
593 delete_idxfmt_hooks();
594 mutt_ch_lookup_remove();
595 }
596 else
597 {
598 HookFlags type = mutt_get_hook_type(buf->data);
599
600 if (type == MUTT_HOOK_NO_FLAGS)
601 {
602 buf_printf(err, _("unhook: unknown hook type: %s"), buf->data);
603 return MUTT_CMD_ERROR;
604 }
605 if (type & (MUTT_CHARSET_HOOK | MUTT_ICONV_HOOK))
606 {
607 mutt_ch_lookup_remove();
608 return MUTT_CMD_SUCCESS;
609 }
610 if (CurrentHookType == type)
611 {
612 buf_printf(err, _("unhook: Can't delete a %s from within a %s"),
613 buf->data, buf->data);
614 return MUTT_CMD_WARNING;
615 }
616 if (type == MUTT_IDXFMTHOOK)
617 delete_idxfmt_hooks();
618 else
619 mutt_delete_hooks(type);
620 }
621 }
622 return MUTT_CMD_SUCCESS;
623}

631{
632 if (!path && !desc)
633 return;

637
638 CurrentHookType = MUTT_FOLDER_HOOK;
639
640 TAILQ_FOREACH(hook, &Hooks, entries)
641 {
642 if (!hook->command)
643 continue;
644
645 if (!(hook->type & MUTT_FOLDER_HOOK))
646 continue;
647
648 const char *match = NULL;

652 match = desc;

654 if (match)
655 {
656 mutt_debug(LL_DEBUG1, "folder-hook '%s' matches '%s'\n", hook->regex.pattern, match);

659 {
660 mutt_error("%s", buf_string(err));
661 break;
662 }
663 }
664 }
665 buf_pool_release(&err);
666
667 CurrentHookType = MUTT_HOOK_NO_FLAGS;

699void mutt_message_hook(struct Mailbox *m, struct Email *e, HookFlags type)
700{
701 struct Hook *hook = NULL;
702 struct PatternCache cache = { 0 };
703 struct Buffer *err = buf_pool_get();
704
705 CurrentHookType = type;
706
707 TAILQ_FOREACH(hook, &Hooks, entries)
708 {
709 if (!hook->command)
710 continue;
711
712 if (hook->type & type)
713 {
714 if ((mutt_pattern_exec(SLIST_FIRST(hook->pattern), 0, m, e, &cache) > 0) ^
715 hook->regex.pat_not)
716 {
717 if (parse_rc_line_cwd(hook->command, hook->source_file, err) == MUTT_CMD_ERROR)
718 {
719 mutt_error("%s", buf_string(err));
720 CurrentHookType = MUTT_HOOK_NO_FLAGS;
721 buf_pool_release(&err);
722
723 return;
724 }
725 /* Executing arbitrary commands could affect the pattern results,
726 * so the cache has to be wiped */
727 memset(&cache, 0, sizeof(cache));
728 }
729 }
730 }
731 buf_pool_release(&err);
732
733 CurrentHookType = MUTT_HOOK_NO_FLAGS;
734}

816{

825 const bool c_force_name = cs_subset_bool(NeoMutt->sub, "force_name");

827 if ((c_save_name || c_force_name) && (to || cc || bcc))

834 buf_pool_release(&buf);

836 buf_strcpy(path, c_record);
837 }
838 else

886void mutt_account_hook(const char *url)
887{
888 /* parsing commands with URLs in an account hook can cause a recursive
889 * call. We just skip processing if this occurs. Typically such commands
890 * belong in a folder-hook -- perhaps we should warn the user. */
891 static bool inhook = false;
892 if (inhook)
893 return;
894
895 struct Hook *hook = NULL;
896 struct Buffer *err = buf_pool_get();
897
898 TAILQ_FOREACH(hook, &Hooks, entries)
899 {
900 if (!(hook->command && (hook->type & MUTT_ACCOUNT_HOOK)))
901 continue;
902
903 if (mutt_regex_match(&hook->regex, url))
904 {
905 inhook = true;
906 mutt_debug(LL_DEBUG1, "account-hook '%s' matches '%s'\n", hook->regex.pattern, url);
907 mutt_debug(LL_DEBUG5, " %s\n", hook->command);
908
909 if (parse_rc_line_cwd(hook->command, hook->source_file, err) == MUTT_CMD_ERROR)
910 {
911 mutt_error("%s", buf_string(err));
912 buf_pool_release(&err);
913
914 inhook = false;
915 goto done;
916 }
917
918 inhook = false;
919 }
920 }
921done:
922 buf_pool_release(&err);
923}

991{
992 if (!IdxFmtHooks)

996 if (!hl)
997 return NULL;

999 CurrentHookType = MUTT_IDXFMTHOOK;

1003 struct Hook *hook = NULL;

1009 {
1010 exp = hook->expando;
1011 break;
1012 }
1013 }
1014
1015 CurrentHookType = MUTT_HOOK_NO_FLAGS;
1016
1017 return exp;
