51static void curses_signal_handler(int sig)
52{
53 int save_errno = errno;
54 enum MuttCursorState old_cursor = MUTT_CURSOR_VISIBLE;
55
56 switch (sig)
57 {
58 case SIGTSTP: /* user requested a suspend */
59 {
60 const bool c_suspend = cs_subset_bool(NeoMutt->sub, "suspend");
61 if (!c_suspend)
62 break;
63 IsEndwin = isendwin();
64 old_cursor = mutt_curses_set_cursor(MUTT_CURSOR_VISIBLE);
65 if (!IsEndwin)
66 endwin();
67 kill(0, SIGSTOP);
68 }
69 FALLTHROUGH;
70
71 case SIGCONT:
72 if (!IsEndwin)
73 refresh();
74 mutt_curses_set_cursor(old_cursor);
75 /* We don't receive SIGWINCH when suspended; however, no harm is done by
76 * just assuming we received one, and triggering the 'resize' anyway. */
77 SigWinch = true;
78 break;
79
80 case SIGWINCH:
81 SigWinch = true;
82 break;
83
84 case SIGINT:
85 SigInt = true;
86 break;
87 }
88 errno = save_errno;
89}

107static void curses_segv_handler(int sig)
108{
109 mutt_curses_set_cursor(MUTT_CURSOR_VISIBLE);
110 endwin(); /* just to be safe */
111#ifdef USE_DEBUG_BACKTRACE
112 show_backtrace();
113#endif
114#ifdef USE_DEBUG_GRAPHVIZ
115 dump_graphviz("segfault", NULL);
116#endif
117
118 // Chain the old SEGV handler, it could have been set by ASAN
119 if (OldSegvHandler)
120 OldSegvHandler(sig);
121
122 struct sigaction act = { 0 };
123 sigemptyset(&act.sa_mask);
124 act.sa_flags = 0;
125 act.sa_handler = SIG_DFL;
126 sigaction(sig, &act, NULL);
127
128 // Re-raise the signal to give outside handlers a chance to deal with it
129 raise(sig);
130}
