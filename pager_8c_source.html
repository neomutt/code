94 static int config_pager_index_lines(struct MuttWindow *win)

101  struct MuttWindow *win_index = mutt_window_find(panel_index, WT_MENU);
102  if (!win_index)
103  return -1;
104 
105  const short c_pager_index_lines =
106  cs_subset_number(NeoMutt->sub, "pager_index_lines");
107 
108  if (c_pager_index_lines > 0)
109  {
110  struct IndexSharedData *shared = dlg->wdata;
111  int vcount = shared->mailbox ? shared->mailbox->vcount : 0;
112  win_index->req_rows = MIN(c_pager_index_lines, vcount);
113  win_index->size = MUTT_WIN_SIZE_FIXED;
114 
115  panel_index->size = MUTT_WIN_SIZE_MINIMISE;
116  panel_index->state.visible = (c_pager_index_lines != 0);
117  }
118  else
119  {
120  win_index->req_rows = MUTT_WIN_SIZE_UNLIMITED;
121  win_index->size = MUTT_WIN_SIZE_MAXIMISE;
122 
123  panel_index->size = MUTT_WIN_SIZE_MAXIMISE;
124  panel_index->state.visible = true;
125  }
126 
127  mutt_window_reflow(dlg);
128  mutt_debug(LL_DEBUG5, "config, request WA_REFLOW\n");
129  return 0;
