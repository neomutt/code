148static enum CommandResult parse_object(const struct Command *cmd,
149 struct Buffer *token, struct Buffer *line,
150 enum ColorId *cid, struct Buffer *err)
151{
152 if (!MoreArgs(line))
153 {
154 buf_printf(err, _("%s: too few arguments"), cmd->name);
155 return MUTT_CMD_WARNING;
156 }
157
158 if (mutt_istr_equal(buf_string(token), "compose"))
159 {
160 if (!MoreArgs(line))
161 {
162 buf_printf(err, _("%s: too few arguments"), cmd->name);
163 return MUTT_CMD_WARNING;
164 }
165
166 struct Buffer *tmp = buf_pool_get();
167 parse_extract_token(tmp, line, TOKEN_NO_FLAGS);
168 buf_fix_dptr(token);
169 buf_add_printf(token, "_%s", buf_string(tmp));
170 buf_pool_release(&tmp);
171 }
172
173 int rc = mutt_map_get_value(buf_string(token), ColorFields);
174 if (rc == -1)
175 {
176 buf_printf(err, _("%s: no such object"), buf_string(token));
177 return MUTT_CMD_WARNING;
178 }
179 else
180 {
181 color_debug(LL_DEBUG5, "object: %s\n", mutt_map_get_name(rc, ColorFields));
182 }
183
184 *cid = rc;
185 return MUTT_CMD_SUCCESS;
186}

194static enum CommandResult parse_uncolor_command(const struct Command *cmd,
195 struct Buffer *token,
196 struct Buffer *line, struct Buffer *err)
197{
198 if (!MoreArgs(line))
199 {
200 buf_printf(err, _("%s: too few arguments"), cmd->name);
201 return MUTT_CMD_WARNING;
202 }
203
204 parse_extract_token(token, line, TOKEN_NO_FLAGS);
205
206 if (mutt_str_equal(buf_string(token), "*"))
207 {
208 colors_reset();
209 return MUTT_CMD_SUCCESS;
210 }
211
212 unsigned int cid = MT_COLOR_NONE;
213 color_debug(LL_DEBUG5, "uncolor: %s\n", buf_string(token));
214 enum CommandResult rc = parse_object(cmd, token, line, &cid, err);
215 if (rc != MUTT_CMD_SUCCESS)
216 return rc;
217
218 if (cid == -1)
219 {
220 buf_printf(err, _("%s: no such object"), buf_string(token));
221 return MUTT_CMD_ERROR;
222 }
223
224 if ((cid == MT_COLOR_STATUS) && !MoreArgs(line))
225 {
226 color_debug(LL_DEBUG5, "simple\n");
227 simple_color_reset(cid); // default colour for the status bar
228 return MUTT_CMD_SUCCESS;
229 }
230
231 if (!mutt_color_has_pattern(cid))
232 {
233 color_debug(LL_DEBUG5, "simple\n");
234 simple_color_reset(cid);
235 return MUTT_CMD_SUCCESS;
236 }
237
238 if (!MoreArgs(line))
239 {
240 if (regex_colors_parse_uncolor(cid, NULL))
241 return MUTT_CMD_SUCCESS;
242 else
243 return MUTT_CMD_ERROR;
244 }
245
246 do
247 {
248 parse_extract_token(token, line, TOKEN_NO_FLAGS);
249 if (mutt_str_equal("*", buf_string(token)))
250 {
251 if (regex_colors_parse_uncolor(cid, NULL))
252 return MUTT_CMD_SUCCESS;
253 else
254 return MUTT_CMD_ERROR;
255 }
256
257 regex_colors_parse_uncolor(cid, buf_string(token));
258
259 } while (MoreArgs(line));
260
261 return MUTT_CMD_SUCCESS;
262}

277static enum CommandResult parse_color_command(const struct Command *cmd,
278 struct Buffer *token, struct Buffer *line,
279 struct Buffer *err, parser_callback_t callback)
280{
281 unsigned int match = 0;
282 enum ColorId cid = MT_COLOR_NONE;
283 enum CommandResult rc = MUTT_CMD_ERROR;
284 struct AttrColor *ac = NULL;
285
286 if (!MoreArgs(line))
287 {
288 if (StartupComplete)
289 {
290 color_dump();
291 rc = MUTT_CMD_SUCCESS;
292 }
293 else
294 {
295 buf_printf(err, _("%s: too few arguments"), cmd->name);
296 rc = MUTT_CMD_WARNING;
297 }
298
299 goto done;
300 }
301
302 parse_extract_token(token, line, TOKEN_NO_FLAGS);
303 color_debug(LL_DEBUG5, "color: %s\n", buf_string(token));
304
305 rc = parse_object(cmd, token, line, &cid, err);
306 if (rc != MUTT_CMD_SUCCESS)
307 goto done;
308
309 ac = attr_color_new();
310 rc = callback(cmd, token, line, ac, err);
311 if (rc != MUTT_CMD_SUCCESS)
312 goto done;
313
314 //------------------------------------------------------------------
315 // Business Logic
316
317 if ((ac->fg.type == CT_RGB) || (ac->bg.type == CT_RGB))
318 {
319#ifndef NEOMUTT_DIRECT_COLORS
320 buf_printf(err, _("Direct colors support not compiled in: %s"), buf_string(line));
321 return MUTT_CMD_ERROR;
322#endif
323
324 const bool c_color_directcolor = cs_subset_bool(NeoMutt->sub, "color_directcolor");
325 if (!c_color_directcolor)
326 {
327 buf_printf(err, _("Direct colors support disabled: %s"), buf_string(line));
328 return MUTT_CMD_ERROR;
329 }
330 }
331
332 if ((ac->fg.color >= COLORS) || (ac->bg.color >= COLORS))
333 {
334 buf_printf(err, _("%s: color not supported by term"), buf_string(line));
335 return MUTT_CMD_ERROR;
336 }
337
338 //------------------------------------------------------------------
339
340 /* extract a regular expression if needed */
341
342 if (mutt_color_has_pattern(cid) && (cid != MT_COLOR_STATUS))
343 {
344 color_debug(LL_DEBUG5, "regex needed\n");
345 if (MoreArgs(line))
346 {
347 parse_extract_token(token, line, TOKEN_NO_FLAGS);
348 }
349 else
350 {
351 buf_strcpy(token, ".*");
352 }
353 }
354
355 if (MoreArgs(line) && (cid != MT_COLOR_STATUS))
356 {
357 buf_printf(err, _("%s: too many arguments"), cmd->name);
358 rc = MUTT_CMD_WARNING;
359 goto done;
360 }
361
362 if (regex_colors_parse_color_list(cid, buf_string(token), ac, &rc, err))
363 {
364 color_debug(LL_DEBUG5, "regex_colors_parse_color_list done\n");
365 goto done;
366 // do nothing
367 }
368 else if ((cid == MT_COLOR_STATUS) && MoreArgs(line))
369 {
370 color_debug(LL_DEBUG5, "status\n");
371 /* 'color status fg bg' can have up to 2 arguments:
372 * 0 arguments: sets the default status color (handled below by else part)
373 * 1 argument : colorize pattern on match
374 * 2 arguments: colorize nth submatch of pattern */
375 parse_extract_token(token, line, TOKEN_NO_FLAGS);
376
377 if (MoreArgs(line))
378 {
379 struct Buffer *tmp = buf_pool_get();
380 parse_extract_token(tmp, line, TOKEN_NO_FLAGS);
381 if (!mutt_str_atoui_full(buf_string(tmp), &match))
382 {
383 buf_printf(err, _("%s: invalid number: %s"), cmd->name, buf_string(tmp));
384 buf_pool_release(&tmp);
385 rc = MUTT_CMD_WARNING;
386 goto done;
387 }
388 buf_pool_release(&tmp);
389 }
390
391 if (MoreArgs(line))
392 {
393 buf_printf(err, _("%s: too many arguments"), cmd->name);
394 rc = MUTT_CMD_WARNING;
395 goto done;
396 }
397
398 rc = regex_colors_parse_status_list(cid, buf_string(token), ac, match, err);
399 goto done;
400 }
401 else // Remaining simple colours
402 {
403 color_debug(LL_DEBUG5, "simple\n");
404 if (simple_color_set(cid, ac))
405 rc = MUTT_CMD_SUCCESS;
406 else
407 rc = MUTT_CMD_ERROR;
408 }
409
410 if (rc == MUTT_CMD_SUCCESS)
411 {
412 get_colorid_name(cid, token);
413 color_debug(LL_DEBUG5, "NT_COLOR_SET: %s\n", buf_string(token));
414 struct EventColor ev_c = { cid, NULL };
415 notify_send(ColorsNotify, NT_COLOR, NT_COLOR_SET, &ev_c);
416 }
417
418done:
419 attr_color_free(&ac);
420 return rc;
421}
