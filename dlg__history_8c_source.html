147void dlg_history(char *buf, size_t buflen, char **matches, int match_count)
148{
149 struct MuttWindow *dlg = simple_dialog_new(MENU_GENERIC, WT_DLG_HISTORY, HistoryHelp);
150
151 struct MuttWindow *sbar = window_find_child(dlg, WT_STATUS_BAR);
152 char title[256] = { 0 };
153 snprintf(title, sizeof(title), _("History '%s'"), buf);
154 sbar_set_title(sbar, title);
155
156 struct Menu *menu = dlg->wdata;
157 menu->make_entry = history_make_entry;
158 menu->max = match_count;
159 menu->mdata = matches;
160 menu->mdata_free = NULL; // Menu doesn't own the data
161
162 struct HistoryData hd = { false, false, buf, buflen,
163 menu, matches, match_count };
164 dlg->wdata = &hd;
165
166 struct MuttWindow *old_focus = window_set_focus(menu->win);
167 // ---------------------------------------------------------------------------
168 // Event Loop
169 int op = OP_NULL;
170 do
171 {
172 menu_tagging_dispatcher(menu->win, op);
173 window_redraw(NULL);
174
175 op = km_dokey(MENU_DIALOG, GETCH_NO_FLAGS);
176 mutt_debug(LL_DEBUG1, "Got op %s (%d)\n", opcodes_get_name(op), op);
177 if (op < 0)
178 continue;
179 if (op == OP_NULL)
180 {
181 km_error_key(MENU_GENERIC);
182 continue;
183 }
184 mutt_clear_error();
185
186 int rc = history_function_dispatcher(dlg, op);
187 if (rc == FR_UNKNOWN)
188 rc = menu_function_dispatcher(menu->win, op);
189 if (rc == FR_UNKNOWN)
190 rc = global_function_dispatcher(NULL, op);
191 } while (!hd.done);
192 // ---------------------------------------------------------------------------
193
194 window_set_focus(old_focus);
195 simple_dialog_free(&dlg);
196}
