59 static int pbar_recalc(struct MuttWindow *win)
60 {
61  if (!mutt_window_is_visible(win))
62  return 0;
63 
64  char buf[1024] = { 0 };
65 
66  struct PBarPrivateData *pbar_data = win->wdata;
67  struct IndexSharedData *shared = pbar_data->shared;
68  struct PagerPrivateData *priv = pbar_data->priv;
69 
70  char pager_progress_str[65]; /* Lots of space for translations */
71  if (priv->last_pos < priv->sb.st_size - 1)
72  {
73  snprintf(pager_progress_str, sizeof(pager_progress_str), OFF_T_FMT "%%",
74  (100 * priv->last_offset / priv->sb.st_size));
75  }
76  else
77  {
78  const char *msg = (priv->topline == 0) ?
79  /* L10N: Status bar message: the entire email is visible in the pager */
80  _("all") :
81  /* L10N: Status bar message: the end of the email is visible in the pager */
82  _("end");
83  mutt_str_copy(pager_progress_str, msg, sizeof(pager_progress_str));
84  }
85 
86  if ((priv->pview->mode == PAGER_MODE_EMAIL) || (priv->pview->mode == PAGER_MODE_ATTACH_E))
87  {
88  int msg_in_pager = shared->ctx ? shared->ctx->msg_in_pager : -1;
89 
90  const char *c_pager_format = cs_subset_string(shared->sub, "pager_format");
91  mutt_make_string(buf, sizeof(buf), win->state.cols, NONULL(c_pager_format),
92  shared->mailbox, msg_in_pager, shared->email,
93  MUTT_FORMAT_NO_FLAGS, pager_progress_str);
94  }
95  else
96  {
97  snprintf(buf, sizeof(buf), "%s (%s)", priv->pview->banner, pager_progress_str);
98  }
99 
100  if (!mutt_str_equal(buf, pbar_data->pager_format))
101  {
102  mutt_str_replace(&pbar_data->pager_format, buf);
103  win->actions |= WA_REPAINT;
104  }
105 
106  return 0;
107 }

220  return 0;

223  struct IndexSharedData *new_shared = pbar_data->shared;
224 
225  if (nc->event_subtype & NT_PAGER_MAILBOX)
