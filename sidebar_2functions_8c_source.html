201static int op_sidebar_first(struct SidebarFunctionData *fdata, const struct KeyEvent *event)
202{
203 struct SidebarWindowData *wdata = fdata->wdata;
204 if (!mutt_window_is_visible(wdata->win))
205 return FR_NO_ACTION;
206
207 if (ARRAY_EMPTY(&wdata->entries) || (wdata->hil_index < 0))
208 return FR_NO_ACTION;

211

216
217 if (orig_hil_index == wdata->hil_index)
218 return FR_NO_ACTION;
219
220 wdata->win->actions |= WA_RECALC;
221 return FR_SUCCESS;

436static void sidebar_matcher_cb(const char *text, void *data)
437{
438 struct MuttWindow *win = data;
439 struct SidebarWindowData *wdata = win->wdata;
440 wdata->hil_index = -1;
441 wdata->repage = true;
442
443 struct SbEntry **sbep = NULL;
444
445 if (text[0] == '\0')
446 {
447 ARRAY_FOREACH(sbep, &wdata->entries)
448 {
449 struct SbEntry *sbe = *sbep;
450 sbe->mailbox->visible = true;
451 sbe->score = -1;
452 if (wdata->hil_index == -1)
453 wdata->hil_index = ARRAY_FOREACH_IDX_sbep;
454 }
455 wdata->win->actions |= WA_RECALC;
456 return;
457 }
458
459 struct FuzzyOptions opts = { .smart_case = true };
460 struct FuzzyResult result = { 0 };
461 int best_score = -1;
462 int best_index = -1;
463 struct Buffer *buf = buf_pool_get();
464
465 ARRAY_FOREACH(sbep, &wdata->entries)
466 {
467 struct SbEntry *sbe = *sbep;
468 buf_printf(buf, "%s %s", sbe->box, sbe->display);
469 int score = fuzzy_match(text, buf_string(buf), FUZZY_ALGO_SUBSEQ, &opts, &result);
470 if (score > 0)
471 {
472 // Extra 2 points for new (unseen) mail
473 // 1 point for old (seen) mail
474 score += (2 * sbe->mailbox->msg_new);
475 score += (sbe->mailbox->msg_unread - sbe->mailbox->msg_new);
476 }
477 sbe->score = score;
478 if (score >= 0)
479 {
480 if ((best_score == -1) || (score > best_score))
481 {
482 best_score = score;
483 best_index = ARRAY_FOREACH_IDX_sbep;
484 }
485 sbe->mailbox->visible = true;
486 }
487 else
488 {
489 sbe->mailbox->visible = false;
490 }
491 }
492
493 if (best_index != -1)
494 wdata->hil_index = best_index;
495
496 wdata->win->actions |= WA_RECALC;
497 buf_pool_release(&buf);
498}

518static int op_sidebar_start_search(struct SidebarFunctionData *fdata,
519 const struct KeyEvent *event)
520{
521 struct SidebarWindowData *wdata = fdata->wdata;
522 if (ARRAY_EMPTY(&wdata->entries) || (wdata->hil_index < 0))
523 {
524 mutt_warning(_("There are no mailboxes"));
525 return FR_NO_ACTION;
526 }
527
528 const bool was_visible = cs_subset_bool(fdata->n->sub, "sidebar_visible");
529 if (!was_visible)
530 {
531 cs_subset_str_native_set(fdata->n->sub, "sidebar_visible", true, NULL);
532 mutt_window_reflow(NULL);
533 }
534
535 struct Buffer *buf = buf_pool_get();
536 int orghlidx = wdata->hil_index;
537
538 struct SbEntry **sbep = NULL;
539 ARRAY_FOREACH(sbep, &wdata->entries)
540 {
541 struct SbEntry *sbe = *sbep;
542 if (sbe->box[0] == '\0')
543 sb_entry_set_display_name(sbe);
544 }
545
546 int rc = FR_NO_ACTION;
547 wdata->search_active = true;
548
549 if (mw_get_field_notify(_("Sidebar search: "), buf, MUTT_COMP_UNBUFFERED,
550 HC_NONE, NULL, NULL, sidebar_matcher_cb, wdata->win,
551 fdata->mod_data->md_sidebar, sb_function_dispatcher) != 0)
552 {
553 wdata->hil_index = orghlidx;
554 goto done;
555 }
556
557 if (!buf || buf_is_empty(buf) || (wdata->hil_index == -1))
558 {
559 wdata->hil_index = orghlidx;
560 goto done;
561 }
562
563 ARRAY_FOREACH(sbep, &wdata->entries)
564 {
565 struct SbEntry *sbe = *sbep;
566 sbe->score = -1;
567 }
568 rc = FR_SUCCESS;
569
570done:
571 ARRAY_FOREACH(sbep, &wdata->entries)
572 {
573 (*sbep)->mailbox->visible = true;
574 }
575 wdata->search_active = false;
576 wdata->repage = false;
577 wdata->win->actions |= WA_RECALC;
578
579 if (rc == FR_SUCCESS)
580 {
581 struct MuttWindow *dlg = dialog_find(wdata->win);
582 index_change_folder(dlg, sb_get_highlight(wdata->win));
583 }
584
585 if (!was_visible)
586 {
587 cs_subset_str_native_set(fdata->n->sub, "sidebar_visible", false, NULL);
588 mutt_window_reflow(NULL);
589 }
590
591 buf_pool_release(&buf);
592 return rc;
593}

622int sb_function_dispatcher(struct MuttWindow *win, const struct KeyEvent *event)
623{
624 if (!event || !win || !win->wdata)
625 return FR_UNKNOWN;
626
627 const int op = event->op;
628
629 struct SidebarModuleData *mod_data = neomutt_get_module_data(NeoMutt, MODULE_ID_SIDEBAR);
630
631 struct SidebarFunctionData fdata = {
632 .n = NeoMutt,
633 .mod_data = mod_data,
634 .wdata = win->wdata,
635 };
636
637 int rc = FR_UNKNOWN;
638 for (size_t i = 0; SidebarFunctions[i].op != OP_NULL; i++)
639 {
640 const struct SidebarFunction *fn = &SidebarFunctions[i];
641 if (fn->op == op)
642 {
643 rc = fn->function(&fdata, event);
644 break;
645 }
646 }
647
648 if (rc == FR_UNKNOWN) // Not our function
649 return rc;
650
651 const char *result = dispatcher_get_retval_name(rc);
652 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
653
654 return rc;
655}
