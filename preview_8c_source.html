133static void draw_preview(struct MuttWindow *win, struct PreviewWindowData *wdata)
134{
135 struct Email *e = wdata->email;
136
137 // Reset preview window and status bar.
138 mutt_window_clear(win);
139 sbar_set_title(wdata->bar, _("-- Preview"));
140
141 // Check for valid content type and disposition
142 if ((e->body->disposition != DISP_INLINE) || (e->body->type != TYPE_TEXT))
143 {
144 mutt_error(_("Only inline attachments with content-type text/* can be previewed"));
145 return;
146 }
147
148 // Ensure file isn't to too large.
149 long file_size = mutt_file_get_size(e->body->filename);
150 if (file_size > MAX_PREVIEW_BODY_SIZE)
151 {
152 mutt_error(_("Email too large to preview"));
153 return;
154 }
155
156 FILE *fp = mutt_file_fopen(e->body->filename, "r");
157 if (!fp)
158 {
159 mutt_perror("%s", e->body->filename);
160 return;
161 }
162
163 wdata->more_content = false;
164
165 int content_lines = 0; // number of (wrapped) content lines
166 int row = 0; // window row to print
167 char *line = NULL;
168 size_t line_len = 0;
169 while ((line = mutt_file_read_line(line, &line_len, fp, NULL, MUTT_RL_NO_FLAGS)))
170 {
171 size_t pos = 0;
172 bool text_left = true;
173 while (text_left)
174 {
175 /* Text wrapping loop
176 *
177 * Note: We need to do the text wrapping also for text outside the visible
178 * area to ensure the scrolling works correctly.
179 */
180
181 content_lines++;
182
183 line = mutt_str_expand_tabs(line, &line_len, 8);
184
185 // Check how much of the string fits into the window width.
186 size_t width = 0;
187 size_t bytes = mutt_wstr_trunc(&line[pos], line_len - pos, win->state.cols, &width);
188
189 // If it doesn't fill the full width we're done wrapping.
190 if ((win->state.cols - width) > 0)
191 text_left = false;
192
193 // Only move the cursor and print if this line is currently visible.
194 if ((content_lines >= wdata->scroll_offset) && (row < win->state.rows))
195 {
196 int rc = mutt_window_move(win, row, 0);
197 if (rc == ERR)
198 mutt_warning(_("Failed to move cursor!"));
199
200 mutt_paddstr(win, win->state.cols, &line[pos]);
201
202 row++;
203 }
204
205 // Advance position in string.
206 pos += bytes;
207 }
208 }
209
210 FREE(&line);
211 mutt_file_fclose(&fp);
212
213 // Show the scroll percentage in the status bar
214 if ((content_lines != 0) && (content_lines > win->state.rows))
215 {
216 char title[256] = { 0 };
217 double percent = 100.0;
218 if ((wdata->scroll_offset + row) < content_lines)
219 percent = 100.0 / content_lines * (wdata->scroll_offset + row);
220
221 // TODO: having the percentage right-aligned would be nice
222 snprintf(title, sizeof(title), _("-- Preview (%.0f%%)"), percent);
223 sbar_set_title(wdata->bar, title);
224
225 if (content_lines > (wdata->scroll_offset + row))
226 wdata->more_content = true;
227 }
228}

233static int preview_color_observer(struct NotifyCallback *nc)
234{
235 if (nc->event_type != NT_COLOR)
236 return 0;
237 if (!nc->global_data || !nc->event_data)
238 return -1;
239
240 struct EventColor *ev_c = nc->event_data;
241 struct MuttWindow *win = nc->global_data;
242
243 enum ColorId cid = ev_c->cid;
244
245 switch (cid)
246 {
247 case MT_COLOR_BOLD:
248 case MT_COLOR_NORMAL:
249 case MT_COLOR_STATUS:
250 case MT_COLOR_MAX: // Sent on `uncolor *`
251 mutt_debug(LL_DEBUG5, "color done, request WA_REPAINT\n");
252 win->actions |= WA_REPAINT;
253 break;
254
255 default:
256 break;
257 }
258 return 0;
259}

281static int preview_window_observer(struct NotifyCallback *nc)
282{
283 if (nc->event_type != NT_WINDOW)
284 return 0;
285 if (!nc->global_data || !nc->event_data)
286 return -1;
287
288 struct MuttWindow *win = nc->global_data;
289 struct EventWindow *ev_w = nc->event_data;
290 if (ev_w->win != win)
291 return 0;
292
293 if (nc->event_subtype == NT_WINDOW_STATE)
294 {
295 win->actions |= WA_RECALC;
296 mutt_debug(LL_DEBUG5, "window state done, request WA_RECALC\n");
297 }
298 else if (nc->event_subtype == NT_WINDOW_DELETE)
299 {
300 struct PreviewWindowData *wdata = win->wdata;
301
302 mutt_color_observer_remove(preview_color_observer, win);
303 notify_observer_remove(win->notify, preview_window_observer, win);
304 notify_observer_remove(wdata->email->notify, preview_email_observer, win);
305 mutt_debug(LL_DEBUG5, "window delete done\n");
306 }
307
308 return 0;
309}

407int preview_function_dispatcher(struct MuttWindow *win, int op)
408{
409 if (!win || !win->wdata)
410 return FR_UNKNOWN;
411
412 int rc = FR_UNKNOWN;
413 for (size_t i = 0; PreviewFunctions[i].op != OP_NULL; i++)
414 {
415 const struct PreviewFunction *fn = &PreviewFunctions[i];
416 if (fn->op == op)
417 {
418 struct PreviewWindowData *wdata = win->wdata;
419 rc = fn->function(wdata, op);
420 break;
421 }
422 }
423
424 if (rc == FR_UNKNOWN) // Not our function
425 return rc;
426
427 const char *result = dispatcher_get_retval_name(rc);
428 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
429
430 return rc;
431}
