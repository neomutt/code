134static void draw_preview(struct MuttWindow *win, struct PreviewWindowData *wdata)
135{
136 struct Email *e = wdata->email;
137
138 // Reset preview window and status bar.
139 mutt_window_clear(win);
140 sbar_set_title(wdata->bar, _("-- Preview"));
141
142 // Check for valid content type and disposition
143 if ((e->body->disposition != DISP_INLINE) || (e->body->type != TYPE_TEXT))
144 {
145 mutt_error(_("Only inline attachments with content-type text/* can be previewed"));
146 return;
147 }
148
149 // Ensure file isn't to too large.
150 long file_size = mutt_file_get_size(e->body->filename);
151 if (file_size > MAX_PREVIEW_BODY_SIZE)
152 {
153 mutt_error(_("Email too large to preview"));
154 return;
155 }
156
157 FILE *fp = mutt_file_fopen(e->body->filename, "r");
158 if (!fp)
159 {
160 mutt_perror("%s", e->body->filename);
161 return;
162 }
163
164 wdata->more_content = false;
165
166 int content_lines = 0; // number of (wrapped) content lines
167 int row = 0; // window row to print
168 char *line = NULL;
169 size_t line_len = 0;
170 while ((line = mutt_file_read_line(line, &line_len, fp, NULL, MUTT_RL_NO_FLAGS)))
171 {
172 size_t pos = 0;
173 bool text_left = true;
174 while (text_left)
175 {
176 /* Text wrapping loop
177 *
178 * Note: We need to do the text wrapping also for text outside the visible
179 * area to ensure the scrolling works correctly.
180 */
181
182 content_lines++;
183
184 line = mutt_str_expand_tabs(line, &line_len, 8);
185
186 // Check how much of the string fits into the window width.
187 size_t width = 0;
188 size_t bytes = mutt_wstr_trunc(&line[pos], line_len - pos, win->state.cols, &width);
189
190 // If it doesn't fill the full width we're done wrapping.
191 if ((win->state.cols - width) > 0)
192 text_left = false;
193
194 // Only move the cursor and print if this line is currently visible.
195 if ((content_lines >= wdata->scroll_offset) && (row < win->state.rows))
196 {
197 int rc = mutt_window_move(win, row, 0);
198 if (rc == ERR)
199 mutt_warning(_("Failed to move cursor!"));
200
201 mutt_paddstr(win, win->state.cols, &line[pos]);
202
203 row++;
204 }
205
206 // Advance position in string.
207 pos += bytes;
208 }
209 }
210
211 FREE(&line);
212 mutt_file_fclose(&fp);
213
214 // Show the scroll percentage in the status bar
215 if ((content_lines != 0) && (content_lines > win->state.rows))
216 {
217 char title[256] = { 0 };
218 double percent = 100.0;
219 if ((wdata->scroll_offset + row) < content_lines)
220 percent = 100.0 / content_lines * (wdata->scroll_offset + row);
221
222 // TODO: having the percentage right-aligned would be nice
223 snprintf(title, sizeof(title), _("-- Preview (%.0f%%)"), percent);
224 sbar_set_title(wdata->bar, title);
225
226 if (content_lines > (wdata->scroll_offset + row))
227 wdata->more_content = true;
228 }
229}

234static int preview_color_observer(struct NotifyCallback *nc)
235{
236 if (nc->event_type != NT_COLOR)
237 return 0;
238 if (!nc->global_data || !nc->event_data)
239 return -1;
240
241 struct EventColor *ev_c = nc->event_data;
242 struct MuttWindow *win = nc->global_data;
243
244 enum ColorId cid = ev_c->cid;
245
246 switch (cid)
247 {
248 case MT_COLOR_BOLD:
249 case MT_COLOR_NORMAL:
250 case MT_COLOR_STATUS:
251 case MT_COLOR_MAX: // Sent on `uncolor *`
252 mutt_debug(LL_DEBUG5, "color done, request WA_REPAINT\n");
253 win->actions |= WA_REPAINT;
254 break;
255
256 default:
257 break;
258 }
259 return 0;
260}

282static int preview_window_observer(struct NotifyCallback *nc)
283{
284 if (nc->event_type != NT_WINDOW)
285 return 0;
286 if (!nc->global_data || !nc->event_data)
287 return -1;
288
289 struct MuttWindow *win = nc->global_data;
290 struct EventWindow *ev_w = nc->event_data;
291 if (ev_w->win != win)
292 return 0;
293
294 if (nc->event_subtype == NT_WINDOW_STATE)
295 {
296 win->actions |= WA_RECALC;
297 mutt_debug(LL_DEBUG5, "window state done, request WA_RECALC\n");
298 }
299 else if (nc->event_subtype == NT_WINDOW_DELETE)
300 {
301 struct PreviewWindowData *wdata = win->wdata;
302
303 mutt_color_observer_remove(preview_color_observer, win);
304 notify_observer_remove(win->notify, preview_window_observer, win);
305 notify_observer_remove(wdata->email->notify, preview_email_observer, win);
306 mutt_debug(LL_DEBUG5, "window delete done\n");
307 }
308
309 return 0;
310}

408int preview_function_dispatcher(struct MuttWindow *win, int op)
409{
410 if (!win || !win->wdata)
411 return FR_UNKNOWN;
412
413 int rc = FR_UNKNOWN;
414 for (size_t i = 0; PreviewFunctions[i].op != OP_NULL; i++)
415 {
416 const struct PreviewFunction *fn = &PreviewFunctions[i];
417 if (fn->op == op)
418 {
419 struct PreviewWindowData *wdata = win->wdata;
420 rc = fn->function(wdata, op);
421 break;
422 }
423 }
424
425 if (rc == FR_UNKNOWN) // Not our function
426 return rc;
427
428 const char *result = dispatcher_get_retval_name(rc);
429 mutt_debug(LL_DEBUG1, "Handled %s (%d) -> %s\n", opcodes_get_name(op), op, NONULL(result));
430
431 return rc;
432}
