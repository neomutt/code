97int check_help_mode(const char *mode)
98{
99 if (mutt_istr_equal(mode, "shared"))
100 return HM_SHARED;
101 if (mutt_istr_equal(mode, "help"))
102 return HM_HELP;
103 if (mutt_istr_equal(mode, "info"))
104 return HM_INFO;
105 if (mutt_istr_equal(mode, "send"))
106 return HM_SEND;
107 if (mutt_istr_equal(mode, "tui"))
108 return HM_TUI;
109 if (mutt_istr_equal(mode, "all"))
110 return HM_ALL;
111 return 0;
112}

127static int mop_up(int argc, char *const *argv, int index, struct StringArray *sa)
128{
129 int count = 0;
130 for (int i = index; i < argc; i++, count++)
131 {
132 // Stop if we find '--' or '-X'
133 if ((argv[i][0] == '-') && (argv[i][1] != '\0'))
134 break;
135
136 ARRAY_ADD(sa, mutt_str_dup(argv[i]));
137 }
138
139 return count;
140}

149bool cli_parse(int argc, char *const *argv, struct CommandLine *cli)
150{
151 if ((argc < 1) || !argv || !cli)
152 return false;
153
154 // Check for unsupported '=' syntax in options
155 for (int i = 1; i < argc; i++)
156 {
157 if ((argv[i][0] == '-') && (argv[i][1] != '\0'))
158 {
159 const char *equals = strchr(argv[i], '=');
160 if (equals)
161 {
162 // L10N: Neomutt doesn't support `-X=VALUE` or `--OPTION=VALUE`
163 // Use `-X VALUE` or `--OPTION VALUE` instead
164 mutt_warning(_("Invalid option syntax: %s (use space instead of '=')"), argv[i]);
165 cli->help.help = true;
166 cli->help.is_set = true;
167 return false;
168 }
169 }
170 }
171
172 // Any leading non-options must be addresses
173 int count = mop_up(argc, argv, 1, &cli->send.addresses);
174 if (count > 0)
175 {
176 cli->send.is_set = true;
177 argc -= count;
178 argv += count;
179 }
180
181 bool rc = true;
182
183 opterr = 0; // We'll handle the errors
184// Always initialise getopt() or the tests will fail
185#if defined(BSD) || defined(__APPLE__)
186 optreset = 1;
187 optind = 1;
188#else
189 optind = 0;
190#endif
191
192 while (rc && (argc > 1) && (optind < argc))
193 {
194 int opt = getopt_long(argc, argv, "+:A:a:b:Cc:Dd:Ee:F:f:Gg:H:hi:l:m:nOpQ:RSs:vyZz",
195 LongOptions, NULL);
196 switch (opt)
197 {
198 // ------------------------------------------------------------
199 // Shared
200 case 'F': // user config file
201 {
202 ARRAY_ADD(&cli->shared.user_files, mutt_str_dup(optarg));
203 cli->shared.is_set = true;
204 break;
205 }
206 case 'n': // no system config file
207 {
208 cli->shared.disable_system = true;
209 cli->shared.is_set = true;
210 break;
211 }
212 case 'e': // enter commands
213 {
214 ARRAY_ADD(&cli->shared.commands, mutt_str_dup(optarg));
215 cli->shared.is_set = true;
216 break;
217 }
218 case 'm': // mbox type

220 buf_strcpy(&cli->shared.mbox_type, optarg);
221 cli->shared.is_set = true;
222 break;
223 }
224 case 'd': // log level
225 {
226 buf_strcpy(&cli->shared.log_level, optarg);
227 cli->shared.is_set = true;
228 break;
229 }
230 case 'l': // log file
231 {
232 buf_strcpy(&cli->shared.log_file, optarg);
233 cli->shared.is_set = true;
234 break;
235 }
236
237 // ------------------------------------------------------------
238 // Help
239 case 'h': // help
240 {
241 if (optind < argc)
242 {
243 cli->help.mode = check_help_mode(argv[optind]);
244 if (cli->help.mode != HM_NONE)
245 optind++;
246 }
247 cli->help.help = true;
248 cli->help.is_set = true;
249 break;
250 }
251 case 'L': // license
252 {
253 cli->help.license = true;
254 cli->help.is_set = true;

265 break;
266 }
267
268 // ------------------------------------------------------------
269 // Info
270 case 'A': // alias lookup
271 {
272 ARRAY_ADD(&cli->info.alias_queries, mutt_str_dup(optarg));
273 // '-A' can take multiple arguments
274 optind += mop_up(argc, argv, optind, &cli->info.alias_queries);
275 cli->info.is_set = true;
276 break;

278 case 'D': // dump config, dump changed

402 case 'y': // browser

404 cli->tui.start_browser = true;

406 break;

409 {
410 cli->tui.start_new_mail = true;
411 cli->tui.is_set = true;
412 break;
413 }
414 case 'z': // any mail
415 {
416 cli->tui.start_any_mail = true;
417 cli->tui.is_set = true;
418 break;
419 }
420
421 // ------------------------------------------------------------
422 case -1: // end of options
423 {
424 for (int i = optind; i < argc; i++)
425 {
426 ARRAY_ADD(&cli->send.addresses, mutt_str_dup(argv[i]));
427 }
428 cli->send.is_set = true;
429 optind = argc; // finish parsing
430 break;
431 }
432 default: // error
433 {
434 if (opt == '?')
435 {
436 // L10N: e.g. `neomutt -X`
437 mutt_warning(_("Invalid option: %c"), optopt);
438 }
439 else if (opt == ':')
440 {
441 // L10N: e.g. `neomutt -F`
442 mutt_warning(_("Option %c requires an argument"), optopt);
443 }
444
445 cli->help.help = true;
446 cli->help.is_set = true;
447 rc = false; // stop parsing
448 break;
449 }
450 }
451 }
452
453 return rc;
