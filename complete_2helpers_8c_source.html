54void matches_ensure_morespace(struct CompletionData *cd, int new_size)
55{
56 if (new_size <= (cd->match_list_len - 2))
57 return;
58
59 new_size = ROUND_UP(new_size + 2, 512);
60
61 MUTT_MEM_REALLOC(&cd->match_list, new_size, const char *);
62 memset(&cd->match_list[cd->match_list_len], 0, new_size - cd->match_list_len);
63
64 cd->match_list_len = new_size;
65}

78bool candidate(struct CompletionData *cd, char *user, const char *src, char *dest, size_t dlen)
79{
80 if (!dest || !user || !src)
81 return false;
82
83 if (strstr(src, user) != src)
84 return false;
85
86 matches_ensure_morespace(cd, cd->num_matched);
87 cd->match_list[cd->num_matched++] = src;
88 if (dest[0] == '\0')
89 {
90 mutt_str_copy(dest, src, dlen);
91 }
92 else
93 {
94 int l;
95 for (l = 0; (src[l] != '\0') && (src[l] == dest[l]); l++)
96 ; // do nothing
97
98 dest[l] = '\0';
99 }
100 return true;
101}

428int mutt_var_value_complete(struct CompletionData *cd, struct Buffer *buf, int pos)
429{
430 char *pt = buf->data;
431
432 if (pt[0] == '\0')
433 return 0;
434
435 SKIPWS(pt);
436 const int spaces = pt - buf->data;
437
438 pt = buf->data + pos - spaces;
439 while ((pt > buf->data) && !mutt_isspace(*pt))
440 pt--;
441 pt++; /* move past the space */
442 if (*pt == '=') /* abort if no var before the '=' */
443 return 0;
444
445 if (buf_startswith(buf, "set"))
446 {
447 char var[256] = { 0 };
448 mutt_str_copy(var, pt, sizeof(var));
449 /* ignore the trailing '=' when comparing */
450 int vlen = mutt_str_len(var);
451 if (vlen == 0)
452 return 0;
453
454 var[vlen - 1] = '\0';
455
456 struct HashElem *he = cs_subset_lookup(NeoMutt->sub, var);
457 if (!he)
458 return 0; /* no such variable. */
459
460 struct Buffer *value = buf_pool_get();
461 struct Buffer *pretty = buf_pool_get();
462 int rc = cs_subset_he_string_get(NeoMutt->sub, he, value);
463 if (CSR_RESULT(rc) == CSR_SUCCESS)
464 {
465 pretty_var(value->data, pretty);
466 snprintf(pt, buf->dsize - (pt - buf->data), "%s=%s", var, pretty->data);
467 buf_pool_release(&value);
468 buf_pool_release(&pretty);
469 return 0;
470 }
471 buf_pool_release(&value);
472 buf_pool_release(&pretty);
473 return 1;
474 }
475 return 0;
476}

516 }

525
