188int mutt_any_key_to_continue(const char *s)
189{
190 struct termios term = { 0 };
191 struct termios old = { 0 };
192
193 int fd = open("/dev/tty", O_RDONLY);
194 if (fd < 0)
195 return EOF;
196
197 tcgetattr(fd, &old); // Save the current tty settings
198
199 term = old;
200 term.c_lflag &= ~(ICANON | ECHO); // Canonical (not line-buffered), don't echo the characters
201 term.c_cc[VMIN] = 1; // Wait for at least one character
202 term.c_cc[VTIME] = 255; // Wait for 25.5s
203 tcsetattr(fd, TCSANOW, &term);
204
205 if (s)
206 fputs(s, stdout);
207 else
208 fputs(_("Press any key to continue..."), stdout);
209 fflush(stdout);
210
211 char ch = '\0';
212 // Wait for a character. This might timeout, so loop.
213 while (read(fd, &ch, 1) == 0)
214 ; // do nothing
215
216 // Change the tty settings to be non-blocking
217 term.c_cc[VMIN] = 0; // Returning with zero characters is acceptable
218 term.c_cc[VTIME] = 0; // Don't wait
219 tcsetattr(fd, TCSANOW, &term);
220
221 char buf[64] = { 0 };
222 while (read(fd, buf, sizeof(buf)) > 0)
223 ; // Mop up any remaining chars
224
225 tcsetattr(fd, TCSANOW, &old); // Restore the previous tty settings
226 close(fd);
227
228 fputs("\r\n", stdout);
229 mutt_clear_error();
230 return (ch >= 0) ? ch : EOF;
231}

366void mutt_simple_format(char *buf, size_t buflen, int min_width, int max_width,
367 enum FormatJustify justify, char pad_char,
368 const char *s, size_t n, bool arboreal)
369{
370 wchar_t wc = 0;
371 int w;
372 size_t k;
373 char scratch[MB_LEN_MAX] = { 0 };
374 mbstate_t mbstate1 = { 0 };
375 mbstate_t mbstate2 = { 0 };
376 bool escaped = false;
377
378 buflen--;
379 char *p = buf;
380 for (; n && (k = mbrtowc(&wc, s, n, &mbstate1)); s += k, n -= k)
381 {
382 if ((k == ICONV_ILLEGAL_SEQ) || (k == ICONV_BUF_TOO_SMALL))
383 {
384 if ((k == ICONV_ILLEGAL_SEQ) && (errno == EILSEQ))
385 memset(&mbstate1, 0, sizeof(mbstate1));
386
387 k = (k == ICONV_ILLEGAL_SEQ) ? 1 : n;
388 wc = ReplacementChar;
389 }
390 if (escaped)
391 {
392 escaped = false;
393 w = 0;
394 }
395 else if (arboreal && (wc == MUTT_SPECIAL_INDEX))
396 {
397 escaped = true;
398 w = 0;
399 }
400 else if (arboreal && (wc < MUTT_TREE_MAX))
401 {
402 w = 1; /* hack */
403 }
404 else
405 {
406#ifdef HAVE_ISWBLANK
407 if (iswblank(wc))
408 wc = ' ';
409 else
410#endif
411 if (!IsWPrint(wc))
412 wc = '?';
413 w = wcwidth(wc);
414 }
415 if (w >= 0)
416 {
417 if (w > max_width)
418 continue;
419 size_t k2 = wcrtomb(scratch, wc, &mbstate2);
420 if ((k2 == ICONV_ILLEGAL_SEQ) || (k2 > buflen))
421 continue;
422
423 min_width -= w;
424 max_width -= w;
425 strncpy(p, scratch, k2);
426 p += k2;
427 buflen -= k2;
428 }
429 }
430 w = ((int) buflen < min_width) ? buflen : min_width;
431 if (w <= 0)
432 {
433 *p = '\0';
434 }
435 else if (justify == JUSTIFY_RIGHT) /* right justify */
436 {
437 p[w] = '\0';
438 while (--p >= buf)
439 p[w] = *p;
440 while (--w >= 0)
441 buf[w] = pad_char;
442 }
443 else if (justify == JUSTIFY_CENTER) /* center */
444 {
445 char *savedp = p;
446 int half = (w + 1) / 2; /* half of cushion space */
447
448 p[w] = '\0';
449
450 /* move str to center of buffer */
451 while (--p >= buf)
452 p[half] = *p;
453
454 /* fill rhs */
455 p = savedp + half;
456 while (--w >= half)
457 *p++ = pad_char;
458
459 /* fill lhs */
460 while (half--)
461 buf[half] = pad_char;
462 }
463 else /* left justify */
464 {
465 while (--w >= 0)
466 *p++ = pad_char;
467 *p = '\0';
468 }
469}

545void mutt_paddstr(struct MuttWindow *win, int n, const char *s)
546{
547 wchar_t wc = 0;
548 size_t k;
549 size_t len = mutt_str_len(s);
550 mbstate_t mbstate = { 0 };
551
552 for (; len && (k = mbrtowc(&wc, s, len, &mbstate)); s += k, len -= k)
553 {
554 if ((k == ICONV_ILLEGAL_SEQ) || (k == ICONV_BUF_TOO_SMALL))
555 {
556 if (k == ICONV_ILLEGAL_SEQ)
557 memset(&mbstate, 0, sizeof(mbstate));
558 k = (k == ICONV_ILLEGAL_SEQ) ? 1 : len;
559 wc = ReplacementChar;
560 }
561 if (!IsWPrint(wc))
562 wc = '?';
563 const int w = wcwidth(wc);
564 if (w >= 0)
565 {
566 if (w > n)
567 break;
568 mutt_window_addnstr(win, (char *) s, k);
569 n -= w;
570 }
571 }
572 while (n-- > 0)
573 mutt_window_addch(win, ' ');

587size_t mutt_wstr_trunc(const char *src, size_t maxlen, size_t maxwid, size_t *width)
588{
589 wchar_t wc = 0;
590 size_t n, w = 0, l = 0, cl;
591 int cw;
592 mbstate_t mbstate = { 0 };
593
594 if (!src)
595 goto out;
596
597 n = mutt_str_len(src);
598
599 for (w = 0; n && (cl = mbrtowc(&wc, src, n, &mbstate)); src += cl, n -= cl)
600 {
601 if (cl == ICONV_ILLEGAL_SEQ)
602 {
603 memset(&mbstate, 0, sizeof(mbstate));
604 cl = 1;
605 wc = ReplacementChar;
606 }
607 else if (cl == ICONV_BUF_TOO_SMALL)
608 {
609 cl = n;
610 wc = ReplacementChar;
611 }
612
613 cw = wcwidth(wc);
614 /* hack because MUTT_TREE symbols aren't turned into characters
615 * until rendered by print_enriched_string() */
616 if ((cw < 0) && (src[0] == MUTT_SPECIAL_INDEX))
617 {
618 cl = 2; /* skip the index coloring sequence */
619 cw = 0;
620 }
621 else if ((cw < 0) && (cl == 1) && (src[0] != '\0') && (src[0] < MUTT_TREE_MAX))
622 {
623 cw = 1;
624 }
625 else if (cw < 0)
626 {
627 cw = 0; /* unprintable wchar */
628 }
629 if (wc == '\n')
630 break;
631 if (((cl + l) > maxlen) || ((cw + w) > maxwid))
632 break;
633 l += cl;
634 w += cw;
635 }
636out:
637 if (width)
638 *width = w;
639 return l;
640}

703void mw_what_key(void)
704{
705 struct MuttWindow *win = msgwin_new(true);
706 if (!win)
707 return;
708
709 char prompt[256] = { 0 };
710 snprintf(prompt, sizeof(prompt), _("Enter keys (%s to abort): "), km_keyname(AbortKey));
711 msgwin_set_text(win, prompt, MT_COLOR_PROMPT);
712
713 msgcont_push_window(win);
714 struct MuttWindow *old_focus = window_set_focus(win);
715 window_redraw(win);
716
717 char keys[256] = { 0 };
718 const struct AttrColor *ac_normal = simple_color_get(MT_COLOR_NORMAL);
719 const struct AttrColor *ac_prompt = simple_color_get(MT_COLOR_PROMPT);
720
721 // ---------------------------------------------------------------------------
722 // Event Loop
723 timeout(1000); // 1 second
724 while (true)
725 {
726 int ch = getch();
727 if (ch == AbortKey)
728 break;
729
730 if (ch == KEY_RESIZE)
731 {
732 timeout(0);
733 while ((ch = getch()) == KEY_RESIZE)
734 {
735 // do nothing
736 }
737 }
738
739 if (ch == ERR)
740 {
741 if (!isatty(0)) // terminal was lost
742 mutt_exit(1);
743
744 if (SigWinch)
745 {
746 SigWinch = false;
747 notify_send(NeoMutt->notify_resize, NT_RESIZE, 0, NULL);
748 window_redraw(NULL);
749 }
750 else
751 {
752 notify_send(NeoMutt->notify_timeout, NT_TIMEOUT, 0, NULL);
753 }
754
755 continue;
756 }
757
758 msgwin_clear_text(win);
759 snprintf(keys, sizeof(keys), _("Char = %s, Octal = %o, Decimal = %d\n"),
760 km_keyname(ch), ch, ch);
761 msgwin_add_text(win, keys, ac_normal);
762 msgwin_add_text(win, prompt, ac_prompt);
763 msgwin_add_text(win, NULL, NULL);
764 window_redraw(NULL);
765 }
766 // ---------------------------------------------------------------------------
767
768 win = msgcont_pop_window();
769 window_set_focus(old_focus);
770 mutt_window_free(&win);
771}
