129 static int smtp_get_resp(struct Connection *conn)
130 {
131  int n;
132  char buf[1024];
133 
134  do
135  {
136  n = mutt_socket_readln(buf, sizeof(buf), conn);
137  if (n < 4)
138  {
139  /* read error, or no response code */
140  return SMTP_ERR_READ;
141  }
142  const char *s = buf + 4; /* Skip the response code and the space/dash */
143  size_t plen;
144 
145  if (mutt_str_startswith(s, "8BITMIME", CASE_IGNORE))
146  Capabilities |= SMTP_CAP_EIGHTBITMIME;
147  else if ((plen = mutt_str_startswith(s, "AUTH ", CASE_IGNORE)))
148  {
149  Capabilities |= SMTP_CAP_AUTH;
150  FREE(&AuthMechs);
151  AuthMechs = mutt_str_strdup(s + plen);
152  }
153  else if (mutt_str_startswith(s, "DSN", CASE_IGNORE))
154  Capabilities |= SMTP_CAP_DSN;
155  else if (mutt_str_startswith(s, "STARTTLS", CASE_IGNORE))
156  Capabilities |= SMTP_CAP_STARTTLS;
157  else if (mutt_str_startswith(s, "SMTPUTF8", CASE_IGNORE))
158  Capabilities |= SMTP_CAP_SMTPUTF8;
159 
160  if (!valid_smtp_code(buf, n, &n))
161  return SMTP_ERR_CODE;
162 
163  } while (buf[3] == '-');
164 
165  if (smtp_success(n) || (n == SMTP_CONTINUE))
166  return 0;
167 
168  mutt_error(_("SMTP session failed: %s"), buf);
169  return -1;
