303const struct Command *command_find_by_id(const struct CommandArray *ca, enum CommandId id)
304{
305 if (!ca || (id == CMD_NONE))
306 return NULL;
307
308 const struct Command **cp = NULL;
309 ARRAY_FOREACH(cp, ca)
310 {
311 const struct Command *cmd = *cp;
312
313 if (cmd->id == id)
314 return cmd;
315 }
316
317 return NULL;
318}
319
328const struct Command *command_find_by_name(const struct CommandArray *ca, const char *name)
329{
330 if (!ca || !name)
331 return NULL;
332
333 const struct Command **cp = NULL;
334 ARRAY_FOREACH(cp, ca)
335 {
336 const struct Command *cmd = *cp;
337
338 if (mutt_str_equal(cmd->name, name))
339 {
340 // If this is a synonym, look up the real Command
341 if ((cmd->flags & CF_SYNONYM) && cmd->data)
342 {
343 const char *real_name = (const char *) cmd->data;
344 // Search for the real command (non-recursive, single pass)
345 const struct Command **rp = NULL;
346 ARRAY_FOREACH(rp, ca)
347 {
348 const struct Command *real_cmd = *rp;
349 if (mutt_str_equal(real_cmd->name, real_name) && !(real_cmd->flags & CF_SYNONYM))
350 return real_cmd;
351 }
352 return NULL; // Real command not found
353 }
354 return cmd;
355 }
356 }
357
358 return NULL;
