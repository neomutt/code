75static StoreHandle *store_rocksdb_open(const char *path)
76{
77 if (!path)
78 return NULL;
79
80 struct RocksDbStoreData *sdata = rocksdb_sdata_new();
81
82 /* RocksDB stores errors in form of strings */
83 sdata->err = NULL;
84
85 /* setup generic options, create new db and limit log to one file */
86 sdata->options = rocksdb_options_create();
87 rocksdb_options_set_create_if_missing(sdata->options, 1);
88 rocksdb_options_set_keep_log_file_num(sdata->options, 1);
89
90 /* setup read options, we verify with checksums */
91 sdata->read_options = rocksdb_readoptions_create();
92 rocksdb_readoptions_set_verify_checksums(sdata->read_options, 1);
93
94 /* setup write options, no sync needed, disable WAL */
95 sdata->write_options = rocksdb_writeoptions_create();
96 rocksdb_writeoptions_set_sync(sdata->write_options, 0);
97 rocksdb_writeoptions_disable_WAL(sdata->write_options, 1);
98
99 rocksdb_options_set_compression(sdata->options, rocksdb_no_compression);
100
101 /* open database and check for error in sdata->error */
102 sdata->db = rocksdb_open(sdata->options, path, &sdata->err);
103 if (sdata->err)
104 {
105 rocksdb_options_destroy(sdata->options);
106 rocksdb_readoptions_destroy(sdata->read_options);
107 rocksdb_writeoptions_destroy(sdata->write_options);
108 rocksdb_sdata_free(&sdata);
109 return NULL;
110 }
111
112 // Return an opaque pointer
113 return (StoreHandle *) sdata;
114}
